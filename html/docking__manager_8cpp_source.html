1030 bool DockingManager::scan_charging_station(sensor_msgs::msg::LaserScan scan)
1031 {
1032  bool result = true;
1033  pcl::PointCloud<pcl::PointXYZ> scan_lidar =
1034  geometry_calculation_lib::to_pcl(robot_base_frame_id_, scan, *tf_buffer_);
1035 
1036  double x = 0.0;
1037  double y = 0.0;
1038  try {
1039  auto robot_base_to_marker = tf_buffer_->lookupTransform(
1040  robot_base_frame_id_, marker_frame_id_, rclcpp::Time(0),
1041  rclcpp::Duration::from_seconds(1));
1042  x = robot_base_to_marker.transform.translation.x;
1043  y = robot_base_to_marker.transform.translation.y;
1044  } catch (tf2::TransformException & ex) {
1045  RCLCPP_ERROR_STREAM(
1046  this->get_logger(),
1047  "Failed to get the TF : " << ex.what());
1048  return false;
1049  }
1050 
1051  // Outlier
1052  scan_lidar = geometry_calculation_lib::statistical_outlier_removal(
1053  scan_lidar, icp_.outlier_mean_k_, icp_.outlier_seddev_multi_);
1054  if (scan_lidar.size() == 0) {
1055  return false;
1056  }
1057  // Pass Through
1058  scan_lidar = filtering_lidar_range(scan_lidar, x, y);
1059  if (scan_lidar.size() == 0) {
1060  return false;
1061  }
1062  // Voxel Grid
1063  scan_lidar =
1064  geometry_calculation_lib::voxel_grid(scan_lidar, icp_.marker_leaf_size_);
1065  if (scan_lidar.size() == 0) {
1066  return false;
1067  }
1068  // Moving Least Square
1069  scan_lidar = geometry_calculation_lib::moving_least_square(
1070  scan_lidar, icp_.mls_search_radius_, icp_.mls_upsampling_radius_,
1071  icp_.mls_upsampling_step_size_);
1072  if (scan_lidar.size() == 0) {
1073  return false;
1074  }
1075 
1076  auto station_cloud_ptr =
1077  std::make_shared<pcl::PointCloud<pcl::PointXYZ>>(get_station_template());
1078  auto marker_pose_ptr = std::make_shared<geometry_msgs::msg::Pose>();
1079 
1080  rclcpp::Time icp_start = now();
1081  double score = icp_.score_threshold_ + 1.0;
1082  if (icp_.multi_search_) {
1083  score = geometry_calculation_lib::multi_iterative_closest_point(
1084  scan_lidar, station_cloud_ptr, marker_pose_ptr, icp_.search_range_m_,

1661 std::vector<Waypoint> DockingManager::heuristic_path(
1662  std::vector<float> & wp_x,
1663  std::vector<float> & wp_y)
1664 {
1665  std::vector<int> n_wp;
1666  for (int i = 0; i < static_cast<int>(wp_x.size() - 1); i++) {
1667  float d = std::sqrt(
1668  std::pow(wp_x[i + 1] - wp_x[i], 2) +
1669  std::pow(wp_y[i + 1] - wp_y[i], 2));
1670  n_wp.push_back(static_cast<int>(d / pt_.path_resolution_));
1671  }
1672 
1673  // straight path
1674  std::vector<float> wp_xs, wp_ys;
1675  for (int i = 0; i < static_cast<int>(wp_x.size() - 1); i++) {
1676  std::vector<float> seg_wp_x = this->linspace(wp_x[i], wp_x[i + 1], n_wp[i]);
1677  wp_xs.insert(wp_xs.end(), seg_wp_x.begin(), seg_wp_x.end());
1678  std::vector<float> seg_wp_y = this->linspace(wp_y[i], wp_y[i + 1], n_wp[i]);
1679  wp_ys.insert(wp_ys.end(), seg_wp_y.begin(), seg_wp_y.end());
1680  }
1681 
1682  // smooth path
1683  std::vector<float> wp_xs_sm;
1684  std::vector<float> wp_ys_sm;
1685  for (int wp_id = 0; wp_id < static_cast<int>(wp_xs.size() - 1); wp_id++) {
1686  if (wp_id < smoothing_dist_) {
1687  wp_xs_sm.push_back(wp_xs[wp_id]);
1688  wp_ys_sm.push_back(wp_ys[wp_id]);
1689  } else if (wp_id > static_cast<int>(wp_xs.size() - 1) - smoothing_dist_) {
1690  wp_xs_sm.push_back(wp_xs[wp_id]);
1691  wp_ys_sm.push_back(wp_ys[wp_id]);
1692  } else {
1693  wp_xs_sm.push_back(
1694  mean(wp_xs, wp_id - smoothing_dist_, wp_id + smoothing_dist_));
1695  wp_ys_sm.push_back(
1696  mean(wp_ys, wp_id - smoothing_dist_, wp_id + smoothing_dist_));
1697  }
1698  }
1699 
1700  // wrapping in waypoint class
1701  std::vector<Waypoint> waypoints;
1702  for (int wp_id = 0; wp_id < static_cast<int>(wp_xs_sm.size() - 1); wp_id++) {
1703  float x_curr = wp_xs_sm[wp_id];
1704  float x_next = wp_xs_sm[wp_id + 1];
1705  float y_curr = wp_ys_sm[wp_id];
1706  float y_next = wp_ys_sm[wp_id + 1];
1707  const float dx_ahead = x_next - x_curr;
1708  const float dy_ahead = y_next - y_curr;
1709  float theta = std::atan2(dy_ahead, dx_ahead);
1710  const float dist_ahead = std::hypot(dx_ahead, dy_ahead);
1711  float kappa;
1712  if (wp_id == 0) {
1713  kappa = 0;
1714  } else {
1715  float x_prev = wp_xs_sm[wp_id - 1];
1716  float y_prev = wp_ys_sm[wp_id - 1];
1717  float dx_behind = x_curr - x_prev;
1718  float dy_behind = y_curr - y_prev;
1719  float angle_behind = std::atan2(dy_behind, dx_behind);
1720  float angle_diff = fmod(theta - angle_behind + M_PI, 2 * M_PI) - M_PI;
1721  kappa = angle_diff / dist_ahead;
1722  }
1723  waypoints.push_back(Waypoint(x_curr, y_curr, theta, kappa, wp_id));
1724  }
1725  return waypoints;
1726 }
