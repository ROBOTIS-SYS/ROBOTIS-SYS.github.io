55  radius * cos(installed_angle), radius * sin(installed_angle), installed_angle,
56  params_.base_radius);
57 
58  // Back Right (ultrasonic4)
59  installed_angle = -M_PI * 3.0 / 4.0;
60  ultrasonic_back_right_.set_params(
61  radius * cos(installed_angle), radius * sin(installed_angle), installed_angle,

99 SafetyFilter::FilterResult SafetyFilter::filtering_cmd_vel(
100  geometry_msgs::msg::TwistStamped raw_data)
101 {
102  FilterResult filtered_result;
103  geometry_msgs::msg::TwistStamped filtered_cmd_vel;
104 
105  // Init
106  filtered_result.cmd_vel = raw_data;
107  filtered_result.decelration = false;
108  filtered_result.clash_time = params_.check_time;
109  filtered_result.deceleration_rate = 1.0;
110 
111  if (safety_enable_ &&
112  (filtered_result.cmd_vel.twist.linear.x != 0.0 ||
113  filtered_result.cmd_vel.twist.angular.z != 0.0))
114  {
115  // Intersection of circles
116  double v = filtered_result.cmd_vel.twist.linear.x;
117  double w = filtered_result.cmd_vel.twist.angular.z;
118 
119  if (v >= 0.0) { // Front Side
120  // front
121  UltrasonicData::ClashResult result_f =
122  ultrasonic_front_.get_min_clash_time(v, w, params_.check_time);
123  if (params_.check_time > result_f.clash_time) {
124  filtered_result.detected_sensor_data.insert(
125  std::make_pair(
126  "F-" + result_f.clash_circle_name,
127  ultrasonic_front_.range_));
128  std::cout <<
129  " F-" << result_f.clash_circle_name << ":" << result_f.clash_time <<
130  "(" << ultrasonic_front_.range_ << ")";
131  }
132  // front right
133  UltrasonicData::ClashResult result_fr =
134  ultrasonic_front_right_.get_min_clash_time(v, w, params_.check_time);
135  if (params_.check_time > result_fr.clash_time) {
136  filtered_result.detected_sensor_data.insert(
137  std::make_pair(
138  "FR-" + result_fr.clash_circle_name,
139  ultrasonic_front_right_.range_));
140  std::cout <<
141  "FR-" << result_fr.clash_circle_name << ":" << result_fr.clash_time <<
142  "(" << ultrasonic_front_right_.range_ << ")";
143  }
144  // front left
145  UltrasonicData::ClashResult result_fl =
146  ultrasonic_front_left_.get_min_clash_time(v, w, params_.check_time);
147  if (params_.check_time > result_fl.clash_time) {
148  filtered_result.detected_sensor_data.insert(
149  std::make_pair(
150  "FL-" + result_fl.clash_circle_name,
151  ultrasonic_front_left_.range_));
152  std::cout <<
153  " FL-" << result_fl.clash_circle_name << ":" << result_fl.clash_time <<
154  "(" << ultrasonic_front_left_.range_ << ")";
155  }
156  filtered_result.clash_time =
157  std::min(result_f.clash_time, std::min(result_fr.clash_time, result_fl.clash_time));
158  } else if (v < 0.0) { // Back Side
159  // Back
160  UltrasonicData::ClashResult result_b =
161  ultrasonic_back_.get_min_clash_time(v, w, params_.check_time);
162  if (params_.check_time > result_b.clash_time) {
163  filtered_result.detected_sensor_data.insert(
164  std::make_pair(
165  "B-" + result_b.clash_circle_name,
166  ultrasonic_back_.range_));
167  std::cout <<
168  " B-" << result_b.clash_circle_name << ":" << result_b.clash_time <<
169  "(" << ultrasonic_back_.range_ << ")";
170  }
171  // back right
172  UltrasonicData::ClashResult result_br =
173  ultrasonic_back_right_.get_min_clash_time(v, w, params_.check_time);
174  if (params_.check_time > result_br.clash_time) {
175  filtered_result.detected_sensor_data.insert(
176  std::make_pair(
177  "BR-" + result_br.clash_circle_name,
178  ultrasonic_back_right_.range_));
179  std::cout <<
180  " BR-" << result_br.clash_circle_name << ":" << result_br.clash_time <<
181  "(" << ultrasonic_back_right_.range_ << ")";
182  }
183  // back left
184  UltrasonicData::ClashResult result_bl =
185  ultrasonic_back_left_.get_min_clash_time(v, w, params_.check_time);
186  if (params_.check_time > result_bl.clash_time) {
187  filtered_result.detected_sensor_data.insert(
188  std::make_pair(
189  "BL-" + result_bl.clash_circle_name,
190  ultrasonic_back_left_.range_));
191  std::cout <<
192  " BL-" << result_bl.clash_circle_name << ":" << result_bl.clash_time <<
193  "(" << ultrasonic_back_left_.range_ << ")";
194  }
195  filtered_result.clash_time =
196  std::min(result_b.clash_time, std::min(result_br.clash_time, result_bl.clash_time));
197  }
198 
199  // Deccelation
200  if (params_.stop_time >= filtered_result.clash_time) {
201  std::cout << " <" << v << "," << w << ">->";
202  v = 0.0;
203  w = 0.0;
204  filtered_result.deceleration_rate = 0.0;
205  filtered_result.decelration = true;
206  std::cout << " <" << v << "," << w << ">" << std::endl;
207  } else if (params_.check_time > filtered_result.clash_time) {
208  std::cout << " <" << v << "," << w << ">->";
209  v = (filtered_result.clash_time / params_.check_time) * v;
210  w = (filtered_result.clash_time / params_.check_time) * w;
211  filtered_result.deceleration_rate = filtered_result.clash_time / params_.check_time;
212  filtered_result.decelration = true;
213  std::cout << " <" << v << "," << w << ">" << std::endl;
214  }
215  filtered_result.cmd_vel.twist.linear.x = v;
216  filtered_result.cmd_vel.twist.angular.z = w;
217  }
218  return filtered_result;
219 }
