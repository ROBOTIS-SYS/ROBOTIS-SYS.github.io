84  def perform(self, context: launch.LaunchContext) -> Text:
85  yaml_filename = launch.utilities.perform_substitutions(context, self.name)

108  def substitute_params(self, yaml, param_rewrites):
109  # substitute leaf-only parameters
110  for key in self.getYamlLeafKeys(yaml):
111  if key.key() in param_rewrites:
112  raw_value = param_rewrites[key.key()]
113  key.setValue(self.convert(raw_value))
114 
115  # substitute total path parameters
116  yaml_paths = self.pathify(yaml)
117  for path in yaml_paths:
118  if path in param_rewrites:
119  # this is an absolute path (ex. 'key.keyA.keyB.val')
120  rewrite_val = param_rewrites[path]
121  yaml_keys = path.split('.')
122  yaml = self.updateYamlPathVals(yaml, yaml_keys, rewrite_val)

125  def updateYamlPathVals(self, yaml, yaml_key_list, rewrite_val):
126  for key in yaml_key_list:
127  if key == yaml_key_list[-1]:
128  yaml[key] = rewrite_val
129  break
130  key = yaml_key_list.pop(0)
131  yaml[key] = self.updateYamlPathVals(yaml.get(key, {}), yaml_key_list, rewrite_val)
132 
133  return yaml

135  def substitute_keys(self, yaml, key_rewrites):
136  if len(key_rewrites) != 0:
137  for key, val in yaml.items():
138  if isinstance(val, dict) and key in key_rewrites:
139  new_key = key_rewrites[key]
140  yaml[new_key] = yaml[key]
141  del yaml[key]
142  self.substitute_keys(val, key_rewrites)

153  def pathify(self, d, p=None, paths=None, joinchar='.'):
154  if p is None:
155  paths = {}
156  self.pathify(d, "", paths, joinchar=joinchar)
157  return paths
158  pn = p
159  if p != "":
160  pn += '.'
161  if isinstance(d, dict):
162  for k in d:
163  v = d[k]
164  self.pathify(v, pn + k, paths, joinchar=joinchar)
165  elif isinstance(d, list):
166  for idx, e in enumerate(d):
167  self.pathify(e, pn + str(idx), paths, joinchar=joinchar)
168  else:
169  paths[p] = d

171  def convert(self, text_value):
172  if self.__convert_types:
173  # try converting to int or float
174  try:
175  return float(text_value) if '.' in text_value else int(text_value)
176  except ValueError:
177  pass
178 
179  # try converting to bool
180  if text_value.lower() == "true":
181  return True
182  if text_value.lower() == "false":
183  return False
184 
185  # nothing else worked so fall through and return text
186  return text_value
