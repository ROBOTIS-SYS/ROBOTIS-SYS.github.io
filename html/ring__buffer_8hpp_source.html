127  bool dequeue()
128  {
129  if (empty_) {
130  return false;
131  } else {
132  ++l_;
133  if (l_ == arr_.size()) {

144  bool enqueue(const T & value)
145  {
146  if (is_full()) {
147  dequeue();

152  if (r_ == arr_.size()) {r_ = 0;}
153  return true;

158  std::vector<T> rtn;

160  return rtn;
161  }
162 
163  if (r_ > l_) {
164  // resize return vector
165  // rtn.resize(r_ - l_);
166  rtn.insert(rtn.begin(), arr_.begin() + l_, arr_.begin() + r_);
167  } else { // l_ >= r_
168  rtn.insert(rtn.begin(), arr_.begin() + l_, arr_.end());
169  rtn.insert(rtn.end(), arr_.begin(), arr_.begin() + r_);
170  }
171  return rtn;
172  }
