25 TaskWaypoint FkSolverRecursive::recursive(
26  const JointValues joint_values,
27  const ComponentMap::const_iterator current_it)
28 {
29  const ConnectedComponent & component = current_it->second;
30  TaskValues from_parent;
31 
32  if (component.component.get_joint().get_type() != Joint::Fixed) {
33  from_parent = TaskValues(
34  component.component.get_pose(
35  joint_values.position(
36  component.num_in_joint_array)),
37  component.component.get_twist(
38  joint_values.position(component.num_in_joint_array),
39  joint_values.velocity(component.num_in_joint_array)));
40  } else {
41  from_parent = TaskValues(
42  component.component.get_pose(0.0),
43  component.component.get_twist(0.0, 0.0));
44  }
45 
46  if (current_it == robot_model_.get_base_component()) {
47  return from_parent;
48  } else {
49  ComponentMap::const_iterator parent_it = component.parent;
50  TaskWaypoint result = recursive(joint_values, parent_it, from_parent.pose) * from_parent;
51  return result;
52  }
53 }
