33 void PageWebcall::user_imread_thread()
34 {
35  std::cout << "user imread thread start" << std::endl;
36  cv::VideoCapture cap("/dev/video0");
37  cap.set(cv::CAP_PROP_FRAME_WIDTH, 320);
38  cap.set(cv::CAP_PROP_FRAME_HEIGHT, 240);
39  if (!cap.isOpened()) {
40  std::cerr << "(User)카메라를 열 수 없습니다." << std::endl;
41  user_thread_running = false;
42  }
43  cv::Mat frame;
44  QImage imgIn;
45  while (user_thread_running) {
46  cap >> frame;
47  if (frame.empty()) {
48  std::cerr << "(User)프레임을 읽을 수 없습니다." << std::endl;
49  user_thread_running = false;
50  } else {
51  // cv::imshow("카메라 영상", frame);
52  imgIn =
53  QImage(
54  reinterpret_cast<uchar *>(frame.data), frame.cols, frame.rows, frame.step,
55  QImage::Format_RGB888);
56  ui->lblCamUser->setPixmap(QPixmap::fromImage(imgIn.rgbSwapped()));
57  }
58  }
59  cap.release();
60  ui->lblCamUser->clear();
61  // cv::destroyAllWindows();
62  std::cout << "user imread thread end" << std::endl;
63 }
