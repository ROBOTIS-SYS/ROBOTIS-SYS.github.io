78  typedef enum RobotStatus
79  {
80  GO_TO_EV = 0b10000000, // 층 내 이동, 로봇이 승강장으로 이동 중
81  WAITING_EV = 0b01000000, // 탑승대기, 로봇이 승강장에서 콜 요청
82  ENTERING_EV = 0b00100000, // 탑승 중, 로봇이 엘리베이터 탑승 중
83  TAKE_ON_EV = 0b00010000, // 탑승완료, 로봇이 엘리베이터 탑승 완료
84  EXITING_EV = 0b00001000, // 하차 중, 로봇이 엘리베이터에서 하차 중
85  GET_OFF_EV = 0b00000100, // 하차완료, 로봇이 엘리베이터에서 하차완료
86  TAKING_ON_CHANGE = 0b00000010,
87  // 탑승변경, 로봇이 엘리베이터의 제어 및 통신상태가 정상임에도 불구하고
88  // 탑승 불가상태 또는 다른 호기를 탑승해야 하는 경우
89  CHARGING = 0b00000001 // 충전 중, 로봇의 배터리 충전 중인 상태
90  } robot_status;
