58  bool wait_for_twist(
59  rclcpp::Executor & executor,
60  const std::chrono::milliseconds & timeout = std::chrono::milliseconds(500))
61  {
62  rclcpp::WaitSet wait_set;
63  wait_set.add_subscription(velocity_command_subscriber_);
64 
65  if (wait_set.wait(timeout).kind() == rclcpp::WaitResultKind::Ready) {
66  executor.spin_some();
67  return true;
68  }
69  return false;

94  void publish(double linear, double angular)
95  {
96  int wait_count = 0;
97  auto topic = velocity_publisher->get_topic_name();
98  while (pub_node->count_subscribers(topic) == 0) {
99  if (wait_count >= 5) {
100  auto error_msg = std::string("publishing to ") + topic + " but no node subscribes to it";
101  throw std::runtime_error(error_msg);
102  }
103  std::this_thread::sleep_for(std::chrono::milliseconds(100));
104  ++wait_count;
105  }
106 
107  geometry_msgs::msg::TwistStamped velocity_message;
108  velocity_message.header.stamp = pub_node->get_clock()->now();
109  velocity_message.twist.linear.x = linear;
110  velocity_message.twist.angular.z = angular;
111  velocity_publisher->publish(velocity_message);
112  }
113 
115  void waitForSetup()
116  {
117  constexpr std::chrono::seconds TIMEOUT{2};
118  auto clock = pub_node->get_clock();
119  auto start = clock->now();
120  while (velocity_publisher->get_subscription_count() <= 0) {
121  if ((clock->now() - start) > TIMEOUT) {
122  FAIL();
123  }
124  rclcpp::spin_some(pub_node);
125  }
126  }

255 TEST_F(TestAimbotDiffDriveController, cleanup)
256 {
257  const auto ret = controller_->init(controller_name);
258  ASSERT_EQ(ret, controller_interface::return_type::OK);
259 
260  controller_->get_node()->set_parameter(
261  rclcpp::Parameter("left_wheel_names", rclcpp::ParameterValue(left_wheel_names)));
262  controller_->get_node()->set_parameter(
263  rclcpp::Parameter("right_wheel_names", rclcpp::ParameterValue(right_wheel_names)));
264  controller_->get_node()->set_parameter(rclcpp::Parameter("wheel_separation", 0.4));
265  controller_->get_node()->set_parameter(rclcpp::Parameter("wheel_radius", 0.1));
266 
267  rclcpp::executors::SingleThreadedExecutor executor;
268  executor.add_node(controller_->get_node()->get_node_base_interface());
269  auto state = controller_->configure();
270  ASSERT_EQ(State::PRIMARY_STATE_INACTIVE, state.id());
271  assignResources();
272 
273  state = controller_->activate();
274  ASSERT_EQ(State::PRIMARY_STATE_ACTIVE, state.id());
275 
276  waitForSetup();
277 
278  // send msg
279  const double linear = 1.0;
280  const double angular = 1.0;
281  publish(linear, angular);
282  controller_->wait_for_twist(executor);
283 
284  ASSERT_EQ(controller_->update(), controller_interface::return_type::OK);
285 
286  state = controller_->deactivate();
287  ASSERT_EQ(State::PRIMARY_STATE_INACTIVE, state.id());
288  ASSERT_EQ(controller_->update(), controller_interface::return_type::OK);
289 
290  state = controller_->cleanup();
291  ASSERT_EQ(State::PRIMARY_STATE_UNCONFIGURED, state.id());
292 
293  // should be stopped
294  EXPECT_EQ(0.0, left_wheel_vel_cmd_.get_value());
295  EXPECT_EQ(0.0, right_wheel_vel_cmd_.get_value());
296 
297  executor.cancel();
298 }

300 TEST_F(TestAimbotDiffDriveController, correct_initialization_using_parameters)
301 {
302  const auto ret = controller_->init(controller_name);
303  ASSERT_EQ(ret, controller_interface::return_type::OK);
304 
305  controller_->get_node()->set_parameter(
306  rclcpp::Parameter("left_wheel_names", rclcpp::ParameterValue(left_wheel_names)));
307  controller_->get_node()->set_parameter(
308  rclcpp::Parameter("right_wheel_names", rclcpp::ParameterValue(right_wheel_names)));
309  controller_->get_node()->set_parameter(rclcpp::Parameter("wheel_separation", 0.4));
310  controller_->get_node()->set_parameter(rclcpp::Parameter("wheel_radius", 1.0));
311 
312  rclcpp::executors::SingleThreadedExecutor executor;
313  executor.add_node(controller_->get_node()->get_node_base_interface());
314 
315  auto state = controller_->configure();
316  assignResources();
317 
318  ASSERT_EQ(State::PRIMARY_STATE_INACTIVE, state.id());
319  EXPECT_EQ(0.01, left_wheel_vel_cmd_.get_value());
320  EXPECT_EQ(0.02, right_wheel_vel_cmd_.get_value());
321 
322  state = controller_->activate();
323  ASSERT_EQ(State::PRIMARY_STATE_ACTIVE, state.id());
324 
325  // send msg
326  const double linear = 1.0;
327  const double angular = 0.0;
328  publish(linear, angular);
329  // wait for msg is be published to the system
330  ASSERT_TRUE(controller_->wait_for_twist(executor));
331 
332  ASSERT_EQ(controller_->update(), controller_interface::return_type::OK);
333  EXPECT_EQ(1.0, left_wheel_vel_cmd_.get_value());
334  EXPECT_EQ(1.0, right_wheel_vel_cmd_.get_value());
335 
336  // deactivated
337  // wait so controller process the second point when deactivated
338  std::this_thread::sleep_for(std::chrono::milliseconds(500));
339  state = controller_->deactivate();
340  ASSERT_EQ(state.id(), State::PRIMARY_STATE_INACTIVE);
341  ASSERT_EQ(controller_->update(), controller_interface::return_type::OK);
342 
343  EXPECT_EQ(0.0, left_wheel_vel_cmd_.get_value()) << "Wheels are halted on deactivate()";
344  EXPECT_EQ(0.0, right_wheel_vel_cmd_.get_value()) << "Wheels are halted on deactivate()";
345 
346  // cleanup
347  state = controller_->cleanup();
348  ASSERT_EQ(State::PRIMARY_STATE_UNCONFIGURED, state.id());
349  EXPECT_EQ(0.0, left_wheel_vel_cmd_.get_value());
350  EXPECT_EQ(0.0, right_wheel_vel_cmd_.get_value());
351 
352  state = controller_->configure();
353  ASSERT_EQ(State::PRIMARY_STATE_INACTIVE, state.id());
354  executor.cancel();
355 }
