96 bool Visualizer::get_closest_patch_image(
97  std::string button_label,
98  std::shared_ptr<std::map<std::string, ButtonPose>> btn_poses,
99  std::shared_ptr<std::vector<RecognizedText>> btns,
100  cv::cuda::GpuMat & out_img,
101  cv::cuda::GpuMat & whole_img,
102  float btn_len_x,
103  float btn_len_y)
104 {
105  CameraGeometry * geo = CameraGeometry::get_instance();
106  Eigen::Matrix3f to_optical = geo->get_cam_to_opt().transpose();
107  if (btn_poses->find(button_label) != btn_poses->end()) {
108  Eigen::Vector3f btn_cen = to_optical * btn_poses->at(button_label).center;
109  Eigen::Matrix<float, 3, 4> proj_mat =
110  geo->get_cam_proj_mat(CameraGeometry::left_zed);
111  Eigen::Vector2f left_reproj_pt =
112  geo->project_to_image(proj_mat, btn_cen);
113  int left_closest_id = find_closest_button(left_reproj_pt, btns);
114  if (left_closest_id > -1) {
115  out_img = btns->at(left_closest_id).d_patch_img;
116  return true;
117  // } else {
118  // return false;
119  // }
120  } else {
121  Eigen::Vector3f btn_nor = to_optical * btn_poses->at(button_label).normal;
122  Eigen::Matrix3f rot = convert_to_quat(btn_nor).normalized().toRotationMatrix();
123  std::vector<Eigen::Vector2f> proj_bbox;
124  Eigen::Vector3f p1 = rot * Eigen::Vector3f(btn_len_x, btn_len_y, 0) + btn_cen;
125  Eigen::Vector3f p2 = rot * Eigen::Vector3f(-btn_len_x, btn_len_y, 0) + btn_cen;
126  Eigen::Vector3f p3 = rot * Eigen::Vector3f(-btn_len_x, -btn_len_y, 0) + btn_cen;
127  Eigen::Vector3f p4 = rot * Eigen::Vector3f(btn_len_x, -btn_len_y, 0) + btn_cen;
128  proj_bbox.push_back(geo->project_to_image(proj_mat, p1));
129  proj_bbox.push_back(geo->project_to_image(proj_mat, p2));
130  proj_bbox.push_back(geo->project_to_image(proj_mat, p3));
131  proj_bbox.push_back(geo->project_to_image(proj_mat, p4));
132 
133  if (compensate_for_missing_patch_image(
134  proj_bbox, whole_img, out_img))
135  {
136  return true;
137  } else {
138  return false;
139  }
140  }
141  } else {
142  return false;
143  }
144 }

209 int Visualizer::find_closest_button(
210  Eigen::Vector2f pt,
211  std::shared_ptr<std::vector<RecognizedText>> btns)
212 {
213  for (size_t i = 0; i < btns->size(); i++) {
214  int x_diff = btns->at(i).features.at(0).x - btns->at(i).features.at(2).x;
215  int y_diff = btns->at(i).features.at(0).y - btns->at(i).features.at(2).y;
216  float radius = static_cast<float>(sqrt(x_diff * x_diff + y_diff * y_diff));
217  cv::Point2i cv_btn_cen = btns->at(i).features.at(2);
218  Eigen::Vector2f btn_cen(cv_btn_cen.x, cv_btn_cen.y);
219  float dist = (pt - btn_cen).norm();
220  if (dist < radius) {
221  return i;
222  }
223  }
224  return -1;
225 }

227 bool Visualizer::compensate_for_missing_patch_image(
228  std::vector<Eigen::Vector2f> crop_pts,
229  cv::cuda::GpuMat & whole_img,
230  cv::cuda::GpuMat & out_img)
231 {
232  if (&whole_img == nullptr || crop_pts.size() == 0) {return false;}
233  CameraGeometry * geo = CameraGeometry::get_instance();
234  int img_res_x = geo->get_cam_img_size(CameraGeometry::left_zed).width;
235  int img_res_y = geo->get_cam_img_size(CameraGeometry::left_zed).height;
236 
237  int margin = 0;
238  int margin_x = static_cast<int>(crop_pts[2](0) - margin);
239  int margin_y = static_cast<int>(crop_pts[2](1) - margin);
240  int margin_width = static_cast<int>((crop_pts[2] - crop_pts[3]).norm()) + margin * 2;
241  int margin_height = static_cast<int>((crop_pts[2] - crop_pts[1]).norm()) + margin * 2;
242 
243  // boundary check
244  if (margin_x > img_res_x - 1) {return false;}
245  if (margin_y > img_res_y - 1) {return false;}
246  if (margin_x < 0) {return false;}
247  if (margin_y < 0) {return false;}
248  if (margin_width + margin_x > img_res_x) {margin_width = img_res_x - 1 - margin_x;}
249  if (margin_height + margin_y > img_res_y) {margin_height = img_res_y - 1 - margin_y;}
250  if (margin_width < 0) {return false;}
251  if (margin_height < 0) {return false;}
252 
253  // std::cout << "x: " << margin_x << " y: " <<
254  // margin_y << " w: " << margin_width << " h: " << margin_height << std::endl;
255  cv::Rect margin_bbox = cv::Rect(margin_x, margin_y, margin_width, margin_height);
256  cv::cuda::resize(whole_img(margin_bbox), out_img, cv::Size(224, 224), 0, 0);
257 
258  return true;
259 }

261 std::vector<Eigen::Vector3f> Visualizer::create_btn_pts(
262  int btn_shape,
263  float btn_len_x,
264  float btn_len_y)
265 {
266  std::vector<Eigen::Vector3f> spatial_btn_pts;
267  int circle_resol = 50;

269  switch (btn_shape) {

272  spatial_btn_pts.push_back(Eigen::Vector3f(-btn_len_x, btn_len_y, 0));

276  case 1:
277  for (int i = 0; i < circle_resol; i++) {
278  float theta = PerceptionUtils::pi_ / circle_resol * i * 2;
279  spatial_btn_pts.push_back(
280  Eigen::Vector3f(
281  btn_len_x * cos(theta), btn_len_y * sin(theta),
282  0));
283  }
284  break;
285  case 2:
286  // float length = btn_len_x;
287  // float radius = btn_len_y;
288  // for (int i = 0; i < circle_resol + 1; i++) {
289  // float theta = PerceptionUtils::pi_ / circle_resol * i;
290  // spatial_btn_pts.push_back(
291  // Eigen::Vector3f(
292  // length / 2 + radius * sin(theta), radius * cos(theta), 0));
293  // }
294  // for (int i = 0; i < circle_resol + 1; i++) {
295  // float theta = PerceptionUtils::pi_ / circle_resol * i;
296  // spatial_btn_pts.push_back(
297  // Eigen::Vector3f(
298  // - length / 2 - radius * sin(theta), - radius * cos(theta), 0));

325 void Visualizer::overlay_btns_to_stereo_image(
326  std::shared_ptr<std::map<std::string, ButtonPose>> btn_pose,
327  cv::Mat & overlaid_img,
328  Eigen::Matrix<float, 3, 4> left_proj_mat,
329  Eigen::Matrix<float, 3, 4> right_proj_mat,
330  std::vector<Eigen::Vector3f> btn_pts,
331  std::string trg_label,
332  bool trg_state)
333 {
334  CameraGeometry * geo = CameraGeometry::get_instance();
335  Eigen::Matrix3f to_optical = geo->get_cam_to_opt().transpose();
336 
337 #pragma omp parallel num_threads(8)
338  {
339 #pragma omp for
340  for (size_t i = 0; i < btn_pose->size(); i++) {
341  auto it = btn_pose->begin();
342  std::advance(it, i);
343  std::string label = it->first;
344  bool state = (label == trg_label) ? trg_state : false;
345 
346  Eigen::Vector3f btn_cen = to_optical * it->second.center;
347  Eigen::Vector3f btn_nor = to_optical * it->second.normal;
348  Eigen::Matrix3f rot = convert_to_quat(btn_nor).normalized().toRotationMatrix();
349 
350  std::vector<cv::Point2i> left_btn_pts;
351  std::vector<cv::Point2i> right_btn_pts;
352  for (size_t j = 0; j < btn_pts.size(); j++) {
353  Eigen::Vector3f cam_pt = rot * btn_pts[j] + btn_cen;
354  Eigen::Vector2f left_proj_pt = geo->project_to_image(left_proj_mat, cam_pt);
355  Eigen::Vector2f right_proj_pt = geo->project_to_image(right_proj_mat, cam_pt);
356 
357  if (check_bounds(left_proj_pt)) {
358  left_btn_pts.push_back(cv::Point2i(left_proj_pt(0), left_proj_pt(1)));
359  }
360  if (check_bounds(right_proj_pt)) {
361  right_btn_pts.push_back(
362  cv::Point2i(
363  right_proj_pt(0),
364  right_proj_pt(1) + geo->get_cam_img_size(CameraGeometry::left_zed).height));
365  }
366  }
367 
368  Eigen::Vector2f left_text_pos = geo->project_to_image(left_proj_mat, btn_cen);
369  Eigen::Vector2f right_text_pos = geo->project_to_image(right_proj_mat, btn_cen);
370 
371  if (left_btn_pts.size() == btn_pts.size()) {
372  draw_btns_with_convex_poly(
373  overlaid_img,
374  label,
375  cv::Point2i(left_text_pos(0), left_text_pos(1)),
376  left_btn_pts,
377  state);
378  }
379 
380  if (right_btn_pts.size() == btn_pts.size()) {
381  draw_btns_with_convex_poly(
382  overlaid_img,
383  label,
384  cv::Point2i(
385  right_text_pos(0),
386  right_text_pos(1) + geo->get_cam_img_size(CameraGeometry::left_zed).height),
387  right_btn_pts,
388  state);
389  }
390  }
391  }
392 }

395  std::shared_ptr<std::map<std::string, ButtonPose>> btn_pose,
396  cv::Mat & overlaid_img,
397  Eigen::Matrix<float, 3, 4> proj_mat,

408  std::vector<cv::Point2i> proj_pts;

427 Eigen::Quaternionf Visualizer::convert_to_quat(Eigen::Vector3f vec)
428 {
429  Eigen::Vector3f up_vec(0.0, 0.0, -1.0);
430  Eigen::Vector3f norm_vec = vec.normalized();
431 
432  Eigen::Vector3f right_axis_vector = up_vec.cross(norm_vec);
433  float theta = norm_vec.dot(up_vec);
434  float angle = acos(theta);
435 
436  float x = right_axis_vector(0) * sin(0.5 * angle);
437  float y = right_axis_vector(1) * sin(0.5 * angle);
438  float z = right_axis_vector(2) * sin(0.5 * angle);
439  float w = cos(0.5 * angle);
440 
441  return Eigen::Quaternionf(w, x, y, z);
442 }

475 {

485  cv::vconcat(merged_image, target_image, merged_image);
486  } else {
487  cv::resize(
488  target_image,
489  target_image,
490  cv::Size(
491  static_cast<int>(target_image.cols * merged_image.cols / target_image.cols),
492  static_cast<int>(target_image.rows * merged_image.cols / target_image.cols)));
493  image_area.push_back(
494  cv::Rect(0, merged_image.rows, target_image.cols, target_image.rows));
495  cv::vconcat(merged_image, target_image, merged_image);
496  }
497  } else {
498  if (target_image.rows == merged_image.rows) {
499  image_area.push_back(
500  cv::Rect(merged_image.cols, 0, target_image.cols, target_image.rows));
501  cv::hconcat(merged_image, target_image, merged_image);
502  } else {
503  cv::resize(
504  target_image,
505  target_image,
506  cv::Size(
507  static_cast<int>(target_image.cols * merged_image.rows / target_image.rows),
508  static_cast<int>(target_image.rows * merged_image.rows / target_image.rows)));
509  image_area.push_back(
510  cv::Rect(merged_image.cols, 0, target_image.cols, target_image.rows));
511  cv::hconcat(merged_image, target_image, merged_image);
512  }
513  }
514  }

517 bool Visualizer::check_bounds(Eigen::Vector2f pt)
518 {
519  CameraGeometry * cb = CameraGeometry::get_instance();
520  float height = static_cast<float>(cb->get_cam_img_size(CameraGeometry::left_zed).height);
521  float width = static_cast<float>(cb->get_cam_img_size(CameraGeometry::left_zed).width);
522  float u = pt(0);
523  float v = pt(1);
524 
525  if (u >= 0 && u < width) {
526  if (v >= 0 && v < height) {
527  return true;
528  } else {
529  return false; // invaild v value
530  }
531  } else {
532  return false; // invaild u value
533  }
534 }

649 void Visualizer::visualize_cliff_state(
650  cv::Mat & result_img,
651  cv::Mat & trajectory_mat_inv,
652  bool cliff_state,
653  float max_width,
654  int max_width_index,
655  const std::vector<std::vector<TrjPoint>> & cliff_detect_view_trajectories)

659  auto point = cliff_detect_view_trajectories[i][j].point;
660  auto color = cliff_detect_view_trajectories[i][j].color;
661 
662  int x = static_cast<int>(point.x);
663  int y = static_cast<int>(point.y);
664 
665  if (x >= 0 && x < result_img.cols && y >= 0 && y < result_img.rows) {
666  if (cliff_state && i == max_width_index) {
667  color = cv::Scalar(255, 255, 0);
668  }
669  cv::circle(result_img, point, 1, color, 1, 8, 0);
670  }
671  }
672  }
673 
674  if (cliff_state) {
675  cv::Point textPosition(result_img.cols / 4, result_img.rows / 4);
676  std::string max_wid = std::to_string(max_width);
677  std::string message = "Detected Cliff Zone " + max_wid;
678  cv::Scalar textColor(255);
679  int fontFace = cv::FONT_HERSHEY_SIMPLEX;
680  double fontScale = 0.6;
681  int thickness = 2;
682  cv::putText(result_img, message, textPosition, fontFace, fontScale, textColor, thickness);
683  }
684 
685  cv::Mat mask = (trajectory_mat_inv == 1);
686 
687  trajectory_mat_inv.convertTo(trajectory_mat_inv, CV_8UC1);
688  cv::cvtColor(trajectory_mat_inv, trajectory_mat_inv, cv::COLOR_GRAY2RGB);
689 
690  std::vector<cv::Mat> channels;
691  cv::split(trajectory_mat_inv, channels);
692 
693  channels[0].setTo(153);
694  channels[1].setTo(255, mask);
695  channels[2].setTo(255, mask);
696 
697  cv::merge(channels, trajectory_mat_inv);
698  trajectory_mat_inv = trajectory_mat_inv * 0.2;
699  result_img = result_img + trajectory_mat_inv;
700 
701  cv::resize(
702  result_img,
703  result_img,
704  cv::Size(
705  result_img.cols * 0.5,
706  result_img.rows * 0.5));
707 }
