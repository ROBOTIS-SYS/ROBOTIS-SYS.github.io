48  min_velocity_ = -max_velocity_;
49  }
50  }

53  throw std::runtime_error(

55  }
56  if (std::isnan(min_acceleration_)) {
57  min_acceleration_ = -max_acceleration_;
58  }
59  }

63  }

92 double SpeedLimiter::limit_acceleration(double & v, double v0, double dt)
93 {
94  const double tmp = v;
95 
96  if (has_acceleration_limits_) {
97  const double dv_min = min_acceleration_ * dt;
98  const double dv_max = max_acceleration_ * dt;
99 
100  const double dv = rcppmath::clamp(v - v0, dv_min, dv_max);
101 
102  v = v0 + dv;
103  }
104 
105  return tmp != 0.0 ? v / tmp : 1.0;
106 }

108 double SpeedLimiter::limit_jerk(double & v, double v0, double v1, double dt)
109 {
110  const double tmp = v;
111 
112  if (has_jerk_limits_) {
113  const double dv = v - v0;
114  const double dv0 = v0 - v1;
115 
116  const double dt2 = 2. * dt * dt;
117 
118  const double da_min = min_jerk_ * dt2;
119  const double da_max = max_jerk_ * dt2;
120 
121  const double da = rcppmath::clamp(dv - dv0, da_min, da_max);
122 
123  v = v0 + dv0 + da;
124  }
125 
126  return tmp != 0.0 ? v / tmp : 1.0;
127 }
