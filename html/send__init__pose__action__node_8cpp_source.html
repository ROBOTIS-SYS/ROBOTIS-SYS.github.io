49 BT::NodeStatus SendInitPoseAction::tick()
50 {
51  std::cout << std::endl;
52  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
53  RCLCPP_INFO_STREAM(get_logger(), "<TICK> SendInitPoseAction");
54  }
55 
56  geometry_msgs::msg::Pose2D init_pose;
57  try {
58  if (!getInput<geometry_msgs::msg::Pose2D>("init_pose", init_pose)) {
59  publish_tree_status(name(), "위치 설정에 실패했습니다.");
60  RCLCPP_ERROR(get_logger(), "failed to get init pose : " + name());
61  return BT::NodeStatus::SUCCESS;
62  }
63 
64  // init publisher
65  std::string topic_name = "initialpose";
66  if (send_init_pose_pub_ == nullptr) {
67  bool on_blackboard =
68  config().blackboard->get<decltype(send_init_pose_pub_)>(
69  topic_name, send_init_pose_pub_);
70 
71  if (on_blackboard == false) {
72  send_init_pose_pub_ =
73  node_->create_publisher<geometry_msgs::msg::PoseWithCovarianceStamped>(
74  topic_name, 10);
75  config().blackboard->set<decltype(send_init_pose_pub_)>(
76  topic_name, send_init_pose_pub_);
77  }
78  }
79 
80  send_init(init_pose);
81  } catch (std::exception & ex) {
82  RCLCPP_ERROR_STREAM(get_logger(), "Error : " << ex.what());
83  return BT::NodeStatus::FAILURE;
84  }
85  return BT::NodeStatus::SUCCESS;
86 }

88 void SendInitPoseAction::send_init(const geometry_msgs::msg::Pose2D & init_pose)
89 {
90  publish_tree_status(name(), "층 이동을 위해 초기 위치를 새로 설정합니다.");
91 
92  // publish initpose
93 
94  geometry_msgs::msg::PoseWithCovarianceStamped init_msg;
95 
96  // header
97  init_msg.header.frame_id = "map_current";
98  init_msg.header.stamp = node_->get_clock()->now();
99 
100  // position
101  init_msg.pose.pose.position.x = init_pose.x;
102  init_msg.pose.pose.position.y = init_pose.y;
103 
104  // orientation
105  tf2::Quaternion quat;
106  quat.setRPY(0.0, 0.0, init_pose.theta);
107  tf2::convert(quat, init_msg.pose.pose.orientation);
108 
109  // covariance
110  init_msg.pose.covariance.at(0) = 0.01 * 0.01;
111  init_msg.pose.covariance.at(7) = 0.01 * 0.01;
112  init_msg.pose.covariance.at(35) = (M_PI / 120.0) * (M_PI / 120.0);
113 
114  // publish
115  send_init_pose_pub_->publish(init_msg);
116  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
117  RCLCPP_INFO_STREAM(
118  get_logger(),
119  "send init pose - x : " << init_pose.x << ", y : " << init_pose.y << ", theta : " <<
120  init_pose.theta);
121  }
122 }
