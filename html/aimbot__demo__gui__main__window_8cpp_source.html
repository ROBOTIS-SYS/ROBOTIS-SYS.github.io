250 void AimbotDemoGUIMainWindow::changePage(std::string page, std::string option)
251 {
252  RCLCPP_INFO(
253  ros_node->get_logger(), "changePage : (page:%s)(option:%s)",
254  page.c_str(), option.c_str());
255 
256  ros_node->publish_log(
257  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
258  aimbot_log_interfaces::msg::Log::TYPE_INFO,
259  "GUI",
260  "changePage() requested [page:" + page + "] [option:" + option + "]");
261 
262  if (prevent_page_change == true) {
263  RCLCPP_INFO(ros_node->get_logger(), "changePage : popup prevents page changes");
264  return;
265  }
266 
267  // 옵션으로 task_info 가 넘어올 경우 처리
268  if (page.compare("Waiting") == 0 || page.compare("WaitingWith") == 0) {
269  if (is_task_info_option(option)) {
270  save_last_task_info(option);
271  option = "";
272  }
273  }
274 
275  // Arrived_0 , Arrived_1 , Arrived_2 에 대한 처리. 그냥 Arrived 면 처리 안함.
276  std::string word = "Arrived";
277  if (page.compare(0, word.length(), word) == 0) {
278  QStringList list = QString(page.c_str()).split("_");
279  if (list.size() > 1) {
280  page = list[0].toStdString();
281 
282  if (option == "CallWaiting") {
283  option = list[1].toStdString();
284  } else {
285  option += list[1].toStdString();
286  }
287  }
288  }
289 
290  // page 가 없으면 그냥 return
291  auto find_it = page_map.find(page);
292  if (find_it == page_map.end()) {
293  RCLCPP_ERROR(ros_node->get_logger(), "Page(%s) is not exist!", page.c_str());
294  return;
295  }
296 
297  // page 가 curr_page 와 같으면 option 만 적용
298  if (curr_page == page_map[page]) {
299  if (option != "") {
300  emit optionChanged(option);
301  }
302  return;
303  }
304 
305  // E-STOP 이 눌려 있으면 특정 페이지만 띄워 줌. (설정, 패스워드 입력, 비상정지화면)
306  if (page.compare("Language") != 0 &&
307  page.compare("InputPassNumber") != 0 &&
308  curr_page == page_map["Warning"] &&
309  ros_node->estop_status_ == true)
310  {
311  RCLCPP_WARN(
312  ros_node->get_logger(),
313  "changePage : The E-STOP button is pressed. Can't change to page [%s] ",
314  page.c_str());
315  return;
316  }
317 
318  set_curr_page(page);
319 
320  if (option != "") {
321  emit optionChanged(option);
322  }
323 }

325 void AimbotDemoGUIMainWindow::changeLanguage(std::string lang)
326 {
327  bool ret = translator.load(
328  ("aimbot_demo_gui_" + lang + ".qm").c_str(),
329  (ament_index_cpp::get_package_prefix("aimbot_demo_gui_g18") + "/translations").c_str());
330 
331  if (ret) {
332  QApplication::installTranslator(&translator);
333  } else {
334  QApplication::removeTranslator(&translator);
335  RCLCPP_WARN(
336  ros_node->get_logger(),
337  "translator : (%s) qm file load failed. set will be default language.",
338  lang.c_str());
339  }
340 
341  ros_node->curr_language_ = lang;
342 
343  ui->retranslateUi(this);
344  curr_page->hide();
345  curr_page->show();
346 }
