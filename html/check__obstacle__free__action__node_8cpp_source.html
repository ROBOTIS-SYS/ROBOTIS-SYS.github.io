61  tf_buffer_ = std::make_shared<tf2_ros::Buffer>(node_->get_clock());
62  config().blackboard->set<std::shared_ptr<tf2_ros::Buffer>>("tf_buffer", tf_buffer_);

65  // tf listener
66  if (tf_listener_ == nullptr) {
67  bool on_blackboard = config().blackboard->get<std::shared_ptr<tf2_ros::TransformListener>>(
68  "tf_listener", tf_listener_);
69  if (on_blackboard == false) {
70  tf_listener_ = std::make_shared<tf2_ros::TransformListener>(*tf_buffer_);
71  config().blackboard->set<std::shared_ptr<tf2_ros::TransformListener>>(

83 BT::NodeStatus CheckObstacleFreeAction::tick()
84 {
85  std::cout << std::endl;
86 
87  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
88  RCLCPP_INFO_STREAM(get_logger(), "<TICK> CheckObstacleAction");
89  }
90 
91  // check input port
92  double check_time;
93  geometry_msgs::msg::Pose2D target_pose;
94  std::string map_frame, lidar_frame;
95  if (!getInput<double>("check_time", check_time)) {
96  check_time = 1.0;
97  }
98  if (
99  !getInput<double>("check_radius", check_radius_) ||
100  !getInput<double>("threshold", threshold_) ||
101  !getInput<geometry_msgs::msg::Pose2D>("check_point", target_pose) ||
102  !getInput<std::string>("map_frame", map_frame) ||
103  !getInput<std::string>("lidar_frame", lidar_frame))
104  {
105  RCLCPP_ERROR(get_logger(), "Missing input port items");
106  return BT::NodeStatus::FAILURE;
107  }
108 
109  is_free_ = false;
110  local_check_point_.setZero(2);
111 
112  // transform check point to local(lidar_link)
113  geometry_msgs::msg::Pose2D local_target;
114 
115  bool result = get_local_target_pose(target_pose, map_frame, lidar_frame, local_target);
116  if (isHaltRequested()) {
117  return BT::NodeStatus::IDLE;
118  }
119  if (result == false) {
120  RCLCPP_ERROR(get_logger(), "Failed to transform the target pose");
121  return BT::NodeStatus::FAILURE;
122  }
123 
124  local_check_point_.coeffRef(0) = local_target.x;
125  local_check_point_.coeffRef(1) = local_target.y;
126 
127  // s -> ms
128  check_time *= 1000.0;
129  double loop_rate = 100.0; // ms
130  int loop_count = 0;
131 
132  rclcpp::WallRate loopRate(std::chrono::milliseconds(100));
133 
134  while (rclcpp::ok() && is_free_ == false && isHaltRequested() == false) {
135  // sleep
136  loopRate.sleep();
137 
138  if ((loop_rate * ++loop_count) > check_time) {
139  break;
140  }
141  }
142 
143  if (isHaltRequested()) {
144  return BT::NodeStatus::IDLE;
145  }
146 
147  if (is_free_ == false) {
148  publish_tree_status(name(), "주어진 위치에 장애물이 있습니다.");
149  return BT::NodeStatus::FAILURE;
150  }
151 
152  publish_tree_status(name(), "주어진 위치에 장애물이 없습니다.");
153  return BT::NodeStatus::SUCCESS;
154 }

156 bool CheckObstacleFreeAction::get_local_target_pose(
157  const geometry_msgs::msg::Pose2D & target, const std::string & map_frame_id,
158  const std::string & robot_frame_id, geometry_msgs::msg::Pose2D & local_target)
159 {
160  try {
161  geometry_msgs::msg::TransformStamped base_link_transform;
162 
163  base_link_transform = tf_buffer_->lookupTransform(
164  robot_frame_id, map_frame_id, rclcpp::Time(0), rclcpp::Duration::from_seconds(0.5));
165 
166  auto dur = node_->get_clock()->now() - base_link_transform.header.stamp;
167  RCLCPP_WARN_STREAM(
168  get_logger(),
169  "[" << name() << "] Transform diff : " << dur.seconds() << "s");
170 
171  geometry_msgs::msg::Pose current_pose;
172  current_pose.position.x = base_link_transform.transform.translation.x;
173  current_pose.position.y = base_link_transform.transform.translation.y;
174  current_pose.position.z = base_link_transform.transform.translation.z;
175  current_pose.orientation = base_link_transform.transform.rotation;
176 
177  std::cout << "current x : " << current_pose.position.x <<
178  ", y : " << current_pose.position.y << ", theta : " <<
179  (tf2::getYaw(current_pose.orientation) * 180.0 / M_PI) << std::endl;
180 
181  // set before pose
182  geometry_msgs::msg::PoseStamped before_pose, after_pose;
183  before_pose.pose.position.x = target.x;
184  before_pose.pose.position.y = target.y;
185  // orientation
186  tf2::Quaternion before_q;
187  before_q.setRPY(0.0, 0.0, target.theta);
188  tf2::convert(before_q, before_pose.pose.orientation);
189 
190  // transform
191  tf2::doTransform(before_pose, after_pose, base_link_transform);
192 
193  // set result to pose2d
194  local_target.x = after_pose.pose.position.x;
195  local_target.y = after_pose.pose.position.y;
196  local_target.theta = tf2::getYaw(after_pose.pose.orientation);
197  } catch (tf2::TransformException & ex) {
198  RCLCPP_ERROR(get_logger(), "Failed to get the local pose");
199 
200  return false;
201  }
202 
203 
204  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
205  RCLCPP_INFO_STREAM(
206  get_logger(), "Obstacle point : " << local_target.x << ", " << local_target.y);
207  }
208 
209  return true;
210 }

212 bool CheckObstacleFreeAction::check_obstacle(const sensor_msgs::msg::LaserScan & scan)
213 {
214  if (local_check_point_.norm() == 0) {
215  return false;
216  }
217 
218  // get param
219  double min_angle = -M_PI, max_angle = M_PI;
220  double max_range = 5.0;
221  int dimension = 2;
222  int obstacle_count = 0;
223  int free_count = 0;
224 
225  // range filter : angle
226  bool include_zero = false;
227  for (size_t ix = 0; ix < scan.ranges.size(); ix++) {
228  double angle = scan.angle_min + scan.angle_increment * static_cast<double>(ix);
229  if (angle < min_angle || angle > max_angle) {
230  continue;
231  }
232 
233  if (isnan(scan.ranges.at(ix)) || scan.ranges.at(ix) >= max_range) {
234  continue;
235  }
236 
237  // change to point
238  Eigen::VectorXd point(dimension);
239  point.coeffRef(0) = scan.ranges.at(ix) * cos(angle);
240  point.coeffRef(1) = scan.ranges.at(ix) * sin(angle);
241 
242  Eigen::VectorXd diff = local_check_point_ - point;
243  if (diff.norm() < check_radius_) {
244  obstacle_count++;
245  } else {
246  free_count++;
247  }
248  }
249 
250  if (obstacle_count > threshold_) {
251  RCLCPP_WARN_STREAM(
252  get_logger(), "Free Count : " << free_count << ", Obstacle Count : " << obstacle_count);
253  return false;
254  }
255 
256 
257  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
258  RCLCPP_INFO_STREAM(
259  get_logger(), "Free Count : " << free_count << ", Obstacle Count : " << obstacle_count);
260  }
261  return true;
262 }
