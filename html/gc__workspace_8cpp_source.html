28 bool GcWorkspace::configure()
29 {
30  RCLCPP_INFO_STREAM(node_->get_logger(), "Ready to Workspace check module.");
31 
32  // ros param
33  // node_->declare_parameter<std::string>(ros_param_root_ + "srv_name", "/workspace");
34  // node_->declare_parameter<double>(ros_param_root_ + "base_size", 0.4);
35  // node_->declare_parameter<double>(ros_param_root_ + "joint5_remap_rate", 0.5);
36 
37  // Robot Model
38  rm2_robot_model::RobotModel model = robot_handler_->get_robot_model();
39  for (auto it = model.get_component("manipulator_base");
40  it != model.get_base_component();
41  it = it->second.parent)
42  {
43  if (it->second.component.get_joint().get_type() != rm2_robot_model::Joint::RotAxis ||
44  it->second.component.get_joint().get_type() != rm2_robot_model::Joint::TransAxis)
45  {
46  base_pose_ = it->second.component.get_pose(0.0) * base_pose_;
47  }
48  }
49  RCLCPP_INFO_STREAM(
50  node_->get_logger(), "Manipulator base pose : " << base_pose_.position.data.transpose());
51  joint1 = robot_handler_->get_robot_model().get_component("link1")->second.component;
52  joint2 = robot_handler_->get_robot_model().get_component("link2")->second.component;
53  joint3 = robot_handler_->get_robot_model().get_component("link3")->second.component;
54  joint4 = robot_handler_->get_robot_model().get_component("link4")->second.component;
55  joint5 = robot_handler_->get_robot_model().get_component("link5")->second.component;
56  tip = robot_handler_->get_robot_model().get_component("tip")->second.component;
57  knock_tip = robot_handler_->get_robot_model().get_component("knock_tip")->second.component;
58 
59  // Data useful to calculate IK
60  link2_length_ = joint3.get_joint().get_origin_from_parent().position.norm();
61  link3_length_ = joint4.get_joint().get_origin_from_parent().position.norm();
62  delta_joint3_ = atan2(
63  joint4.get_joint().get_origin_from_parent().position.z(),
64  sqrt(
65  joint4.get_joint().get_origin_from_parent().position.x() *
66  joint4.get_joint().get_origin_from_parent().position.x() +
67  joint4.get_joint().get_origin_from_parent().position.y() *
68  joint4.get_joint().get_origin_from_parent().position.y()));
69  RCLCPP_INFO_STREAM(node_->get_logger(), "Success to load robot model.");
70 
71  // Base Distance XY
72  base_distance_ = sqrt(
73  base_pose_.position.x() * base_pose_.position.x() +
74  base_pose_.position.y() * base_pose_.position.y());
75  RCLCPP_INFO_STREAM(node_->get_logger(), "Manipulator base distance : " << base_distance_);
76 
77  // base Size
78  base_size_ =
79  node_->get_parameter(ros_param_root_ + "base_size").as_double();
80  RCLCPP_INFO_STREAM(node_->get_logger(), "Mobile Base size is '" << base_size_ << "'.");
81 
82  // min xy distance
83  min_xy_distance_ = sqrt(
84  base_size_ * base_size_ -
85  base_pose_.position.x() * base_pose_.position.x() -
86  base_pose_.position.y() * base_pose_.position.y());
87  RCLCPP_INFO_STREAM(node_->get_logger(), "minimum xy distance is '" << min_xy_distance_ << "'.");
88 
89  // max distance
90  double max_distance = sqrt(
91  link2_length_ * link2_length_ +
92  link3_length_ * link3_length_ -
93  2 * link2_length_ * link3_length_ *
94  cos(M_PI_2 - joint3.get_joint().get_min_position() + delta_joint3_));
95  RCLCPP_INFO_STREAM(node_->get_logger(), "Maximum distance is '" << max_distance << "'.");
96 
97  max_z_distance_ = sqrt(max_distance * max_distance - min_xy_distance_ * min_xy_distance_);
98  RCLCPP_INFO_STREAM(node_->get_logger(), "Maximum Z distance is '" << max_z_distance_ << "'.");
99 
100  // Service server
101  std::string srv_name = node_->get_parameter(ros_param_root_ + "srv_name").as_string();
102  using namespace std::placeholders;
103  workspace_srv_ = node_->create_service<manipulator_interfaces::srv::GcWorkspace>(
104  std::string(node_->get_name()) + srv_name,
105  std::bind(&GcWorkspace::check_workspace_callback, this, _1, _2));
106  RCLCPP_INFO_STREAM(node_->get_logger(), "Create service '" << srv_name << "'.");

114 // }

117 // ref_min_distance_ = sqrt(

120 // cos(M_PI_2 + delta_joint3_ - max_joint3_angle));

129 // }
130 // RCLCPP_INFO_STREAM(node_->get_logger(), "Minimum Joint3 angle is '"
131 // << min_joint3_angle << "'.");
132 // ref_max_distance_ = sqrt(
133 // link2_length_ * link2_length_ + link3_length_ * link3_length_ -

135 // cos(M_PI_2 + delta_joint3_ - min_joint3_angle));
136 // RCLCPP_INFO_STREAM(
137 // node_->get_logger(), "Maximum Reference Distance is '" << ref_max_distance_ << "'.");
138 
139 
140  // joint5 remap rate
141  joint5_remap_rate_ = node_->get_parameter(ros_param_root_ + "joint5_remap_rate").as_double();
142  if (joint5_remap_rate_ >= 1.0) {

199 void GcWorkspace::check_workspace_callback(
200  const std::shared_ptr<manipulator_interfaces::srv::GcWorkspace::Request> request,
201  std::shared_ptr<manipulator_interfaces::srv::GcWorkspace::Response> response)
202 {
203  RCLCPP_INFO_STREAM(node_->get_logger(), "Received request to check workspace.");
204 
205  if (robot_handler_->is_debug_mode()) {
206  // Publish Goal pose for visualization.
207  if (goal_pose_pub_->get_subscription_count() > 0) {
208  geometry_msgs::msg::PoseStamped goal_pose_msg;
209  goal_pose_msg.header.stamp = node_->now();
210  goal_pose_msg.header.frame_id = request->header.frame_id;
211  goal_pose_msg.pose = request->pose;
212  goal_pose_pub_->publish(goal_pose_msg);
213  RCLCPP_INFO_STREAM(node_->get_logger(), "Goal Pose Pub");
214  }
215  }
216 
217  // Convert target according to frame and time
218  rclcpp::Time target_ref_time = request->header.stamp;
219  std::string target_ref_frame = request->header.frame_id;
220  rm2_common::TaskValues target_value
221  ((rm2_common::Pose(
222  rm2_common::Vector(
223  request->pose.position.x,
224  request->pose.position.y,
225  request->pose.position.z),
226  rm2_common::Rotation(
227  request->pose.orientation.w,
228  request->pose.orientation.x,
229  request->pose.orientation.y,
230  request->pose.orientation.z)))
231  );
232  RCLCPP_INFO_STREAM(
233  node_->get_logger(),
234  "Reference time : " << target_ref_time.seconds() << ", " <<
235  "Reference frame name : " << target_ref_frame);
236 
237  // Calculation goal pose from root
238  rm2_common::TaskValues target_pose_from_root;
239  if (target_ref_frame == "" ||
240  target_ref_frame == robot_handler_->get_robot_model().get_base_component()->first)
241  {
242  target_pose_from_root = target_value;
243  } else if (robot_handler_->get_robot_model().check_component(target_ref_frame)) {
244  rm2_common::TaskValues standard_frame_pose_from_root =
245  robot_handler_->get_present_task_value(target_ref_time.seconds(), target_ref_frame);
246  RCLCPP_INFO_STREAM(
247  node_->get_logger(),
248  "Reference frame pose from root : " <<
249  standard_frame_pose_from_root.pose.position.data.transpose());
250  target_pose_from_root = standard_frame_pose_from_root * target_value;
251  } else {
252  RCLCPP_INFO_STREAM(
253  node_->get_logger(),
254  "Reference frame name '" << target_ref_frame << "' is not included in robot model.");
255  response->success = false;
256  return;
257  }
258 
259  // Reset target orientation
260  if (request->component_name == "tip") {
261  double goal_roll, goal_pitch, goal_yaw;
262  rm2_common::Rotation(
263  target_pose_from_root.pose.orientation.w(),
264  target_pose_from_root.pose.orientation.x(),
265  target_pose_from_root.pose.orientation.y(),
266  target_pose_from_root.pose.orientation.z()).get_rpy(
267  goal_roll, goal_pitch, goal_yaw);
268  target_pose_from_root.pose.orientation = rm2_common::Rotation::RPY(0.0, 0.0, goal_yaw);
269  }
270 
271  // Delta Pose
272  rm2_common::Pose delta_pose
273  (rm2_common::Vector(
274  request->delta_position.x,
275  request->delta_position.y,
276  request->delta_position.z));
277  rm2_common::Pose second_target_pose = target_pose_from_root.pose * delta_pose;
278 
279  if (robot_handler_->is_debug_mode()) {
280  // Publish Goal pose for visualization.
281  if (second_goal_pose_pub_->get_subscription_count() > 0) {
282  geometry_msgs::msg::PoseStamped second_goal_pose_msg;
283  second_goal_pose_msg.header.stamp = node_->now();
284  second_goal_pose_msg.header.frame_id =
285  robot_handler_->get_robot_model().get_base_component()->first;
286  second_goal_pose_msg.pose = transport_pose_to_msg(second_target_pose);
287  second_goal_pose_pub_->publish(second_goal_pose_msg);
288  RCLCPP_INFO_STREAM(node_->get_logger(), "Second Goal Pose Pub");
289  }
290  }
291 
292  // Calculation
293  rm2_common::Pose root_pose;
294  rm2_common::JointValues target_joint_value(robot_handler_->get_dof());
295  bool moved = false;
296  if (!check_workspace(
297  request->component_name,
298  second_target_pose, root_pose,
299  target_joint_value, moved))
300  {
301  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to calcaulate delta pose workspace.");
302  response->success = false;
303  return;
304  }
305 
306  moved = false;
307  if (!check_workspace(
308  request->component_name,
309  target_pose_from_root.pose, root_pose,
310  target_joint_value, moved))
311  {
312  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to Calcaulate target psoe workspace.");
313  response->success = false;
314  return;
315  }
316 
317  moved = false;
318  if (!check_workspace(
319  request->component_name,
320  second_target_pose, root_pose,
321  target_joint_value, moved))
322  {
323  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to calcaulate delta pose workspace.");
324  response->success = false;
325  return;
326  }
327 
328  if (moved) {
329  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to calcaulate delta pose workspace.");
330  response->success = false;
331  return;
332  }
333 
334 
335  if (!moved) {
336  // Root base
337  Pose2D mobile_distance;
338  mobile_distance.x = root_pose.position.x();
339  mobile_distance.y = root_pose.position.y();
340  double roll, pitch;
341  root_pose.orientation.get_rpy(roll, pitch, mobile_distance.theta);
342 
343  // Filnal moved target Pose
344  rm2_common::Pose moved_pose;
345  if (request->component_name == "tip") {
346  moved_pose = base_pose_ *
347  joint1.get_pose(target_joint_value.position(0)) *
348  joint2.get_pose(target_joint_value.position(1)) *
349  joint3.get_pose(target_joint_value.position(2)) *
350  joint4.get_pose(target_joint_value.position(3)) *
351  joint5.get_pose(target_joint_value.position(4)) *
352  tip.get_pose(0.0);
353  } else if (request->component_name == "knock_tip") {
354  moved_pose = base_pose_ *
355  joint1.get_pose(target_joint_value.position(0)) *
356  joint2.get_pose(target_joint_value.position(1)) *
357  joint3.get_pose(target_joint_value.position(2)) *
358  joint4.get_pose(target_joint_value.position(3)) *
359  joint5.get_pose(target_joint_value.position(4)) *
360  knock_tip.get_pose(0.0);
361  }
362 
363  response->success = true;
364  response->distance = mobile_distance;
365  response->frame_id = robot_handler_->get_robot_model().get_base_component()->first;
366  response->moved_pose = transport_pose_to_msg(moved_pose);
367  }
368 }

370 bool GcWorkspace::check_workspace(
371  std::string component_name, rm2_common::Pose target_pose,
372  rm2_common::Pose & root_pose,
373  rm2_common::JointValues & target_joint_value, bool moved)
374 {
375  // print_pose("base position", base_pose_);
376  // print_pose("goal position", target_pose_from_root.pose);
377 
378  int cnt = 0;
379  while (true) {
380  // cnt
381  if (cnt > 10) {
382  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to Calcaulate work space.");
383  return false;
384  } else {cnt++;}
385 
386  // std::cout << "-------------------------[ " << cnt << " ]-------------------------" <<
387  // std::endl;
388 
389  // trans root to manipulator base
390  rm2_common::Pose goal_pose_from_mani_base =
391  base_pose_.inverse() * root_pose.inverse() * target_pose;
392  // std::cout << "goal position from manipulator base : " <<
393  // goal_pose_from_mani_base.position.data.transpose() << std::endl;
394 
395  // Link5 Pose
396  rm2_common::Pose link5_pose;
397  if (component_name == "tip") {
398  link5_pose = goal_pose_from_mani_base * tip.get_pose(0.0).inverse();
399  tip_length_ = tip.get_pose(0.0).position.norm();
400  // print_pose("tip", tip.get_pose(0.0));
401  } else if (component_name == "knock_tip") {
402  link5_pose = goal_pose_from_mani_base * knock_tip.get_pose(0.0).inverse();
403  tip_length_ = knock_tip.get_pose(0.0).position.norm();
404  // print_pose("knock_tip", knock_tip.get_pose(0.0));
405  } else {
406  RCLCPP_ERROR_STREAM(
407  node_->get_logger(), "[IK Solver Failure] please set tip name 'knock_tip' to 'tip'.");
408  return false;
409  }
410  // print_pose("link5_pose", link5_pose);
411 
412  // Joint 1 angle
413  target_joint_value.position(0) = atan2(link5_pose.position.y(), link5_pose.position.x());
414  // std::cout << "joint1 angle : " << target_joint_value.position(0) << std::endl;
415  if (!joint1.check_joint_limit(target_joint_value.position(0), 0.0, 0.0)) {
416  rm2_common::Pose link5_pose_from_root = base_pose_ * link5_pose;
417  // print_pose("link5_pose_from_root", link5_pose_from_root);
418  // new root pose
419  double theta = atan2(link5_pose_from_root.position.y(), link5_pose_from_root.position.x());
420  rm2_common::Pose base_move0 =
421  rm2_common::Pose(
422  rm2_common::Vector(
423  -base_size_ - link5_pose_from_root.position.x(),
424  0.0,
425  0.0),
426  rm2_common::Rotation::Yaw(theta));
427  // print_pose("base_move", base_move0);
428  // print_pose("root_pose", root_pose);
429  root_pose = root_pose * base_move0;
430  // print_pose("new root_pose", root_pose);
431  // std::cout << "\t--> Joint1 Limit " << target_joint_value.position(0) <<
432  // " -> rotate " << theta * 180.0 / M_PI << " deg." << std::endl;
433  // std::cout << "\t--> The target is 180 degrees behind the base. -> trans " <<
434  // base_move0.position.x() << ", " << base_move0.position.y() << std::endl;
435  RCLCPP_INFO_STREAM(node_->get_logger(), "[WorkSpace] joint1 limit.");
436  moved = true;
437  continue;
438  }
439 
440  // Joint2 Pose
441  rm2_common::Pose joint2_pose =
442  joint1.get_pose(target_joint_value.position(0)) *
443  joint2.get_joint().get_origin_from_parent();
444  // print_pose("joint2_pose", joint2_pose);
445 
446  // Pose from joint2 to link5
447  rm2_common::Pose pose_form_joint2_to_link5 = joint2_pose.inverse() * link5_pose;
448  // print_pose("pose_form_joint2_to_link5", pose_form_joint2_to_link5);
449 
450  // Gc Manipulator tip RPY
451  // pitch + yaw
452  double tip_roll, tip_pitch, tip_yaw;
453  if (!gc_manipulator_orientation(
454  pose_form_joint2_to_link5.orientation,
455  tip_roll, tip_pitch, tip_yaw))
456  {
457  RCLCPP_ERROR_STREAM(
458  node_->get_logger(), "[IK Solver Failure] Wrong goal orientation. (" <<
459  tip_roll << ",\t " << tip_pitch << ",\t " << tip_yaw << ")\n" <<
460  pose_form_joint2_to_link5.orientation.data);
461  return false;
462  }
463 
464  // Joint5 angle
465  target_joint_value.position(4) = tip_yaw;
466  // std::cout << "joint5 angle : " << target_joint_value.position(4) << std::endl;
467  if (!joint5.check_joint_limit(target_joint_value.position(4), 0.0, 0.0)) {
468  rm2_common::Pose goal_pose_from_root = root_pose.inverse() * target_pose;
469  rm2_common::Pose base_move1 = goal_pose_from_root *
470  (rm2_common::Rotation::Yaw(tip_yaw * joint5_remap_rate_) * goal_pose_from_root.inverse());
471  // print_pose("base_move", base_move1);
472 
473  // new root pose
474  root_pose = root_pose * base_move1;
475  // std::cout << "\t--> Joint5 Limit -> trans " <<
476  // base_move1.position.x() << ", " << base_move1.position.y() <<
477  // " / rotate " << tip_yaw * joint5_remap_rate_ * 180.0 / M_PI << " deg." << std::endl;
478  RCLCPP_INFO_STREAM(node_->get_logger(), "[WorkSpace] joint5 limit.");
479  moved = true;
480  continue;
481  }
482 
483 
484  // link4 Pose
485  rm2_common::Pose link4_pose =
486  link5_pose * joint5.get_pose(target_joint_value.position(4)).inverse();
487  // print_pose("link4_pose", link4_pose);
488 
489  // Link4 from Root
490  rm2_common::Pose link4_pose_from_root = base_pose_ * link4_pose;
491  // print_pose("link4_pose_from_root", link4_pose_from_root);
492  double xy_distance_from_root = sqrt(
493  link4_pose_from_root.position.x() * link4_pose_from_root.position.x() +
494  link4_pose_from_root.position.y() * link4_pose_from_root.position.y());
495  // std::cout << "xy_distance_from_root : " << xy_distance_from_root << std::endl;
496  // std::cout << "base_size_ : " << base_size_ << std::endl;
497 
498  // Pose from joint2 to joint4
499  rm2_common::Pose pose_from_joint2_to_link4 = joint2_pose.inverse() * link4_pose;
500  // print_pose("pose_from_joint2_to_link4", pose_from_joint2_to_link4);
501 
502  if (xy_distance_from_root < base_size_ && pose_from_joint2_to_link4.position.z() < 0.0) {
503  rm2_common::Pose base_move2 =
504  rm2_common::Pose(
505  ((xy_distance_from_root - base_size_ - tip_length_ ) / xy_distance_from_root) *
506  rm2_common::Vector(
507  link4_pose_from_root.position.x(),
508  link4_pose_from_root.position.y(),
509  0.0));
510  // std::cout <<
511  // "(xy_distance_from_root - base_size_ - tip_length_) / xy_distance_from_root : " <<
512  // (xy_distance_from_root - base_size_ - tip_length_ ) / xy_distance_from_root << std::endl;
513  // print_pose("base_move", base_move2);
514 
515  // new root pose
516  root_pose = root_pose * base_move2;
517  // std::cout << "\t--> Distance too close than base size -> trans " <<
518  // base_move2.position.x() << ", " << base_move2.position.y() << std::endl;
519  RCLCPP_INFO_STREAM(node_->get_logger(), "[WorkSpace] base size limit.");
520  moved = true;
521  continue;
522  }
523 
524  if (max_z_distance_ < fabs(pose_from_joint2_to_link4.position.z())) {
525  RCLCPP_ERROR_STREAM(
526  node_->get_logger(),
527  "Can not approach the target highet '" << pose_from_joint2_to_link4.position.z() <<
528  "'. maximum is '" << max_z_distance_ << "'.");
529  return false;
530  }
531 
532  double alpha = atan2(
533  pose_from_joint2_to_link4.position.z(),
534  sqrt(
535  pose_from_joint2_to_link4.position.x() * pose_from_joint2_to_link4.position.x() +
536  pose_from_joint2_to_link4.position.y() * pose_from_joint2_to_link4.position.y()));
537  // std::cout << "alpha: " << alpha * 180 / M_PI << std::endl;
538 
539  // Calculate beta angle
540  // Beta : angle between goal position vector and link2
541  // l3^2 = d^2 + l2^2 - 2*d*l2*cos(beta)
542  double distance = pose_from_joint2_to_link4.position.norm();
543  // std::cout << "distance : " << distance << std::endl;
544 
545  double xy_distance = sqrt(
546  pose_from_joint2_to_link4.position.x() * pose_from_joint2_to_link4.position.x() +
547  pose_from_joint2_to_link4.position.y() * pose_from_joint2_to_link4.position.y());
548  // std::cout << "xy_distance : " << xy_distance << std::endl;
549 
550  double z_distance = pose_from_joint2_to_link4.position.z();
551  // std::cout << "z_distance : " << z_distance << std::endl;
552 
553  double beta = acos(
554  (distance * distance + link2_length_ * link2_length_ - link3_length_ * link3_length_) /
555  (2 * distance * link2_length_));
556  // std::cout << "beta : " << beta <<
557  // " (" << M_PI_2 + delta_joint3_ - joint3.get_joint().get_min_position() << ")" << std::endl;
558 
559  if (std::isnan(beta) ||
560  std::isinf(beta) ||
561  beta > M_PI_2 + delta_joint3_ - joint3.get_joint().get_min_position() ||
562  beta < M_PI_2 + delta_joint3_ - joint3.get_joint().get_max_position())
563  {
564  rm2_common::Pose base_move3 =
565  rm2_common::Pose(
566  rm2_common::Vector(
567  ((xy_distance + tip_length_ - min_xy_distance_) / xy_distance) *
568  pose_from_joint2_to_link4.position.x(),
569  ((xy_distance + tip_length_ - min_xy_distance_) / xy_distance) *
570  pose_from_joint2_to_link4.position.y(),
571  0.0));
572  // std::cout << "min_xy_distance_ : " << min_xy_distance_ << std::endl;
573  // std::cout << "(xy_distance+ tip_length_ - min_xy_distance_ / xy_distance) : " <<
574  // ((xy_distance + tip_length_ - min_xy_distance_) / xy_distance) << std::endl;
575 
576  // new root pose
577  root_pose = root_pose * base_move3;
578  // std::cout << "\t--> Distance Limit -> trans " <<
579  // base_move3.position.x() << ", " << base_move3.position.y() << "." << std::endl;
580  RCLCPP_INFO_STREAM(node_->get_logger(), "[WorkSpace] distance limit.");
581  moved = true;
582  continue;
583  }
584 
585  // Joint 2 angle
586  // link2 is not bended
587  target_joint_value.position(1) = M_PI_2 - (alpha + beta);
588  // std::cout << "joint2 angle : " << target_joint_value.position(1) << std::endl;
589 
590  // Calculate phi angle
591  // Phi : angle between link2 and link3
592  // d^2 = l2^2 + l3^2 - 2*l2*l3*cos(Phi)
593  double phi = acos(
594  (link2_length_ * link2_length_ + link3_length_ * link3_length_ - distance * distance) /
595  (2 * link2_length_ * link3_length_));
596  // std::cout << "phi : " << phi << std::endl;
597 
598  // Joint 3 angle
599  target_joint_value.position(2) = M_PI_2 - phi + delta_joint3_;
600  // std::cout << "joint3 angle : " << target_joint_value.position(2) << std::endl;
601 
602  // Joint 4 angle
603  target_joint_value.position(3) =
604  tip_pitch - target_joint_value.position(1) - target_joint_value.position(2);
605  // std::cout << "joint4 angle : " << target_joint_value.position(3) << std::endl;
606 
607  if (target_joint_value.position(2) + target_joint_value.position(3) > M_PI_2) {
608  double delta_z = link3_length_ * cos(delta_joint3_) - z_distance;
609  double delta_xy = sqrt(link2_length_ * link2_length_ - delta_z * delta_z) +
610  link3_length_ * sin(delta_joint3_);
611 
612  rm2_common::Pose base_move4 =
613  rm2_common::Pose(
614  rm2_common::Vector(
615  ((delta_xy - xy_distance - tip_length_) / xy_distance) *
616  pose_from_joint2_to_link4.position.x(),
617  ((delta_xy - xy_distance - tip_length_) / xy_distance) *
618  pose_from_joint2_to_link4.position.y(),
619  0.0));
620 
621  // std::cout << "delta_xy : " << delta_xy << std::endl;
622  // std::cout << "(delta_xy - xy_distance- tip_length_) / xy_distance : " <<
623  // ((delta_xy - xy_distance - tip_length_) / xy_distance) << std::endl;
624 
625  // new root pose
626  root_pose = root_pose * base_move4;
627  // std::cout << "\t--> Joint3 and Joint4 Limit -> trans " <<
628  // base_move4.position.x() << ", " << base_move4.position.y() << "." << std::endl;
629  RCLCPP_INFO_STREAM(node_->get_logger(), "[WorkSpace] link3 angle limit.");
630  moved = true;
631  continue;
632  }
633 
634  if (robot_handler_->is_debug_mode()) {
635  publish_component_poses(root_pose, target_joint_value);
636  }
637  return true;
638  }

643  double & roll,
644  double & pitch,
645  double & yaw)

648  // (cosP 0 sinP) (cosY -sinY 0) (cosPcosY -cosPsinY sinP)
649  // (0 1 0 ) x (sinY cosY 0) = (sinY cosY 0 )
650  // (-sinP 0 cosP) (0 0 1) (-sinPcosY sinPsinY cosP)
651  // Yaw * Pitch * Roll(E)
652  // (cosY -sinY 0) (cosP 0 sinP) (cosYcosP -sinY cosYsinP)
653  // (sinY cosY 0) x (0 1 0 ) = (sinYcosP cosY sinYsinP)
654  // (0 0 1) (-sinP 0 cosP) (-sinP 0 cosP)
655  if (orientation(2, 1) < 1e-3) {
656  roll = 0.0;
657  pitch = atan2(-orientation(2, 0), orientation(2, 2));
658  yaw = atan2(-orientation(0, 1), orientation(1, 1));
659  return true;
660  } else {
661  orientation.get_rpy(roll, pitch, yaw);
662 // if (roll < 1e-3) {
663 // return true;
664 // }
665  return false;
666  }
