33 return_type HardwareInterface::configure(
34  const hardware_interface::HardwareInfo & info)
35 {
36  if (configure_default(info) != return_type::OK) {
37  return return_type::ERROR;
38  }
39 
40  uint8_t id =
41  static_cast<uint8_t>(stoi(info_.hardware_parameters["id"]));
42  uint32_t baud_rate =
43  static_cast<uint32_t>(stoi(info_.hardware_parameters["baud_rate"]));
44  std::string port_name =
45  info_.hardware_parameters["port_name"];
46  double protocol_version =
47  stod(info_.hardware_parameters["protocol_version"]);
48  uint32_t joint_acceleration =
49  static_cast<uint32_t>(stoi(info_.hardware_parameters["joint_acceleration"]));
50  bool loging_velocity =
51  (info_.hardware_parameters["loging_velocity"] == "true" ||
52  info_.hardware_parameters["loging_velocity"] == "True");
53  std::string log_file_path =
54  info_.hardware_parameters["log_file_path"];
55 
56  MotorCommModule::init(
57  id, baud_rate, port_name,
58  protocol_version, joint_acceleration,
59  loging_velocity, log_file_path);
60 
61  hw_stop_sec_ = stod(info_.hardware_parameters["hw_stop_duration_sec"]);
62  hw_states_.resize(info_.joints.size(), std::numeric_limits<double>::quiet_NaN());
63  hw_commands_.resize(info_.joints.size(), std::numeric_limits<double>::quiet_NaN());
64 
65  for (const hardware_interface::ComponentInfo & joint : info_.joints) {
66  // RRBotSystemPositionOnly has exactly one state and command interface on each joint
67  if (joint.command_interfaces.size() != 1) {
68  RCLCPP_FATAL(
69  rclcpp::get_logger("HardwareInterface"),
70  "Joint '%s' has %d command interfaces found. 1 expected.",
71  joint.name.c_str(), joint.command_interfaces.size());
72  return return_type::ERROR;
73  }
74 
75  if (joint.command_interfaces[0].name != hardware_interface::HW_IF_VELOCITY) {
76  RCLCPP_FATAL(
77  rclcpp::get_logger("HardwareInterface"),
78  "Joint '%s' have %s command interfaces found. '%s' expected.",
79  joint.name.c_str(), joint.command_interfaces[0].name.c_str(),
80  hardware_interface::HW_IF_VELOCITY);
81  return return_type::ERROR;
82  }
83 
84  if (joint.state_interfaces.size() != 1) {
85  RCLCPP_FATAL(
86  rclcpp::get_logger("HardwareInterface"),
87  "Joint '%s' has %d state interface. 1 expected.",
88  joint.name.c_str(), joint.state_interfaces.size());
89  return return_type::ERROR;
90  }
91 
92  if (joint.state_interfaces[0].name != hardware_interface::HW_IF_POSITION) {
93  RCLCPP_FATAL(
94  rclcpp::get_logger("HardwareInterface"),
95  "Joint '%s' have %s state interface. '%s' expected.",
96  joint.name.c_str(), joint.state_interfaces[0].name.c_str(),
97  hardware_interface::HW_IF_POSITION);
98  return return_type::ERROR;
99  }
100  }
101 
102  ultrasonic_.clear();
103  ultrasonic_error_.resize(8);
104  ultrasonic_range_.resize(8, 2.0);
105  ultrasonic_fov_.resize(8);
106  for (const hardware_interface::ComponentInfo & sensor : info_.sensors) {
107  sensor_msgs::msg::Range range;
108  range.header.frame_id = sensor.parameters.at("frame_id");
109  range.max_range = std::stof(sensor.parameters.at("max_range"));
110  range.min_range = std::stof(sensor.parameters.at("min_range"));
111  ultrasonic_.insert(std::make_pair(sensor.name.c_str(), range));
112  }
113 
114  status_ = hardware_interface::status::CONFIGURED;
115  return return_type::OK;
116 }

118 std::vector<hardware_interface::StateInterface>
119 HardwareInterface::export_state_interfaces()
120 {
121  std::vector<hardware_interface::StateInterface> state_interfaces;
122  for (uint i = 0; i < info_.joints.size(); i++) {
123  state_interfaces.emplace_back(
124  hardware_interface::StateInterface(
125  info_.joints[i].name, hardware_interface::HW_IF_POSITION, &hw_states_[i]));
126  }
127 
128  // Ultrasonic data
129  for (uint i = 0; i < info_.sensors.size(); i++) {
130  state_interfaces.emplace_back(
131  hardware_interface::StateInterface(
132  info_.sensors[i].name,
133  "range",
134  &ultrasonic_range_[
135  static_cast<uint32_t>(
136  std::stol(info_.sensors[i].parameters.at("num"))) - 1]));
137  state_interfaces.emplace_back(
138  hardware_interface::StateInterface(
139  info_.sensors[i].name,
140  "field_of_view",
141  &ultrasonic_fov_[
142  static_cast<uint32_t>(
143  std::stol(info_.sensors[i].parameters.at("num"))) - 1]));
144  }
145 
146  return state_interfaces;
147 }

162 return_type HardwareInterface::start()
163 {
164  RCLCPP_INFO(
165  rclcpp::get_logger("HardwareInterface"),
166  "Starting ...please wait...");
167 
168  // set some default values
169  for (uint i = 0; i < hw_states_.size(); i++) {
170  if (std::isnan(hw_states_[i])) {
171  hw_states_[i] = 0;
172  hw_commands_[i] = 0;
173  }
174  }
175 
176  // time sync
177  write_time_sync();
178 
179  status_ = hardware_interface::status::STARTED;
180 
181  RCLCPP_INFO(
182  rclcpp::get_logger("HardwareInterface"),
183  "System Sucessfully started!");
184 
185  return return_type::OK;
186 }

188 return_type HardwareInterface::stop()
189 {
190  RCLCPP_INFO(
191  rclcpp::get_logger("HardwareInterface"),
192  "Stopping ...please wait...");
193 
194  for (int i = 0; i <= hw_stop_sec_; i++) {
195  rclcpp::sleep_for(std::chrono::seconds(1));
196  RCLCPP_INFO(
197  rclcpp::get_logger("HardwareInterface"),
198  "%.1f seconds left...", hw_stop_sec_ - i);
199  }
200 
201  status_ = hardware_interface::status::STOPPED;
202  return return_type::OK;
203 }
