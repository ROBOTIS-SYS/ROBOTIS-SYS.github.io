73 void TrapezoidalControl::update_cancel(rm2_control_module::ControlModule::CancelResult result)
74 {
75  TaskGoalAction::Result::SharedPtr action_result = std::make_shared<TaskGoalAction::Result>();
76  manipulator_interfaces::msg::Result result_state;
77 
78  if (result == CancelResult::JOINT_LIMIT) {
79  RCLCPP_INFO_STREAM(
80  node_->get_logger(),
81  "Task trapezoidal contorl canceled : joint limit over.");
82  result_state.set__val(manipulator_interfaces::msg::Result::COLLISION_CHECKING_UNAVAILABLE);
83  } else if (result == CancelResult::IK_FAIL) {
84  RCLCPP_INFO_STREAM(
85  node_->get_logger(),
86  "Task trapezoidal contorl canceled : ik solving failure");
87  result_state.set__val(manipulator_interfaces::msg::Result::NO_IK_SOLUTION);
88  } else if (result == CancelResult::PREEMPTED) {
89  RCLCPP_INFO_STREAM(
90  node_->get_logger(),
91  "Task trapezoidal contorl canceled : preempted by the other controller");
92  result_state.set__val(manipulator_interfaces::msg::Result::PREEMPTED);
93  } else if (result == CancelResult::CHECKER) {
94  result_state.set__val(manipulator_interfaces::msg::Result::GOAL_IN_COLLISION);
95  RCLCPP_INFO_STREAM(
96  node_->get_logger(),
97  "Task trapezoidal contorl canceled : lidar");
98  } else {
99  result_state.set__val(manipulator_interfaces::msg::Result::FAILURE);
100  RCLCPP_INFO_STREAM(
101  node_->get_logger(),
102  "Task trapezoidal contorl canceled : fail.");
103  }
104  action_result->set__result_state(result_state);
105 
106  if (actived_goal_) {
107  actived_goal_->abort(action_result);
108  }
109  actived_goal_.reset();
110 }

119 rclcpp_action::GoalResponse TrapezoidalControl::goal_callback(
120  const rclcpp_action::GoalUUID & uuid,
121  std::shared_ptr<const TaskGoalAction::Goal> goal)
122 {
123  RCLCPP_INFO_STREAM(node_->get_logger(), "Received new task trajectory(trapezoidal) action goal");
124  std::lock_guard<std::mutex> lock(mutex_);
126  /* Validate Goal Time */
128  RCLCPP_INFO_STREAM(
129  node_->get_logger(), "start time : " <<
130  goal->header.stamp.sec << "." << goal->header.stamp.nanosec);
131  rclcpp::Time start_time = static_cast<rclcpp::Time>(goal->header.stamp);
132  // If the starting time it set to 0.0, it means the controller should start it now.
133  // Otherwise we check if the trajectory ends before the current time,
134  // in which case it can be ignored.
135  if (start_time.seconds() != 0.0) {
136  rclcpp::Time end_time = start_time +
137  rclcpp::Duration::from_seconds(goal->goal_waypoint.time_from_start);
138  if (end_time < node_->now()) {
139  RCLCPP_ERROR_STREAM(
140  node_->get_logger(),
141  "Received trajectory with non zero time start time (" << start_time.seconds() <<
142  ") that ends on the past (" << end_time.seconds() << ")");
143  return rclcpp_action::GoalResponse::REJECT;
144  }
145  } else {
146  start_time = node_->now();
147  }
148 
149  RCLCPP_INFO_STREAM(node_->get_logger(), "Component name is '" << goal->component_name << "'.");
150  if (!robot_handler_->check_component_name(goal->component_name)) {
151  RCLCPP_ERROR_STREAM(
152  node_->get_logger(),
153  "The manipulator has not compnoent '" << goal->component_name.c_str() << "'.");
154  return rclcpp_action::GoalResponse::REJECT;
155  }
156 
157  // start waypoint
158  rm2_common::TaskValues start_waypoint =
159  robot_handler_->get_previous_goal_task_value(goal->component_name);
160 
161  // goal waypoint
162  rm2_common::TaskValues goal_waypoint_from_ref(
163  rm2_common::Pose(
164  rm2_common::Vector(
165  goal->goal_waypoint.poses.position.x,
166  goal->goal_waypoint.poses.position.y,
167  goal->goal_waypoint.poses.position.z),
168  rm2_common::Rotation(
169  goal->goal_waypoint.poses.orientation.w,
170  goal->goal_waypoint.poses.orientation.x,
171  goal->goal_waypoint.poses.orientation.y,
172  goal->goal_waypoint.poses.orientation.z)),
173  rm2_common::Twist(
174  goal->goal_waypoint.velocities.linear.x,
175  goal->goal_waypoint.velocities.linear.y,
176  goal->goal_waypoint.velocities.linear.z,
177  goal->goal_waypoint.velocities.angular.x,
178  goal->goal_waypoint.velocities.angular.y,
179  goal->goal_waypoint.velocities.angular.z));
180 
181  rm2_common::TaskValues goal_waypoint;
182  if (goal->goal_waypoint.ref_frame == "" ||
183  goal->goal_waypoint.ref_frame == robot_handler_->get_robot_model().get_base_component()->first)
184  {
185  goal_waypoint = goal_waypoint_from_ref;
186  } else if (robot_handler_->get_robot_model().check_component(goal->goal_waypoint.ref_frame)) {
187  rm2_common::TaskValues ref_frame_pose_from_root =
188  robot_handler_->get_previous_goal_task_value(goal->goal_waypoint.ref_frame);
189  RCLCPP_INFO_STREAM(
190  node_->get_logger(),
191  "Reference frame pose from root : " <<
192  ref_frame_pose_from_root.pose.position.data.transpose());
193  goal_waypoint = ref_frame_pose_from_root * goal_waypoint_from_ref;
194  } else {
195  RCLCPP_ERROR_STREAM(
196  node_->get_logger(),
197  "The manipulator has not compnoent '" << goal->goal_waypoint.ref_frame.c_str() <<
198  "' set as ref_frame.");
199  return rclcpp_action::GoalResponse::REJECT;
200  }
201 
202  // set start time
203  robot_handler_->set_start_time(start_time.seconds());
204 
205  if (!task_trajectory_ptr_->set_profile(
206  goal->profile.velocity,
207  goal->profile.acceleration,
208  goal->profile.tip_link_length))
209  {
210  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to set profile (vel, acc, tip length).");
211  return rclcpp_action::GoalResponse::REJECT;
212  }
213 
214  if (!task_trajectory_ptr_->calculation(
215  start_waypoint.pose, goal_waypoint.pose,
216  start_waypoint.velocity, goal_waypoint.velocity))
217  {
218  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to Calculate trajectory.");
219  return rclcpp_action::GoalResponse::REJECT;
220  }
221 
222  current_controlled_component_name_ = goal->component_name;
223 
224  double end_time = task_trajectory_ptr_->get_move_time();
225  RCLCPP_INFO_STREAM(
226  node_->get_logger(), "Success to calculate trajectory. total move time is " << end_time);
227 
228  // set start time
229  robot_handler_->set_start_time(start_time.seconds());
230  goal_for_feedback_ = trans_handler_to_msg(goal_waypoint, end_time);
231 
232  return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
233 }

235 rclcpp_action::CancelResponse TrapezoidalControl::cancel_callback(
236  const std::shared_ptr<rclcpp_action::ServerGoalHandle<TrapezoidalControl::TaskGoalAction>>
237  goal_handle)
238 {
239  RCLCPP_INFO_STREAM(node_->get_logger(), "Got request to cancel goal");
240  std::lock_guard<std::mutex> lock(mutex_);
241  try {
242  if (actived_goal_ == goal_handle) {
243  // Controller uptime
244  // Enter hold current position mode
245  stop_update();
246 
247  RCLCPP_DEBUG_STREAM(
248  node_->get_logger(), "Canceling active action goal because cancel callback received");
249 
250  // Mark the current goal as canceled
251  auto action_res = std::make_shared<TaskGoalAction::Result>();
252  manipulator_interfaces::msg::Result result_state;
253  result_state.set__val(manipulator_interfaces::msg::Result::CANCELED);
254  action_res->set__result_state(result_state);
255  if (actived_goal_) {
256  actived_goal_->canceled(action_res);
257  }
258  } else {
259  // Reset current goal
260  actived_goal_.reset();
261  RCLCPP_ERROR_STREAM(
262  node_->get_logger(), "Fail to Canceling active action goal due to goal handle mismatch.");
263  return rclcpp_action::CancelResponse::REJECT;
264  }
265  } catch (std::runtime_error e) {
266  // Reset current goal
267  actived_goal_.reset();
268  RCLCPP_ERROR_STREAM(node_->get_logger(), "Runtime error : " << e.what());
269  return rclcpp_action::CancelResponse::REJECT;
270  }
271  // Reset current goal
272  actived_goal_.reset();
273  return rclcpp_action::CancelResponse::ACCEPT;
274 }

319  result.poses.position.y = waypoint.pose.position.y();
320  result.poses.position.z = waypoint.pose.position.z();
321  result.poses.orientation.x = waypoint.pose.orientation.x();
322  result.poses.orientation.y = waypoint.pose.orientation.y();
323  result.poses.orientation.z = waypoint.pose.orientation.z();
324  result.poses.orientation.w = waypoint.pose.orientation.w();
325 
326  result.velocities.linear.x = waypoint.velocity.linear.x();
327  result.velocities.linear.y = waypoint.velocity.linear.y();
328  result.velocities.linear.z = waypoint.velocity.linear.z();
329  result.velocities.angular.x = waypoint.velocity.angular.x();
