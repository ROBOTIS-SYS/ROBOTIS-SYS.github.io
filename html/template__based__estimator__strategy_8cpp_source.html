37 int TemplateBasedEstimatorStrategy::calculate(
38  std::shared_ptr<std::vector<RecognizedText>> & left_buttons,
39  std::shared_ptr<std::vector<RecognizedText>> & right_buttons,
40  std::shared_ptr<std::map<std::string, ButtonPose>> & button_poses)
41 {
42  button_poses->clear();
43  if (left_buttons->size() < 2 || right_buttons->size() < 2) {
44  return PERCEPTION_STATE::ERROR_RGG_LESS_NUM_OF_PTS;
45  }
46 
47  // reconstruct points after matching left and right button's features
48  std::map<std::string, Eigen::MatrixXf> cal_poses;
49  std::vector<std::string> labels(left_buttons->size());
50  std::vector<Eigen::MatrixXf> poses(left_buttons->size());
51 
52 #pragma omp parallel num_threads(MAX_THREADS)
53  {
54 #pragma omp for
55  for (size_t i = 0; i < left_buttons->size(); i++) {
56  std::string left_label = left_buttons->at(i).class_name;
57  auto corresponding_one =
58  std::find_if(
59  right_buttons->begin(), right_buttons->end(),
60  [&left_label](const RecognizedText btn) {
61  return btn.class_name == left_label;
62  });
63  if (corresponding_one != right_buttons->end()) {
64  labels[i] = left_label;
65  poses[i] = triangulate(left_buttons->at(i), *corresponding_one);
66  }
67  }
68  }
69 
70  for (size_t i = 0; i < labels.size(); i++) {
71  if (labels[i].empty() == false) {
72  cal_poses[labels[i]] = poses[i];
73  }
74  }
75 
76  // make a pair between recon. pts and template
77  Eigen::MatrixXf src_pts(cal_poses.size() * 5, 3);
78  Eigen::MatrixXf trg_pts(cal_poses.size() * 5, 3);
79  int next_id = 0;
80  std::map<std::string, Eigen::MatrixXf>::iterator iter;
81  for (iter = template_.begin(); iter != template_.end(); iter++) {
82  std::string label = iter->first;
83  if (cal_poses.find(label) != cal_poses.end()) {
84  src_pts.block<5, 3>(next_id * 5, 0) = template_[label];
85  trg_pts.block<5, 3>(next_id * 5, 0) = cal_poses[label];
86  next_id++;
87  }
88  }
89 
90  // for (int i = 0; i < src_pts.rows(); i++) {
91  // std::cout << i << "-th [src]: " <<
92  // src_pts.row(i)(0) << ", " <<
93  // src_pts.row(i)(1) << ", "<<
94  // src_pts.row(i)(2) << ", [trg]: " <<
95  // trg_pts.row(i)(0) << ", " <<
96  // trg_pts.row(i)(1) << ", "<<
97  // trg_pts.row(i)(2) << std::endl;
98  // }
99  // std::cout << "check point 3" << std::endl;
100 
101  // compensate for button poses with templates
102  Eigen::Matrix3f R_best;
103  Eigen::Vector3f t_best;
104  int state =
105  register_->paired_point_registration_using_ransac_cuda(
106  src_pts,
107  trg_pts,
108  R_best,
109  t_best,
110  5.0);
111  switch (state) {
112  case PERCEPTION_STATE::SUCCESS_NO_ISSUE:
113  case PERCEPTION_STATE::SUCCESS_REG_RANSAC_TOO_SMALL_PTS:
114  case PERCEPTION_STATE::SUCCESS_REG_RANSAC_TOO_SMALL_INLIERS:
115  if (is_valid(t_best, 100, 1000)) {
116  Eigen::Vector3f tf_normal = R_best * normal_;
117  for (auto it = template_.begin(); it != template_.end(); it++) {
118  ButtonPose pose;
119  Eigen::VectorXf tmp_pt = it->second.row(2);
120  pose.center = R_best * tmp_pt + t_best;
121  pose.normal = tf_normal;
122  button_poses->insert(std::make_pair(it->first, pose));
123  }
124  return state;
125  } else {
126  return PERCEPTION_STATE::ERROR_INVALID_TRANSLATION;
127  }
128  case PERCEPTION_STATE::ERROR_REG_RANSAC_CUDA:
129  case PERCEPTION_STATE::ERROR_REG_INVALID_PAIR:
130  case PERCEPTION_STATE::ERROR_RGG_LESS_NUM_OF_PTS:
131  case PERCEPTION_STATE::ERROR_WRONG_INITIALIZATION:
132  return state;
133  default:
134  return PERCEPTION_STATE::ERROR_UNKNOWN;
135  }
136 }
