96  auto reset_bumper_callback = [this](
97  const std_srvs::srv::Trigger::Request::SharedPtr request,
98  std_srvs::srv::Trigger::Response::SharedPtr response) {
99  (void) request;
100  uint8_t command = 81;
101  uint8_t dxl_error = 0;
102  int dxl_comm_result = COMM_TX_FAIL;
103 
104  port_mutex_.lock();
105  dxl_comm_result =
106  packet_handler_->write1ByteTxRx(
107  port_handler_, id_, ADDR_OPERATION_CMD, command,
108  &dxl_error);
109  port_mutex_.unlock();
110 
111  if (dxl_comm_result != COMM_SUCCESS) {
112  response->success = false;
113  }
114  response->success = true;
115  };
116  reset_bumper_server_ = this->create_service<std_srvs::srv::Trigger>(
117  "reset_bumper",
118  reset_bumper_callback);

138  port_mutex_.unlock();
139 
140  if (dxl_comm_result != COMM_SUCCESS) {
141  response->success = false;

144  };

241 bool MotorCommModule::init_dynamixel_sdk_handlers()
242 {

244  packet_handler_ =
245  dynamixel::PacketHandler::getPacketHandler(static_cast<float>(protocol_version_));
246

248  RCLCPP_INFO_STREAM(this->get_logger(), "Succeeded in opening the port!");

251  return false;
252  }
253

255  RCLCPP_INFO_STREAM(this->get_logger(), "Succeeded in setting the baudrate!");
256  } else {
257  RCLCPP_ERROR_STREAM(this->get_logger(), "Failed to set the baudrate!");
258  return false;
259  }
260 
261  if (write_joint_acceleration(static_cast<int32_t>(joint_acceleration_)) == true) {
262  RCLCPP_INFO_STREAM(this->get_logger(), "Succeeded in writing the joint acceleration");
263  } else {
264  RCLCPP_ERROR_STREAM(this->get_logger(), "Failed to write the joint acceleration");
265 // return false;
266  }
267 
268  return true;

271 double * MotorCommModule::read_register()
272 {
273  int32_t previous_tick[2] = {present_tick_[0], present_tick_[1]};
274 
275  // READ
276  static uint8_t * data = reinterpret_cast<uint8_t *>(calloc(READ_LENGTH, sizeof(uint8_t)));
277 
278  uint8_t dxl_error = 0;
279  int dxl_comm_result = COMM_TX_FAIL;
280 
281  port_mutex_.lock();
282  dxl_comm_result = packet_handler_->readTxRx(
283  port_handler_, id_, READ_START_ADDR, READ_LENGTH, data, &dxl_error);
284  port_mutex_.unlock();
285 
286  if (dxl_comm_result != COMM_SUCCESS) {
287  RCLCPP_INFO_STREAM(this->get_logger(), "result: " << dxl_comm_result);
288 // return pos;
289  }
290 
291  // check logging
292  if (loging_velocity_ == true) {
293  union Data {
294  int32_t * dword;
295  uint8_t * byte;
296  } union_data;
297 
298  // Millis
299  union_data.byte = &data[ADDR_MILLIS - READ_START_ADDR];
300  int millis = union_data.dword[0];
301 
302  // Goal RPM
303  int goal_vels[2] = {0, 0};
304  int start_goal_addr = ADDR_LEFT_GOAL_RPM - READ_START_ADDR;
305  union_data.byte = &data[start_goal_addr];
306  goal_vels[0] = union_data.dword[0];
307  goal_vels[1] = union_data.dword[1];
308 
309  // Goal acceleration
310  int goal_accs[2] = {0, 0};
311  int start_goal_accs_addr = ADDR_LEFT_GOAL_ACC - READ_START_ADDR;
312  union_data.byte = &data[start_goal_accs_addr];
313  goal_accs[0] = union_data.dword[0];
314  goal_accs[1] = union_data.dword[1];
315 
316  // present RPM
317  int present_vels[2] = {0, 0};
318  int start_present_addr = ADDR_LEFT_PRESENT_RPM - READ_START_ADDR;
319  union_data.byte = &data[start_present_addr];
320  present_vels[0] = union_data.dword[0];
321  present_vels[1] = union_data.dword[1];
322 
323  // Alive
324  int alive_count[2] = {0, 0};
325  int start_alive_count_addr = ADDR_LEFT_ALIVE_COUNT - READ_START_ADDR;
326  union_data.byte = &data[start_alive_count_addr];
327  alive_count[0] = union_data.dword[0];
328  alive_count[1] = union_data.dword[1];
329 
330  // Hall count
331  int present_pos[2] = {0, 0};
332  int start_hall_count_addr = ADDR_LEFT_PRESENT_POSITION - READ_START_ADDR;
333  union_data.byte = &data[start_hall_count_addr];
334  present_pos[0] = union_data.dword[0];
335  present_pos[1] = union_data.dword[1];
336 
337  loging_velocities(
338  millis, present_vels, goal_vels, goal_accs,
339  alive_count, present_pos);
340  }
341 
342  // firmware version
343  fw_major_version_ = data[ADDR_FW_VER_MAJOR - READ_START_ADDR];
344  fw_minor_version_ = data[ADDR_FW_VER_MINOR - READ_START_ADDR];
345 
346  // Publish battery state
347  sensor_msgs::msg::BatteryState battery_state_msg;
348  battery_state_msg.voltage = DXL_MAKEWORD(
349  data[ADDR_BATTERY_VOLTAGE - READ_START_ADDR],
350  data[ADDR_BATTERY_VOLTAGE - READ_START_ADDR + 1]);
351 
352  float current = static_cast<float>(static_cast<int>(DXL_MAKEWORD(
353  data[ADDR_BATTERY_CHARGE_CURRENT - READ_START_ADDR],
354  data[ADDR_BATTERY_CHARGE_CURRENT - READ_START_ADDR + 1])) / 100.0);
355  current += static_cast<float>(static_cast<int>(DXL_MAKEWORD(
356  data[ADDR_BATTERY_DISCHARGE_CURRENT - READ_START_ADDR],
357  data[ADDR_BATTERY_DISCHARGE_CURRENT - READ_START_ADDR + 1])) / 100.0);
358  battery_state_msg.current = current;
359 
360  battery_state_msg.charge = NAN;
361  battery_state_msg.capacity = NAN;
362 
363  battery_state_msg.design_capacity = DXL_MAKEWORD(
364  data[ADDR_BATTERY_CAPACITY - READ_START_ADDR],
365  data[ADDR_BATTERY_CAPACITY - READ_START_ADDR + 1]);
366 
367  battery_state_msg.percentage =
368  static_cast<float>(data[ADDR_BATTERY_PERCENTAGE - READ_START_ADDR] / 100.0f);
369 
370  if (data[ADDR_BATTERY_CHARGING - READ_START_ADDR] == 0) {
371  if (data[ADDR_DOCKING_STATUS - READ_START_ADDR] == 0) { // Docking (FALSE) && Charging (FALSE)
372  battery_state_msg.power_supply_status =
373  sensor_msgs::msg::BatteryState::POWER_SUPPLY_STATUS_NOT_CHARGING;
374  } else { // Docking (TRUE) && Charging (FALSE)
375  battery_state_msg.power_supply_status =
376  sensor_msgs::msg::BatteryState::POWER_SUPPLY_STATUS_UNKNOWN;
377  }
378  } else if (data[ADDR_BATTERY_CHARGING - READ_START_ADDR] == 1) {
379  battery_state_msg.power_supply_status =
380  sensor_msgs::msg::BatteryState::POWER_SUPPLY_STATUS_CHARGING;
381  }
382 
383  battery_state_msg.power_supply_health =
384  sensor_msgs::msg::BatteryState::POWER_SUPPLY_HEALTH_UNKNOWN;
385  battery_state_msg.power_supply_technology =
386  sensor_msgs::msg::BatteryState::POWER_SUPPLY_TECHNOLOGY_UNKNOWN;
387 
388  battery_state_msg.present = true;
389 
390  battery_state_pub_->publish(battery_state_msg);
391 
392  // publish estop status
393  std_msgs::msg::Bool estop_msg;
394  estop_msg.data = data[ADDR_ESTOP - READ_START_ADDR];
395  estop_status_pub_->publish(estop_msg);
396 
397  estop_pressed_ = data[ADDR_ESTOP - READ_START_ADDR];
398 
399  // publish docking switch
400  std_msgs::msg::Bool docking_switch_msg;
401  docking_switch_msg.data = data[ADDR_DOCKING_SWITCH - READ_START_ADDR];
402  docking_switch_pub_->publish(docking_switch_msg);
403 
404  // publish motor state
405  aimbot_hw_interfaces::msg::MotorState motor_state_msg;
406  motor_state_ = static_cast<motor_state>(data[ADDR_MOTOR_STATE - READ_START_ADDR]);
407  motor_state_msg.state = motor_state_;
408  motor_state_msg.motor_brake_source = data[ADDR_P_MOTOR_BRAKE_SOURCE - READ_START_ADDR];
409  motor_state_pub_->publish(motor_state_msg);
410 
411  // publish bumper state
412  aimbot_hw_interfaces::msg::BumperState bumper_state_msg;
413  bumper_state_msg.front_state = data[ADDR_FRONT_BUMPER_STATE - READ_START_ADDR];
414  bumper_state_msg.rear_state = data[ADDR_REAR_BUMPER_STATE - READ_START_ADDR];
415  bumper_state_pub_->publish(bumper_state_msg);
416 
417  // TOF
418  sensor_msgs::msg::Range tof_msg;
419  tof_msg.header.set__stamp(this->get_clock()->now());
420  tof_msg.set__radiation_type(sensor_msgs::msg::Range::INFRARED);
421 
422  // right tof
423  tof_.at("right_tof").distance = tof_.at("right_tof").max_distance;
424  tof_.at("right_tof").confidence = 0.0;
425  tof_.at("right_tof").error_state = TOF_ERROR::SENSOR_ERROR;
426  for (uint detector_num = 0; detector_num < tof_.at("right_tof").size; detector_num++) {
427  double confidence = DXL_MAKEWORD(
428  data[ADDR_TOF1_CONFIDENCE_1 + (detector_num * 2) - READ_START_ADDR],
429  data[ADDR_TOF1_CONFIDENCE_1 + (detector_num * 2) - READ_START_ADDR + 1]);
430  double distance = DXL_MAKEWORD(
431  data[ADDR_TOF1_DISTANCE_1 + (detector_num * 2) - READ_START_ADDR],
432  data[ADDR_TOF1_DISTANCE_1 + (detector_num * 2) - READ_START_ADDR + 1]) * 0.001;
433 
434  // confidence , 9
435  if (confidence > tof_.at("right_tof").confidence_threshold &&
436  distance < tof_.at("right_tof").distance)
437  {
438  tof_.at("right_tof").distance = distance;
439  tof_.at("right_tof").confidence = confidence;
440  tof_.at("right_tof").error_state = TOF_ERROR::OK;
441  }
442  }
443  // for visualization
444  tof_msg.header.frame_id = tof_["right_tof"].frame_id;
445  tof_msg.max_range = tof_["right_tof"].max_distance;
446  tof_msg.min_range = tof_["right_tof"].min_distance;
447  tof_msg.field_of_view = tof_["right_tof"].fov;
448  tof_msg.range = tof_.at("right_tof").distance;
449  tof_pub_array_[1]->publish(tof_msg);
450 
451  // left tof
452  tof_.at("left_tof").distance = tof_.at("left_tof").max_distance;
453  tof_.at("left_tof").confidence = 0.0;
454  tof_.at("left_tof").error_state = TOF_ERROR::SENSOR_ERROR;
455  for (uint detector_num = 0; detector_num < tof_.at("left_tof").size; detector_num++) {
456  double confidence = DXL_MAKEWORD(
457  data[ADDR_TOF2_CONFIDENCE_1 + (detector_num * 2) - READ_START_ADDR],
458  data[ADDR_TOF2_CONFIDENCE_1 + (detector_num * 2) - READ_START_ADDR + 1]);
459  double distance = DXL_MAKEWORD(
460  data[ADDR_TOF2_DISTANCE_1 + (detector_num * 2) - READ_START_ADDR],
461  data[ADDR_TOF2_DISTANCE_1 + (detector_num * 2) - READ_START_ADDR + 1]) * 0.001;
462 
463  // confidence , 9
464  if (confidence > tof_.at("left_tof").confidence_threshold &&
465  distance < tof_.at("left_tof").distance)
466  {
467  tof_.at("left_tof").distance = distance;
468  tof_.at("left_tof").confidence = confidence;
469  tof_.at("left_tof").error_state = TOF_ERROR::OK;
470  }
471  }
472  // for visualization
473  tof_msg.header.frame_id = tof_["left_tof"].frame_id;
474  tof_msg.max_range = tof_["left_tof"].max_distance;
475  tof_msg.min_range = tof_["left_tof"].min_distance;
476  tof_msg.field_of_view = tof_["left_tof"].fov;
477  tof_msg.range = tof_.at("left_tof").distance;
478  tof_pub_array_[0]->publish(tof_msg);
479 
480  // publish ATM_pressure
481  sensor_msgs::msg::FluidPressure atm_msg;
482  atm_msg.header.frame_id = "base_link";
483  atm_msg.header.stamp = this->now();
484  int atm_index = ADDR_ATM_PRESSURE - READ_START_ADDR;
485  uint32_t pascal_value = DXL_MAKEDWORD(
486  DXL_MAKEWORD(data[atm_index + 0], data[atm_index + 1]),
487  DXL_MAKEWORD(data[atm_index + 2], data[atm_index + 3]));
488  atm_msg.fluid_pressure = static_cast<double>(pascal_value);
489  atm_msg.variance = 0;
490  atm_pressure_pub_->publish(atm_msg);
491 
492  // read docking station ID
493  docking_station_id = data[ADDR_DOCKING_STATION_ID - READ_START_ADDR];
494 
495  // check motor state
496  // READY, RUNNING
497  if (motor_state_ == motor_state::READY || motor_state_ == motor_state::RUNNING) {
498  // Calc position
499  int pos_index = ADDR_LEFT_PRESENT_POSITION - READ_START_ADDR;
500  present_tick_[0] = DXL_MAKEDWORD(
501  DXL_MAKEWORD(data[pos_index + 0], data[pos_index + 1]),
502  DXL_MAKEWORD(data[pos_index + 2], data[pos_index + 3]));
503  present_tick_[1] = DXL_MAKEDWORD(
504  DXL_MAKEWORD(data[pos_index + 4], data[pos_index + 5]),
505  DXL_MAKEWORD(data[pos_index + 6], data[pos_index + 7]));
506  } else {
507  present_tick_[0] = previous_tick[0];
508  present_tick_[1] = previous_tick[1];
509  RCLCPP_ERROR_STREAM(
510  this->get_logger(),
511  "Fail to get present position, please check MOTOR STATE : " <<
512  static_cast<motor_state>(motor_state_));
513 // wheel_position_[0] = (static_cast<double>(previous_tick[0]) * TICK2RAD) +
514 // wheel_position_offset_[0];
515 // wheel_position_[1] = (static_cast<double>(previous_tick[1]) * TICK2RAD) +
516 // wheel_position_offset_[1];
517 // return wheel_position_;
518  }
519 
520  // . tick
521  bool encoder_reboot = data[ADDR_MOTOR_REBOOT_CHECK - READ_START_ADDR];
522  if (encoder_reboot) {
523  wheel_position_offset_[0] = (static_cast<double>(previous_tick[0]) * TICK2RAD) +
524  wheel_position_offset_[0];
