87 void TkeControllerNode::service_start(uint8_t num_of_robot, int retry_count)
88 {
89  if (retry_count > 10) {
90  std::cout << RED << "Service Start 실패, 프로그램 종료" << C_END << std::endl;
91  exit(-1);
92  }
93 
94  uint8_t tx_packet[13] = {
95  0xF0, 0x52, 0x00, 0x00, 0x00, 0x04, 0x53, 0x01, 0x00, 0x00, 0xD7, 0x5E, 0xF4};
96 
97  tx_packet[7] = num_of_robot;
98  std::cout << GREEN << "로봇 대수 : " << std::to_string(num_of_robot) << C_END << std::endl;
99 
100  uint8_t error = 0;
101  int comm_result = COMM_TX_FAIL;
102 
103  uint8_t cnt = 0;
104  uint8_t cnt_max = 10;
105 
106  while (true) {
107  if (cnt > cnt_max) {
108  std::cout << RED << "service start failed (" << retry_count << ")" << C_END << std::endl;
109  port_handler_->close_port();
110  conn();
111  service_start(num_of_robot, ++retry_count);
112  return;
113  }
114  comm_result =
115  packet_handler_->txrx_packet(port_handler_, tx_packet, rx_packet_, &error);
116 
117  if (comm_result != COMM_SUCCESS) {
118  std::cout << packet_handler_->get_txrx_result(comm_result) << std::endl;
119  cnt++;
120  usleep(100000);
121  continue;
122  }
123 
124  num_of_ev_ = rx_packet_[7];
125  std::cout << GREEN << "엘리베이터 개수 : " << num_of_ev_ << C_END << std::endl;
126  break;
127  }
128 
129  std::vector<EvInfo>().swap(ev_info_); // ev_info 벡터 clear
130 
131  for (unsigned int i = 0; i < num_of_ev_; i++) {
132  ev_info_.push_back(EvInfo(static_cast<uint8_t>(i + 1)));
133  }
134  set_bottom_floor_info();
135 
136  robot_data_.robot_status_2 = robot_status_2::COMM_NORMAL;
137 
138  // 로봇 서비스(층간이동) 진행 중 서버 응답시간(10초) 초과로
139  // 다시 Service Start 한 경우, 이전 콜 재등록 시도
140  if (sequence_ == "Waiting For Elevator" || sequence_ == "Preparation For Taking On") {
141  publish_log(
142  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
143  aimbot_log_interfaces::msg::Log::TYPE_WARN,
144  "EV Controller",
145  " 홀 콜 재등록 시도 (call_floor:" + std::to_string(call_floor_) + ") (dir:" +
146  ((robot_service_dir_ == call_dir::UP) ? "Up" : "Down") + ") (last seq:" + sequence_);
147 
148  rclcpp::sleep_for(std::chrono::milliseconds(500));
149 
150  std::string log_contents;
151 
152  if (hall_call(ev_num_, robot_service_dir_, call_floor_)) {
153  log_contents = " 홀 콜 재등록 성공";
154  } else {
155  log_contents = " 홀 콜 재등록 실패";
156  }
157  publish_log(
158  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
159  aimbot_log_interfaces::msg::Log::TYPE_WARN,
160  "EV Controller", log_contents);
161  } else if (sequence_ == "Moving" || sequence_ == "Preparation For Getting Off") {
162  publish_log(
163  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
164  aimbot_log_interfaces::msg::Log::TYPE_WARN,
165  "EV Controller",
166  " 카 콜 재등록 시도 (dest_floor:" + std::to_string(dest_floor_) + ") (ev_num:" +
167  std::to_string(allocated_ev_num_) + ") (last seq:" + sequence_);
168 
169  rclcpp::sleep_for(std::chrono::milliseconds(500));
170 
171  std::string log_contents;
172 
173  if (car_call(allocated_ev_num_, dest_floor_)) {
174  log_contents = " 카 콜 재등록 성공";
175  } else {
176  log_contents = " 카 콜 재등록 실패";
177  }
178  publish_log(
179  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
180  aimbot_log_interfaces::msg::Log::TYPE_WARN,
181  "EV Controller", log_contents);
182  }
183 
184  std::thread robot_info_thread(&TkeControllerNode::send_robot_info_thread, this);
185  std::thread ev_info_thread(&TkeControllerNode::read_ev_info_thread, this);
186  robot_info_thread.detach();
187  ev_info_thread.detach();
188 
189  service_started_ = true;
190  resting_sleep_timer_->reset();

319 void TkeControllerNode::update_ev_info(uint8_t * rxpacket)
320 {
321  if (rxpacket[TYPE] != 'e') {
322  return;
323  }
324  uint32_t data_length = get_packet_length(rxpacket) - 9;
325  if (data_length < 134 * num_of_ev_) {
326  std::cout << RED << "recived not enough data" << C_END << std::endl;
327  return;
328  }
329  if (rxpacket[data_length + 8] != 0xF4) { // ETX
330  std::cout << RED << "recived not enough data (2)" << C_END << std::endl;
331  return;
332  }
333  const int ADDR_START = 6;
334  for (unsigned int i = 0; i < num_of_ev_; i++) {
335  memcpy(ev_info_[i].data_, rxpacket + (i * 134) + ADDR_START, 134);
336  }
337 
338  // Print For Check
339  std::cout << GREEN << "dir : " << enum_direction_to_string(ev_info_[0].direction()) <<
340  std::endl;
341  std::cout << "flr : " << ev_info_[0].current_floor() << std::endl;
342  std::cout << "door : " << enum_door_state_to_string(ev_info_[0].ev_door_state()) << C_END <<
343  std::endl;
344  if (robot_data_.control_cmd & control_cmd::DOOR_OPEN) {
345  std::cout << YELLOW << "PDO | ";
346  }
347  if (robot_data_.control_cmd & control_cmd::DOOR_CLOSE) {
348  std::cout << YELLOW << "PDC | ";
349  }
350  if (robot_data_.control_cmd & control_cmd::CARCALL) {
351  std::cout << YELLOW << "CARCALL | ";
352  }
353  if (robot_data_.control_cmd & control_cmd::UP_HALLCALL) {
354  std::cout << YELLOW << "UP_HALLCALL | ";
355  }
356  if (robot_data_.control_cmd & control_cmd::DOWN_HALLCALL) {
357  std::cout << YELLOW << "DOWN_HALLCALL | ";
358  }
359  std::cout << C_END << std::endl;
360 
361  std::cout << "-----------" << std::endl;
362 
363  check_sequence();
364 }

366 void TkeControllerNode::check_sequence()
367 {
368  // * 모든 EV에 대해 일치 검사
369  for (auto iter = ev_info_.begin(); iter != ev_info_.end(); iter++) {
370  if (sequence_ == "No service" || sequence_ == "Waiting For Elevator" ||
371  sequence_ == "Preparation For Taking On")
372  {
373  if (iter->current_floor() == call_floor_) { // 1. 층 일치
374  if ((robot_service_dir_ == call_dir::UP &&
375  iter->direction() == EvInfo::ev_direction::UP_STOP) ||
376  (robot_service_dir_ == call_dir::DOWN &&
377  iter->direction() == EvInfo::ev_direction::DOWN_STOP)) // 2. 방향 일치 & 정지상태
378  {
379  if (sequence_ == "No service") {
380  std::cout << std::endl << BLUE << "<Preparation For Taking On>" << C_END << std::endl;
381  sequence_ = "Preparation For Taking On";
382  }
383  if (iter->ev_door_state() == EvInfo::door_state::OPEN) { // 3. 문 열림
384  if (iter->is_available()) {
385  allocated_ev_num_ = iter->ev_id_;
386  std::cout << std::endl << GREEN << "<Waiting For Robot>" << C_END << std::endl;
387  sequence_ = "Waiting For Robot";
388  robot_data_.robot_status = robot_status::ENTERING_EV;
389  push_door_open(allocated_ev_num_); // PDO
390  }
391  call_cancel_check_cnt_ = 0;
392  }
393  }
394  }
395  } else if (iter->ev_id_ == allocated_ev_num_ && iter->current_floor() == dest_floor_) {
396  // 1. 층 일치, 할당 호기 일치
397  if (iter->direction() == EvInfo::ev_direction::DOWN_STOP ||
398  iter->direction() == EvInfo::ev_direction::UP_STOP ||
399  iter->direction() == EvInfo::ev_direction::STOP) // 2. 정지 상태
400  {
401  if (sequence_ == "Moving") {
402  std::cout << std::endl << BLUE << "<Preparation For Getting Off>" << C_END << std::endl;
403  sequence_ = "Preparation For Getting Off";
404  }
405  if (iter->ev_door_state() == EvInfo::door_state::OPEN) { // 3. 문 열림
406  if (iter->is_available()) {
407  if (sequence_ == "Moving" || sequence_ == "Preparation For Getting Off") {
408  std::cout << std::endl << GREEN << "<Arrived>" << C_END << std::endl;
409  sequence_ = "Arrived";
410  robot_data_.robot_status = robot_status::EXITING_EV;
411  push_door_open(allocated_ev_num_); // PDO
412  }
413  }
414  call_cancel_check_cnt_ = 0;
415  }
416  }
417  }
418  }
419 
420  // print ev status to ternimal
421  if (ev_num_to_print_status > 0) {
422  ev_info_.at(ev_num_to_print_status - 1).print_registered_call();
423  }
424 
425  // 콜 등록 취소 체크
426  if (sequence_ == "Waiting For Elevator" || sequence_ == "Preparation For Taking On") {
427  // 출발층에서 stop 상태가 아닌 ev가 하나라도 있다면 콜 재등록 검사 x
428  // 출발층 도착해서 문열리기 직전 구간
429  for (auto iter = ev_info_.begin(); iter != ev_info_.end(); iter++) {
430  if (iter->current_floor() == call_floor_) {
431  if (iter->direction() != EvInfo::ev_direction::STOP) {
432  return;
433  }
434  }
435  }
436  if (robot_service_dir_ == call_dir::UP) {
437  if (up_hallcall_check(call_floor_)) {
438  call_cancel_check_cnt_ = 0;
439  } else if (++call_cancel_check_cnt_ > call_cancel_check_cnt_max_) {
440  hall_call(robot_data_.el_id, call_dir::UP, call_floor_);
441  } else {
442  std::cout << YELLOW << "up hallcall not registered (" << call_cancel_check_cnt_ << "/" <<
443  call_cancel_check_cnt_max_ << ")" << C_END << std::endl;
444  }
445  } else if (robot_service_dir_ == call_dir::DOWN) {
446  if (down_hallcall_check(call_floor_)) {
447  call_cancel_check_cnt_ = 0;
448  } else if (++call_cancel_check_cnt_ > call_cancel_check_cnt_max_) {
449  hall_call(robot_data_.el_id, call_dir::DOWN, call_floor_);
450  } else {
451  std::cout << YELLOW << "down hallcall not registered (" << call_cancel_check_cnt_ << "/" <<
452  call_cancel_check_cnt_max_ << ")" << C_END << std::endl;
453  }
454  }
455  } else if (sequence_ == "Moving") {
456  // 목적층에서 stop 상태가 아닌 ev가 하나라도 있다면 콜 재등록 검사 x
457  // 목적층 도착해서 문열리기 직전 구간
458  for (auto iter = ev_info_.begin(); iter != ev_info_.end(); iter++) {
459  if (iter->current_floor() == dest_floor_) {
460  if (iter->direction() != EvInfo::ev_direction::STOP) {
461  return;
462  }
463  }
464  }
465  if (carcall_check(dest_floor_, allocated_ev_num_)) {
466  call_cancel_check_cnt_ = 0;
467  } else if (++call_cancel_check_cnt_ > call_cancel_check_cnt_max_) {
468  car_call(allocated_ev_num_, dest_floor_);
469  } else {
470  std::cout << YELLOW << "carcall not registered (" << C_END << call_cancel_check_cnt_ << "/" <<
471  call_cancel_check_cnt_max_ << ")" << std::endl;
472  }
473  }
474  // Todo
475  // 콜 재등록 일정카운트 넘어가면 Service End ->Start 등 처리
476  // 무한히 재등록하면서 기다리는 것 방지

481 bool TkeControllerNode::hall_call(int ev_num, call_dir direction, int floor, int retry_count)
482 {
483  if (!is_available_check()) {
484  std::cout << RED << "(hallcall) 호출 가능한 EV가 없습니다." << C_END << std::endl;
485  return false;
486  }
487  if (all_ev_locked_check(floor)) {
488  std::cout << RED << "(" << floor << ") 층의 모든 호기가 잠김" << C_END << std::endl;
489  return false;
490  }
491  uint8_t tx_packet[17]{
492  0xF0, 0x72, 0x00, 0x00, 0x00, 0x08,
493  0x01, 0x00, 0x00, 0x00, 0x00, 0x00,
494  0x00, 0x00, 0x00, 0x00, 0xF4};
495 
496  int comm_result = COMM_TX_FAIL;
497 
498  robot_data_.el_id = static_cast<uint8_t>(ev_num);
499  robot_data_.control_cmd =
500  (direction == UP) ? control_cmd::UP_HALLCALL : control_cmd::DOWN_HALLCALL;
501  if (ev_num == 0) {
502  robot_data_.floor = floor;
503  } else {
504  try {
505  if (ev_info_.at(static_cast<unsigned int>(ev_num - 1)).ev_id_ == ev_num) {
506  robot_data_.floor = floor;
507  } else {
508  std::cout << RED << "ev num does not match.." << C_END << std::endl;
509  return false;
510  }
511  } catch (std::out_of_range & e) {
512  std::cout << RED << "ERROR ev_num out of range" << C_END << std::endl;
513  return false;
514  }
515  }
516 
517  tx_packet[6] = robot_data_.robot_id;
518  tx_packet[7] = robot_data_.robot_status;
519  tx_packet[8] = robot_data_.robot_status_2;
520  tx_packet[9] = robot_data_.el_id;
521  tx_packet[10] = ev_info_.at(ev_num - 1).protocol_floor(robot_data_.floor);
522  tx_packet[11] = robot_data_.control_cmd;
523 
524  comm_result = packet_handler_->tx_packet(port_handler_, tx_packet);
525 
526  if (comm_result != COMM_SUCCESS) {
