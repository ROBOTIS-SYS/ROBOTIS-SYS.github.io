46 void OdomSmoother::odomCallback(const nav_msgs::msg::Odometry::SharedPtr msg)
47 {
48  std::lock_guard<std::mutex> lock(odom_mutex_);
49 
50  // update cumulated odom only if history is not empty
51  if (!odom_history_.empty()) {
52  // to store current time
53  auto current_time = rclcpp::Time(msg->header.stamp);
54 
55  // to store time of the first odom in history
56  auto front_time = rclcpp::Time(odom_history_.front().header.stamp);
57 
58  // update cumulated odom when duration has exceeded and pop earliest msg
59  while (current_time - front_time > odom_history_duration_) {
60  const auto & odom = odom_history_.front();
61  odom_cumulate_.twist.twist.linear.x -= odom.twist.twist.linear.x;
62  odom_cumulate_.twist.twist.linear.y -= odom.twist.twist.linear.y;
63  odom_cumulate_.twist.twist.linear.z -= odom.twist.twist.linear.z;
64  odom_cumulate_.twist.twist.angular.x -= odom.twist.twist.angular.x;
65  odom_cumulate_.twist.twist.angular.y -= odom.twist.twist.angular.y;
66  odom_cumulate_.twist.twist.angular.z -= odom.twist.twist.angular.z;
67  odom_history_.pop_front();
68 
69  if (odom_history_.empty()) {
70  break;
71  }
72 
73  // update with the timestamp of earliest odom message in history
74  front_time = rclcpp::Time(odom_history_.front().header.stamp);
75  }
76  }
77 
78  odom_history_.push_back(*msg);
79  updateState();
80 }
