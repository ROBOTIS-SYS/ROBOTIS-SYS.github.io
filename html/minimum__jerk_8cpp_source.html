33 bool JointMinimumJerk::calculation(
34  rm2_common::DataArray start_position,
35  rm2_common::DataArray goal_position,
36  rm2_common::DataArray start_velocity,
37  rm2_common::DataArray goal_velocity)
38 {
39  if (start_position.size() != goal_position.size()) {
40  return false;
41  }
42 
43  if (start_velocity.size() == 0) {
44  start_velocity.set_zero(start_position.size());
45  } else if (start_velocity.size() != start_position.size()) {
46  return false;
47  }
48  if (goal_velocity.size() == 0) {
49  goal_velocity.set_zero(goal_position.size());
50  } else if (goal_velocity.size() != goal_position.size()) {
51  return false;
52  }
53 
54  dof_ = start_position.size();
55  coefficient_.resize(6, dof_);
56 
57  rm2_common::DataArray start_acceleration(start_position.size());
58  rm2_common::DataArray goal_acceleration(start_position.size());
59  coefficient_.row(0) = start_position.data.transpose();
60  coefficient_.row(1) = start_velocity.data.transpose();
61  coefficient_.row(2) = 0.5 * start_acceleration.data.transpose();
62 
63  Eigen::Matrix3d A = Eigen::Matrix3d::Identity();
64  Eigen::MatrixXd x = Eigen::MatrixXd::Zero(3, dof_);
65  Eigen::MatrixXd b = Eigen::MatrixXd::Zero(3, dof_);
66 
67  A << pow(move_time_, 3), pow(move_time_, 4), pow(move_time_, 5),
68  3 * pow(move_time_, 2), 4 * pow(move_time_, 3), 5 * pow(move_time_, 4),
69  6 * pow(move_time_, 1), 12 * pow(move_time_, 2), 20 * pow(move_time_, 3);
70 
71  b <<
72  (goal_position - start_position -
73  (start_velocity * move_time_ + 0.5 * start_acceleration * pow(move_time_, 2))).data.transpose(),
74  (goal_velocity - start_velocity - (start_acceleration * move_time_)).data.transpose(),
75  (goal_acceleration - start_acceleration).data.transpose();
76 
77  Eigen::ColPivHouseholderQR<Eigen::Matrix3d> dec(A);
78  x = dec.solve(b);
79 
80  coefficient_.block(3, 0, 3, dof_) = x;
81 
82  return true;
83 }

120 bool TaskMinimumJerk::calculation(
121  rm2_common::Pose start_pose, rm2_common::Pose goal_pose,
122  rm2_common::Twist start_twist, rm2_common::Twist goal_twist)
123 {
124  rm2_common::Vector start_acceleration = rm2_common::Vector::Zero();
125  rm2_common::Vector goal_acceleration = rm2_common::Vector::Zero();
126 
127  position_coefficient_.resize(6, 3);
128 
129  position_coefficient_.row(0) = start_pose.position.data.transpose();
130  position_coefficient_.row(1) = start_twist.linear.data.transpose();
131  position_coefficient_.row(2) = 0.5 * start_acceleration.data.transpose();
132 
133 
134  orientation_coefficient_.resize(6);
135 
136  start_orientation_ = start_pose.orientation;
137  goal_orientation_ = goal_pose.orientation;
138  start_angular_velocity_ = rm2_common::skewSymmetricMatrix(start_twist.angular) *
139  start_orientation_.data;
140  goal_angular_velocity_ = rm2_common::skewSymmetricMatrix(goal_twist.angular) *
141  goal_orientation_.data;
142 
143  double start_point_p = 0.0;
144  double start_point_v = start_twist.angular.norm();
145  double start_point_a = 0.0;
146  double end_point_p = 1.0;
147  double end_point_v = goal_twist.angular.norm();
148  double end_point_a = 0.0;
149  orientation_coefficient_(0) = start_point_p;
150  orientation_coefficient_(1) = start_point_v;
151  orientation_coefficient_(2) = 0.5 * start_point_a;
152 
153 
154  Eigen::Matrix3d A = Eigen::Matrix3d::Identity();
155  A << pow(move_time_, 3), pow(move_time_, 4), pow(move_time_, 5),
156  3 * pow(move_time_, 2), 4 * pow(move_time_, 3), 5 * pow(move_time_, 4),
157  6 * pow(move_time_, 1), 12 * pow(move_time_, 2), 20 * pow(move_time_, 3);
158 
159  Eigen::Matrix3d b = Eigen::Matrix3d::Zero();
160  b <<
161  (goal_pose.position - start_pose.position -
162  (start_twist.linear * move_time_ + 0.5 * start_acceleration *
163  pow(move_time_, 2))).data.transpose(),
164  (goal_twist.linear - start_twist.linear - (start_acceleration * move_time_)).data.transpose(),
165  (goal_acceleration - start_acceleration).data.transpose();
166 
167  Eigen::Matrix3d x = Eigen::Matrix3d::Zero();
168  Eigen::ColPivHouseholderQR<Eigen::Matrix3d> dec(A);
169  x = dec.solve(b);
170  position_coefficient_.block(3, 0, 3, 3) = x;
171 
172  Eigen::Vector3d rot_b = Eigen::Vector3d::Zero();
173  rot_b <<
174  (end_point_p - start_point_p -
175  (start_point_v * move_time_ + 0.5 * start_point_a * pow(move_time_, 2))),
176  (end_point_v - start_point_v - (start_point_a * move_time_)),
177  (end_point_a - start_point_a);
178 
179  Eigen::Vector3d rot_x = Eigen::Vector3d::Zero();
180  Eigen::ColPivHouseholderQR<Eigen::Matrix3d> rot_dec(A);
181  rot_x = rot_dec.solve(rot_b);
182 
183  orientation_coefficient_(3) = rot_x(0);
184  orientation_coefficient_(4) = rot_x(1);
185  orientation_coefficient_(5) = rot_x(2);
186 
187  return true;
188 }

190 rm2_common::TaskValues TaskMinimumJerk::get(
191  double time,
192  bool & finish)
193 {
194  // Check Time
195  if (time >= move_time_) {
196  finish = true;
197  time = move_time_;
198  } else if (time < 0) {
199  finish = false;
200  time = 0.0;
201  } else {
202  finish = false;
203  }
204 
205  rm2_common::Pose pose;
206  rm2_common::Twist twist;
207 
208  // Position
209  Eigen::MatrixXd A(2, 6);
210  A << 1, pow(time, 1), pow(time, 2), pow(time, 3), pow(time, 4), pow(time, 5),
211  0, 1, 2 * pow(time, 1), 3 * pow(time, 2), 4 * pow(time, 3), 5 * pow(time, 4);
212  auto linear_result_ = A * position_coefficient_;
213 
214  // Goal
215  pose.position = linear_result_.row(0).transpose();
216  twist.linear = linear_result_.row(1).transpose();
217 
218  // Orientation
219  double point_pos = orientation_coefficient_(0) +
220  orientation_coefficient_(1) * pow(time, 1) +
221  orientation_coefficient_(2) * pow(time, 2) +
222  orientation_coefficient_(3) * pow(time, 3) +
223  orientation_coefficient_(4) * pow(time, 4) +
224  orientation_coefficient_(5) * pow(time, 5);
225 
226  double point_vel = orientation_coefficient_(1) +
227  2 * orientation_coefficient_(2) * pow(time, 1) +
228  3 * orientation_coefficient_(3) * pow(time, 2) +
229  4 * orientation_coefficient_(4) * pow(time, 3) +
230  5 * orientation_coefficient_(5) * pow(time, 4);
231 
232  // Goal
233  // orientation
234  pose.orientation = start_orientation_.slerp(point_pos, goal_orientation_);
235 
236  // omega (velocity)
237  // R.dot
238  rm2_common::Rotation r_dot;
239  r_dot = start_angular_velocity_.slerp(point_vel, goal_angular_velocity_);
240 
241  rm2_common::Rotation omega_skew = r_dot * pose.orientation.transpose();
242  if (omega_skew(0, 0) == 0.0 && omega_skew(1, 1) == 0.0 && omega_skew(2, 2) == 0.0 &&
243  rm2_common::equal(omega_skew(2, 1), omega_skew(1, 2)) &&
244  rm2_common::equal(omega_skew(0, 2), omega_skew(2, 0)) &&
245  rm2_common::equal(omega_skew(1, 0), omega_skew(1, 0)))
246  {
247  // omega
248  twist.angular = rm2_common::Vector(
249  omega_skew(2, 1),
250  omega_skew(0, 2),
251  omega_skew(1, 0));
252  } else {
253  twist.angular = rm2_common::Vector(); // Todo fix
254  }
255 
256  return rm2_common::TaskValues(pose, twist);
257 }
