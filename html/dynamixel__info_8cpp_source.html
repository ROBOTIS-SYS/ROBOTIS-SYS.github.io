31 void DynamixelInfo::InitDxlModelInfo()

55 void DynamixelInfo::ReadDxlModelFile(uint8_t id, uint16_t model_num)
56 {
57  std::string path = dxl_model_file_dir + "/";
58 
59  // find model file path
60  auto it = dxl_model_list_.find(model_num);
61  if (it != dxl_model_list_.end()) {
62  path += it->second;
63  } else {
64  fprintf(stderr, "[ERROR] CANNOT FIND THE DXL MODEL FROM FILE LIST.\n");
65  return;
66  }
67 
68  // fprintf(stderr, "[DBG] find model file\n");
69 
70  // file open
71  std::ifstream open_file(path);
72  if (open_file.is_open() != 1) {
73  fprintf(stderr, "[ERROR] CANNOT FIND DXL [%s] MODEL FILE.\n", path.c_str());
74  exit(-1);
75  }
76  // fprintf(stderr, "[DBG] find model file open\n");
77 
78  DxlInfo temp_dxl_info;
79  std::string line;
80 
81  temp_dxl_info.model_num = model_num;
82 
83  // read type info
84  while (!open_file.eof() ) {
85  getline(open_file, line);
86  if (strcmp(line.c_str(), "[control table]") == 0) {
87  break;
88  }
89 
90  std::vector<std::string> strs;
91  boost::split(strs, line, boost::is_any_of("\t"));
92 
93  if (strs.at(0) == "value_of_zero_radian_position") {
94  temp_dxl_info.value_of_zero_radian_position = static_cast<int32_t>(stoi(strs.at(1)));
95  } else if (strs.at(0) == "value_of_max_radian_position") {
96  temp_dxl_info.value_of_max_radian_position = static_cast<int32_t>(stoi(strs.at(1)));
97  } else if (strs.at(0) == "value_of_min_radian_position") {
98  temp_dxl_info.value_of_min_radian_position = static_cast<int32_t>(stoi(strs.at(1)));
99  } else if (strs.at(0) == "min_radian") {
100  temp_dxl_info.min_radian = static_cast<double>(stod(strs.at(1)));
101  } else if (strs.at(0) == "max_radian") {
102  temp_dxl_info.max_radian = static_cast<double>(stod(strs.at(1)));
103  } else if (strs.at(0) == "torque_constant") {
104  temp_dxl_info.torque_constant = static_cast<double>(stod(strs.at(1)));
105  }
106  }
107  // fprintf(stderr, "[DBG] read type info\n");
108 
109  // read control table item
110  getline(open_file, line);
111  while (!open_file.eof() ) {
112  getline(open_file, line);
113  if (!open_file.good()) {
114  break;
115  }
116 
117  std::vector<std::string> strs;
118  boost::split(strs, line, boost::is_any_of("\t"));
119 
120  // fprintf(stderr, "[DBG] address %s.\n", strs.at(0).c_str());
121  // fprintf(stderr, "[DBG] size %s.\n", strs.at(1).c_str());
122  // fprintf(stderr, "[DBG] name %s.\n", strs.at(2).c_str());
123 
124  ControlItem temp;
125  temp.address = static_cast<uint16_t>(stoi(strs.at(0)));
126  temp.size = static_cast<uint8_t>(stoi(strs.at(1)));
127  temp.item_name = strs.at(2);
128  temp_dxl_info.item.push_back(temp);
129  }
130  // fprintf(stderr, "[DBG] read control item\n");
131 
132  dxl_info_[id] = temp_dxl_info;
133  open_file.close();
134 }

166 int32_t DynamixelInfo::ConvertRadianToValue(uint8_t id, double radian)
167 {
168  if (radian > 0) {
169  return static_cast<int32_t>(radian *
170  (dxl_info_[id].value_of_max_radian_position -
171  dxl_info_[id].value_of_zero_radian_position) / dxl_info_[id].max_radian) +
172  dxl_info_[id].value_of_zero_radian_position;
173  } else if (radian < 0) {
174  return static_cast<int32_t>(radian *
175  (dxl_info_[id].value_of_min_radian_position -
176  dxl_info_[id].value_of_zero_radian_position) / dxl_info_[id].min_radian) +
177  dxl_info_[id].value_of_zero_radian_position;
178  } else {
179  return dxl_info_[id].value_of_zero_radian_position;
180  }
181 }

183 double DynamixelInfo::ConvertValueToRadian(uint8_t id, int32_t value)
184 {
185  if (value > dxl_info_[id].value_of_zero_radian_position) {
186  return static_cast<double>(value - dxl_info_[id].value_of_zero_radian_position) *
187  dxl_info_[id].max_radian /
188  static_cast<double>(dxl_info_[id].value_of_max_radian_position -
189  dxl_info_[id].value_of_zero_radian_position);
190  } else if (value < dxl_info_[id].value_of_zero_radian_position) {
191  return static_cast<double>(value - dxl_info_[id].value_of_zero_radian_position) *
192  dxl_info_[id].min_radian /
193  static_cast<double>(dxl_info_[id].value_of_min_radian_position -
194  dxl_info_[id].value_of_zero_radian_position);
195  } else {
196  return 0.0;
197  }
