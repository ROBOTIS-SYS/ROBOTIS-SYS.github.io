56 void TeleopJoystick::joy_callback(const sensor_msgs::msg::Joy::SharedPtr msg)
57 {
58  static uint8_t status = 0;
59  static double target_linear_velocity = 0.0;
60  static double target_angular_velocity = 0.0;
61 
62  if (status != 0 && status % 20 == 0) {
63  print_msgs();
64  }
65 
66  if (msg->axes.at(7) == 1.0) {
67  target_linear_velocity =
68  check_linear_velocity_limit(target_linear_velocity + LIN_VEL_STEP_SIZE);
69  status += 1;
70  print_vels(target_linear_velocity, target_angular_velocity);
71  } else if (msg->axes.at(7) == -1.0) {
72  target_linear_velocity =
73  check_linear_velocity_limit(target_linear_velocity - LIN_VEL_STEP_SIZE);
74  status += 1;
75  print_vels(target_linear_velocity, target_angular_velocity);
76  } else if (msg->axes.at(6) == 1.0) {
77  target_angular_velocity =
78  check_angular_velocity_limit(target_angular_velocity + ANG_VEL_STEP_SIZE);
79  status = status + 1;
80  print_vels(target_linear_velocity, target_angular_velocity);
81  } else if (msg->axes.at(6) == -1.0) {
82  target_angular_velocity =
83  check_angular_velocity_limit(target_angular_velocity - ANG_VEL_STEP_SIZE);
84  status = status + 1;
85  print_vels(target_linear_velocity, target_angular_velocity);
86  } else if (msg->buttons.at(0) == 1) {
87  target_linear_velocity = 0.0;
88  target_angular_velocity = 0.0;
89  print_vels(target_linear_velocity, target_angular_velocity);
90  }
91 
92  geometry_msgs::msg::Twist twist;
93 
94  twist.linear.x = target_linear_velocity;
95  twist.linear.y = 0.0;
96  twist.linear.z = 0.0;
97 
98  twist.angular.x = 0.0;
99  twist.angular.y = 0.0;
100  twist.angular.z = target_angular_velocity;
101 
102  cmd_vel_pub_->publish(twist);
103 }

126 double TeleopJoystick::constrain(double input_vel, double low_bound, double high_bound)
127 {
128  if (input_vel < low_bound) {
129  input_vel = low_bound;
130  } else if (input_vel > high_bound) {
131  input_vel = high_bound;
132  } else {
133  input_vel = input_vel;
134  }
135 
136  return input_vel;
137 }
