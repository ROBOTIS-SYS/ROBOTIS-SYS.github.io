28 bool SingleTrapezoidal::calculation(
29  double start_position, double goal_position,
30  double start_velocity, double goal_velocity,
31  double max_vel, double max_acc)
32 {
33  if (max_vel < fabs(start_velocity)) {
34  start_velocity = max_vel;
35  }
36  if (max_vel < fabs(goal_velocity)) {
37  goal_velocity = max_vel;
38  }
39 // std::cout << "---CALCULATION---" << std::endl;
40 
41 // std::cout << "<Param>" << std::endl;
42 // std::cout << "start_position : " << start_position << std::endl;
43 // std::cout << "goal_position : " << goal_position << std::endl;
44 // std::cout << "start_velocity : " << start_velocity << std::endl;
45 // std::cout << "goal_velocity : " << goal_velocity << std::endl;
46 // std::cout << "max_vel : " << max_vel << std::endl;
47 // std::cout << "max_acc : " << max_acc << std::endl;
48 
49 // std::cout << "<Calc>" << std::endl;
50  // Deredtion
51  double sig = sign(goal_position - start_position);
52 // std::cout << "Deredtion : " << sig << std::endl;
53 
54  // Acceleration time
55  double acc_time = (max_vel - sig * start_velocity) / max_acc;
56  double delta_position_to_time_1 = sig *
57  max_acc * (acc_time * acc_time) / 2.0 + start_velocity * acc_time;
58 
59 // std::cout << "acc_time : " << acc_time << std::endl;
60 // std::cout << "delta_position_to_time_1_ : " << delta_position_to_time_1 << std::endl;
61 
62  // Deceleration time
63  double dec_time = (max_vel - sig * goal_velocity) / max_acc;
64  double delta_position_from_time_2 = sig *
65  max_acc * (dec_time * dec_time) / 2.0 + goal_velocity * dec_time;
66 // std::cout << "dec_time : " << dec_time << std::endl;
67 // std::cout << "delta_position_from_time_2_ : " << delta_position_from_time_2 << std::endl;
68 
69  // Constant velocity time
70  double const_vel_time =
71  (goal_position - start_position - delta_position_to_time_1 - delta_position_from_time_2) /
72  (sig * max_vel);
73 // std::cout << "const_vel_time : " << const_vel_time << std::endl;
74 
75  // set times
76  if (const_vel_time > 0.0) {
77  // plan a complete profile : trapezoidal
78  time_1_ = acc_time;
79  time_2_ = acc_time + const_vel_time;
80  move_time_ = acc_time + const_vel_time + dec_time;
81  } else {
82  // plan an incomplete profile : triangle
83  double over_time = (1.0 - sqrt(1.0 + 1.0 * (max_acc / max_vel) * const_vel_time)) /
84  (1.0 * (max_acc / max_vel));
85  if (std::isnan(over_time)) {
86  std::cout << "over_time : " << over_time << std::endl;
87  return false;
88  }
89  time_1_ = acc_time - over_time;
90  move_time_ = acc_time + dec_time - 2.0 * over_time;
91  time_2_ = time_1_;
92  }
93 // std::cout << "time_1_ : " << time_1_ << std::endl;
94 // std::cout << "time_2_ : " << time_2_ << std::endl;
95 // std::cout << "move_time_ : " << move_time_ << std::endl;
96 
97  // Set coefficient
98  coefficient_1_ <<
99  sig * max_acc / 2.0,
100  start_velocity,
101  start_position;
102  coefficient_2_ <<
103  0.0,
104  start_velocity + sig * max_acc * time_1_,
105  start_position - (sig * max_acc * time_1_ * time_1_) / 2.0;
106  coefficient_3_ <<
107  -sig * max_acc / 2.0,
108  goal_velocity + sig * max_acc * move_time_,
109  goal_position - goal_velocity * move_time_ - (sig * max_acc * move_time_ * move_time_) / 2.0;
110 
111 // std::cout << "coefficient_1_ : " << coefficient_1_.transpose() << std::endl;
112 // std::cout << "coefficient_2_ : " << coefficient_2_.transpose() << std::endl;
113 // std::cout << "coefficient_3_ : " << coefficient_3_.transpose() << std::endl;
114 
115 // std::cout << "------------------\n\n" << std::endl;
116  return true;
117 }

119 bool SingleTrapezoidal::change_move_time(double new_move_time)
120 {
121  double factor = move_time_ / new_move_time;
122  if (factor > 1) {
123  return false;
124  } else if (rm2_common::equal(factor, 1.0, 1e-6)) {
125  return true;
126  } else if (rm2_common::equal(move_time_, 0.0, 1e-6)) {
127  return true;
128  }
129 // std::cout << "------Change time-----" << std::endl;
130 // std::cout << factor << std::endl;
131  Eigen::Matrix3d factor_mat;
132  factor_mat <<
133  factor * factor, 0.0, 0.0,
134  0.0, factor, 0.0,
135  0.0, 0.0, 1.0;
136  coefficient_1_ = factor_mat * coefficient_1_;
137  coefficient_2_ = factor_mat * coefficient_2_;
138  coefficient_3_ = factor_mat * coefficient_3_;
139  time_1_ = time_1_ / factor;
140  time_2_ = time_2_ / factor;
141  move_time_ = new_move_time;
142 
143 // std::cout << "coefficient_1_ : " << coefficient_1_.transpose() << std::endl;
144 // std::cout << "coefficient_2_ : " << coefficient_2_.transpose() << std::endl;
145 // std::cout << "coefficient_3_ : " << coefficient_3_.transpose() << std::endl;
146 
147 // std::cout << "time_1_ : " << time_1_ << std::endl;
148 // std::cout << "time_2_ : " << time_2_ << std::endl;
149 
150 // std::cout << "------------------\n\n" << std::endl;
151 
152  return true;
153 }

158  if (tick_time < 0.0) {
159  tick_time = 0.0;

161  tick_time = move_time_;

165  position_t << tick_time * tick_time, tick_time, 1.0;

170 // std::cout << "goal : " << coefficient_2_.dot(position_t) << std::endl;

332 bool JointTrapezoidal::calculation(
333  rm2_common::DataArray start_position,
334  rm2_common::DataArray goal_position,
335  rm2_common::DataArray start_velocity,
336  rm2_common::DataArray goal_velocity)
337 {
338  // Check move time and goal size
339  if (start_position.size() != goal_position.size()) {
340  return false;
341  }
342 
343  // Check velocity sizee
344  if (start_velocity.size() == 0) {
345  start_velocity.set_zero(start_position.size());
346  } else if (start_velocity.size() != start_position.size()) {
347  return false;
348  }
349  if (goal_velocity.size() == 0) {
350  goal_velocity.set_zero(goal_position.size());
351  } else if (goal_velocity.size() != goal_position.size()) {
352  return false;
353  }
354 
355  // Set size
356  dof_ = start_position.size();
357 
358  trap_.clear();
359  for (int joint_num = 0; joint_num < dof_; ++joint_num) {
360  SingleTrapezoidal trap;
361  auto res = trap.calculation(
362  start_position(joint_num), goal_position(joint_num),
363  start_velocity(joint_num), goal_velocity(joint_num),
364  max_vel_, max_acc_);
365  if (res == false) {
366  return false;
367  }
368  trap_.push_back(trap);
369  }
370 
371  total_move_time_ = 0.0;
372  for (auto trap : trap_) {
373  total_move_time_ = std::max(total_move_time_, trap.move_time());
374  }
375 
376  if (time_sync_) {
377  for (int index = 0; index < trap_.size(); index++) {
378  if (!trap_.at(index).change_move_time(total_move_time_)) {
379  return false;
380  }
381  }
382  }
383 
384  return true;
385 }
