33 return_type HardwareInterface::configure(
34  const hardware_interface::HardwareInfo & info)
35 {
36  if (configure_default(info) != return_type::OK) {
37  return return_type::ERROR;
38  }
39 
40  // Motor Comm Module
41  uint8_t motor_id =
42  static_cast<uint8_t>(stoi(info_.hardware_parameters["motor_id"]));
43  uint32_t motor_baud_rate =
44  static_cast<uint32_t>(stoi(info_.hardware_parameters["motor_baud_rate"]));
45  std::string motor_port_name =
46  info_.hardware_parameters["motor_port_name"];
47  double motor_protocol_version =
48  stod(info_.hardware_parameters["motor_protocol_version"]);
49  uint32_t joint_acceleration =
50  static_cast<uint32_t>(stoi(info_.hardware_parameters["joint_acceleration"]));
51  bool loging_velocity =
52  (info_.hardware_parameters["loging_velocity"] == "true" ||
53  info_.hardware_parameters["loging_velocity"] == "True");
54  std::string velocity_log_file_path =
55  info_.hardware_parameters["velocity_log_file_path"];
56  motor_comm_module_.init(
57  motor_id, motor_baud_rate, motor_port_name,
58  motor_protocol_version, joint_acceleration,
59  loging_velocity, velocity_log_file_path);
60 
61  // Body Comm Module
62  uint8_t body_id =
63  static_cast<uint8_t>(stoi(info_.hardware_parameters["body_id"]));
64  uint32_t body_baud_rate =
65  static_cast<uint32_t>(stoi(info_.hardware_parameters["body_baud_rate"]));
66  std::string body_port_name =
67  info_.hardware_parameters["body_port_name"];
68  double body_protocol_version =
69  stod(info_.hardware_parameters["body_protocol_version"]);
70  body_comm_module_.init(
71  body_id, body_baud_rate, body_port_name,
72  body_protocol_version);
73 
74 
75  hw_stop_sec_ = stod(info_.hardware_parameters["hw_stop_duration_sec"]);
76  hw_states_.resize(info_.joints.size(), std::numeric_limits<double>::quiet_NaN());
77  hw_commands_.resize(info_.joints.size(), std::numeric_limits<double>::quiet_NaN());
78 
79  for (const hardware_interface::ComponentInfo & joint : info_.joints) {
80  // RRBotSystemPositionOnly has exactly one state and command interface on each joint
81  if (joint.command_interfaces.size() != 1) {
82  RCLCPP_FATAL(
83  rclcpp::get_logger("HardwareInterface"),
84  "Joint '%s' has %d command interfaces found. 1 expected.",
85  joint.name.c_str(), joint.command_interfaces.size());
86  return return_type::ERROR;
87  }
88 
89  if (joint.command_interfaces[0].name != hardware_interface::HW_IF_VELOCITY) {
90  RCLCPP_FATAL(
91  rclcpp::get_logger("HardwareInterface"),
92  "Joint '%s' have %s command interfaces found. '%s' expected.",
93  joint.name.c_str(), joint.command_interfaces[0].name.c_str(),
94  hardware_interface::HW_IF_VELOCITY);
95  return return_type::ERROR;
96  }
97 
98  if (joint.state_interfaces.size() != 1) {
99  RCLCPP_FATAL(
100  rclcpp::get_logger("HardwareInterface"),
101  "Joint '%s' has %d state interface. 1 expected.",
102  joint.name.c_str(), joint.state_interfaces.size());
103  return return_type::ERROR;
104  }
105 
106  if (joint.state_interfaces[0].name != hardware_interface::HW_IF_POSITION) {
107  RCLCPP_FATAL(
108  rclcpp::get_logger("HardwareInterface"),
109  "Joint '%s' have %s state interface. '%s' expected.",
110  joint.name.c_str(), joint.state_interfaces[0].name.c_str(),
111  hardware_interface::HW_IF_POSITION);
112  return return_type::ERROR;
113  }
114  }
115 
116  for (const hardware_interface::ComponentInfo & sensor : info_.sensors) {
117  motor_comm_module::MotorCommModule::TOF tof;
118  tof.frame_id = sensor.parameters.at("frame_id");
119  tof.max_distance = std::stof(sensor.parameters.at("max_distance"));
120  tof.min_distance = std::stof(sensor.parameters.at("min_distance"));
121  tof.fov = std::stof(sensor.parameters.at("field_of_view"));
122  tof.size = static_cast<uint8_t>(std::stoi(sensor.parameters.at("size")));
123  tof.confidence_threshold = std::stoi(sensor.parameters.at("confidence_threshold"));
124  motor_comm_module_.tof_.insert(std::make_pair(sensor.name.c_str(), tof));
125  }
126 
127  status_ = hardware_interface::status::CONFIGURED;
128  return return_type::OK;
129 }

131 std::vector<hardware_interface::StateInterface>
132 HardwareInterface::export_state_interfaces()
133 {
134  std::vector<hardware_interface::StateInterface> state_interfaces;
135  for (uint i = 0; i < info_.joints.size(); i++) {
136  state_interfaces.emplace_back(
137  hardware_interface::StateInterface(
138  info_.joints[i].name, hardware_interface::HW_IF_POSITION, &hw_states_[i]));
139  }
140 
141  state_interfaces.emplace_back(
142  hardware_interface::StateInterface(
143  "left_tof", "distance",
144  &motor_comm_module_.tof_["left_tof"].distance));
145  state_interfaces.emplace_back(
146  hardware_interface::StateInterface(
147  "left_tof", "confidence",
148  &motor_comm_module_.tof_.at("left_tof").confidence));
149  state_interfaces.emplace_back(
150  hardware_interface::StateInterface(
151  "right_tof", "distance",
152  &motor_comm_module_.tof_.at("right_tof").distance));
153  state_interfaces.emplace_back(
154  hardware_interface::StateInterface(
155  "right_tof", "confidence",
156  &motor_comm_module_.tof_.at("right_tof").confidence));
157 
158  return state_interfaces;
159 }

174 return_type HardwareInterface::start()
175 {
176  RCLCPP_INFO(
177  rclcpp::get_logger("HardwareInterface"),
178  "Starting ...please wait...");
179 
180  // set some default values
181  for (uint i = 0; i < hw_states_.size(); i++) {
182  if (std::isnan(hw_states_[i])) {
183  hw_states_[i] = 0;
184  hw_commands_[i] = 0;
185  }
186  }
187 
188  // time sync
189  motor_comm_module_.write_time_sync();
190  body_comm_module_.write_time_sync();
191 
192  status_ = hardware_interface::status::STARTED;
193 
194  RCLCPP_INFO(
195  rclcpp::get_logger("HardwareInterface"),
196  "System Sucessfully started!");
197 
198  return return_type::OK;
199 }

201 return_type HardwareInterface::stop()
202 {
203  RCLCPP_INFO(
204  rclcpp::get_logger("HardwareInterface"),
205  "Stopping ...please wait...");
206 
207  for (int i = 0; i <= hw_stop_sec_; i++) {
208  rclcpp::sleep_for(std::chrono::seconds(1));
209  RCLCPP_INFO(
210  rclcpp::get_logger("HardwareInterface"),
211  "%.1f seconds left...", hw_stop_sec_ - i);
212  }
213 
214  status_ = hardware_interface::status::STOPPED;
215  return return_type::OK;
216 }
