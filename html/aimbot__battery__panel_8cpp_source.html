129 void AimbotBatteryPanel::battery_status_callback(
130  const sensor_msgs::msg::BatteryState::SharedPtr msg)
131 {
132  double voltage = msg->voltage * 1e-2;
133  std::stringstream voltage_msg;
134  voltage_msg << ((voltage < 24.0) ? "<font color=red>" : "<font color=green>") <<
135  voltage << "v (" << (msg->percentage * 100.0) << " %)</color>";
136 
137  QString status;
138  switch (msg->power_supply_status) {
139  case sensor_msgs::msg::BatteryState::POWER_SUPPLY_STATUS_CHARGING:
140  status = "<font color=red>Charging</color>";
141  break;
142 
143  case sensor_msgs::msg::BatteryState::POWER_SUPPLY_STATUS_NOT_CHARGING:
144  status = "<font color=green>Standby</color>";
145  break;
146 
147  default:
148  status = "Unknown";
149  break;
150  }
151 
152  QString battery_status(
153  "<table><tr><td width=100><b>Status:</b></td>"
154  "<td>" +
155  status + "</td></tr></table>");
156  QString battery_voltage(
157  "<table><tr><td width=100><b>Voltage:</b></td>"
158  "<td>" +
159  QString::fromStdString(voltage_msg.str()) + "</td></tr></table>");
160 
161  battery_voltage_->setText(battery_voltage);
162  battery_charging_status_->setText(battery_status);
163 }

165 void AimbotBatteryPanel::save_screen_callback(
166  const std_msgs::msg::Empty::SharedPtr msg)
167 {
168  (void)msg;
169 
170  // set file name
171  std::string home = std::string(getenv("HOME"));
172 
173  std::string default_file_path = home + "/log";
174  // make directory
175  DIR * dir = opendir(default_file_path.c_str());
176  if (dir) {
177  RCLCPP_INFO_STREAM(ros_node_->get_logger(), "Directory exists : " << default_file_path);
178  } else if (ENOENT == errno) {
179  RCLCPP_INFO_STREAM(ros_node_->get_logger(), "make dir : " << default_file_path);
180  mkdir(default_file_path.c_str(), 0777);
181  } else {
182  RCLCPP_ERROR_STREAM(ros_node_->get_logger(), "opendir() failed for some other reason.");
183  }
184  closedir(dir);
185 
186  std::string default_cap_file_path = default_file_path + "/cap";
187  dir = opendir(default_cap_file_path.c_str());
188  if (dir) {
189  RCLCPP_INFO_STREAM(ros_node_->get_logger(), "Directory exists : " << default_cap_file_path);
190  } else if (ENOENT == errno) {
191  RCLCPP_INFO_STREAM(ros_node_->get_logger(), "make dir : " << default_cap_file_path);
192  mkdir(default_cap_file_path.c_str(), 0777);
193  } else {
194  RCLCPP_ERROR_STREAM(ros_node_->get_logger(), "opendir() failed for some other reason.");
195  }
196  closedir(dir);
197 
198  rclcpp::Time ros_time = ros_node_->now();
199  time_t now = ros_time.seconds();
200  struct tm tstruct;
201  localtime_r(&now, &tstruct);
202 
203  char time_buf[100];
204  strftime(
205  time_buf, sizeof(time_buf), "%Y-%m-%d-%H-%M-%S",
206  &tstruct); // YYYY-MM-DD
207  std::string file_name = default_cap_file_path + "/rviz-" + time_buf + ".png";
208 
209  QPixmap original_pixmap;
210 
211  QScreen * screen = QGuiApplication::primaryScreen();
212  if (const QWindow * window = windowHandle()) {
213  screen = window->screen();
214  }
215  if (!screen) {
216  return;
217  }
218 
219  QWidget * parent_wid = this; // Current widget
220  while (parent_wid->parent()) { // find parent
221  parent_wid = parent_wid->parentWidget();
222  }
223 
224  // pop rviz window
225  QMessageBox m_box(parent_wid);
226  m_box.setText("Save Screen");
227  QPushButton * ok_button = m_box.addButton(QMessageBox::Ok);
228  QTimer::singleShot(150, ok_button, [&] {ok_button->click();});
229  int ret = m_box.exec();
230 
231  // save screen
232  if (ret == QMessageBox::Ok) {
233  rclcpp::sleep_for(std::chrono::milliseconds(150));
234  original_pixmap = screen->grabWindow(parent_wid->winId());
235  original_pixmap.save(file_name.c_str());
236  parent_wid->showMinimized(); // minimize window
237  return;
238  } else {
239  std::cout << "failed to save a screen" << std::endl;
240  return;
241  }
242 }
