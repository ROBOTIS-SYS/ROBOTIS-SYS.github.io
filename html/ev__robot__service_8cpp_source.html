111 bool EVRobotService::set_sequence(ServiceState state)
112 {
113  if (service_sequence_ == state) {
114  return false;
115  }
116 
117  // Waiting For Robot, Arrived인 경우 door timer 구동
118  auto door_timeout = [this](double timeout, ServiceState current_state) {
119  // log
120  std::stringstream log_ss;
121  log_ss << "Set timeout : " << timeout << ", state : " << get_service_name(current_state);
122  print_log(log_ss.str());
123 
124  constexpr int INTERVAL_MS = 100;
125  for (int ix = 0; ix < static_cast<int>(timeout * 1000.0); ix += INTERVAL_MS) {
126  usleep(INTERVAL_MS * 1000); // 100ms
127 
128  if (service_sequence_ != current_state) {
129  print_log("Timer is expired for changing E/L state.");
130  return;
131  }
132  }
133 
134  print_log("Set TIMEOUT at the end of the thread : " + get_service_name(service_sequence_));
135  set_signal(TIMEOUT_OUTSIDE); // 임시로 내외부 구분 안함
136  };
137 
138  if (state == WAITING_FOR_ROBOT) {
139  std::thread(door_timeout, timeout_outside_, state).detach();
140  } else if (state == ARRIVED) {
141  std::thread(door_timeout, timeout_inside_, state).detach();
142  }
143 
144  last_sequence_ = service_sequence_;
145  service_sequence_ = state;
146  check_closing_ = true;
147 
148  return true;
149 }

183 // 로봇으로 부터 오는 메시지 처리
184 bool EVRobotService::set_robot_action(ServiceState state)
185 {
186  if (service_sequence_ == NO_SERVICE) {
187  return false;
188  }
189 
190  switch (state) {
191  case TAKING_ON:
192  if (service_sequence_ == WAITING_FOR_ROBOT ||
193  service_sequence_ == PREPARATION_TAKING_ON)
194  {
195  set_sequence(MOVING);
196  return true;
197  }
198  break;
199 
200  case GETTING_OFF:
201  if (service_sequence_ == ARRIVED ||
202  service_sequence_ == PREPATATION_GETTING_OFF)
203  {
204  set_sequence(NO_SERVICE);
205  return true;
206  }
207  break;
208 
209  default:
210  return false;
211  }
212 
213  return false;
214 }

341 // EV 상태에 따라 로봇 서비스 상태 변경
342 bool EVRobotService::handle_ev_state(std::map<std::string, EVState::SharedPtr> & ev_state_list)
343 {
344  bool arrived = false, opened = false;
345  uint8_t ev_door_status;
346  auto allocated_it = ev_state_list.find(allocated_ev_num_);
347 
348  switch (service_sequence_) {
349  case NO_SERVICE:
350  return false;
351 
352  case COMMING:
353  case PREPARATION_TAKING_ON:
354  // EV별 도착 확인
355  for (auto it : ev_state_list) {
356  if (it.second->is_arrived_req(
357  call_floor_, direction_, false, use_hall_lantern_, arrived, opened))
358  {
359  if (arrived) {
360  allocated_ev_num_ = it.first;
361  std::cout << "Allocated EV num : " << allocated_ev_num_ << std::endl;
362  return set_sequence(opened ? WAITING_FOR_ROBOT : PREPARATION_TAKING_ON);
363 // } else {
364 // if (opened) {
365 // return set_sequence(DOOR_OPENED_OUTSIDE);
366 // }
367  }
368  }
369  }
370 
371  break;
372 
373  case WAITING_FOR_ROBOT:
374  // 할당 된 EV 체크
375  if (allocated_it == ev_state_list.end()) {
376  break;
377  }
378 
379  if (allocated_it->second->get_door_state(opened, ev_door_status)) {
380  if (check_closing_ && ev_door_status == EVStatusMsg::DOOR_CLOSING) {
381  // 로봇 반응 기다림, 문이 닫히는 중을 로봇에 알림
382  std::cout << "Sequence : DOOR_CLOSING" << std::endl;
383  check_closing_ = false;
384 // set_signal(DOOR_CLOSING);
385  }
386  if (ev_door_status == EVStatusMsg::DOOR_CLOSED) {
387  // 로봇 반응 기다림, 문이 닫힌 경우 서비스 실패
388  std::cout << "Sequence : NO_SERVICE" << std::endl;
389  return set_sequence(NO_SERVICE);
390  }
391  }
392  break;
393 
394  case TAKING_ON:
395  // 로봇 반응
396  for (auto ev_status : ev_state_list) {
397  ev_status.second->clear_ev_state();
398  }
399  break;
400 
401  case MOVING:
402  case PREPATATION_GETTING_OFF:
403  // EV 도착 확인(방향 체크 안함)
404  if (allocated_it->second->is_arrived_req(
405  dest_floor_, EVStatusMsg::DIRECTION_UNKNOWN,
406  true, false, arrived, opened) == false)
407  {
408  return false;
409  }
410  if (arrived) {
411  return set_sequence(opened ? ARRIVED : PREPATATION_GETTING_OFF);
412  // return true;
413 // } else {
414 // if (opened) {
415 // set_signal(DOOR_OPENED_INSIDE);
416 // }
417  }
418  break;
419 
420  case ARRIVED:
421  // 할당 된 EV 체크
422  if (allocated_it == ev_state_list.end()) {
423  break;
424  }
425 
426  if (allocated_it->second->get_door_state(opened, ev_door_status)) {
427  if (check_closing_ && ev_door_status == EVStatusMsg::DOOR_CLOSING) {
428  // 로봇 반응 기다림, 문이 닫히는 중을 로봇에 알림
429  std::cout << "Sequence : DOOR_CLOSING" << std::endl;
430  check_closing_ = false;
431 // set_signal(DOOR_CLOSING);
432  }
433  if (ev_door_status == EVStatusMsg::DOOR_CLOSED) {
434  // 로봇 반응 기다림, 문이 닫힌 경우 서비스 실패
435  std::cout << "Sequence : NO_SERVICE" << std::endl;
436  return set_sequence(NO_SERVICE);
437  }
438  }
439  break;
440 
441  case GETTING_OFF:
442  // 로봇 반응
443  for (auto ev_status : ev_state_list) {
444  ev_status.second->clear_ev_state();
445  }
446  break;
447 
448  default:
449  return false;
450  }
451 
452  return false;
453 }
