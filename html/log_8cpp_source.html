153 void Log::write_log_file(aimbot_log_interfaces::msg::Log log)
154 {
155  auto log_stream_ptr = std::make_shared<std::ofstream>();
156 
157  try {
158  // Get the current time
159  rclcpp::Time ros_time = this->now();
160  time_t now = ros_time.seconds();
161  struct tm tstruct;
162  localtime_r(&now, &tstruct);
163 
164  char time_buf[100];
165  strftime(time_buf, sizeof(time_buf), "%Y-%m-%d", &tstruct); // YYYY-MM-DD
166  std::string file_name = default_file_path + "/" + time_buf + ".csv";
167 
168  bool file_exist_flag = (access(file_name.c_str(), F_OK) != -1);
169  RCLCPP_INFO_STREAM(get_logger(), "save_path : " << file_name.c_str());
170  RCLCPP_INFO_STREAM(get_logger(), (file_exist_flag ? "File is exist" : "Make a new file"));
171 
172  log_stream_ptr->open(file_name, std::ofstream::out | std::ofstream::app);
173 
174  if (file_exist_flag == false) {
175  *log_stream_ptr <<
176  "Data,Time,UNIX Time,Level,Type,Source,Contents,Current Floor,Current Position," <<
177  "Target POI,Battery Status,Battery Voltage,Capture\n";
178  }
179 
180  char log_time[50];
181  std::strftime(log_time, sizeof(log_time), "%Y-%m-%d,%H:%M:%S", &tstruct);
182  double ms =
183  (ros_time.seconds() - static_cast<double>(static_cast<int>(ros_time.seconds()))) * 1000.0;
184 
185  std::string level;
186  switch (log.level) {
187  case log.LEVEL_HIGH: level = "HIGH"; break;
188  case log.LEVEL_MEDIUM: level = "MEDIUM"; break;
189  case log.LEVEL_LOW: level = "LOW"; break;
190  case log.LEVEL_DEBUG: level = "DEBUG"; break;
191  }
192 
193  std::string type;
194  switch (log.type) {
195  case log.TYPE_INFO: type = "INFO"; break;
196  case log.TYPE_WARN: type = "WARN"; break;
197  case log.TYPE_ERROR: type = "ERROR"; break;
198  }
199 
200  uint32_t precisionVal = 2;
201  std::string current_position =
202  "x: " +
203  std::to_string(current_pose_2d_from_map_.x).substr(
204  0,
205  std::to_string(current_pose_2d_from_map_.x).find(".") + precisionVal + 1) + "/" +
206  "y: " +
207  std::to_string(current_pose_2d_from_map_.y).substr(
208  0,
209  std::to_string(current_pose_2d_from_map_.y).find(".") + precisionVal + 1) + "/" +
210  "theta: " + std::to_string(current_pose_2d_from_map_.theta * (180.0 / M_PI)).substr(
211  0, std::to_string(
212  current_pose_2d_from_map_.theta * (180.0 / M_PI)).find(".") + precisionVal + 1);
213 
214  *log_stream_ptr << log_time << "." <<
215  std::dec << std::setw(3) << std::setfill('0') << static_cast<int>(ms) << "," <<
216  std::fixed << std::setprecision(6) << ros_time.seconds() << "," <<
217  level << "," << type << "," <<
218  log.source << "," << log.contents << "," << current_floor_ << "," <<
219  current_position << "," << log.target_poi << "," << battery_status_ << "," <<
220  battery_voltage_;
221 
222  if (log.capture == true && capture_ && compressed_image_msg_.data.size()) {
223  char log_time[30];
224  {
225  std::time_t raw_time = static_cast<time_t>(compressed_image_msg_.header.stamp.sec);
226  struct tm timeinfo;
227  localtime_r(&raw_time, &timeinfo);
228  std::strftime(log_time, sizeof(log_time), "%Y-%m-%d-%H-%M-%S", &timeinfo);
229  }
230 
231  std::string file_name = default_cap_file_path + "/" + log_time + ".jpg";
232  *log_stream_ptr << "," <<
233  (save_image(file_name) ? std::string(log_time) + ".jpg" : "failure");
234 
235  } else if (log.capture == true && capture_) {
236  *log_stream_ptr << "," << "NO IMAGE";
237  }
238 
239  *log_stream_ptr << "\n";
240 
241  log_stream_ptr->close();
242  log_stream_ptr.reset();
243  } catch (std::exception & ex) {
244  RCLCPP_ERROR_STREAM(get_logger(), "[Aimbot Log] " << ex.what());
245  if (log_stream_ptr != nullptr) {
246  log_stream_ptr->close();
247  log_stream_ptr.reset();
248  }
249  }
250 }

253 bool Log::save_image(std::string file_name)
254 {
255  cv_bridge::CvImagePtr cv_ptr;
256  try {
257  cv_ptr = cv_bridge::toCvCopy(compressed_image_msg_, sensor_msgs::image_encodings::BGR8);
258  } catch (cv_bridge::Exception & e) {
259  RCLCPP_ERROR_STREAM(get_logger(), "cv_bridge exception: " << e.what());
260  return false;
261  }
262 
263  std::vector<int> compression_params;
264  compression_params.push_back(cv::IMWRITE_JPEG_QUALITY);
265  compression_params.push_back(10);
266 
267  try {
268  cv::imwrite(file_name.c_str(), cv_ptr->image, compression_params);
269  RCLCPP_INFO_STREAM(get_logger(), "Save the image : " << file_name);
270  } catch (std::runtime_error & ex) {
271  RCLCPP_ERROR_STREAM(get_logger(), "Failed to save the image");
272  return false;
273  }
274  return true;
275 }
