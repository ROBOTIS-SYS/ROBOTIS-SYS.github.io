276 void PageCheckTraySwingDoor::set_title_text(std::string option)
277 {
278  for (int i = 0; i < 4; i++) {
279  set_btn_enabled(i + 1, false);
280  }
281 
282  QString qstr_tray = "";
283  QStringList list = QString(option.c_str()).split("|");
284  for (const QString & str : list) {
285  bool ok;
286  int tray_num = str.toInt(&ok);
287  if (ok && tray_num >= 0 && tray_num < 5) {
288  if (tray_num == 0) {
289  tray_num_ = 0b1111;
290  } else {
291  if ( (/*ros_node_->config_cargo_merge_12_.compare("Y") == 0 &&*/ tray_num == 2) ||
292  (/*ros_node_->config_cargo_merge_34_.compare("Y") == 0 &&*/ tray_num == 4))
293  {
294  // do nothing
295  } else {
296  tray_num_ |= 1 << (tray_num - 1);
297  }
298  }
299  set_btn_enabled(tray_num, true);
300  if (tray_num == 0) {
301  if (true /*ros_node_->config_cargo_merge_12_.compare("Y") == 0*/) {
302  qstr_tray += " ①+② /";
303  } else {
304  qstr_tray += " ① / ② /";
305  }
306  if (true /*ros_node_->config_cargo_merge_34_.compare("Y") == 0*/) {
307  qstr_tray += " ③+④ /";
308  } else {
309  qstr_tray += " ③ / ④ /";
310  }
311  } else if (tray_num == 1) {
312  if (true /*ros_node_->config_cargo_merge_12_.compare("Y") == 0*/) {
313  qstr_tray += " ①+② /";
314  } else {
315  qstr_tray += " ① /";
316  }
317  } else if (tray_num == 2) {
318  if (true /*ros_node_->config_cargo_merge_12_.compare("Y") != 0*/) {
319  qstr_tray += " ② /";
320  }
321  } else if (tray_num == 3) {
322  if (true /*ros_node_->config_cargo_merge_34_.compare("Y") == 0*/) {
323  qstr_tray += " ③+④ /";
324  } else {
325  qstr_tray += " ③ /";
326  }
327  } else if (tray_num == 4) {
328  if (true /*ros_node_->config_cargo_merge_34_.compare("Y") != 0*/) {
329  qstr_tray += " ④ /";
330  }
331  }
332  }
333  }
334  qstr_tray.chop(1);
335  ui->lblTitle->setText(tr("<b>%1번 서랍</b>을 열고, 물건을 확인하세요.").arg(qstr_tray));

376 void PageCheckTraySwingDoor::on_timeout()
377 {
378  if (tray_num_ != opened_tray_num_) { // 서랍 안 열었음
379  if (++count_ > 5 &&
380  ros_node_->cargo_lock_state_[0] == true &&
381  ros_node_->cargo_lock_state_[1] == true)
382  {
383  // 서랍도 안 열고 3분 지남.
384  wait_timer_->stop();
385  emit languageChanged(ros_node_->config_language_);
386  ui->retranslateUi(this);
387 
388  // 다시 WaitingWith 로 돌아가서 트레이 체크를 기다림.
389  emit pageChanged("WaitingWith", "");
390  } else {
391  // emit playSound("Arrival_Items.mp3");
392  }
393  } else { // 서랍 열었음
394  if (++count_ > 5) {
395  // 3분 지남. 서랍은 모두 열었었으니 서랍이 열려있으면 닫고, 대기화면으로 돌아감.
396  if (ros_node_->cargo_lock_state_[0] == 0 || ros_node_->cargo_lock_state_[1] == 0 ||
397  ros_node_->cargo_lock_state_[2] == 0 || ros_node_->cargo_lock_state_[3] == 0)
398  {
399  is_tray_closing_ = true;
400  std::thread close_thread(tray_close_thread, this);
401  close_thread.detach();
402 
403  QGraphicsBlurEffect * gbe = new QGraphicsBlurEffect();
404  gbe->setBlurRadius(10);
405  this->setGraphicsEffect(gbe);
406  dlg_tray_moving_->set_closing_text();
407  dlg_tray_moving_->exec();
408 
409  while (is_tray_closing_) {
410  std::this_thread::sleep_for(std::chrono::duration<int>(1));
411  }
412  this->setGraphicsEffect(nullptr);
413  RCLCPP_INFO(ros_node_->get_logger(), "cargo_closed ");
414  }
415 
416  emit guiChanged("CheckTray", "", "");
417  emit pageChanged("Waiting", "");
418  } else {
419  // emit playSound("Received_G18.mp3");
420  }
421  }
422 }
