69 void
70 StaticLayer::onInitialize()
71 {
72  global_frame_ = layered_costmap_->getGlobalFrameID();
73 
74  getParameters();
75 
76  rclcpp::QoS map_qos(10); // initialize to default
77  if (map_subscribe_transient_local_) {
78  map_qos.transient_local();
79  map_qos.reliable();
80  map_qos.keep_last(1);
81  }
82 
83  RCLCPP_INFO(
84  node_->get_logger(),
85  "Subscribing to the map topic (%s) with %s durability",
86  map_topic_.c_str(),
87  map_subscribe_transient_local_ ? "transient local" : "volatile");
88  map_sub_ = node_->create_subscription<nav_msgs::msg::OccupancyGrid>(
89  map_topic_, map_qos,
90  std::bind(&StaticLayer::incomingMap, this, std::placeholders::_1));
91 
92  if (subscribe_to_updates_) {
93  RCLCPP_INFO(node_->get_logger(), "Subscribing to updates");
94  map_update_sub_ = node_->create_subscription<map_msgs::msg::OccupancyGridUpdate>(
95  map_topic_ + "_updates",
96  rclcpp::SystemDefaultsQoS(),
97  std::bind(&StaticLayer::incomingUpdate, this, std::placeholders::_1));
98  }

118 StaticLayer::getParameters()
119 {
120  int temp_lethal_threshold = 0;
121  double temp_tf_tol = 0.0;
122 
123  declareParameter("enabled", rclcpp::ParameterValue(true));
124  declareParameter("subscribe_to_updates", rclcpp::ParameterValue(false));
125  declareParameter("map_subscribe_transient_local", rclcpp::ParameterValue(true));
126  declareParameter("transform_tolerance", rclcpp::ParameterValue(0.0));
127  declareParameter("map_topic", rclcpp::ParameterValue(""));
128 
129  node_->get_parameter(name_ + "." + "enabled", enabled_);
130  node_->get_parameter(name_ + "." + "subscribe_to_updates", subscribe_to_updates_);
131  std::string private_map_topic, global_map_topic;
132  node_->get_parameter(name_ + "." + "map_topic", private_map_topic);
133  node_->get_parameter("map_topic", global_map_topic);
134  if (!private_map_topic.empty()) {
135  map_topic_ = private_map_topic;
136  } else {
137  map_topic_ = global_map_topic;
138  }
139  node_->get_parameter(
140  name_ + "." + "map_subscribe_transient_local",
141  map_subscribe_transient_local_);
142  node_->get_parameter("track_unknown_space", track_unknown_space_);
143  node_->get_parameter("use_maximum", use_maximum_);
144  node_->get_parameter("lethal_cost_threshold", temp_lethal_threshold);
145  node_->get_parameter("unknown_cost_value", unknown_cost_value_);
146  node_->get_parameter("trinary_costmap", trinary_costmap_);
147  node_->get_parameter("transform_tolerance", temp_tf_tol);
148 
149  // Enforce bounds
150  lethal_threshold_ = std::max(std::min(temp_lethal_threshold, 100), 0);
151  map_received_ = false;
152  update_in_progress_.store(false);
153 
154  transform_tolerance_ = tf2::durationFromSec(temp_tf_tol);

157 void
158 StaticLayer::processMap(const nav_msgs::msg::OccupancyGrid & new_map)
159 {
160  RCLCPP_DEBUG(node_->get_logger(), "StaticLayer: Process map");
161 
162  unsigned int size_x = new_map.info.width;
163  unsigned int size_y = new_map.info.height;
164 
165  RCLCPP_DEBUG(
166  node_->get_logger(),
167  "StaticLayer: Received a %d X %d map at %f m/pix", size_x, size_y,
168  new_map.info.resolution);
169 
170  // resize costmap if size, resolution or origin do not match
171  Costmap2D * master = layered_costmap_->getCostmap();
172  if (!layered_costmap_->isRolling() && (master->getSizeInCellsX() != size_x ||
173  master->getSizeInCellsY() != size_y ||
174  master->getResolution() != new_map.info.resolution ||
175  master->getOriginX() != new_map.info.origin.position.x ||
176  master->getOriginY() != new_map.info.origin.position.y ||
177  !layered_costmap_->isSizeLocked()))
178  {
179  // Update the size of the layered costmap (and all layers, including this one)
180  RCLCPP_INFO(
181  node_->get_logger(),
182  "StaticLayer: Resizing costmap to %d X %d at %f m/pix", size_x, size_y,
183  new_map.info.resolution);
184  layered_costmap_->resizeMap(
185  size_x, size_y, new_map.info.resolution,
186  new_map.info.origin.position.x,
187  new_map.info.origin.position.y,
188  true);
189  } else if (size_x_ != size_x || size_y_ != size_y || // NOLINT
190  resolution_ != new_map.info.resolution ||
191  origin_x_ != new_map.info.origin.position.x ||
192  origin_y_ != new_map.info.origin.position.y)
193  {
194  // only update the size of the costmap stored locally in this layer
195  RCLCPP_INFO(
196  node_->get_logger(),
197  "StaticLayer: Resizing static layer to %d X %d at %f m/pix", size_x, size_y,
198  new_map.info.resolution);
199  resizeMap(
200  size_x, size_y, new_map.info.resolution,
201  new_map.info.origin.position.x, new_map.info.origin.position.y);
202  }
203 
204  unsigned int index = 0;
205 
206  // we have a new map, update full size of map
207  std::lock_guard<Costmap2D::mutex_t> guard(*getMutex());
208 
209  // initialize the costmap with static data
210  for (unsigned int i = 0; i < size_y; ++i) {
211  for (unsigned int j = 0; j < size_x; ++j) {
212  unsigned char value = new_map.data[index];
213  costmap_[index] = interpretValue(value);
214  ++index;
215  }
216  }
217 
218  map_frame_ = new_map.header.frame_id;
219 
220  x_ = y_ = 0;
221  width_ = size_x_;
222  height_ = size_y_;
223  has_updated_data_ = true;
224 
225  current_ = true;
226 }

241 unsigned char
242 StaticLayer::interpretValue(unsigned char value)
243 {
244  // check if the static value is above the unknown or lethal thresholds
245  if (track_unknown_space_ && value == unknown_cost_value_) {
246  return NO_INFORMATION;
247  } else if (!track_unknown_space_ && value == unknown_cost_value_) {
248  return FREE_SPACE;
249  } else if (value >= lethal_threshold_) {
250  return LETHAL_OBSTACLE;
251  } else if (trinary_costmap_) {
252  return FREE_SPACE;
253  }
254 
255  double scale = static_cast<double>(value) / lethal_threshold_;
256  return scale * LETHAL_OBSTACLE;
257 }

259 void
260 StaticLayer::incomingMap(const nav_msgs::msg::OccupancyGrid::SharedPtr new_map)
261 {
262  std::lock_guard<Costmap2D::mutex_t> guard(*getMutex());
263  if (!map_received_) {
264  map_received_ = true;
265  processMap(*new_map);
266  }
267  if (update_in_progress_.load()) {
268  map_buffer_ = new_map;
269  } else {
270  processMap(*new_map);
271  map_buffer_ = nullptr;
272  }
273 }

275 void
276 StaticLayer::incomingUpdate(map_msgs::msg::OccupancyGridUpdate::ConstSharedPtr update)
277 {
278  std::lock_guard<Costmap2D::mutex_t> guard(*getMutex());
279  if (update->y < static_cast<int32_t>(y_) ||
280  y_ + height_ < update->y + update->height ||
281  update->x < static_cast<int32_t>(x_) ||
282  x_ + width_ < update->x + update->width)
283  {
284  RCLCPP_WARN(
285  node_->get_logger(),
286  "StaticLayer: Map update ignored. Exceeds bounds of static layer.\n"
287  "Static layer origin: %d, %d bounds: %d X %d\n"
288  "Update origin: %d, %d bounds: %d X %d",
289  x_, y_, width_, height_, update->x, update->y, update->width,
290  update->height);
291  return;
292  }
293 
294  if (update->header.frame_id != map_frame_) {
295  RCLCPP_WARN(
296  node_->get_logger(),
297  "StaticLayer: Map update ignored. Current map is in frame %s "
298  "but update was in frame %s",
299  map_frame_.c_str(), update->header.frame_id.c_str());
300  }
301 
302  unsigned int di = 0;
303  for (unsigned int y = 0; y < update->height; y++) {
304  unsigned int index_base = (update->y + y) * size_x_;
305  for (unsigned int x = 0; x < update->width; x++) {
306  unsigned int index = index_base + x + update->x;
307  costmap_[index] = interpretValue(update->data[di++]);
308  }
309  }
310 
311  x_ = update->x;
312  y_ = update->y;
313  width_ = update->width;
314  height_ = update->height;
315  has_updated_data_ = true;
316 }

319 void
320 StaticLayer::updateBounds(
321  double /*robot_x*/, double /*robot_y*/, double /*robot_yaw*/, double * min_x,
322  double * min_y,
323  double * max_x,
324  double * max_y)
325 {
326  if (!map_received_) {
327  return;
328  }
329 
330  std::lock_guard<Costmap2D::mutex_t> guard(*getMutex());
331  update_in_progress_.store(true);
332 
333  // If there is a new available map, load it.
334  if (map_buffer_) {
335  processMap(*map_buffer_);
336  map_buffer_ = nullptr;
337  }
338 
339  if (!layered_costmap_->isRolling() ) {
340  if (!(has_updated_data_ || has_extra_bounds_)) {
341  return;
342  }
343  }
344 
345  useExtraBounds(min_x, min_y, max_x, max_y);
346 
347  double wx, wy;
348 
349  mapToWorld(x_, y_, wx, wy);
350  *min_x = std::min(wx, *min_x);
351  *min_y = std::min(wy, *min_y);
352 
353  mapToWorld(x_ + width_, y_ + height_, wx, wy);
354  *max_x = std::max(wx, *max_x);
355  *max_y = std::max(wy, *max_y);
356 
357  has_updated_data_ = false;
358 }

360 void
361 StaticLayer::updateCosts(
362  antis_nav_costmap_2d::Costmap2D & master_grid,
363  int min_i, int min_j, int max_i, int max_j)
364 {
365  if (!enabled_) {
366  update_in_progress_.store(false);
367  return;
368  }
369  if (!map_received_) {
370  static int count = 0;
371  // throttle warning down to only 1/10 message rate
372  if (++count == 10) {
373  RCLCPP_WARN(node_->get_logger(), "Can't update static costmap layer, no map received");
374  count = 0;
375  }
376  update_in_progress_.store(false);
377  return;
378  }
379 
380  if (!layered_costmap_->isRolling()) {
381  // if not rolling, the layered costmap (master_grid) has same coordinates as this layer
382  if (!use_maximum_) {
383  updateWithTrueOverwrite(master_grid, min_i, min_j, max_i, max_j);
384  } else {
385  updateWithMax(master_grid, min_i, min_j, max_i, max_j);
386  }
387  } else {
388  // If rolling window, the master_grid is unlikely to have same coordinates as this layer
389  unsigned int mx, my;
390  double wx, wy;
391  // Might even be in a different frame
392  geometry_msgs::msg::TransformStamped transform;
393  try {
394  transform = tf_->lookupTransform(
395  map_frame_, global_frame_, tf2::TimePointZero,
396  transform_tolerance_);
397  } catch (tf2::TransformException & ex) {
398  RCLCPP_ERROR(node_->get_logger(), "StaticLayer: %s", ex.what());
399  update_in_progress_.store(false);
400  return;
401  }
402  // Copy map data given proper transformations
403  tf2::Transform tf2_transform;
404  tf2::fromMsg(transform.transform, tf2_transform);
405 
406  for (int i = min_i; i < max_i; ++i) {
407  for (int j = min_j; j < max_j; ++j) {
408  // Convert master_grid coordinates (i,j) into global_frame_(wx,wy) coordinates
409  layered_costmap_->getCostmap()->mapToWorld(i, j, wx, wy);
410  // Transform from global_frame_ to map_frame_
411  tf2::Vector3 p(wx, wy, 0);
412  p = tf2_transform * p;
413  // Set master_grid with cell from map
414  if (worldToMap(p.x(), p.y(), mx, my)) {
415  if (!use_maximum_) {
416  master_grid.setCost(i, j, getCost(mx, my));
417  } else {
418  master_grid.setCost(i, j, std::max(getCost(mx, my), master_grid.getCost(i, j)));
419  }
420  }
421  }
422  }
423  }
424  update_in_progress_.store(false);
425 }
