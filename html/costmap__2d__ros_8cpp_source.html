122 antis_nav_util::CallbackReturn
123 Costmap2DROS::on_configure(const rclcpp_lifecycle::State & /*state*/)
124 {
125  RCLCPP_INFO(get_logger(), "Configuring");
126  getParameters();
127 
128  // Create the costmap itself
129  layered_costmap_ = new LayeredCostmap(global_frame_, rolling_window_, track_unknown_space_);
130 
131  if (!layered_costmap_->isSizeLocked()) {
132  layered_costmap_->resizeMap(
133  (unsigned int)(map_width_meters_ / resolution_),
134  (unsigned int)(map_height_meters_ / resolution_), resolution_, origin_x_, origin_y_);
135  }
136 
137  // Create the transform-related objects
138  tf_buffer_ = std::make_shared<tf2_ros::Buffer>(rclcpp_node_->get_clock());
139  auto timer_interface = std::make_shared<tf2_ros::CreateTimerROS>(
140  rclcpp_node_->get_node_base_interface(),
141  rclcpp_node_->get_node_timers_interface());
142  tf_buffer_->setCreateTimerInterface(timer_interface);
143  tf_listener_ = std::make_shared<tf2_ros::TransformListener>(*tf_buffer_);
144 
145  // Then load and add the plug-ins to the costmap
146  for (unsigned int i = 0; i < plugin_names_.size(); ++i) {
147  RCLCPP_INFO(get_logger(), "Using plugin \"%s\"", plugin_names_[i].c_str());
148 
149  std::shared_ptr<Layer> plugin = plugin_loader_.createSharedInstance(plugin_types_[i]);
150  layered_costmap_->addPlugin(plugin);
151 
152  // TODO(mjeronimo): instead of get(), use a shared ptr
153  plugin->initialize(
154  layered_costmap_, plugin_names_[i], tf_buffer_.get(),
155  shared_from_this(), client_node_, rclcpp_node_);
156 
157  RCLCPP_INFO(get_logger(), "Initialized plugin \"%s\"", plugin_names_[i].c_str());
158  }
159 
160  // Create the publishers and subscribers
161  footprint_sub_ = create_subscription<geometry_msgs::msg::Polygon>(
162  "footprint",
163  rclcpp::SystemDefaultsQoS(),
164  std::bind(&Costmap2DROS::setRobotFootprintPolygon, this, std::placeholders::_1));
165 
166  footprint_pub_ = create_publisher<geometry_msgs::msg::PolygonStamped>(
167  "published_footprint", rclcpp::SystemDefaultsQoS());
168 
169  costmap_publisher_ = new Costmap2DPublisher(
170  shared_from_this(),
171  layered_costmap_->getCostmap(), global_frame_,
172  "costmap", always_send_full_costmap_);
173 
174  // Set the footprint
175  if (use_radius_) {
176  setRobotFootprint(makeFootprintFromRadius(robot_radius_));
177  } else {
178  std::vector<geometry_msgs::msg::Point> new_footprint;
179  makeFootprintFromString(footprint_, new_footprint);
180  setRobotFootprint(new_footprint);
181  }
182 
183  // Add cleaning service
184  clear_costmap_service_ = std::make_unique<ClearCostmapService>(shared_from_this(), *this);
185 
186  // parameter
187  parameters_client_ = std::make_shared<rclcpp::AsyncParametersClient>(
188  get_node_base_interface(),
189  get_node_topics_interface(),
190  get_node_graph_interface(),
191  get_node_services_interface());
192 
193  parameter_sub_ = parameters_client_->on_parameter_event(
194  std::bind(&Costmap2DROS::on_parameter_event_callback, this, std::placeholders::_1));
195 
196  return antis_nav_util::CallbackReturn::SUCCESS;
197 }

199 antis_nav_util::CallbackReturn
200 Costmap2DROS::on_activate(const rclcpp_lifecycle::State & /*state*/)
201 {
202  RCLCPP_INFO(get_logger(), "Activating");
203 
204  costmap_publisher_->on_activate();
205  footprint_pub_->on_activate();
206 
207  // First, make sure that the transform between the robot base frame
208  // and the global frame is available
209 
210  std::string tf_error;
211 
212  RCLCPP_INFO(get_logger(), "Checking transform");
213  rclcpp::Rate r(2);
214  while (rclcpp::ok() &&
215  !tf_buffer_->canTransform(
216  global_frame_, robot_base_frame_, tf2::TimePointZero, &tf_error))
217  {
218  RCLCPP_INFO(
219  get_logger(), "Timed out waiting for transform from %s to %s"
220  " to become available, tf error: %s",
221  robot_base_frame_.c_str(), global_frame_.c_str(), tf_error.c_str());
222 
223  // The error string will accumulate and errors will typically be the same, so the last
224  // will do for the warning above. Reset the string here to avoid accumulation
225  tf_error.clear();
226  r.sleep();
227  }
228 
229  // Create a thread to handle updating the map
230  stopped_ = true; // to active plugins
231  stop_updates_ = false;
232  map_update_thread_shutdown_ = false;
233 
234  map_update_thread_ = new std::thread(
235  std::bind(
236  &Costmap2DROS::mapUpdateLoop, this, map_update_frequency_));
237 
238  start();
239 
240  return antis_nav_util::CallbackReturn::SUCCESS;
241 }

294 void Costmap2DROS::on_parameter_event_callback(
295  const rcl_interfaces::msg::ParameterEvent::SharedPtr event)
296 {
297  using rcl_interfaces::msg::ParameterType;
298 
299  for (auto & changed_parameter : event->changed_parameters) {
300  const auto & type = changed_parameter.value.type;
301  const auto & name = changed_parameter.name;
302  const auto & value = changed_parameter.value;
303 
304  if (name == "plugins" && type == ParameterType::PARAMETER_STRING_ARRAY) {
305  plugin_names_ = value.string_array_value;
306  auto node = shared_from_this();
307  if (plugin_names_ == default_plugins_) {
308  for (size_t i = 0; i < default_plugins_.size(); ++i) {
309  antis_nav_util::declare_parameter_if_not_declared(
310  node, default_plugins_[i] + ".plugin", rclcpp::ParameterValue(default_types_[i]));
311  }
312  }
313  plugin_types_.resize(plugin_names_.size());
314 
315  for (size_t i = 0; i < plugin_names_.size(); ++i) {
316  plugin_types_[i] = antis_nav_util::get_plugin_type_param(node, plugin_names_[i]);
317  }
318  }
319 
320  if (name == "always_send_full_costmap" && type == ParameterType::PARAMETER_BOOL) {
321  always_send_full_costmap_ = value.bool_value;
322  }
323  if (name == "footprint" && type == ParameterType::PARAMETER_STRING) {
324  footprint_ = value.string_value;
325 
326  if (footprint_ != "" && footprint_ != "[]") {
327  // Footprint parameter has been specified, try to convert it
328  std::vector<geometry_msgs::msg::Point> new_footprint;
329  if (makeFootprintFromString(footprint_, new_footprint)) {
330  // The specified footprint is valid, so we'll use that instead of the radius
331  use_radius_ = false;
332  } else {
333  // Footprint provided but invalid, so stay with the radius
334  RCLCPP_ERROR(
335  get_logger(), "The footprint parameter is invalid: \"%s\", using radius (%lf) instead",
336  footprint_.c_str(), robot_radius_);
337  }
338  }
339  }
340  if (name == "footprint_padding" && type == ParameterType::PARAMETER_DOUBLE) {
341  footprint_padding_ = value.double_value;
342  }
343  if (name == "global_frame" && type == ParameterType::PARAMETER_STRING) {
344  global_frame_ = value.string_value;
345  }
346  if (name == "height" && type == ParameterType::PARAMETER_INTEGER) {
347  map_height_meters_ = value.integer_value;
348  }
349  if (name == "origin_x" && type == ParameterType::PARAMETER_DOUBLE) {
350  origin_x_ = value.double_value;
351  }
352  if (name == "origin_y" && type == ParameterType::PARAMETER_DOUBLE) {
353  origin_y_ = value.double_value;
354  }
355  if (name == "publish_frequency" && type == ParameterType::PARAMETER_DOUBLE) {
356  map_publish_frequency_ = value.double_value;
357  }
358  if (name == "resolution" && type == ParameterType::PARAMETER_DOUBLE) {
359  resolution_ = value.double_value;
360  }
361  if (name == "robot_base_frame" && type == ParameterType::PARAMETER_STRING) {
362  robot_base_frame_ = value.string_value;
363  }
364  if (name == "robot_radius" && type == ParameterType::PARAMETER_DOUBLE) {
365  robot_radius_ = value.double_value;
366  }
367  if (name == "rolling_window" && type == ParameterType::PARAMETER_BOOL) {
368  rolling_window_ = value.bool_value;
369  }
370  if (name == "track_unknown_space" && type == ParameterType::PARAMETER_BOOL) {
371  track_unknown_space_ = value.bool_value;
372  }
373  if (name == "transform_tolerance" && type == ParameterType::PARAMETER_DOUBLE) {
374  transform_tolerance_ = value.double_value;
375  }
376  if (name == "update_frequency" && type == ParameterType::PARAMETER_DOUBLE) {
377  map_update_frequency_ = value.double_value;
378  }
379  if (name == "width" && type == ParameterType::PARAMETER_INTEGER) {
380  map_width_meters_ = value.integer_value;
381  }
382  }

385 void
386 Costmap2DROS::getParameters()
387 {
388  RCLCPP_DEBUG(get_logger(), " getParameters");
389 
390  // Get all of the required parameters
391  get_parameter("always_send_full_costmap", always_send_full_costmap_);
392  get_parameter("footprint", footprint_);
393  get_parameter("footprint_padding", footprint_padding_);
394  get_parameter("global_frame", global_frame_);
395  get_parameter("height", map_height_meters_);
396  get_parameter("origin_x", origin_x_);
397  get_parameter("origin_y", origin_y_);
398  get_parameter("publish_frequency", map_publish_frequency_);
399  get_parameter("resolution", resolution_);
400  get_parameter("robot_base_frame", robot_base_frame_);
401  get_parameter("robot_radius", robot_radius_);
402  get_parameter("rolling_window", rolling_window_);
403  get_parameter("track_unknown_space", track_unknown_space_);
404  get_parameter("transform_tolerance", transform_tolerance_);
405  get_parameter("update_frequency", map_update_frequency_);
406  get_parameter("width", map_width_meters_);
407  get_parameter("plugins", plugin_names_);
408 
409  auto node = shared_from_this();
410 
411  if (plugin_names_ == default_plugins_) {
412  for (size_t i = 0; i < default_plugins_.size(); ++i) {
413  antis_nav_util::declare_parameter_if_not_declared(
414  node, default_plugins_[i] + ".plugin", rclcpp::ParameterValue(default_types_[i]));
415  }
416  }
417  plugin_types_.resize(plugin_names_.size());
418 
419  // 1. All plugins must have 'plugin' param defined in their namespace to define the plugin type
420  for (size_t i = 0; i < plugin_names_.size(); ++i) {
421  plugin_types_[i] = antis_nav_util::get_plugin_type_param(node, plugin_names_[i]);
422  }
423 
424  // 2. The map publish frequency cannot be 0 (to avoid a divde-by-zero)
425  if (map_publish_frequency_ > 0) {
426  publish_cycle_ = rclcpp::Duration::from_seconds(1 / map_publish_frequency_);
427  } else {
428  publish_cycle_ = rclcpp::Duration(-1);
429  }
430 
431  // 3. If the footprint has been specified, it must be in the correct format
432  use_radius_ = true;
433 
434  if (footprint_ != "" && footprint_ != "[]") {
435  // Footprint parameter has been specified, try to convert it
436  std::vector<geometry_msgs::msg::Point> new_footprint;
437  if (makeFootprintFromString(footprint_, new_footprint)) {
438  // The specified footprint is valid, so we'll use that instead of the radius
439  use_radius_ = false;
440  } else {
441  // Footprint provided but invalid, so stay with the radius
442  RCLCPP_ERROR(
443  get_logger(), "The footprint parameter is invalid: \"%s\", using radius (%lf) instead",
444  footprint_.c_str(), robot_radius_);
445  }
446  }
447 }

479 void
480 Costmap2DROS::mapUpdateLoop(double frequency)
481 {
482  RCLCPP_DEBUG(get_logger(), "mapUpdateLoop frequency: %lf", frequency);
483 
484  // the user might not want to run the loop every cycle
485  if (frequency == 0.0) {
486  return;
487  }
488 
489  RCLCPP_DEBUG(get_logger(), "Entering loop");
490 
491  rclcpp::WallRate r(frequency); // 200ms by default
492 
493  while (rclcpp::ok() && !map_update_thread_shutdown_) {
494  antis_nav_util::ExecutionTimer timer;
495 
496  // Measure the execution time of the updateMap method
497  timer.start();
498  updateMap();
499  timer.end();
500 
501  RCLCPP_DEBUG(get_logger(), "Map update time: %.9f", timer.elapsed_time_in_seconds());
502  if (publish_cycle_ > rclcpp::Duration(0) && layered_costmap_->isInitialized()) {
503  unsigned int x0, y0, xn, yn;
504  layered_costmap_->getBounds(&x0, &xn, &y0, &yn);
505  costmap_publisher_->updateBounds(x0, xn, y0, yn);
506 
507  auto current_time = now();
508  if ((last_publish_ + publish_cycle_ < current_time) || // publish_cycle_ is due
509  (current_time < last_publish_)) // time has moved backwards, probably due to a switch to sim_time // NOLINT
510  {
511  RCLCPP_DEBUG(get_logger(), "Publish costmap at %s", name_.c_str());
512  costmap_publisher_->publishCostmap();
513  last_publish_ = current_time;
514  }
515  }
516 
517  // Make sure to sleep for the remainder of our cycle time
518  r.sleep();
519 
520 #if 0
521  // TODO(bpwilcox): find ROS2 equivalent or port for r.cycletime()
522  if (r.period() > tf2::durationFromSec(1 / frequency)) {
523  RCLCPP_WARN(
524  get_logger(),
525  "Costmap2DROS: Map update loop missed its desired rate of %.4fHz... "
526  "the loop actually took %.4f seconds", frequency, r.period());
527  }
528 #endif
529  }
530 }

532 void
533 Costmap2DROS::updateMap()
534 {
535  RCLCPP_DEBUG(get_logger(), "Updating map...");
536 
537  if (!stop_updates_) {
538  // get global pose
539  geometry_msgs::msg::PoseStamped pose;
540  if (getRobotPose(pose)) {
541  const double & x = pose.pose.position.x;
542  const double & y = pose.pose.position.y;
543  const double yaw = tf2::getYaw(pose.pose.orientation);
544  layered_costmap_->updateMap(x, y, yaw);
545  is_updated_ = true;
546 
547  auto footprint = std::make_unique<geometry_msgs::msg::PolygonStamped>();
548  footprint->header.frame_id = global_frame_;
549  footprint->header.stamp = now();
550  transformFootprint(x, y, yaw, padded_footprint_, *footprint);
551 
552  RCLCPP_DEBUG(get_logger(), "Publishing footprint");
553  footprint_pub_->publish(std::move(footprint));
554  initialized_ = true;
555  }
556  }
557 }

559 void
560 Costmap2DROS::start()
561 {
562  RCLCPP_INFO(get_logger(), "start");
563  std::vector<std::shared_ptr<Layer>> * plugins = layered_costmap_->getPlugins();
564 
565  // check if we're stopped or just paused
566  if (stopped_) {
567  // if we're stopped we need to re-subscribe to topics
568  for (std::vector<std::shared_ptr<Layer>>::iterator plugin = plugins->begin();
569  plugin != plugins->end();
570  ++plugin)
571  {
572  (*plugin)->activate();
573  }
574  stopped_ = false;
575  }
576  stop_updates_ = false;
577 
578  // block until the costmap is re-initialized.. meaning one update cycle has run
579  rclcpp::Rate r(20.0);
580  while (rclcpp::ok() && !initialized_) {
581  RCLCPP_DEBUG(get_logger(), "Sleeping, waiting for initialized_");
582  r.sleep();
583  }
584 }

586 void
587 Costmap2DROS::stop()
588 {
589  stop_updates_ = true;
590  std::vector<std::shared_ptr<Layer>> * plugins = layered_costmap_->getPlugins();
591  // unsubscribe from topics
592  for (std::vector<std::shared_ptr<Layer>>::iterator plugin = plugins->begin();
593  plugin != plugins->end(); ++plugin)
594  {
595  (*plugin)->deactivate();
596  }
597  initialized_ = false;
598  stopped_ = true;
599 }
