27 bool TrackingControl::configure()
28 {
29  RCLCPP_INFO_STREAM(node_->get_logger(), "Ready Tracking Control.");
30 
31  /* Parametor setting */
32  std::vector<bool> tracking_activated_orientation;
33  tracking_activated_orientation.push_back(true); // roll
34  tracking_activated_orientation.push_back(true); // pitch
35  tracking_activated_orientation.push_back(true); // yaw
36 
37  // Action Name
38  std::string action_name =
39  node_->get_parameter(ros_param_root_ + "action_name").get_value<std::string>();
40  RCLCPP_INFO_STREAM(node_->get_logger(), "Tracking action name is '" << action_name << "'.");
41 
42  // Check time
43  checking_time_ = node_->get_parameter(ros_param_root_ + "checking_time").get_value<double>();
44  RCLCPP_INFO_STREAM(
45  node_->get_logger(),
46  "Check the Stability of the target pose during '" <<
47  checking_time_ <<
48  "s' before start to tracking.");
49 
50  // control period
51  control_period_ = node_->get_parameter(ros_param_root_ + "control_period").get_value<double>();
52  RCLCPP_INFO_STREAM(node_->get_logger(), "Control period set to '" << control_period_ << "s'.");
53 
54  // Filter
55  double x_filter_tau =
56  node_->get_parameter(ros_param_root_ + "x_filter_tau").get_value<double>();
57  RCLCPP_INFO_STREAM(
58  node_->get_logger(), "Position filter tau set to '" << x_filter_tau << ".");
59  x_filter_.set_tau(x_filter_tau);
60  double y_filter_tau =
61  node_->get_parameter(ros_param_root_ + "y_filter_tau").get_value<double>();
62  RCLCPP_INFO_STREAM(
63  node_->get_logger(), "Position filter tau set to '" << y_filter_tau << ".");
64  y_filter_.set_tau(y_filter_tau);
65  double z_filter_tau =
66  node_->get_parameter(ros_param_root_ + "z_filter_tau").get_value<double>();
67  RCLCPP_INFO_STREAM(
68  node_->get_logger(), "Position filter tau set to '" << z_filter_tau << ".");
69  z_filter_.set_tau(z_filter_tau);
70 
71  double orientation_filter_hbias = node_->get_parameter(
72  ros_param_root_ + "orientation_filter_hbias").get_value<double>();
73  double orientation_filter_hrange = node_->get_parameter(
74  ros_param_root_ + "orientation_filter_hrange").get_value<double>();
75  RCLCPP_INFO_STREAM(
76  node_->get_logger(),
77  "Orientation filter h bias set to '" << orientation_filter_hbias << "'," <<
78  "Orientation filter h range set to '" << orientation_filter_hrange << "'.");
79  orientation_filter_.set_h_bias(orientation_filter_hbias);
80  orientation_filter_.set_h_range(orientation_filter_hrange);
81 
82  // Orientation Checker
83  tracking_activated_orientation_ = node_->get_parameter(
84  ros_param_root_ + "tracking_activated_orientation").get_value<std::vector<bool>>();
85  while (tracking_activated_orientation_.size() < 3)
86  {tracking_activated_orientation_.push_back(true);}
87  RCLCPP_INFO_STREAM(node_->get_logger(), "Activated orientation is set");
88  RCLCPP_INFO_STREAM(node_->get_logger(), "Roll : " << tracking_activated_orientation_.at(0));
89  RCLCPP_INFO_STREAM(node_->get_logger(), "Pitch : " << tracking_activated_orientation_.at(1));
90  RCLCPP_INFO_STREAM(node_->get_logger(), "Yaw : " << tracking_activated_orientation_.at(2));
91 
92  time_out_ = node_->get_parameter(ros_param_root_ + "time_out").get_value<double>();
93  RCLCPP_INFO_STREAM(
94  node_->get_logger(), "time_out time is " << time_out_ << "s.");
95 
96  target_pose_pub_frame_id_ =
97  node_->get_parameter(ros_param_root_ + "target_pose_pub_frame_id").get_value<std::string>();
98  RCLCPP_INFO_STREAM(
99  node_->get_logger(), "target_pose_pub_frame_id is " << target_pose_pub_frame_id_);
100 
101  first_input_ = false;
102 
103  /* Publisher */
104  rclcpp::QoS qos(rclcpp::KeepLast(1));
105  target_pose_pub_ = node_->create_publisher<PoseMsg>(
106  std::string(node_->get_name()) + "/tracking/target_pose", qos);
107  tick_goal_pose_pub_ = node_->create_publisher<PoseMsg>(
108  std::string(node_->get_name()) + "/tracking/goal_pose", qos);
109 
110  /* Action server */
111  using namespace std::placeholders;
112  action_server_ = rclcpp_action::create_server<TrackingAction>(
113  node_->get_node_base_interface(),
114  node_->get_node_clock_interface(),
115  node_->get_node_logging_interface(),
116  node_->get_node_waitables_interface(),
117  std::string(node_->get_name()) + "/" + action_name,
118  std::bind(&TrackingControl::goal_callback, this, _1, _2),
119  std::bind(&TrackingControl::cancel_callback, this, _1),
120  std::bind(&TrackingControl::aceepted_callback, this, _1)
121  );
122  RCLCPP_INFO_STREAM(node_->get_logger(), "Succeess to create the action server.");
123 
124  return true;
125 }

132 void TrackingControl::update()
133 {
134  using rm2_common::TaskValues;
135  using rm2_common::Twist;
136  using rm2_common::Vector;
137  using rm2_common::Rotation;
138  using rm2_common::Pose;
139 
140  // std::cout << std::fixed;
141  // std::cout.precision(9);
142 // std::cout << "\n----[" << robot_handler_->get_tick_time() << "]----" << std::endl;
143 
144  // Goal_callback 함수가 실행되고부터 time_out 시간동안 트래킹이 끝나지 않았을때 타임아웃
145  if (robot_handler_->get_tick_time() > time_out_) {
146  RCLCPP_ERROR_STREAM(
147  node_->get_logger(),
148  "Total Time Out : Tracking failed to reach the target point during the timeout period of '" <<
149  robot_handler_->get_tick_time() << "'seconds.");
150  update_cancel(rm2_control_module::ControlModule::CancelResult::TIME_OUT);
151  stop_update();
152  return;
153  }
154 
155  // expected pose가 없을 경우에만 첫 번째 detection 값이 들어올 때까지 기다림.
156  if (!has_expected_pose_ && !first_input_) {
157  RCLCPP_WARN_STREAM(node_->get_logger(), "Waiting for the first target pose data to receive.");
158  return;
159  }
160 
161  // If target is lost over checking time, the tracking control will be stoped
162  if (robot_handler_->get_present_time() - target_detected_time_ > checking_time_) {
163  RCLCPP_ERROR_STREAM(
164  node_->get_logger(),
165  "Checking Time Out : The target pose was not updated for '" <<
166  robot_handler_->get_present_time() - target_detected_time_ << "'seconds.");
167  update_cancel(rm2_control_module::ControlModule::CancelResult::TIME_OUT);
168  stop_update();
169  return;
170  }
171 
172  // present Value ?? previous goal??
173  TaskValues present_task_value =
174  robot_handler_->get_previous_goal_task_value(current_control_end_effector_);
175  present_task_value.velocity = previous_twist_;
176 // std::cout << "present_position : " << present_task_value.pose.position.data.transpose() <<
177 // std::endl;
178 
179 
180  // Final goal Position
181  rm2_common::Pose target_pose_with_offset(target_pose_from_base_ * target_offset_,
182  target_pose_from_base_.orientation);
183 // std::cout << "Final Goal Position : " << final_goal_position.data.transpose() << std::endl;
184  Vector final_goal_position = target_pose_with_offset * goal_position_from_target_;
185 // std::cout << "Final Goal Position : " << final_goal_position.data.transpose() << std::endl;
186 
187  // Position distance
188  Vector delta_position = final_goal_position - present_task_value.pose.position;
189 
190  // Goal Orientaion
191  Vector direction = target_pose_with_offset.position - present_task_value.pose.position;
192  Rotation goal_orientation = Rotation::FromTwoVectors(Vector(1, 0, 0), direction);
193 
194  // Orientaion distance
195  Vector delta_orientation = goal_orientation - present_task_value.pose.orientation;
196 
197  if (fabs(delta_position(0)) < tolerance_.x &&
198  fabs(delta_position(1)) < tolerance_.y &&
199  fabs(delta_position(2)) < tolerance_.z &&
200  fabs(delta_orientation(0)) < tolerance_.roll &&
201  fabs(delta_orientation(1)) < tolerance_.pitch &&
202  fabs(delta_orientation(2)) < tolerance_.yaw &&
203  first_input_)
204  {
205  RCLCPP_INFO_STREAM(
206  node_->get_logger(), "GOAL!! " << "(" <<
207  delta_position(0) << "/" << tolerance_.x << ", " <<
208  delta_position(1) << "/" << tolerance_.y << ", " <<
209  delta_position(2) << "/" << tolerance_.z << ", " <<
210  delta_orientation(0) << "/" << tolerance_.roll << ", " <<
211  delta_orientation(1) << "/" << tolerance_.pitch << ", " <<
212  delta_orientation(2) << "/" << tolerance_.yaw << ").");
213 
214  stop_update();
215  send_success();
216  return;
217  }
218 
219  rm2_trajectory::TaskTrapezoidal trap;
220  trap.set_time_sync(false);
221  if (!trap.set_profile(profile_.velocity, profile_.acceleration, profile_.tip_link_length)) {
222  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to set profile vel.");
223  update_cancel(rm2_control_module::ControlModule::CancelResult::PATH_PLANNING_FAIL);
224  stop_update();
225  return;
226  }
227  if (!trap.calculation(
228  present_task_value.pose, Pose(final_goal_position, goal_orientation),
229  present_task_value.velocity, Twist()))
230  {
231  RCLCPP_ERROR_STREAM(
232  node_->get_logger(), "Fail to calculate trajectory." <<
233  "\npresent position : " << present_task_value.pose.position.data.transpose() <<
234  "\nfinal goal position : " << final_goal_position.data.transpose());
235  update_cancel(rm2_control_module::ControlModule::CancelResult::PATH_PLANNING_FAIL);
236  stop_update();
237  return;
238  }
239 
240  bool finished = false;
241  rm2_common::TaskValues tick_goal = trap.get(control_period_, finished);
242  if (finished) {
243  RCLCPP_INFO_STREAM(node_->get_logger(), "GOAL!!!!!!!");
244  stop_update();
245  send_success();
246  return;
247  }
248 
249  previous_twist_ = tick_goal.velocity;
250  if (!robot_handler_->set_goal_task_value(current_control_end_effector_, tick_goal)) {
251  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to set Task values!!!!!!");
252  return;
253  }
254 
255  // send feedback
256  auto feedback = std::make_shared<TrackingAction::Feedback>();
257 
258  // time
259  feedback->header.stamp = node_->now();
260  // frame id
261  feedback->header.frame_id = robot_handler_->get_robot_model().get_base_component()->first;
262  // goal pose
263  feedback->goal_pose = trans_pose_to_msg(
264  target_pose_from_base_.position,
265  target_pose_from_base_.orientation);
266  // Present
267  feedback->presnet_pose = trans_pose_to_msg(
268  present_task_value.pose.position,
269  present_task_value.pose.orientation);
270  // distanse
271  rm2_common::Pose pose_error = rm2_common::Pose(
272  delta_position,
273  Rotation::RPY(
274  delta_orientation[0],
275  delta_orientation[1],
276  delta_orientation[2]));
277  feedback->distance = trans_pose_to_msg(pose_error.position, pose_error.orientation);
278 
279  if (!target_pose_pub_frame_id_.empty()) {
280  auto frame_pose_from_base =
281  robot_handler_->get_present_task_value(
282  robot_handler_->get_present_time(), target_pose_pub_frame_id_).pose;
283  auto target_pose_from_detected_frame = frame_pose_from_base.inverse() * target_pose_from_base_;
284 
285  PoseMsg target_pose_msg;
286  target_pose_msg.header.stamp = node_->now();
287  target_pose_msg.header.frame_id = target_pose_pub_frame_id_;
288  target_pose_msg.pose = trans_pose_to_msg(
289  target_pose_from_detected_frame.position,
290  target_pose_from_detected_frame.orientation);
291  target_pose_pub_->publish(target_pose_msg);
292  feedback->target_pose = target_pose_msg;
293  }
294  // debug
295  PoseMsg tick_pose_msg;
296  tick_pose_msg.header.stamp = node_->now();
297  tick_pose_msg.header.frame_id = robot_handler_->get_robot_model().get_base_component()->first;
298  tick_pose_msg.pose = trans_pose_to_msg(tick_goal.pose.position, tick_goal.pose.orientation);
299  tick_goal_pose_pub_->publish(tick_pose_msg);
300 
301  if (actived_goal_) {
302  actived_goal_->publish_feedback(feedback);
303  }
304 }

307 void TrackingControl::update_cancel(rm2_control_module::ControlModule::CancelResult result)
308 {
309  TrackingAction::Result::SharedPtr action_result = std::make_shared<TrackingAction::Result>();
310  manipulator_interfaces::msg::Result result_state;
311 
312  if (result == CancelResult::JOINT_LIMIT) {
313  result_state.set__val(manipulator_interfaces::msg::Result::COLLISION_CHECKING_UNAVAILABLE);
314  RCLCPP_INFO_STREAM(
315  node_->get_logger(),
316  "Tracking control canceled : joint limit over.");
317  } else if (result == CancelResult::TOO_FAST_MOVE) {
318  result_state.set__val(manipulator_interfaces::msg::Result::CONTROL_FAILED);
319  RCLCPP_INFO_STREAM(
320  node_->get_logger(),
321  "Tracking contorl canceled : too fast move failure");
322  } else if (result == CancelResult::INVALID_GOAL_VALUES) {
323  result_state.set__val(manipulator_interfaces::msg::Result::CONTROL_FAILED);
324  RCLCPP_INFO_STREAM(
325  node_->get_logger(),
326  "Tracking contorl canceled : invalid goal values failure");
327  } else if (result == CancelResult::IK_FAIL) {
328  RCLCPP_INFO_STREAM(
329  node_->get_logger(),
330  "Tracking control canceled : ik solving failure");
331  result_state.set__val(manipulator_interfaces::msg::Result::NO_IK_SOLUTION);
332  } else if (result == CancelResult::PREEMPTED) {
333  RCLCPP_INFO_STREAM(
334  node_->get_logger(),
335  "Tracking control canceled : preempted by the other controller");
336  result_state.set__val(manipulator_interfaces::msg::Result::PREEMPTED);
337  } else if (result == CancelResult::TIME_OUT) {
338  RCLCPP_INFO_STREAM(
339  node_->get_logger(),
340  "Tracking control canceled : time out");
341  result_state.set__val(manipulator_interfaces::msg::Result::TIMED_OUT);
342  } else if (result == CancelResult::PATH_PLANNING_FAIL) {
343  RCLCPP_INFO_STREAM(
344  node_->get_logger(),
345  "Tracking control canceled : trajectory planning fail");
346  result_state.set__val(manipulator_interfaces::msg::Result::PLANNING_FAILED);
347  } else if (result == CancelResult::CHECKER) {
348  result_state.set__val(manipulator_interfaces::msg::Result::GOAL_IN_COLLISION);
349  RCLCPP_INFO_STREAM(
350  node_->get_logger(),
351  "Tracking contorl canceled : lidar");
352  } else {
353  result_state.set__val(manipulator_interfaces::msg::Result::FAILURE);
354  RCLCPP_INFO_STREAM(
355  node_->get_logger(),
356  "Tracking contorl canceled : fail.");
357  }
358 
359  action_result->set__result_state(result_state);
360 
361  if (actived_goal_) {
362  actived_goal_->abort(action_result);
363  }
364  actived_goal_.reset();
365  target_pose_sub_.reset();
366 }

385 rclcpp_action::GoalResponse TrackingControl::goal_callback(
386  const rclcpp_action::GoalUUID & uuid,
387  std::shared_ptr<const manipulator_interfaces::action::Tracking::Goal> goal)
388 {
389  RCLCPP_INFO_STREAM(node_->get_logger(), "Received new Tracking action goal");
390  std::lock_guard<std::mutex> lock(mutex_);
391 
393  // Set Control End Effector Id //
395  current_control_end_effector_ = goal->end_effector_frame_id;
396  RCLCPP_INFO_STREAM(
397  node_->get_logger(), "End effector name is '" << current_control_end_effector_.c_str() << "'.");
398 
400  // Set Goal position from target //
402  goal_position_from_target_ = rm2_common::Vector(
403  goal->goal_position_from_target.x,
404  goal->goal_position_from_target.y,
405  goal->goal_position_from_target.z);
406  RCLCPP_INFO_STREAM(
407  node_->get_logger(), "The end effector will approach to '(" <<
408  goal_position_from_target_.data.transpose() << ")' from target pose.");
409 
411  // Set target offset //
413  target_offset_ = rm2_common::Vector(
414  goal->target_offset.x,
415  goal->target_offset.y,
416  goal->target_offset.z);
417  RCLCPP_INFO_STREAM(
418  node_->get_logger(), "The target offset is '(" << target_offset_.data.transpose() << ")'.");
419 
421  // Set profile data //
423  // Check profile
424  if (goal->profile.velocity < 0.0 ||
425  goal->profile.acceleration < 0.0 ||
426  goal->profile.tip_link_length < 0.0)
427  {
428  RCLCPP_ERROR_STREAM(node_->get_logger(), "Please set profile data to over '0'.");
429  return rclcpp_action::GoalResponse::REJECT;
430  }
431  // Set trajectory profile data
432  profile_ = goal->profile;
433  RCLCPP_INFO_STREAM(
434  node_->get_logger(), "Approach speed : '" <<
435  profile_.velocity << "'m/s and '" <<
436  profile_.velocity / profile_.tip_link_length << "'rad/s.");
437  RCLCPP_INFO_STREAM(
438  node_->get_logger(), "Approach acceleration : '" <<
439  profile_.acceleration << "'m/s^2 and '" <<
440  profile_.acceleration / profile_.tip_link_length << "'rad/s^2.");
441 
443  // Set pose tolerance //
445  tolerance_ = goal->tolerance;
446  RCLCPP_INFO_STREAM(
447  node_->get_logger(), "The Tolerance '" <<
448  "(" << tolerance_.x << ", " << tolerance_.y << ", " << tolerance_.z << ")" << "'m and '" <<
449  "(" << tolerance_.roll << ", " << tolerance_.pitch << ", " << tolerance_.yaw << ")" "'rad.");
450 
452  // Expected Target pose //
454  has_expected_pose_ = true;
455  double expected_target_detected_time =
456  rclcpp::Time(goal->expected_target_pose.header.stamp).seconds();
457  std::string expected_target_base_frame = goal->expected_target_pose.header.frame_id;
458  RCLCPP_INFO_STREAM(
459  node_->get_logger(),
460  "- [expected] Target detected time : " << expected_target_detected_time << " -\n" <<
461  "Detected frame name : " << expected_target_base_frame);
462  if (expected_target_base_frame.empty()) {
463  RCLCPP_WARN_STREAM(
464  node_->get_logger(),
465  "Time or frame_id is empty. time : " << expected_target_detected_time <<
466  ", frame : " << expected_target_base_frame);
467  has_expected_pose_ = false;
468  }
469 
470  // expected target pose
471  rm2_common::Pose first_target_pose_from_base;
472  if (has_expected_pose_) {
473  double expected_roll, expected_pitch, expected_yaw;
474  rm2_common::Rotation(
475  goal->expected_target_pose.pose.orientation.w,
476  goal->expected_target_pose.pose.orientation.x,
477  goal->expected_target_pose.pose.orientation.y,
478  goal->expected_target_pose.pose.orientation.z).get_rpy(
479  expected_roll,
480  expected_pitch,
481  expected_yaw);
482  if (!tracking_activated_orientation_.at(0)) {
483  expected_roll = 0.0;
484  }
485  if (!tracking_activated_orientation_.at(1)) {
486  expected_pitch = 0.0;
487  }
488  if (!tracking_activated_orientation_.at(2)) {
489  expected_yaw = 0.0;
490  }
491 
492  // Expected Target Pose
493  rm2_common::Pose expected_target_pose_without_offset(
494  rm2_common::Vector(
495  goal->expected_target_pose.pose.position.x,
496  goal->expected_target_pose.pose.position.y,
497  goal->expected_target_pose.pose.position.z),
498  rm2_common::Rotation::RPY(expected_roll, expected_pitch, expected_yaw));
499  rm2_common::Pose expected_target_pose(
500  expected_target_pose_without_offset * target_offset_,
501  expected_target_pose_without_offset.orientation);
502 
503  // if expected_target_base_frame is base or not
504  if (expected_target_base_frame !=
505  robot_handler_->get_robot_model().get_base_component()->first)
506  {
507  // base_frame
508  auto frame_pose_from_base =
509  robot_handler_->get_present_task_value(
510  expected_target_base_frame).pose;
511  RCLCPP_INFO_STREAM(
512  node_->get_logger(),
513  "Detected frame pose : " << frame_pose_from_base.position.data.transpose());
514 
515  // Translate
516  first_target_pose_from_base = frame_pose_from_base * expected_target_pose;
517  } else {
518  first_target_pose_from_base = expected_target_pose;
519  }
520  RCLCPP_INFO_STREAM(
521  node_->get_logger(),
522  "Expected tracking target position : " <<
523  first_target_pose_from_base.position.data.transpose());
524  RCLCPP_INFO_STREAM(
525  node_->get_logger(),
526  "Expected tracking target orientation : " <<
527  first_target_pose_from_base.orientation.data.transpose());
528  }
529 
530  first_input_ = false;
531  time_from_first_detection_ = node_->now();
532 
533  // Subscriber msg name
534  std::string tracking_msg_name = goal->tracking_target_msg_name;
535  RCLCPP_INFO_STREAM(
536  node_->get_logger(), "Tracking message name is '" << tracking_msg_name.c_str() << "'.");
537  // Create subscriber
538  target_pose_sub_ =
539  node_->create_subscription<PoseMsg>(
540  tracking_msg_name, rclcpp::SystemDefaultsQoS(), [this](
541  const std::shared_ptr<PoseMsg> cmd) -> void {
542  // using namespace rm2_common;
543  // Subscriber Callback
544  rclcpp::Time target_detected_time = cmd->header.stamp;
545  double delta_time = target_detected_time.seconds() - target_detected_time_;
546  target_detected_time_ = target_detected_time.seconds();
547  std::string target_base_frame = cmd->header.frame_id;
548 
549  double roll;
550  double pitch;
551  double yaw;
552  rm2_common::Rotation(
553  cmd->pose.orientation.w,
554  cmd->pose.orientation.x,
555  cmd->pose.orientation.y,
556  cmd->pose.orientation.z).get_rpy(roll, pitch, yaw);
557  if (!tracking_activated_orientation_.at(0)) {

569  rm2_common::Vector(
570  cmd->pose.position.x,
571  cmd->pose.position.y,

574 
575  // Calculation goal pose from base
576  RCLCPP_INFO_STREAM(
577  node_->get_logger(),
578  "- Target detected time : " << target_detected_time.seconds() << " -\n" <<
579  "Detected frame name : " << target_base_frame);
580  auto frame_pose_from_base =
581  robot_handler_->get_present_task_value(
582  target_detected_time_,
583  target_base_frame).pose;
584 
585  RCLCPP_INFO_STREAM(

587  "Detected frame pose : " << frame_pose_from_base.position.data.transpose());
588

596  target_pose_from_base_ =
597  rm2_common::Pose(

601  z_filter_.get_filtering_data(target_pose_from_base_raw.position.z(), delta_time)),
602  orientation_filter_.get_filtering_data(target_pose_from_base_raw.orientation));
603  RCLCPP_INFO_STREAM(
604  node_->get_logger(),
605  "Tracking target pose(filtered) : " << target_pose_from_base_.position.data.transpose());
606 
607  first_input_ = true;
608  });
609 
610  // Check Publisher size
611  if (target_pose_sub_->get_publisher_count() == 0) {
612  RCLCPP_ERROR_STREAM(node_->get_logger(), "tracking message publisher is not created.");
613  target_pose_sub_.reset();
614  return rclcpp_action::GoalResponse::REJECT;
615  }
616  RCLCPP_INFO_STREAM(node_->get_logger(), "Success to create tracking message subscriber.");
617 
618  // Set First Data to filter
619  x_filter_.set_first_input_flag(true);
620  y_filter_.set_first_input_flag(true);
621  z_filter_.set_first_input_flag(true);
622  orientation_filter_.set_first_input_flag(true);
623 
624  if (has_expected_pose_) {
625  target_pose_from_base_ =
626  rm2_common::Pose(
627  rm2_common::Vector(
628  x_filter_.get_filtering_data(first_target_pose_from_base.position.x()),
629  y_filter_.get_filtering_data(first_target_pose_from_base.position.y()),
630  z_filter_.get_filtering_data(first_target_pose_from_base.position.z())),
631  orientation_filter_.get_filtering_data(first_target_pose_from_base.orientation));
632  RCLCPP_INFO_STREAM(
633  node_->get_logger(),
634  "Tracking target pose(filtered) : " << target_pose_from_base_.position.data.transpose());
635  target_detected_time_ = node_->now().seconds();
636  }
637  robot_handler_->set_start_time();
638  previous_twist_ = rm2_common::Twist();
639 
640  return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;

643 rclcpp_action::CancelResponse TrackingControl::cancel_callback(
644  const TrackingControl::GoalHandlePtr goal_handle)
645 {
646  RCLCPP_INFO_STREAM(node_->get_logger(), "Got request to cancel goal");
647  std::lock_guard<std::mutex> lock(mutex_);
648  try {
649  if (actived_goal_ == goal_handle) {
650  // Controller uptime
651  // Enter hold current position mode
652  stop_update();
653  RCLCPP_DEBUG_STREAM(
654  node_->get_logger(), "Canceling active action goal because cancel callback received");
655 
656  // Mark the current goal as canceled
657  auto action_res = std::make_shared<TrackingAction::Result>();
658  manipulator_interfaces::msg::Result result_state;
659  result_state.set__val(manipulator_interfaces::msg::Result::CANCELED);
660  action_res->set__result_state(result_state);
661  if (actived_goal_) {
662  actived_goal_->canceled(action_res);
663  }
664  } else {
665  // Reset current goal
666  actived_goal_.reset();
667  RCLCPP_ERROR_STREAM(
668  node_->get_logger(), "Fail to Canceling active action goal due to goal handle mismatch.");
669  return rclcpp_action::CancelResponse::REJECT;
670  }
671  } catch (std::runtime_error e) {
672  // Reset current goal
673  actived_goal_.reset();
674  RCLCPP_ERROR_STREAM(node_->get_logger(), "Runtime error : " << e.what());
675  return rclcpp_action::CancelResponse::REJECT;
676  }
677  // Reset current goal
678  actived_goal_.reset();
679  return rclcpp_action::CancelResponse::ACCEPT;
680 }
