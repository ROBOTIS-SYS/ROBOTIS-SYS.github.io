145 void GcKnockControlWithImu::update()
146 {
148  // Prepare
150  if (knocking_state_ == KnockMotionState::Prepare) {
151  bool goal_finish = false;
152  rm2_common::JointWaypoint tick_goal = preprare_trajectory_.get(
153  robot_handler_->get_tick_time(), goal_finish);
154  // Check Joint Limit
155  auto limit_over_joints = robot_handler_->check_joint_limit(tick_goal);
156  if (!limit_over_joints.empty()) {
157  RCLCPP_WARN_STREAM(
158  node_->get_logger(),
159  "[Prepare Motion] Failed in joint limit checker: " << limit_over_joints.at(0) <<
160  ", Move to Returning Motion");
161  if (!make_returning_trajectory()) {
162  RCLCPP_ERROR_STREAM(
163  node_->get_logger(), "[Prepare Motion] Fail To calculate Returning trajectory.");
164  send_failure();
165  stop_update();
166  return;
167  }
168  success_flag_ = false;
169  knocking_state_ = KnockMotionState::Finish;
170  robot_handler_->set_start_time();
171  } else {
172  robot_handler_->set_goal_joint_value_unlimited(tick_goal);
173  }
174  // Check Over Motion
175  if (goal_finish) {
176  RCLCPP_INFO_STREAM(
177  node_->get_logger(),
178  "[Prepare Motion] Prepare trajectory is finished, Move to Approach Motion");
179  knocking_state_ = KnockMotionState::Approach;
180  last_approach_start_time_ = robot_handler_->get_present_time();
181  robot_handler_->set_start_time();
182  }
184  // Approach
186  } else if (knocking_state_ == KnockMotionState::Approach) {
187  bool goal_finish = false;
188  rm2_common::TaskWaypoint tick_goal = approach_trajectory_.get(
189  robot_handler_->get_tick_time(), goal_finish);
190  get_imu_sensor_data(); // get imu data to imu_data_
191  // RCLCPP_INFO_STREAM(node_->get_logger(), "<" << robot_handler_->get_tick_time() <<
192  // "> Tick goal pos : " << tick_goal.pose.position.data.transpose() <<
193  // ", vel : " << tick_goal.velocity.linear.data.transpose());
194  auto limit_over_joints = robot_handler_->check_joint_limit(knock_component_name_, tick_goal);
195  if (!limit_over_joints.empty()) {
196  RCLCPP_WARN_STREAM(
197  node_->get_logger(),
198  "[Approach Motion] Failed in joint limit checker: " << limit_over_joints.at(0) <<
199  ", Move to Back Motion");
200  double back_time = time_between_knocks_ -
201  (robot_handler_->get_present_time() - last_approach_start_time_);
202  if (back_time < approach_time_) {back_time = approach_time_;}
203  if (!make_back_trajectory(back_time)) {
204  RCLCPP_ERROR_STREAM(
205  node_->get_logger(), "[Approach Motion] Fail To calculate Back trajectory.");
206  send_failure();
207  stop_update();
208  return;
209  }
210  knocking_state_ = KnockMotionState::Back;
211  success_flag_ = false;
212  robot_handler_->set_start_time();
213  return;
214  } else {
215  robot_handler_->set_goal_task_value_unlimited(knock_component_name_, tick_goal);
216  }
217 
218  if (imu_data_ > imu_data_threshold_) {
219  RCLCPP_INFO_STREAM(
220  node_->get_logger(),
221  "[Approach Motion] Touches door, Move to Back Motion");
222  double back_time = time_between_knocks_ -
223  (robot_handler_->get_present_time() - last_approach_start_time_);
224  if (back_time < approach_time_) {back_time = approach_time_;}
225  if (!make_back_trajectory(back_time)) {
226  RCLCPP_ERROR_STREAM(
227  node_->get_logger(), "[Approach Motion] Fail To calculate Back trajectory.");
228  send_failure();
229  stop_update();
230  return;
231  }
232  knocking_state_ = KnockMotionState::Back;
233  robot_handler_->set_start_time();
234  } else if (robot_handler_->get_present_time() - last_approach_start_time_ > knocking_timeout_) {
235  RCLCPP_WARN_STREAM(
236  node_->get_logger(),
237  "[Approach Motion] Couldn't reach the door within the timeout: " <<
238  knocking_timeout_ << "s.");
239  double back_time = time_between_knocks_ -
240  (robot_handler_->get_present_time() - last_approach_start_time_);
241  if (back_time < approach_time_) {back_time = approach_time_;}
242  if (!make_back_trajectory(back_time)) {
243  RCLCPP_ERROR_STREAM(
244  node_->get_logger(), "[Approach Motion] Fail To calculate Back trajectory.");
245  send_failure();
246  stop_update();
247  return;
248  }
249  knocking_state_ = KnockMotionState::Back;
250  success_flag_ = false;
251  robot_handler_->set_start_time();
252  } else if (goal_finish) {
253  RCLCPP_INFO_STREAM(
254  node_->get_logger(),
255  "[Approach Motion] Approach trajectory is finished, Move to Knocking Motion");
256  knocking_state_ = KnockMotionState::Knocking;
257  robot_handler_->set_start_time();
258  }
260  // Knocking
262  } else if (knocking_state_ == KnockMotionState::Knocking) {
263  rm2_common::TaskWaypoint tick_goal =
264  robot_handler_->get_previous_goal_task_value(knock_component_name_);
265  tick_goal.velocity = approach_task_values_.velocity;
266  tick_goal.pose.position +=
267  tick_goal.pose.orientation * (tick_goal.velocity.linear * robot_handler_->get_tick_time());
268  robot_handler_->set_start_time();
269  get_imu_sensor_data(); // get imu data to imu_data_
270  // RCLCPP_INFO_STREAM(node_->get_logger(), "<" << robot_handler_->get_tick_time() <<
271  // "> Tick goal pos : " << tick_goal.pose.position.data.transpose() <<
272  // ", vel : " << tick_goal.velocity.linear.data.transpose());
273  auto limit_over_joints = robot_handler_->check_joint_limit(knock_component_name_, tick_goal);
274  if (!limit_over_joints.empty()) {
275  RCLCPP_WARN_STREAM(
276  node_->get_logger(),
277  "[Knocking Motion] Failed in joint limit checker: " << limit_over_joints.at(0) <<
278  ", Move to Back Motion");
279  double back_time = time_between_knocks_ -
280  (robot_handler_->get_present_time() - last_approach_start_time_);
281  if (back_time < approach_time_) {back_time = approach_time_;}
282  if (!make_back_trajectory(back_time)) {
283  RCLCPP_ERROR_STREAM(
284  node_->get_logger(), "[Knocking Motion] Fail To calculate Back trajectory.");
285  send_failure();
286  stop_update();
287  return;
288  }
289  knocking_state_ = KnockMotionState::Back;
290  success_flag_ = false;
291  robot_handler_->set_start_time();
292  return;
293  } else {
294 // RCLCPP_INFO_STREAM(node_->get_logger(), "<" << robot_handler_->get_tick_time() <<
295 // "> Tick goal pos : " << tick_goal.pose.position.data.transpose() <<
296 // ", vel : " << tick_goal.velocity.linear.data.transpose());
297  robot_handler_->set_goal_task_value_unlimited(knock_component_name_, tick_goal);
298  }
299 
300  if (imu_data_ > imu_data_threshold_) {
301  RCLCPP_INFO_STREAM(
302  node_->get_logger(),
303  "[Knocking Motion] Touches door, Move to Back Motion");
304  double back_time = time_between_knocks_ -
305  (robot_handler_->get_present_time() - last_approach_start_time_);
306  if (back_time < approach_time_) {back_time = approach_time_;}
307  if (!make_back_trajectory(back_time)) {
308  RCLCPP_ERROR_STREAM(
309  node_->get_logger(), "[Knocking Motion] Fail To calculate Back trajectory.");
310  send_failure();
311  stop_update();
312  return;
313  }
314  knocking_state_ = KnockMotionState::Back;
315  robot_handler_->set_start_time();
316  } else if (robot_handler_->get_present_time() - last_approach_start_time_ > knocking_timeout_) {
317  RCLCPP_WARN_STREAM(
318  node_->get_logger(),
319  "[Knocking Motion] Couldn't reach the door within the timeout: " <<
320  knocking_timeout_ << "s.");
321  double back_time = time_between_knocks_ -
322  (robot_handler_->get_present_time() - last_approach_start_time_);
323  if (back_time < approach_time_) {back_time = approach_time_;}
324  if (!make_back_trajectory(back_time)) {
325  RCLCPP_ERROR_STREAM(
326  node_->get_logger(), "[Knocking Motion] Fail To calculate Back trajectory.");
327  send_failure();
328  stop_update();
329  return;
330  }
331  knocking_state_ = KnockMotionState::Back;
332  success_flag_ = false;
333  robot_handler_->set_start_time();
334  }
336  // Back
338  } else if (knocking_state_ == KnockMotionState::Back) {
339  bool goal_finish = false;
340  rm2_common::JointWaypoint tick_goal = back_trajectory_.get(
341  robot_handler_->get_tick_time(), goal_finish);
342  // Check Joint Limit
343  auto limit_over_joints = robot_handler_->check_joint_limit(tick_goal);
344  if (!limit_over_joints.empty()) {
345  RCLCPP_WARN_STREAM(
346  node_->get_logger(),
347  "[Back Motion] Failed in joint limit checker: " << limit_over_joints.at(0) <<
348  ", Move to Returning Motion");
349  if (!make_returning_trajectory()) {
350  RCLCPP_ERROR_STREAM(
351  node_->get_logger(), "[Back Motion] Fail To calculate Returning trajectory.");
352  send_failure();
353  stop_update();
354  return;
355  }
356  success_flag_ = false;
357  knocking_state_ = KnockMotionState::Finish;
358  robot_handler_->set_start_time();
359  } else {
360  robot_handler_->set_goal_joint_value_unlimited(tick_goal);
361  }
362 
363  if (goal_finish) {
364  if (total_knocking_count_ > knocking_count_ + 1 && success_flag_) {
365  RCLCPP_INFO_STREAM(
366  node_->get_logger(),
367  "[Back Motion] Back trajectory is finished, knocking count is " << knocking_count_ + 1 <<
368  ". Move to Approach Motion");
369  knocking_state_ = KnockMotionState::Approach;
370  robot_handler_->set_start_time();
371  last_approach_start_time_ = robot_handler_->get_present_time();
372  knocking_count_ += 1;
373  } else {
374  RCLCPP_INFO_STREAM(
375  node_->get_logger(),
376  "[Back Motion] Every knock motion is finished, Move to Finish Motion");
377  if (!make_returning_trajectory()) {
378  RCLCPP_ERROR_STREAM(
379  node_->get_logger(), "[Back Motion] Fail To calculate Returning trajectory.");
380  send_failure();
381  stop_update();
382  return;
383  }
384  knocking_state_ = KnockMotionState::Finish;
385  robot_handler_->set_start_time();
386  }
387  }
389  // Finish
391  } else if (knocking_state_ == KnockMotionState::Finish) {
392  bool goal_finish = false;
393  rm2_common::JointWaypoint tick_goal = returning_trajectory_.get(
394  robot_handler_->get_tick_time(), goal_finish);
395 
396  robot_handler_->set_goal_joint_value(tick_goal);
397 
398  if (goal_finish) {
399  if (success_flag_) {
400  RCLCPP_INFO_STREAM(node_->get_logger(), "[Finish Motion] Success to Knock Motion");
401  send_completed();
402  stop_update();
403  } else {
404  RCLCPP_WARN_STREAM(node_->get_logger(), "[Finish Motion] Fail to Knock Motion");
405  send_failure();
406  stop_update();
407  }
408  return;
409  }
410  }
411  send_feedback();
412 }

414 void GcKnockControlWithImu::update_cancel(rm2_control_module::ControlModule::CancelResult result)
415 {
416  KnockMotionAction::Result::SharedPtr action_result =
417  std::make_shared<KnockMotionAction::Result>();
418  manipulator_interfaces::msg::Result result_state;
419 
420  if (result == CancelResult::JOINT_LIMIT) {
421  result_state.set__val(manipulator_interfaces::msg::Result::COLLISION_CHECKING_UNAVAILABLE);
422  RCLCPP_INFO_STREAM(
423  node_->get_logger(),
424  "Knock contorl With imu canceled : joint limit over.");
425  } else if (result == CancelResult::IK_FAIL) {
426  result_state.set__val(manipulator_interfaces::msg::Result::NO_IK_SOLUTION);
427  RCLCPP_INFO_STREAM(
428  node_->get_logger(),
429  "Knock contorl With imu canceled : ik solving failure");
430  } else if (result == CancelResult::PREEMPTED) {
431  result_state.set__val(manipulator_interfaces::msg::Result::PREEMPTED);
432  RCLCPP_INFO_STREAM(
433  node_->get_logger(),
434  "Knock contorl With imu canceled : preempted by the other controller");
435  } else if (result == CancelResult::CHECKER) {
436  result_state.set__val(manipulator_interfaces::msg::Result::GOAL_IN_COLLISION);
437  RCLCPP_INFO_STREAM(
438  node_->get_logger(),
439  "Knock contorl With imu canceled : lidar");
440  } else {
441  result_state.set__val(manipulator_interfaces::msg::Result::FAILURE);
442  RCLCPP_INFO_STREAM(
443  node_->get_logger(),
444  "Knock contorl With imu canceled : fail.");
445  }
446  action_result->set__result_state(result_state);
447 
448  if (actived_goal_) {
449  actived_goal_->abort(action_result);
450  }
451  actived_goal_.reset();
452 }

468 bool GcKnockControlWithImu::calc_door_pose(
469  rclcpp::Time door_detected_time,
470  std::string door_base_frame,
471  geometry_msgs::msg::Polygon door,
472  rm2_common::Pose mani_base_task_values,
473  rm2_common::Pose & result_knock_pose,
474  double & knock_pose_yaw)
475 {
476  // Translate
477  rm2_common::TaskValues door_base_task_values = robot_handler_->get_present_task_value(
478  door_detected_time.seconds(),
479  door_base_frame);
480 
481  if (door.points.size() != 4) {
482  return false;
483  }
484 
485  std::vector<rm2_common::Vector> door_points_pre;
486  for (auto point : door.points) {
487  door_points_pre.push_back(
488  mani_base_task_values.inverse() *
489  door_base_task_values.pose *
490  rm2_common::Vector(
491  static_cast<double>(point.x),
492  static_cast<double>(point.y),
493  static_cast<double>(point.z)));
494  }
495 
496  // <<< +
497  // left top(0) ^*---* right top(1)
498  // ^| |
499  // left bot(2) +* * right bot(3)
500  std::deque<rm2_common::Vector> door_points;
501 
502  // First point
503  door_points.push_back(door_points_pre.at(0));
504  // set 2 top point (Z)
505  for (size_t index = 1; index < door_points_pre.size(); index++) {
506  if (door_points.at(0).z() <= static_cast<double>(door_points_pre.at(index).z())) {
507  door_points.push_front(
508  mani_base_task_values.inverse() *
509  door_base_task_values.pose *
510  rm2_common::Vector(door_points_pre.at(index)));
511  } else {
512  door_points.push_back(
513  rm2_common::Vector(door_points_pre.at(index)));
514  }
515  }
516 
517  if (door_points.at(0).y() < door_points.at(1).y()) {
518  rm2_common::Vector tmp = door_points.at(0);
519  door_points.at(0) = door_points.at(1);
520  door_points.at(1) = tmp;
521  }
522 
523  // Set bottom left and right
524  if (door_points.at(2).y() < door_points.at(3).y()) {
525  rm2_common::Vector tmp = door_points.at(2);
526  door_points.at(2) = door_points.at(3);
527  door_points.at(3) = tmp;
528  }
529 
530  // door Corners
531  // <<< +
532  // left top(0) ^*---* right top(1)
533  // ^| |
534  // left bot(2) +* * right bot(3)
535  // Set top left and right
536  RCLCPP_INFO_STREAM(node_->get_logger(), "DOOR POINTS");
537  rm2_common::Vector door_left_top = door_points.at(0);
538  rm2_common::Vector door_right_top = door_points.at(1);
539  rm2_common::Vector door_left_bottom = door_points.at(2);
540  rm2_common::Vector door_right_bottom = door_points.at(3);
541  RCLCPP_INFO_STREAM(
542  node_->get_logger(), door_left_top.data.transpose() << " * ---- * " <<
543  door_right_top.data.transpose());
544  RCLCPP_INFO_STREAM(
545  node_->get_logger(), door_left_bottom.data.transpose() << " * ---- * " <<
546  door_right_bottom.data.transpose());
547 
548  double door_angle = atan2(
549  door_left_bottom.y() - door_right_bottom.y(),
550  door_left_bottom.x() - door_right_bottom.x());
551  RCLCPP_INFO_STREAM(node_->get_logger(), "door angle : " << door_angle);
552  rm2_common::Vector door_normal =
553  rm2_common::Rotation::RPY(0.0, 0.0, door_angle - M_PI_2) * rm2_common::Vector(1.0, 0.0, 0.0);
554 
555  double distance = door_normal.dot(door_left_bottom);
556  knock_pose_yaw = door_angle - M_PI_2;
557  rm2_common::Vector door_position = distance * door_normal;
558  door_position.z(knock_h_);

564  double x_min = std::min(door_left_bottom.x(), door_right_bottom.x());
565  double y_max = std::max(door_left_bottom.y(), door_right_bottom.y());
566  double y_min = std::min(door_left_bottom.y(), door_right_bottom.y());
567 
568  if (door_position.x() >= x_min &&
569  door_position.x() <= x_max &&
570  door_position.y() >= y_min &&
571  door_position.y() <= y_max)
572  {
573  result_knock_pose.position = door_position;
574  result_knock_pose.orientation = rm2_common::Rotation::Yaw(knock_pose_yaw);
575  return true;
576  } else {
577  result_knock_pose.position = rm2_common::Vector::Zero();
578  result_knock_pose.orientation = rm2_common::Rotation::Identity();
579  return false;
580  }

598 rclcpp_action::GoalResponse GcKnockControlWithImu::goal_callback(
599  const rclcpp_action::GoalUUID & uuid,
600  std::shared_ptr<const manipulator_interfaces::action::GcKnockMotion::Goal> goal)
601 {
602  (void) uuid;
603  RCLCPP_INFO_STREAM(node_->get_logger(), "Received 'knock' action goal");
604  std::lock_guard<std::mutex> lock(mutex_);
605 
606  // get manipulator base pose at door detected time
607  rclcpp::Time door_detected_time = goal->header.stamp;
608  std::string door_base_frame = goal->header.frame_id;
609  rm2_common::TaskValues mani_base_task_values = robot_handler_->get_present_task_value(
610  door_detected_time.seconds(),
611  manipulator_base_name_);
612 
613 
615  // Start pose (present)
617  // get present pose
618  start_joint_values_ = robot_handler_->get_previous_goal_joint_value();
619  start_joint_values_.stoped(); // Stop
620  RCLCPP_INFO_STREAM(
621  node_->get_logger(), "Start Joint : " << start_joint_values_.position.data.transpose());
622  // check joint limit
623  auto joint_names_over_limit = robot_handler_->check_joint_limit(start_joint_values_);
624  if (!joint_names_over_limit.empty()) {
625  RCLCPP_ERROR_STREAM(
626  node_->get_logger(), "Start Pose: Limit over Joint : " <<
627  merge_string_array(joint_names_over_limit));
628  return rclcpp_action::GoalResponse::REJECT;
629  }
631 
632 
634  // Prepare Pose
636  // Calculate Knock pose from manipulator base
637  rm2_common::Pose door_pose_from_manipulator_base;
638  double door_pose_yaw;
639  if (!calc_door_pose(
640  door_detected_time, door_base_frame,
641  goal->door_position,
642  mani_base_task_values.pose,
643  door_pose_from_manipulator_base,
644  door_pose_yaw
645  ))
646  {
647  RCLCPP_ERROR_STREAM(node_->get_logger(), "Door is out of knocking range.");
648  return rclcpp_action::GoalResponse::REJECT;
649  }
650  // knocking direction
651  rm2_common::Pose prepare_direction(
652  rm2_common::Vector(
653  prepare_distance_,
654  0.0,
655  0.0),
656  rm2_common::Rotation::Pitch(-start_tip_angle_));
657  // Prepare Pose
658  rm2_common::TaskValues prepare_task_values;
659  prepare_task_values.pose =
660  mani_base_task_values.pose *
661  door_pose_from_manipulator_base *
662  prepare_direction.inverse();
663  prepare_task_values.stoped(); // Stop
664  // Prepare Pose joint vavlue
665  rm2_common::JointValues ik_present_joint_values = start_joint_values_;
666  ik_present_joint_values.position(0) = door_pose_yaw;
667  if (!robot_handler_->solve_inverse_kinematics(
668  knock_component_name_, prepare_task_values, &prepare_joint_values_, ik_present_joint_values))
669  {
670  RCLCPP_ERROR_STREAM(
671  node_->get_logger(), "Fail to calculate Inverse kinematics about prepare position.");
672  return rclcpp_action::GoalResponse::REJECT;
673  }
674  prepare_joint_values_.stoped(); // Stop
675  // check joint limit
676  joint_names_over_limit = robot_handler_->check_joint_limit(prepare_joint_values_);
677  if (!joint_names_over_limit.empty()) {
678  RCLCPP_ERROR_STREAM(
679  node_->get_logger(), "Prepare Pose: Limit over Joint : " <<
680  merge_string_array(joint_names_over_limit));
681  return rclcpp_action::GoalResponse::REJECT;
682  }
683  RCLCPP_INFO_STREAM(
684  node_->get_logger(), "Prepare POSE : " << prepare_task_values.pose.position.data.transpose());
685  RCLCPP_INFO_STREAM(
686  node_->get_logger(),
687  "Prepare TWIST : " << prepare_task_values.velocity.linear.data.transpose());
688  RCLCPP_INFO_STREAM(
689  node_->get_logger(), "Prepare Joint : " << prepare_joint_values_.position.data.transpose());
691 
693  // Approach Pose
695  // Approach pose
696  knock_direction_ = rm2_common::Pose(
697  rm2_common::Vector(
698  goal->knock_direction.x,
699  0.0,
700  goal->knock_direction.y),
701  rm2_common::Rotation::Identity());
702  approach_task_values_.pose = prepare_task_values.pose * knock_direction_;
703  approach_task_values_.velocity =
704  rm2_common::Twist(
705  knock_direction_.position / approach_time_,
706  rm2_common::Vector::Zero());
707  RCLCPP_INFO_STREAM(
708  node_->get_logger(),
709  "Approach POSE : " << approach_task_values_.pose.position.data.transpose());
710  RCLCPP_INFO_STREAM(
711  node_->get_logger(),
712  "Approach TWIST : " << approach_task_values_.velocity.linear.data.transpose());
714 
715 
716  if (robot_handler_->is_debug_mode()) {
718  geometry_msgs::msg::PoseStamped door_pose_msg;
719  door_pose_msg.header.stamp = node_->now();
720  door_pose_msg.header.frame_id = manipulator_base_name_;
721  door_pose_msg.pose = trans_pose_to_msg(
722  door_pose_from_manipulator_base.position,
723  door_pose_from_manipulator_base.orientation);
724  door_pose_pub_->publish(door_pose_msg);
725 
726  geometry_msgs::msg::PoseStamped prepare_pose_msg;
727  prepare_pose_msg.header.stamp = node_->now();
728  prepare_pose_msg.header.frame_id =
729  robot_handler_->get_robot_model().get_base_component()->first;
730  prepare_pose_msg.pose = trans_pose_to_msg(
731  prepare_task_values.pose.position,
732  prepare_task_values.pose.orientation);
733  prepare_pose_pub_->publish(prepare_pose_msg);
735  }
736 
738  // Trajectories
740  // Trajectory from start to prepare
741  preprare_trajectory_.set_move_time(prepare_time_);
742  if (!preprare_trajectory_.calculation(
743  start_joint_values_.position, prepare_joint_values_.position,
744  start_joint_values_.velocity, prepare_joint_values_.velocity))
745  {
746  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to calcuate Prepare trajectory.");
747  return rclcpp_action::GoalResponse::REJECT;
748  }
749  // Trajectory from prepare to approach pose
750  approach_trajectory_.set_move_time(approach_time_);
751  if (!approach_trajectory_.calculation(
752  prepare_task_values.pose, approach_task_values_.pose,
753  prepare_task_values.velocity, approach_task_values_.velocity))
754  {
755  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to calcuate Approach trajectory.");
756  return rclcpp_action::GoalResponse::REJECT;
757  }
759 
761  // Parameter Initialize
763 // imu_data_threshold_ = 5000.0; // action .
764  imu_data_ = 0.0;
765  knocking_count_ = 0;
766  total_knocking_count_ = goal->knock_num;
767  knocking_state_ = KnockMotionState::Prepare;
768  success_flag_ = true;
769 
770  // set start time
771  robot_handler_->set_start_time(node_->now().seconds());
773 
774  return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
775 }

777 rclcpp_action::CancelResponse GcKnockControlWithImu::cancel_callback(
778  const std::shared_ptr<rclcpp_action::ServerGoalHandle<KnockMotionAction>> goal_handle)
779 {
780  RCLCPP_INFO_STREAM(node_->get_logger(), "Got request to cancel goal");
781  std::lock_guard<std::mutex> lock(mutex_);
782  try {
783  if (actived_goal_ == goal_handle) {
784  stop_update();
785 
786  RCLCPP_INFO_STREAM(
787  node_->get_logger(),
788  "Canceling active action goal because cancel callback received.");
789 
790  // Mark the current goal as canceled
791  auto action_res = std::make_shared<KnockMotionAction::Result>();
792  manipulator_interfaces::msg::Result result_state;
793  result_state.set__val(manipulator_interfaces::msg::Result::CANCELED);
794  action_res->set__result_state(result_state);
795  if (actived_goal_) {
796  actived_goal_->canceled(action_res);
797  }
798  } else {
799  // Reset current goal
800  actived_goal_.reset();
801  RCLCPP_ERROR_STREAM(
802  node_->get_logger(), "Fail to Canceling active action goal due to goal handle mismatch.");
803  return rclcpp_action::CancelResponse::REJECT;
804  }
805  } catch (std::runtime_error e) {
806  // Reset current goal
807  actived_goal_.reset();
808  RCLCPP_ERROR_STREAM(node_->get_logger(), "Runtime error : " << e.what());
809  return rclcpp_action::CancelResponse::REJECT;
810  }
811  // Reset current goal
812  actived_goal_.reset();
813  return rclcpp_action::CancelResponse::ACCEPT;
814 }
