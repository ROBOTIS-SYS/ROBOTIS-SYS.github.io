68 hardware_interface::return_type DynamixelHardware::configure(
69  const hardware_interface::HardwareInfo & info)
70 {
71  // log msg pub
72  log_pub_ = this->create_publisher<aimbot_log_interfaces::msg::Log>(
73  "/aimbot_log", rclcpp::SystemDefaultsQoS());
74 
75  if (configure_default(info) != hardware_interface::return_type::OK) {
76  return hardware_interface::return_type::ERROR;
77  }
78 
79  num_of_joints_ = static_cast<size_t>(stoi(info_.hardware_parameters["number_of_joints"]));
80  num_of_transmissions_ =
81  static_cast<size_t>(stoi(info_.hardware_parameters["number_of_transmissions"]));
82  SetMatrix();
83 
85  // Dynamixel Communication Setting
86  port_name_ = info_.hardware_parameters["port_name"];
87  baud_rate_ = info_.hardware_parameters["baud_rate"];
88  err_timeout_sec_ = stod(info_.hardware_parameters["error_timeout_sec"]);
89 
90  RCLCPP_INFO_STREAM(
91  logger_,
92  "port_name " << port_name_.c_str() << " / baudrate " << baud_rate_.c_str());
93 
94  // Dynamixel Model Setting
95  std::string dxl_model_folder = info_.hardware_parameters["dynamixel_model_folder"];
96  dxl_comm_ = std::unique_ptr<Dynamixel>(
97  new Dynamixel(
98  (ament_index_cpp::get_package_share_directory("dynamixel_hardware_interface") +
99  dxl_model_folder).c_str()));
100 
102  // init communication
103  RCLCPP_INFO_STREAM(logger_, "$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$");
104  RCLCPP_INFO_STREAM(logger_, "$$$$$ Init Dxl Comm Port");
105  for (const hardware_interface::ComponentInfo & gpio : info_.gpios) {
106  if (gpio.parameters.at("type") == "dxl") {
107  dxl_id_.push_back(static_cast<uint8_t>(stoi(gpio.parameters.at("ID"))));
108  } else if (gpio.parameters.at("type") == "sensor") {
109  sensor_id_.push_back(static_cast<uint8_t>(stoi(gpio.parameters.at("ID"))));
110  } else {
111  RCLCPP_ERROR_STREAM(logger_, "Invalid DXL / Sensoe type");
112  exit(-1);
113  }
114  }
115  // if (dxl_comm_->InitDxlComm(dxl_id_, port_name_, baud_rate_) != DxlError::OK) {
116  // return hardware_interface::return_type::ERROR;
117  // }
118  bool trying_connect = true;
119  int trying_cnt = 60; // second
120  int cnt = 0;
121 
122  while (trying_connect) {
123  std::vector<uint8_t> id_arr;
124  for (auto dxl : dxl_id_) {
125  id_arr.push_back(dxl);
126  }
127  for (auto sensor : sensor_id_) {
128  id_arr.push_back(sensor);
129  }
130  // init communication
131  if (dxl_comm_->InitDxlComm(id_arr, port_name_, baud_rate_) == DxlError::OK) {
132  RCLCPP_INFO_STREAM(logger_, "Trying to connect to the communication port...");
133  trying_connect = false;
134  } else {
135  sleep(1);
136  cnt++;
137  if (cnt > trying_cnt) {
138  RCLCPP_ERROR_STREAM(logger_, "Cannot connect communication port! :(");
139 
140  aimbot_log_interfaces::msg::Log msg;
141  msg.header.stamp = this->now();
142  msg.level = msg.LEVEL_DEBUG;
143  msg.type = msg.TYPE_ERROR;
144  msg.source = "Xavier-Manipulator";
145  msg.contents = "Initial dxl connection fail.";
146  log_pub_->publish(msg);
147 
148  cnt = 0;
149  // exit(-1); // 노드 종료
150  }
151  }
152  } // end of while
153 
154  // item initialization
155  RCLCPP_INFO_STREAM(logger_, "$$$$$ Init Dxl Items");
156  for (const hardware_interface::ComponentInfo & gpio : info_.gpios) {
157  uint8_t id = static_cast<uint8_t>(stoi(gpio.parameters.at("ID")));
158  for (auto it : gpio.parameters) {
159  if (it.first != "ID" && it.first != "type") {
160  if (dxl_comm_->WriteItem(
161  id, it.first,
162  static_cast<uint32_t>(stoi(it.second))) != DxlError::OK)
163  {
164  RCLCPP_ERROR_STREAM(logger_, "[ID:" << std::to_string(id) << "] Wtite Item error");
165  // return hardware_interface::return_type::ERROR;
166  }
167  RCLCPP_INFO_STREAM(
168  logger_,
169  "[ID:" << std::to_string(id) << "] item_name:" << it.first.c_str() << "\tdata:" <<
170  stoi(it.second));
171  }
172  }
173  }
174 
175  // set read items & transmissions handler initialization
176  RCLCPP_INFO_STREAM(logger_, "$$$$$ Init Dxl Read Items");
177  hdl_trans_states_.clear();
178  hdl_gpio_sensor_states_.clear();
179  for (const hardware_interface::ComponentInfo & gpio : info_.gpios) {
180  if (gpio.state_interfaces.size() && gpio.parameters.at("type") == "dxl") {
181  // read item
182  uint8_t id = static_cast<uint8_t>(stoi(gpio.parameters.at("ID")));
183  HandlerVarType temp_read;
184 
185  temp_read.id = id;
186  temp_read.name = gpio.name;
187 
188  // Present Position
189  temp_read.interface_name_vec.push_back("Present Position");
190  temp_read.value_ptr_vec.push_back(std::make_shared<double>(0.0));
191 
192  // Present Velocity
193  temp_read.interface_name_vec.push_back("Present Velocity");
194  temp_read.value_ptr_vec.push_back(std::make_shared<double>(0.0));
195 
196  for (auto it : gpio.state_interfaces) {
197  if (it.name != "Present Position" && it.name != "Present Velocity") {
198  temp_read.interface_name_vec.push_back(it.name);
199  temp_read.value_ptr_vec.push_back(std::make_shared<double>(0.0));
200 
201  // init. hardware error state
202  if (it.name == "Hardware Error Status") {
203  dxl_hw_err_[id] = 0x00;
204  }
205  }
206  }
207  hdl_trans_states_.push_back(temp_read);
208  } else if (gpio.state_interfaces.size() && gpio.parameters.at("type") == "sensor") {
209  HandlerVarType temp_sensor;
210  for (auto it : gpio.state_interfaces) {
211  uint8_t id = static_cast<uint8_t>(stoi(gpio.parameters.at("ID")));
212 
213  temp_sensor.id = id;
214  temp_sensor.name = gpio.name;
215  for (auto it : gpio.state_interfaces) {
216  temp_sensor.interface_name_vec.push_back(it.name);
217  temp_sensor.value_ptr_vec.push_back(std::make_shared<double>(0.0));
218  }
219  }
220  hdl_gpio_sensor_states_.push_back(temp_sensor);
221  }
222  }
223 
224  for (auto it : hdl_trans_states_) {
225  if (dxl_comm_->SetDxlReadItems(
226  it.id, it.interface_name_vec,
227  it.value_ptr_vec) != DxlError::OK)
228  {
229  return hardware_interface::return_type::ERROR;
230  }
231  }
232 
233  if (dxl_comm_->SetMultiDxlRead() != DxlError::OK) {
234  return hardware_interface::return_type::ERROR;
235  }
236 
237  // set write items & transmissions handler initialization
238  RCLCPP_INFO_STREAM(logger_, "$$$$$ Init Dxl Write Items");
239  hdl_trans_commands_.clear();
240  for (const hardware_interface::ComponentInfo & gpio : info_.gpios) {
241  if (gpio.command_interfaces.size()) {
242  // write item
243  uint8_t id = static_cast<uint8_t>(stoi(gpio.parameters.at("ID")));
244  for (auto it : gpio.command_interfaces) {
245  HandlerVarType temp_write;
246  temp_write.id = id;
247  temp_write.name = gpio.name;
248 
249  temp_write.interface_name_vec.push_back(it.name);
250  temp_write.value_ptr_vec.push_back(std::make_shared<double>(0.0));
251  hdl_trans_commands_.push_back(temp_write);
252  }
253  }
254  }
255  for (auto it : hdl_trans_commands_) {
256  if (dxl_comm_->SetDxlWriteItems(
257  it.id, it.interface_name_vec,
258  it.value_ptr_vec) != DxlError::OK)
259  {
260  return hardware_interface::return_type::ERROR;
261  }
262  }
263 
264  if (dxl_comm_->SetMultiDxlWrite() != DxlError::OK) {
265  return hardware_interface::return_type::ERROR;
266  }
267 
268  if (num_of_transmissions_ != hdl_trans_commands_.size() &&
269  num_of_transmissions_ != hdl_trans_states_.size())
270  {
271  RCLCPP_ERROR_STREAM(
272  logger_, "Error: number of transmission " << num_of_transmissions_ << ", " <<
273  hdl_trans_commands_.size() << ", " << hdl_trans_states_.size());
274  return hardware_interface::return_type::ERROR;
275  }
276 
278  comm_reset_flag_ = COMM_OK;
279 
281  hdl_joint_states_.clear();
282  for (const hardware_interface::ComponentInfo & joint : info_.joints) {
283  HandlerVarType temp_state;
284  temp_state.name = joint.name;
285 
286  // position
287  temp_state.interface_name_vec.push_back("position");
288  temp_state.value_ptr_vec.push_back(std::make_shared<double>(0.0));
289 
290  // velocity
291  temp_state.interface_name_vec.push_back("velocity");
292  temp_state.value_ptr_vec.push_back(std::make_shared<double>(0.0));
293 
294  for (auto it : joint.state_interfaces) {
295  if (hardware_interface::HW_IF_POSITION != it.name &&
296  hardware_interface::HW_IF_VELOCITY != it.name &&
297  hardware_interface::HW_IF_ACCELERATION != it.name &&
298  hardware_interface::HW_IF_EFFORT != it.name &&
299  HW_IF_HARDWARE_STATE != it.name)
300  {
301  return hardware_interface::return_type::ERROR;
302  }
303  if (it.name != "position" && it.name != "velocity") {
304  temp_state.interface_name_vec.push_back(it.name);
305  temp_state.value_ptr_vec.push_back(std::make_shared<double>(0.0));
306  }
307  }
308  hdl_joint_states_.push_back(temp_state);
309  }
310 
311 
312  hdl_joint_commands_.clear();
313  for (const hardware_interface::ComponentInfo & joint : info_.joints) {
314  HandlerVarType temp_cmd;
315  temp_cmd.name = joint.name;
316 
317  for (auto it : joint.command_interfaces) {
318  if (hardware_interface::HW_IF_POSITION != it.name &&
319  hardware_interface::HW_IF_VELOCITY != it.name &&
320  hardware_interface::HW_IF_ACCELERATION != it.name &&
321  hardware_interface::HW_IF_EFFORT != it.name)
322  {
323  return hardware_interface::return_type::ERROR;
324  }
325  temp_cmd.interface_name_vec.push_back(it.name);
326  temp_cmd.value_ptr_vec.push_back(std::make_shared<double>(0.0));
327  }
328  hdl_joint_commands_.push_back(temp_cmd);
329  }
330 
331  if (num_of_joints_ != hdl_joint_commands_.size() &&
332  num_of_joints_ != hdl_joint_states_.size())
333  {
334  RCLCPP_ERROR_STREAM(
335  logger_, "Error: number of joints " << num_of_joints_ << ", " <<
336  hdl_joint_commands_.size() << ", " << hdl_joint_commands_.size());
337  return hardware_interface::return_type::ERROR;
338  }
339 
340  if (num_of_joints_ != hdl_joint_commands_.size() &&
341  num_of_joints_ != hdl_joint_states_.size())
342  {
343  RCLCPP_ERROR_STREAM(
344  logger_, "Error: number of joints " << num_of_joints_ << ", " <<
345  hdl_joint_commands_.size() << ", " << hdl_joint_commands_.size());
346  return hardware_interface::return_type::ERROR;
347  }
348 
350  hdl_sensor_states_.clear();
351  for (const hardware_interface::ComponentInfo & sensor : info_.sensors) {
352  HandlerVarType temp_state;
353  temp_state.name = sensor.name;
354 
355  for (auto it : sensor.state_interfaces) {
356  temp_state.interface_name_vec.push_back(it.name);
357  temp_state.value_ptr_vec.push_back(std::make_shared<double>(0.0));
358  }
359  hdl_sensor_states_.push_back(temp_state);
360  }
361 
363  // ros msg pub
364  std::string str_dxl_state_pub_name =
365  info_.hardware_parameters["dynamixel_state_pub_msg_name"];
366  dxl_state_pub_ = this->create_publisher<DynamixelStateMsg>(
367  str_dxl_state_pub_name, rclcpp::SystemDefaultsQoS());
368  dxl_state_pub_uni_ptr_ = std::make_unique<StatePublisher>(dxl_state_pub_);
369 
370  size_t num_of_pub_data = hdl_trans_states_.size();
371  dxl_state_pub_uni_ptr_->lock();
372  dxl_state_pub_uni_ptr_->msg_.id.resize(num_of_pub_data);
373  dxl_state_pub_uni_ptr_->msg_.dxl_hw_state.resize(num_of_pub_data);
374  dxl_state_pub_uni_ptr_->unlock();
375 
376  // ros srv server
377  using namespace std::placeholders;
378  std::string str_get_dxl_data_srv_name =
379  info_.hardware_parameters["get_dynamixel_data_srv_name"];
380  get_dxl_data_srv_ = create_service<dynamixel_interfaces::srv::GetDataFromDxl>(
381  str_get_dxl_data_srv_name,
382  std::bind(&DynamixelHardware::get_dxl_data_srv_callback, this, _1, _2));
383 
384  std::string str_set_dxl_data_srv_name =
385  info_.hardware_parameters["set_dynamixel_data_srv_name"];
386  set_dxl_data_srv_ = create_service<dynamixel_interfaces::srv::SetDataToDxl>(
387  str_set_dxl_data_srv_name,
388  std::bind(&DynamixelHardware::set_dxl_data_srv_callback, this, _1, _2));
389 
390  std::string str_reboot_dxl_srv_name =
391  info_.hardware_parameters["reboot_dxl_srv_name"];
392  reboot_dxl_srv_ = create_service<dynamixel_interfaces::srv::RebootDxl>(
393  str_reboot_dxl_srv_name,
394  std::bind(&DynamixelHardware::reboot_dxl_srv_callback, this, _1, _2));
395 
396 
398  ros_update_freq_ = stoi(info_.hardware_parameters["ros_update_freq"]);
399  ros_update_thread_ = std::thread(
400  [this]() {
401  RCLCPP_INFO_STREAM(logger_, "ros_update rate is " << ros_update_freq_ << "hz");
402  while (rclcpp::ok()) {
403  // dxl state pub
404  size_t index = 0;
405  if (dxl_state_pub_uni_ptr_ && dxl_state_pub_uni_ptr_->trylock()) {
406  dxl_state_pub_uni_ptr_->msg_.header.stamp = this->now();
407  dxl_state_pub_uni_ptr_->msg_.comm_state = dxl_comm_err_;
408  for (auto it : hdl_trans_states_) {
409  dxl_state_pub_uni_ptr_->msg_.id.at(index) = it.id;
410  dxl_state_pub_uni_ptr_->msg_.dxl_hw_state.at(index) = dxl_hw_err_[it.id];
411  index++;
412  }
413  dxl_state_pub_uni_ptr_->unlockAndPublish();
414  }
415 
416  // ros spin
417  rclcpp::spin_some(this->get_node_base_interface());
418 
419  // ros_update thread time sleep
420  struct timespec duration{};
421  duration.tv_nsec = 1000000000 / ros_update_freq_;
422  clock_nanosleep(CLOCK_MONOTONIC, 0, &duration, nullptr);
423  }
424  }
425  );
426 
427  aimbot_log_interfaces::msg::Log msg;
428  msg.header.stamp = this->now();
429  msg.level = msg.LEVEL_DEBUG;
430  msg.type = msg.TYPE_INFO;
431  msg.source = "Xavier-Manipulator";
432  msg.contents = "Initial dxl connection success";
433  log_pub_->publish(msg);
434 
435  status_ = hardware_interface::status::CONFIGURED;
436  return hardware_interface::return_type::OK;
437 }

496 hardware_interface::return_type DynamixelHardware::start()
497 {
498  // read present state from dxl
499  dxl_comm_err_ = dxl_comm_->ReadMultiDxlData();
500  if (dxl_comm_err_ != DxlError::OK) {
501  RCLCPP_ERROR_STREAM(logger_, "Dynamixel Start Fail --> " << dxl_comm_err_);
502  return hardware_interface::return_type::ERROR;
503  }
504 
505  // actuator to handler
506  CalcTransmissionToJoint();
507 
508  // sync commands = states joint
509  for (auto it_states : hdl_joint_states_) {
510  for (auto it_commands : hdl_joint_commands_) {
511  if (it_states.name == it_commands.name) {
512  for (size_t i = 0; i < it_states.interface_name_vec.size(); i++) {
513  if (it_commands.interface_name_vec.at(0) == it_states.interface_name_vec.at(i)) {
514  *it_commands.value_ptr_vec.at(0) = *it_states.value_ptr_vec.at(i);
515  RCLCPP_INFO_STREAM(
516  logger_, "Sync joint state to command (" <<
517  it_commands.interface_name_vec.at(0).c_str() << ", " <<
518  *it_commands.value_ptr_vec.at(0) * 180.0 / 3.141592 << " <- " <<
519  it_states.interface_name_vec.at(i).c_str() << ", " <<
520  *it_states.value_ptr_vec.at(i) * 180.0 / 3.141592);
521  }
522  }
523  }
524  }
525  }
526 
527  usleep(500 * 1000); // 500ms
528 
529  // torque on
530  dxl_comm_->DynamixelEnable(dxl_id_);
531 
532 
533  RCLCPP_INFO_STREAM(logger_, "Dynamixel Hardware Start!");
534 
535  status_ = hardware_interface::status::STARTED;
536  return hardware_interface::return_type::OK;
537 }

550 hardware_interface::return_type DynamixelHardware::read()
551 {
552  if (comm_reset_flag_ == COMM_OK) {
553  // read dxl
554  dxl_comm_err_ = CheckError(dxl_comm_->ReadMultiDxlData());
555  if (dxl_comm_err_ != DxlError::OK) {
556  RCLCPP_ERROR_STREAM(logger_, "Dynamixel Read Fail " << dxl_comm_err_);
557  return hardware_interface::return_type::ERROR;
558  }
559 
560  // actuator to handler
561  CalcTransmissionToJoint();
562 
563  // read sensor
564  for (auto sensor : hdl_gpio_sensor_states_) {
565  for (auto item : sensor.interface_name_vec) {
566  uint32_t data;
567  dxl_comm_->ReadItem(sensor.id, item, data);
568  for (size_t i = 0; i < hdl_sensor_states_.size(); i++) {
569  for (size_t j = 0; j < hdl_sensor_states_.at(i).interface_name_vec.size(); j++) {
570  if (hdl_sensor_states_.at(i).name == sensor.name &&
571  hdl_sensor_states_.at(i).interface_name_vec.at(j) == item)
572  {
573  *hdl_sensor_states_.at(i).value_ptr_vec.at(j) = data;
574  }
575  }
576  }
577  }
578  }
579 
580  // read item buffer

736 void DynamixelHardware::CommReset()
737 {
738  RCLCPP_INFO_STREAM(logger_, "Communication Reset Thread Start");
739 
740  // hardware interface stop
741  stop();
742 
743  aimbot_log_interfaces::msg::Log msg;
744  msg.header.stamp = this->now();
745  msg.level = msg.LEVEL_DEBUG;
746  msg.type = msg.TYPE_ERROR;
747  msg.source = "Xavier-Manipulator";
748  msg.contents = "Manipulator reset start";

750

752  dxl_comm_->RWDataReset();
753 
754  while (1) {

756 
757  // reboot dxl

761  // exit(-1); // 노드 종료
762  is_reset_ok = false;
763  break;
764  }

766  }
767 
768  if (is_reset_ok) {
769  // item initialization
770  RCLCPP_INFO_STREAM(logger_, "$$$$$ Init Dxl Items");
771  for (const hardware_interface::ComponentInfo & gpio : info_.gpios) {
772  uint8_t id = static_cast<uint8_t>(stoi(gpio.parameters.at("ID")));
773  for (auto it : gpio.parameters) {
774  if (it.first != "ID" && it.first != "type") {

776  id, it.first,
777  static_cast<uint32_t>(stoi(it.second))) != DxlError::OK)

779  RCLCPP_ERROR_STREAM(logger_, "[ID:" << std::to_string(id) << "] Wtite Item error");
780  // is_reset_ok = false;

782  RCLCPP_INFO_STREAM(

784  "[ID:" << std::to_string(id) << "] item_name:" << it.first.c_str() << "\tdata:" <<
785  stoi(it.second));
786  }
787  }
788  }
789  }
790 
791  if (is_reset_ok) {
792  // set read items
793  for (auto it : hdl_trans_states_) {
794  if (dxl_comm_->SetDxlReadItems(
795  it.id, it.interface_name_vec,

806  // exit(-1); // 노드 종료
807  is_reset_ok = false;

811  if (is_reset_ok) {
812  for (auto it : hdl_trans_commands_) {
813  if (dxl_comm_->SetDxlWriteItems(
814  it.id, it.interface_name_vec,
815  it.value_ptr_vec) != DxlError::OK)

817  RCLCPP_ERROR_STREAM(logger_, "Cannot connect communication port! :(");
818  // exit(-1); // 노드 종료
819  is_reset_ok = false;
820  }
821  }
822 
823  if (dxl_comm_->SetMultiDxlWrite() != DxlError::OK) {
824  RCLCPP_ERROR_STREAM(logger_, "Cannot connect communication port! :(");
825  // exit(-1); // 노드 종료
826  is_reset_ok = false;
827  }
828  }
829 
830  if (is_reset_ok) {
831  // start (torque on)
832  start();
833 
834  aimbot_log_interfaces::msg::Log msg;
835  msg.header.stamp = this->now();

844  }
845 
846  if (comm_reset_flag_ != WAIT_READ) {
847  RCLCPP_ERROR_STREAM(logger_, "RESET restart");
848  usleep(1000 * 1000); // 1000ms
849  } else {
850  RCLCPP_ERROR_STREAM(logger_, "RESET success");
851  return;
