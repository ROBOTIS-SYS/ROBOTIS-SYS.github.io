28 def generate_launch_description():
29  # Get the launch directory
30  bringup_dir = get_package_share_directory('antis_nav_bringup')
31  launch_dir = os.path.join(bringup_dir, 'launch')
32 
33  # Create the launch configuration variables
34  namespace = LaunchConfiguration('namespace')
35  use_namespace = LaunchConfiguration('use_namespace')
36  slam = LaunchConfiguration('slam')
37  map_yaml_file = LaunchConfiguration('map')
38  use_sim_time = LaunchConfiguration('use_sim_time')
39  params_file = LaunchConfiguration('params_file')
40  default_bt_xml_filename = LaunchConfiguration('default_bt_xml_filename')
41  autostart = LaunchConfiguration('autostart')
42 
43  stdout_linebuf_envvar = SetEnvironmentVariable(
44  'RCUTILS_LOGGING_BUFFERED_STREAM', '1')
45 
46  declare_namespace_cmd = DeclareLaunchArgument(
47  'namespace',
48  default_value='',
49  description='Top-level namespace')
50 
51  declare_use_namespace_cmd = DeclareLaunchArgument(
52  'use_namespace',
53  default_value='false',
54  description='Whether to apply a namespace to the navigation stack')
55 
56  declare_slam_cmd = DeclareLaunchArgument(
57  'slam',
58  default_value='False',
59  description='Whether run a SLAM')
60 
61  declare_map_yaml_cmd = DeclareLaunchArgument(
62  'map',
63  description='Full path to map yaml file to load')
64 
65  declare_use_sim_time_cmd = DeclareLaunchArgument(
66  'use_sim_time',
67  default_value='false',
68  description='Use simulation (Gazebo) clock if true')
69 
70  declare_params_file_cmd = DeclareLaunchArgument(
71  'params_file',
72  default_value=os.path.join(bringup_dir, 'params', 'antis_nav_params.yaml'),
73  description='Full path to the ROS2 parameters file to use for all launched nodes')
74 
75  declare_bt_xml_cmd = DeclareLaunchArgument(
76  'default_bt_xml_filename',
77  default_value=os.path.join(
78  get_package_share_directory('antis_nav_bt_navigator'),
79  'behavior_trees', 'navigate_w_replanning_and_recovery.xml'),
80  description='Full path to the behavior tree xml file to use')
81 
82  declare_autostart_cmd = DeclareLaunchArgument(
83  'autostart', default_value='true',
84  description='Automatically startup the antis_nav_util stack')
85 
86  # Specify the actions
87  bringup_cmd_group = GroupAction([
88  PushRosNamespace(
89  condition=IfCondition(use_namespace),
90  namespace=namespace),
91 
92  IncludeLaunchDescription(
93  PythonLaunchDescriptionSource(os.path.join(launch_dir, 'slam_launch.py')),
94  condition=IfCondition(slam),
95  launch_arguments={'namespace': namespace,
96  'use_sim_time': use_sim_time,
97  'autostart': autostart,
98  'params_file': params_file}.items()),
99 
100  IncludeLaunchDescription(
101  PythonLaunchDescriptionSource(os.path.join(launch_dir,
102  'localization_launch.py')),
103  condition=IfCondition(PythonExpression(['not ', slam])),
104  launch_arguments={'namespace': namespace,
105  'map': map_yaml_file,
106  'use_sim_time': use_sim_time,
107  'autostart': autostart,
108  'params_file': params_file,
109  'use_lifecycle_mgr': 'false'}.items()),
110 
111  IncludeLaunchDescription(
112  PythonLaunchDescriptionSource(os.path.join(launch_dir, 'navigation_launch.py')),
113  launch_arguments={'namespace': namespace,
114  'use_sim_time': use_sim_time,
115  'autostart': autostart,
116  'params_file': params_file,
117  'default_bt_xml_filename': default_bt_xml_filename,
118  'use_lifecycle_mgr': 'false',
119  'map_subscribe_transient_local': 'true'}.items()),
120  ])
121 
122  # Create the launch description and populate
123  ld = LaunchDescription()
124 
125  # Set environment variables
126  ld.add_action(stdout_linebuf_envvar)
127 
128  # Declare the launch options
129  ld.add_action(declare_namespace_cmd)
130  ld.add_action(declare_use_namespace_cmd)
131  ld.add_action(declare_slam_cmd)
132  ld.add_action(declare_map_yaml_cmd)
133  ld.add_action(declare_use_sim_time_cmd)
134  ld.add_action(declare_params_file_cmd)
135  ld.add_action(declare_autostart_cmd)
136  ld.add_action(declare_bt_xml_cmd)
137 
138  # Add the actions to launch all of the navigation nodes
139  ld.add_action(bringup_cmd_group)
140 
141  return ld
