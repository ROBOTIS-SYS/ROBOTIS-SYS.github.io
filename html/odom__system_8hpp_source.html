103  bool update_system_state(uint64_t time_pair_size, uint64_t odom_fused_size)
104  {
105  bool able_to_run = false;
106  bool full_streaming = true;
107  uint stream_count = 0;
108  for (size_t i = 0; i < states_.size(); i++) {
109  bool stream = (states_[i] == BufferState::STREAMING);
110  able_to_run |= stream;
111  full_streaming &= stream;
112  if (stream) {stream_count++;}
113  }
114  if (!able_to_run) {
115  sys_state_ = SystemState::FATAL;
116  } else if (full_streaming) {
117  sys_state_ = SystemState::FULL_FUSION_MODE;
118  } else if (stream_count == 1) {
119  sys_state_ = SystemState::BYPASS_MODE;
120  } else {
121  sys_state_ = SystemState::PARTIAL_FUSION_MODE;
122  }
123  time_pair_size_ = time_pair_size;
124  odom_fused_size_ = odom_fused_size;
125  return able_to_run;
126  }

294 bool OdomSystem::sanitizer(
295  rclcpp::Time now,
296  int timeout_msec)
297 {
298  // state update
299  if (odom_queue_1_.size() > 0) {
300  rclcpp::Duration latency =
301  now - odom_queue_1_.back()->msg.header.stamp;
302  if (latency.nanoseconds() / 10e+6 < timeout_msec) {
303  is_queue1_ok_ = true;
304  buffer_states_.states_[0] = BufferState::STREAMING;
305  } else {
306  is_queue1_ok_ = false;
307  buffer_states_.states_[0] = BufferState::OFFLINE;
308  }
309  } else {
310  is_queue1_ok_ = false;
311  buffer_states_.states_[0] = BufferState::EMPTY;
312  }
313 
314  if (odom_queue_2_.size() > 0) {
315  rclcpp::Duration latency =
316  now - odom_queue_2_.back()->msg.header.stamp;
317  if (latency.nanoseconds() / 10e+6 < timeout_msec) {
318  is_queue2_ok_ = true;
319  buffer_states_.states_[1] = BufferState::STREAMING;
320  } else {
321  is_queue2_ok_ = false;
322  buffer_states_.states_[1] = BufferState::OFFLINE;
323  }
324  } else {
325  is_queue2_ok_ = false;
326  buffer_states_.states_[1] = BufferState::EMPTY;
327  }
328 
329  // return odometry state
330  return buffer_states_.update_system_state(
331  odom_sync_->time_matching_pair_ptr_.size(),
332  odom_fused_ring_.size());
333 }

366  break;

370 bool OdomSystem::manage_queue(rclcpp::Time now, int timeout_msec)
371 {
372  // check the odom_queue drop time
373  rclcpp::Time drop_time = odom_fused_ring_.front().header.stamp;
374  // 하나라도 스트리밍이 안되면
375  bool is_safe_stream = true;
376  for (size_t i = 0; i < buffer_states_.states_.size(); i++) {
377  is_safe_stream = is_safe_stream && (buffer_states_.states_[i] == BufferState::STREAMING);
378  }
379 
380  // 타임아웃에 의한 강제 drop
381  if (!is_safe_stream) {
382  drop_time = now - rclcpp::Duration(0, timeout_msec * 10e+6);
383  }
384 
385  // clearing odom_queue before drop_time
386  for (auto odom_it = odom_queue_1_.begin(); odom_it != odom_queue_1_.end(); ) {
387  if (rclcpp::Time((*odom_it)->msg.header.stamp) >= drop_time) {
388  break;
389  }
390  odom_it = odom_queue_1_.erase(odom_it); // odom_it++ aleady executed
391  }
392 
393  for (auto odom_it = odom_queue_2_.begin(); odom_it != odom_queue_2_.end(); ) {
394  if (rclcpp::Time((*odom_it)->msg.header.stamp) >= drop_time) {
395  break;
396  }
397  odom_it = odom_queue_2_.erase(odom_it); // odom_it++ aleady executed
398  }
399  return true;
400 }
