76 int BellDetector::load_network()
77 {
78  using ModelState = perception_interfaces::srv::GetModelState::Response;
79 
80  if (obj_detector_.use_count() == 0) {
81  RCLCPP_INFO(
82  get_logger(), "Bell detection model loading...");
83  obj_detector_ = std::make_shared<ml_model::ObjectDetection>(bell_det_model_name_);
84  int bell_det_state = obj_detector_->initialize();
85  if (bell_det_state >= PERCEPTION_STATE::SUCCESS_NO_ISSUE) {
86  RCLCPP_INFO(
87  get_logger(), "Bell detection model successfully loaded");
88  } else if (bell_det_state == PERCEPTION_STATE::ERROR_MODEL_READ) {
89  RCLCPP_ERROR(
90  this->get_logger(), "No bell detection model model exists");
91  return ModelState::MODEL_DOES_NOT_EXIST;
92  } else {
93  RCLCPP_ERROR(
94  this->get_logger(), "Failed to load bell detection model * '%s'",
95  get_error_log(bell_det_state));
96  return ModelState::MODEL_LOAD_FAILED;
97  }
98  } else {
99  RCLCPP_INFO(
100  get_logger(), "Already bell detection model loaded");
101  }
102 
103  this->set_model_state(ModelState::LOAD);
104 
105  return ModelState::SUCCESS;
106 }

140 bool BellDetector::update()
141 {
142  cv::Mat cv_left_image, cv_right_image;
143  rclcpp::Time left_stamp, right_stamp;
144  std::string dummy;
145  bool ack_left_img, ack_right_img;
146 
147  if (this->is_data_available(left_img_sub_msg_name_) &&
148  this->is_data_available(right_img_sub_msg_name_))
149  {
150  ack_left_img =
151  this->get_data_image(left_img_sub_msg_name_, cv_left_image, left_stamp, dummy);
152  ack_right_img =
153  this->get_data_image(right_img_sub_msg_name_, cv_right_image, right_stamp, dummy);
154  } else {
155  return false;
156  }
157 
158  if (ack_left_img && ack_right_img) {
159  camera_status_ = true;
160  cv::Mat cv_vconcat_image;
161  cv_vconcat_image = visualizer_->convert_rgba_imgs_to_rgb_img(cv_left_image, cv_right_image);
162  std::vector<ObjectDetectorOutputs> detector_outputs;
163  std::vector<RecognizedText> detected_buttons;
164  std::vector<RecognizedText> left_detected_buttons;
165  std::vector<RecognizedText> right_detected_buttons;
166  offset_ = cv::Point2i(0, static_cast<int>(cv_vconcat_image.size().height / 2));
167 
168  if (obj_detector_.use_count() > 0) {
169  obj_detector_->model_run(cv_vconcat_image, detector_outputs);
170  detected_buttons = convert_to_recognized_text(detector_outputs);
171  divide_left_right_pts(detected_buttons, left_detected_buttons, right_detected_buttons);
172 
173  visualizer_->get_object_detection_result(
174  cv_vconcat_image, cv_vconcat_image, detector_outputs,
175  bell_det_labels_);
176 
177  sensor_msgs::msg::PointCloud2 dummy;
178  dummy.header.stamp = left_stamp;
179  pose_estimator_->set_input(
180  left_detected_buttons,
181  right_detected_buttons,
182  dummy);
183 
184  geometry_msgs::msg::PoseStamped target_btn_pose;
185  bool target_btn_state = false;
186  int calc_state = pose_estimator_->calculate();
187  if (calc_state > -1) {
188  if (pose_estimator_->get_pose(target_btn_str_, target_btn_pose)) {
189  target_btn_pose.header.stamp = left_stamp;
190  target_btn_pose.header.frame_id = frame_id_;
191 
192  Eigen::Quaterniond q(
193  target_btn_pose.pose.orientation.w,
194  target_btn_pose.pose.orientation.x,
195  target_btn_pose.pose.orientation.y,
196  target_btn_pose.pose.orientation.z);
197  q.normalize();
198  target_btn_pose.pose.orientation.w = q.w();
199  target_btn_pose.pose.orientation.x = q.x();
200  target_btn_pose.pose.orientation.y = q.y();
201  target_btn_pose.pose.orientation.z = q.z();
202 
203  RCLCPP_INFO_STREAM(
204  this->get_logger(),
205  "frame time stamp: " << std::fixed << left_stamp.seconds());
206  RCLCPP_INFO_STREAM(
207  this->get_logger(),
208  "target button name: " << target_btn_str_);
209  RCLCPP_INFO_STREAM(
210  this->get_logger(),
211  "target button position: " <<
212  target_btn_pose.pose.position.x << ", " <<
213  target_btn_pose.pose.position.y << ", " <<
214  target_btn_pose.pose.position.z);
215  RCLCPP_INFO_STREAM(
216  this->get_logger(),
217  "target button orientation: " <<
218  target_btn_pose.pose.orientation.w << ", " <<
219  target_btn_pose.pose.orientation.x << ", " <<
220  target_btn_pose.pose.orientation.y << ", " <<
221  target_btn_pose.pose.orientation.z);
222 
223  bell_detected_btn_goal_pose_pub_->publish(target_btn_pose);
224  } else {
225  RCLCPP_WARN(
226  this->get_logger(),
227  "Failed to get target pose (invalid target name, or no target in the acquired image)");
228  }
229 
230  // detected btn array
231  if (bell_detected_btn_arr_pub_->get_subscription_count() > 0) {
232  perception_interfaces::msg::DetectedButtonArray detected_btn_arr;
233  detected_btn_arr =
234  pose_estimator_->get_detected_button_array(target_btn_str_, target_btn_state);
235  detected_btn_arr.header.stamp = left_stamp;
236  detected_btn_arr.header.frame_id = frame_id_;
237  bell_detected_btn_arr_pub_->publish(detected_btn_arr);
238  } else {
239  RCLCPP_WARN(
240  this->get_logger(), "No detected button array subscriber!");
241  }
242  } else {
243  RCLCPP_WARN(
244  this->get_logger(), get_error_log(calc_state));
245  }
246  } else {
247  RCLCPP_WARN(this->get_logger(), "No loaded obj model");
248  }
249 
250  // visualizer_->overlay_stereo_images(
251  // cv_vconcat_image,
252  // pose_estimator_->get_pose_map(),
253  // target_btn_str_,
254  // target_btn_state,
255  // 1,
256  // 40,
257  // 40);
258 
259  this->run_streaming(
260  cv_vconcat_image,
261  left_stamp,
262  frame_id_,
263  zed_streaming_pub_msg_name_,
264  zed_streaming_image_ratio_);
265  }
266 
267  return true;
268 }

366 void BellDetector::divide_left_right_pts(
367  std::vector<RecognizedText> & all_points,
368  std::vector<RecognizedText> & left_points,
369  std::vector<RecognizedText> & right_points)
370 {
371  for (auto det_but : all_points) {
372  if (check_left_right_pts(det_but.features)) {
373  for (size_t j = 0; j < 5; ++j) {
374  det_but.features[j] -= offset_;
375  }
376  right_points.push_back(det_but);
377  } else {
378  left_points.push_back(det_but);
379  }
380  }

383 std::vector<RecognizedText> BellDetector::convert_to_recognized_text(
384  std::vector<ObjectDetectorOutputs> & outputs)
385 {
386  std::vector<RecognizedText> rec_output;
387  for (auto output : outputs) {
388  RecognizedText tmp_rec_output;
389  tmp_rec_output.class_name = bell_det_labels_[output.cls_num];
390  tmp_rec_output.button_state = false;
391  tmp_rec_output.features.push_back(
392  cv::Point2i(
393  static_cast<int>((output.x1 + output.x2) / 2),
394  static_cast<int>(output.y1))
395  );
396  tmp_rec_output.features.push_back(
397  cv::Point2i(
398  static_cast<int>(output.x1),
399  static_cast<int>((output.y1 + output.y2) / 2))
400  );
401  tmp_rec_output.features.push_back(
402  cv::Point2i(
403  static_cast<int>((output.x1 + output.x2) / 2),
404  static_cast<int>((output.y1 + output.y2) / 2))
405  );
406  tmp_rec_output.features.push_back(
407  cv::Point2i(
408  static_cast<int>(output.x2),
409  static_cast<int>((output.y1 + output.y2) / 2))
410  );
411  tmp_rec_output.features.push_back(
412  cv::Point2i(
413  static_cast<int>((output.x1 + output.x2) / 2),
414  static_cast<int>(output.y2))
415  );
416  rec_output.push_back(tmp_rec_output);
417  }
418  return rec_output;
419 }
