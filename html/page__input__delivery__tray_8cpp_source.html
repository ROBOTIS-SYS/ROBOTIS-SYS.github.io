227 void PageInputDeliveryTray::on_btnNext_clicked()
228 {
229  RCLCPP_INFO(ros_node_->get_logger(), "[InputDelivery] NEXT clicked");
230 
231  std::vector<QPushButton *> btn_list = {ui->btnDest1, ui->btnDest2, ui->btnDest3};
232  if (ros_node_->input_dest_page_name_.compare("SelectTarget") == 0) {
233  std::vector<aimbot_state_interfaces::msg::Destination> destination_list;
234  ros_node_->get_target_list(destination_list);
235  if (destination_list.size() == 0) {
236  return;
237  }
238  for (const auto & it : destination_list) {
239  for (size_t i = 0; i < btn_list.size(); i++) {
240  if (it.name.compare(btn_list[i]->text().toStdString()) == 0) {
241  ros_node_->cargo_goal_[i] = it.key;
242  }
243  }
244  }
245  } else {
246  for (size_t i = 0; i < btn_list.size(); i++) {
247  if (btn_list[i]->text().compare(DEFAULT_BTN_TEXT) != 0) {
248  ros_node_->cargo_goal_[i] = btn_list[i]->text().toStdString();
249  }
250  }
251  }
252 
253  std::string opt = "";
254  for (size_t i = 0; i < btn_list.size(); i++) {
255  opt +=
256  ((btn_list[i]->text().compare(DEFAULT_BTN_TEXT)) ? btn_list[i]->text().toStdString() : "") +
257  "|";
258  }
259  emit pageChanged("DeliveryStart", opt);
260 }
