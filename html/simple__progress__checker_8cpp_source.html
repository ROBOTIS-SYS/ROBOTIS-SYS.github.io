51 bool SimpleProgressChecker::check(geometry_msgs::msg::PoseStamped & current_pose)
52 {
53  // relies on short circuit evaluation to not call is_robot_moved_enough if
54  // baseline_pose is not set.
55  geometry_msgs::msg::Pose2D current_pose2d;
56  current_pose2d = nav_2d_utils::poseToPose2D(current_pose.pose);
57 
58  if ((!baseline_pose_set_) || (is_robot_moved_enough(current_pose2d))) {
59  reset_baseline_pose(current_pose2d);
60  RCLCPP_INFO_STREAM(logger_, "[simple progress checker] Robot is moving");
61  return true;
62  }
63  if ((nh_->now() - baseline_time_) > time_allowance_) {
64  RCLCPP_INFO_STREAM(logger_, "[simple progress checker] Robot is not moving");
65  return false;
66  }
67  return true;
68 }

82 bool SimpleProgressChecker::is_robot_moved_enough(const geometry_msgs::msg::Pose2D & pose)
83 {
84  if (pose_distance(pose, baseline_pose_) > radius_) {
85  return true;
86  } else {
87  return false;

91 static double pose_distance(
92  const geometry_msgs::msg::Pose2D & pose1,
93  const geometry_msgs::msg::Pose2D & pose2)
94 {
95  double dx = pose1.x - pose2.x;
96  double dy = pose1.y - pose2.y;
97 
98  return std::hypot(dx, dy);
99 }
