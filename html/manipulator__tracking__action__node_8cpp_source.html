48 BT::NodeStatus ManipulatorTrackingAction::on_tick()
49 {
50  // Set Ros action configure
51  std::string manipulator_namespace = "gc_manipulator";
52  std::string action_name = manipulator_namespace + "/manipulator_controller/tracking";
53 
54  // Create Action [.action]
55  // set Action Timeout = 10s
56  std::chrono::milliseconds action_timeout = std::chrono::seconds(10);
57  if (!create_action_client(action_name, action_timeout)) {
58  return BT::NodeStatus::FAILURE;
59  }
60 
61  // Log [.msg]
62  try {
63  bool on_blackboard = config().blackboard->get<decltype(log_pub_)>("/aimbot_log", log_pub_);
64  if (on_blackboard == false) {
65  log_pub_ = node_->create_publisher<aimbot_log_interfaces::msg::Log>("/aimbot_log", 30);
66  config().blackboard->set<decltype(log_pub_)>("/aimbot_log", log_pub_);
67  }
68  } catch (std::exception & ex) {
69  RCLCPP_ERROR_STREAM(
70  node_->get_logger(),
71  "Failed to create \"/aimbot_log\" action : " << ex.what());
72  return BT::NodeStatus::FAILURE;
73  }
74 
75  set_ros_spin_loop_rate(10);
76 
77  std::string target_msg_name;
78  if (!getInput<std::string>("target_msg_name", target_msg_name)) {
79  RCLCPP_ERROR_STREAM(node_->get_logger(), "Target msg name is empty!!");
80  return BT::NodeStatus::FAILURE;
81  }
82 
83  std::string end_effector_frame_id;
84  if (!getInput<std::string>("end_effector_frame_id", end_effector_frame_id)) {
85  RCLCPP_ERROR_STREAM(node_->get_logger(), "End effector frame id is empty!!");
86  return BT::NodeStatus::FAILURE;
87  }
88 
89  geometry_msgs::msg::Point goal_from_target;
90  if (!getInput<geometry_msgs::msg::Point>("goal_from_target", goal_from_target)) {
91  RCLCPP_ERROR_STREAM(node_->get_logger(), "Goal from target is empty!!");
92  return BT::NodeStatus::FAILURE;
93  }
94 
95  geometry_msgs::msg::Point target_offset;
96  if (!getInput<geometry_msgs::msg::Point>("target_offset", target_offset)) {
97  target_offset.set__x(0.0);
98  target_offset.set__y(0.0);
99  target_offset.set__z(0.0);
100  }
101 
102  double velocity;
103  if (!getInput<double>("velocity", velocity)) {
104  RCLCPP_ERROR_STREAM(node_->get_logger(), "velocity is empty!!");
105  return BT::NodeStatus::FAILURE;
106  }
107 
108  double acceleration;
109  if (!getInput<double>("acceleration", acceleration)) {
110  RCLCPP_ERROR_STREAM(node_->get_logger(), "acceleration is empty!!");
111  return BT::NodeStatus::FAILURE;
112  }
113 
114  double tip_link_length;
115  if (!getInput<double>("tip_link_length", tip_link_length)) {
116  RCLCPP_ERROR_STREAM(node_->get_logger(), "tip_link_length is empty!!");
117  return BT::NodeStatus::FAILURE;
118  }
119 
120  double tolerance_x;
121  if (!getInput<double>("tolerance_x", tolerance_x)) {
122  RCLCPP_ERROR_STREAM(node_->get_logger(), "tolerance_x is empty!!");
123  return BT::NodeStatus::FAILURE;
124  }
125 
126  double tolerance_y;
127  if (!getInput<double>("tolerance_y", tolerance_y)) {
128  RCLCPP_ERROR_STREAM(node_->get_logger(), "tolerance_y is empty!!");
129  return BT::NodeStatus::FAILURE;
130  }
131 
132  double tolerance_z;
133  if (!getInput<double>("tolerance_z", tolerance_z)) {
134  RCLCPP_ERROR_STREAM(node_->get_logger(), "tolerance_z is empty!!");
135  return BT::NodeStatus::FAILURE;
136  }
137 
138  double tolerance_roll;
139  if (!getInput<double>("tolerance_roll", tolerance_roll)) {

143 
144  double tolerance_pitch;

146  RCLCPP_ERROR_STREAM(node_->get_logger(), "tolerance_pitch is empty!!");
147  return BT::NodeStatus::FAILURE;
148  }
149 
150  double tolerance_yaw;
151  if (!getInput<double>("tolerance_yaw", tolerance_yaw)) {
152  RCLCPP_ERROR_STREAM(node_->get_logger(), "tolerance_yaw is empty!!");
153  return BT::NodeStatus::FAILURE;
154  }
155 
156  bool pose_empty = false;
157  geometry_msgs::msg::Pose expected_target_pose;
158  expected_target_pose.position.x = 0.0;
159  expected_target_pose.position.y = 0.0;
160  expected_target_pose.position.y = 0.0;
161  expected_target_pose.orientation.w = 1.0;
162  expected_target_pose.orientation.x = 0.0;
163  expected_target_pose.orientation.y = 0.0;
164  expected_target_pose.orientation.z = 0.0;
165  if (!getInput<geometry_msgs::msg::Pose>("expected_target_pose", expected_target_pose)) {
166  RCLCPP_ERROR_STREAM(node_->get_logger(), "expected_target_pose is empty!!");
167  pose_empty = true;
168  }
169 
170  bool frame_id_empty = false;
171  std::string expected_target_pose_frame_id;
172  if (!getInput<std::string>("expected_target_pose_frame_id", expected_target_pose_frame_id)) {
173  RCLCPP_ERROR_STREAM(node_->get_logger(), "expected_target_pose_frame_id is empty!!");
174  frame_id_empty = true;
175  }
176 
177  goal_.header.stamp = rclcpp::Clock().now();
178 
179  goal_.tracking_target_msg_name = target_msg_name;
180  goal_.end_effector_frame_id = end_effector_frame_id;
181 
182  goal_.goal_position_from_target = goal_from_target;
183  goal_.target_offset = target_offset;
184 
185  goal_.profile.velocity = velocity;
186  goal_.profile.acceleration = acceleration;
187  goal_.profile.tip_link_length = tip_link_length;
188 
189  if (frame_id_empty == false) { // frame id ok
190  if (pose_empty == false) { // pose ok
191  goal_.expected_target_pose.header.stamp = node_->now();
192  goal_.expected_target_pose.header.frame_id = expected_target_pose_frame_id;
193  goal_.expected_target_pose.pose = expected_target_pose;
194  } else { // no pose
195  goal_.expected_target_pose.header.stamp = rclcpp::Time(0);
196  goal_.expected_target_pose.header.frame_id.clear();
197  goal_.expected_target_pose.pose = expected_target_pose;
198  }
199  } else { // no frame id
200  if (pose_empty == false) { // pose ok
201  goal_.expected_target_pose.header.stamp = rclcpp::Time(0);
202  goal_.expected_target_pose.header.frame_id.clear();
203  goal_.expected_target_pose.pose = expected_target_pose;
204  } else { // no pose
205  goal_.expected_target_pose.header.stamp = rclcpp::Time(0);
206  goal_.expected_target_pose.header.frame_id.clear();
207  goal_.expected_target_pose.pose = expected_target_pose;
208  }
209  }
210 
211  goal_.tolerance.x = tolerance_x;
212  goal_.tolerance.y = tolerance_y;
213  goal_.tolerance.z = tolerance_z;
214  goal_.tolerance.roll = tolerance_roll;
215  goal_.tolerance.pitch = tolerance_pitch;
216  goal_.tolerance.yaw = tolerance_yaw;
217 
218  return BT::NodeStatus::RUNNING;

221 BT::NodeStatus ManipulatorTrackingAction::on_success()
222 {
223  auto value = result_.result->result_state.val;
224  if (value == manipulator_interfaces::msg::Result::SUCCESS) {
225  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
226  RCLCPP_INFO_STREAM(node_->get_logger(), "[Tracking] SUCCESS!!!");
227  }
228  publish_log("SUCCESS!!!");
229  return BT::NodeStatus::SUCCESS;
230  } else if (value == manipulator_interfaces::msg::Result::CANCELED) {
231  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
232  RCLCPP_INFO_STREAM(node_->get_logger(), "[Tracking] canceled manipulator action.");
233  }
234  publish_log("CANCELED!");
235  return BT::NodeStatus::SUCCESS;
236  } else if (value == manipulator_interfaces::msg::Result::PREEMPTED) {
237  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
238  RCLCPP_INFO_STREAM(
239  node_->get_logger(), "[Tracking] Accepted another manipulator action.");
240  }
241  publish_log(
242  "PREEMPTED : " + std::to_string(value),
243  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
244  aimbot_log_interfaces::msg::Log::TYPE_ERROR);
245  return BT::NodeStatus::SUCCESS;
246  } else {
247  RCLCPP_ERROR_STREAM(
248  node_->get_logger(), "[Tracking] FAIL :( ERROR : " << value);
249  publish_log(
250  "ERROR : " + std::to_string(value),
251  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
252  aimbot_log_interfaces::msg::Log::TYPE_ERROR);
253  return BT::NodeStatus::FAILURE;
254  }
