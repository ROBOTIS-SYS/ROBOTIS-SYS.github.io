71 int SpButtonDetector::load_network()
72 {
73  using ModelState = perception_interfaces::srv::GetModelState::Response;
74 
75  if (ocr_detection_.use_count() == 0) {
76  RCLCPP_INFO(
77  get_logger(), "OCR detection model loading...");
78  ocr_detection_ = std::make_shared<ml_model::OcrDetection>(sp_btn_det_model_name_);
79  int ocr_det_state = ocr_detection_->initialize();
80  if (ocr_det_state >= PERCEPTION_STATE::SUCCESS_NO_ISSUE) {
81  RCLCPP_INFO(
82  get_logger(), "OCR detection model successfully loaded");
83  } else if (ocr_det_state == PERCEPTION_STATE::ERROR_MODEL_READ) {
84  RCLCPP_ERROR(
85  this->get_logger(), "No OCR detection model exists");
86  return ModelState::MODEL_DOES_NOT_EXIST;
87  } else {
88  RCLCPP_ERROR(
89  this->get_logger(), "Failed to load OCR detection model * '%s'",
90  get_error_log(ocr_det_state));
91  return ModelState::MODEL_LOAD_FAILED;
92  }
93  } else {
94  RCLCPP_INFO(
95  get_logger(), "Already OCR detection model loaded");
96  }
97 
98  if (ocr_recognition_ == nullptr) {
99  RCLCPP_INFO(
100  get_logger(), "OCR recognition model loading...");
101  ocr_recognition_ = std::make_shared<ml_model::OcrRecognition>(
102  sp_btn_rec_model_name_, sp_rec_cls_labels_);
103  int ocr_rec_state = ocr_recognition_->initialize();
104  if (ocr_rec_state >= PERCEPTION_STATE::SUCCESS_NO_ISSUE) {
105  RCLCPP_INFO(
106  get_logger(), "OCR recognition model successfully loaded");
107  } else if (ocr_rec_state == PERCEPTION_STATE::ERROR_MODEL_READ) {
108  RCLCPP_ERROR(
109  this->get_logger(), "No OCR recognition model exists");
110  return ModelState::MODEL_DOES_NOT_EXIST;
111  } else {
112  RCLCPP_ERROR(
113  this->get_logger(), "Failed to load OCR recognition model * '%s'",
114  get_error_log(ocr_rec_state));
115  return ModelState::MODEL_LOAD_FAILED;
116  }
117  } else {
118  RCLCPP_INFO(
119  get_logger(), "Already OCR recognition model loaded");
120  }
121 
122  this->set_model_state(ModelState::LOAD);
123 
124  return ModelState::SUCCESS;
125 }

161 bool SpButtonDetector::update()
162 {
163  cv::Mat cv_left_image, cv_right_image;
164  rclcpp::Time left_stamp, right_stamp;
165  std::string dummy;
166  bool ack_left_img, ack_right_img;
167 
168  if (this->is_data_available(left_img_sub_msg_name_) &&
169  this->is_data_available(right_img_sub_msg_name_))
170  {
171  ack_left_img =
172  this->get_data_image(left_img_sub_msg_name_, cv_left_image, left_stamp, dummy);
173  ack_right_img =
174  this->get_data_image(right_img_sub_msg_name_, cv_right_image, right_stamp, dummy);
175  } else {
176  return false;
177  }
178 
179  if (ack_left_img && ack_right_img) {
180  camera_status_ = true;
181  cv_left_image = visualizer_->convert_rgba_to_rgb_img(cv_left_image);
182  cv_right_image = visualizer_->convert_rgba_to_rgb_img(cv_right_image);
183 
184  if (is_model_loaded() == true) {
185  // Segmentation
186  std::vector<cv::Mat> left_crop_imgs, right_crop_imgs;
187  std::vector<std::vector<std::vector<int>>> left_det_result_boxes, right_det_result_boxes;
188  std::vector<std::vector<std::vector<int>>> left_rec_result_boxes, right_rec_result_boxes;
189  ocr_detection_->model_run(cv_left_image, left_crop_imgs, left_det_result_boxes);
190  ocr_detection_->model_run(cv_right_image, right_crop_imgs, right_det_result_boxes);
191 
192  // Recognition
193  std::vector<std::string> left_rec_result_txt, right_rec_result_txt;
194  std::vector<RecognizedText> left_recognized_txt;
195  std::vector<RecognizedText> right_recognized_txt;
196 
197  ocr_recognition_->model_run(
198  left_crop_imgs,
199  left_rec_result_txt,
200  left_det_result_boxes,
201  left_rec_result_boxes,
202  left_recognized_txt);
203  ocr_recognition_->model_run(
204  right_crop_imgs,
205  right_rec_result_txt,
206  right_det_result_boxes,
207  right_rec_result_boxes,
208  right_recognized_txt);
209 
210  std::vector<std::string> target_list = {target_btn_str_};
211  ocr_recognition_->filter_text(
212  left_recognized_txt,
213  target_list,
214  sp_detected_filter_ratio_);
215  ocr_recognition_->filter_text(
216  right_recognized_txt,
217  target_list,
218  sp_detected_filter_ratio_);
219 
220  visualizer_->draw_ocr_infer_bbox(
221  cv_left_image,
222  left_recognized_txt);
223  visualizer_->draw_ocr_infer_bbox(
224  cv_right_image,
225  right_recognized_txt);
226 
227  // Pose calculation
228  sensor_msgs::msg::PointCloud2 dummy;
229  dummy.header.stamp = left_stamp;
230  pose_estimator_->set_input(
231  left_recognized_txt,
232  right_recognized_txt,
233  dummy);
234 
235  geometry_msgs::msg::PoseStamped target_btn_pose;
236  boundingbox_strategy_->set_pose_calc_location(sp_targets_[target_btn_str_]);
237  int calc_state = pose_estimator_->calculate();
238  if (calc_state > -1) {
239  if (pose_estimator_->get_pose(target_btn_str_, target_btn_pose)) {
240  target_btn_pose.header.stamp = left_stamp;
241  target_btn_pose.header.frame_id = frame_id_;
242 
243  Eigen::Quaterniond q(
244  target_btn_pose.pose.orientation.w,
245  target_btn_pose.pose.orientation.x,
246  target_btn_pose.pose.orientation.y,
247  target_btn_pose.pose.orientation.z);
248  q.normalize();
249  target_btn_pose.pose.orientation.w = q.w();
250  target_btn_pose.pose.orientation.x = q.x();
251  target_btn_pose.pose.orientation.y = q.y();
252  target_btn_pose.pose.orientation.z = q.z();
253 
254  RCLCPP_INFO_STREAM(
255  this->get_logger(),
256  "frame time stamp: " << std::fixed << left_stamp.seconds());
257  RCLCPP_INFO_STREAM(
258  this->get_logger(),
259  "target button name: " << target_btn_str_);
260  RCLCPP_INFO_STREAM(
261  this->get_logger(),
262  "target button position: " <<
263  target_btn_pose.pose.position.x << ", " <<
264  target_btn_pose.pose.position.y << ", " <<
265  target_btn_pose.pose.position.z);
266  RCLCPP_INFO_STREAM(
267  this->get_logger(),
268  "target button orientation: " <<
269  target_btn_pose.pose.orientation.w << ", " <<
270  target_btn_pose.pose.orientation.x << ", " <<
271  target_btn_pose.pose.orientation.y << ", " <<
272  target_btn_pose.pose.orientation.z);
273 
274  sp_detected_btn_goal_pose_pub_->publish(target_btn_pose);
275  } else {
276  RCLCPP_WARN(
277  this->get_logger(),
278  "Failed to get target pose");
279  RCLCPP_WARN_STREAM(
280  this->get_logger(),
281  "target name received from server: " << target_btn_str_);
282  }
283  } else {
284  RCLCPP_WARN(
285  this->get_logger(), get_error_log(calc_state));
286  }
287  }
288 
289  cv::Mat streaming_img =
290  visualizer_->convert_rgba_imgs_to_rgb_img(
291  cv_left_image,
292  cv_right_image);
293  this->run_streaming(
294  streaming_img,
295  left_stamp,
296  frame_id_,
297  zed_streaming_pub_msg_name_,
298  zed_streaming_image_ratio_);
299  }
300 
301  return true;
302 }

427 void SpButtonDetector::sp_detected_btn_server_callback(
428  const std::shared_ptr<perception_interfaces::srv::DetectedButton::Request> req,

436  target_btn_str_ = sp_ev_tag_target_name_;

441  return;
442  }
443  res->camera_status = camera_status_;
444  res->pose_msg_name = sp_detected_button_goal_pose_;
445  RCLCPP_INFO(
446  this->get_logger(), "SP button detected goal pose * '%s'",
447  target_btn_str_.c_str());
448 }
