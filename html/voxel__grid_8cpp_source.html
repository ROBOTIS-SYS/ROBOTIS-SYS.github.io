41 VoxelGrid::VoxelGrid(unsigned int size_x, unsigned int size_y, unsigned int size_z)
42 : logger(rclcpp::get_logger("voxel_grid"))
43 {
44  size_x_ = size_x;
45  size_y_ = size_y;
46  size_z_ = size_z;
47 
48  if (size_z_ > 16) {
49  RCLCPP_INFO(
50  logger, "Error, this implementation can only support up to 16 z values (%d)",
51  size_z_);
52  size_z_ = 16;
53  }
54 
55  data_ = new uint32_t[size_x_ * size_y_];
56  uint32_t unknown_col = ~((uint32_t)0) >> 16;
57  uint32_t * col = data_;
58  for (unsigned int i = 0; i < size_x_ * size_y_; ++i) {
59  *col = unknown_col;
60  ++col;
61  }
62 }

172 VoxelStatus VoxelGrid::getVoxel(unsigned int x, unsigned int y, unsigned int z)
173 {
174  if (x >= size_x_ || y >= size_y_ || z >= size_z_) {
175  RCLCPP_DEBUG(logger, "Error, voxel out of bounds. (%d, %d, %d)\n", x, y, z);
176  return UNKNOWN;
177  }
178  uint32_t full_mask = ((uint32_t)1 << z << 16) | (1 << z);
179  uint32_t result = data_[y * size_x_ + x] & full_mask;
180  unsigned int bits = numBits(result);
181 
182  // known marked: 11 = 2 bits, unknown: 01 = 1 bit, known free: 00 = 0 bits
183  if (bits < 2) {
184  if (bits < 1) {
185  return FREE;
186  }
187 
188  return UNKNOWN;
189  }
190 
191  return MARKED;
192 }

194 VoxelStatus VoxelGrid::getVoxelColumn(
195  unsigned int x, unsigned int y,
196  unsigned int unknown_threshold, unsigned int marked_threshold)
197 {
198  if (x >= size_x_ || y >= size_y_) {
199  RCLCPP_DEBUG(logger, "Error, voxel out of bounds. (%d, %d)\n", x, y);
200  return UNKNOWN;
201  }
202 
203  uint32_t * col = &data_[y * size_x_ + x];
204 
205  unsigned int unknown_bits = uint16_t(*col >> 16) ^ uint16_t(*col);
206  unsigned int marked_bits = *col >> 16;
207 
208  // check if the number of marked bits qualifies the col as marked
209  if (!bitsBelowThreshold(marked_bits, marked_threshold)) {
210  return MARKED;
211  }
212 
213  // check if the number of unkown bits qualifies the col as unknown
214  if (!bitsBelowThreshold(unknown_bits, unknown_threshold)) {
215  return UNKNOWN;
216  }
217 
218  return FREE;
219 }
