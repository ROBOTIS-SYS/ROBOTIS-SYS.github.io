136 controller_interface::return_type ManipulatorController::update()
137 {
138  auto start = rclcpp::Clock().now().seconds();
139  if (get_current_state().id() == lifecycle_msgs::msg::State::PRIMARY_STATE_INACTIVE) {
140  if (!is_halted_) {
141  halt();
142  is_halted_ = true;
143  }
144  pub_heartbeat("StandBy");
145  return controller_interface::return_type::OK;
146  }
147 
148  // set update time
149  robot_handler_->set_present_time(rclcpp::Clock().now().seconds());
150  // update Present robot state (state interface -> robot handler)
151  if (!get_present_values()) {
152  pub_heartbeat("Fail to get Present values.");
153  return controller_interface::return_type::ERROR;
154  }
155 
156  // update present sensor value
157  get_sensor_values();
158 
159  // Checker control methods
160  for (uint i = 0; i < control_modules_list_.size(); i++) {
161  if (!control_modules_list_.at(i).module->checker()) {
162  robot_handler_->update_cancel(rm2_robot_handler::CancelResult::CHECKER);
163  }
164  }
165 
166  // update control method
167  robot_handler_->update();
168  // update goal value
169  set_goal_values();
170 
171  // check dxl error
172  for (uint64_t index = 0; index < robot_handler_->get_joint_names().size(); index++) {
173  double err = joint_hardware_state_interface_.at(index).get().get_value();
174  if (err != 0.0) {
175  // RCLCPP_ERROR_STREAM(node_->get_logger(), "hardware error " << index << " " << err);
176  robot_handler_->update_cancel(rm2_robot_handler::CancelResult::HARDWARE_ERROR);
177  }
178  }
179 
180  auto end = rclcpp::Clock().now().seconds();
181  if (end - start > 0.01) {
182  RCLCPP_WARN_STREAM(node_->get_logger(), "update_time : " << end - start);
183  }
184  pub_heartbeat("Activated");
185  return controller_interface::return_type::OK;
186 }

190 rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn ManipulatorController::
191 on_configure(const rclcpp_lifecycle::State & previous_state)
192 {
193  // get robot file name
194  std::string urdf_file;
195  urdf_file = node_->get_parameter("urdf_file_path").as_string();
196  robot_handler_->load_robot_model(urdf_file);
197 
198  bool debug_mode = node_->get_parameter("debug_mode").as_bool();
199  robot_handler_->set_debug_mode(debug_mode);
200 
201  // set joint deque size
202  int64_t deque_size = node_->get_parameter(JointDequeSizeTag).as_int();
203  robot_handler_->set_joint_deque_size(static_cast<size_t>(deque_size));
204 
205  bool time_based_control = node_->get_parameter(TimeBasedControlTag).as_bool();
206  period_ = static_cast<float>(node_->get_parameter(PeriodSTag).as_double());
207 
208  std::string heartbeat_name = node_->get_parameter(HeartbeatNameTag).as_string();
209  heartbeat_timeout_ = static_cast<float>(node_->get_parameter(HeartbeatTimeoutSTag).as_double());
210  heartbeat_pub_ = node_->create_publisher<aimbot_state_interfaces::msg::Heartbeat>(
211  "/heartbeat/" + heartbeat_name, 0);
212 
213  robot_handler_->set_time_based_control(time_based_control, static_cast<double>(period_));
214 
215  /* FK Solver*/
216  // get fk solver name(type)
217  FKSolverModulePtr fk_solver;
218  std::string fk_solver_type =
219  node_->get_parameter(std::string(FKSolverTag) + "." + std::string(TypeTag)).as_string();
220  RCLCPP_INFO_STREAM(node_->get_logger(), "FK Solver Module : " << fk_solver_type);
221  // load fk solver
222  if (!class_loader_fk_solver_->isClassAvailable(fk_solver_type)) {
223  const std::string error_msg("Loader for FK solver '" + fk_solver_type + "' not found.");
224  RCLCPP_ERROR_STREAM(node_->get_logger(), error_msg.c_str());
225  RCLCPP_ERROR_STREAM(node_->get_logger(), "available classes:");
226  for (const auto & c : class_loader_fk_solver_->getDeclaredClasses()) {
227  RCLCPP_ERROR_STREAM(node_->get_logger(), c.c_str());
228  }
229  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
230  }
231  try {
232  fk_solver = class_loader_fk_solver_->createSharedInstance(fk_solver_type);
233 
234  // set fk solver to handler
235  if (!robot_handler_->set_fk_solver(fk_solver)) {
236  const std::string error_msg("Can not initialize FK solver.");
237  RCLCPP_ERROR_STREAM(node_->get_logger(), error_msg.c_str());
238  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
239  }
240 
241  // init fk solver
242  if (!fk_solver->initialize()) {
243  RCLCPP_ERROR_STREAM(
244  node_->get_logger(),
245  "Fail to initialize forward kinematics solver.");
246  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
247  }
248  } catch (pluginlib::LibraryLoadException & e) {
249  RCLCPP_ERROR_STREAM(
250  node_->get_logger(),
251  "Fail to load forward kinematics solver : " << e.what());
252  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
253  }
254 
255  // set fk solver params
256  std::map<std::string, std::string> fk_params;
257  for (auto param : fk_solver->get_param_list()) {
258  // Declare fk params
259  auto_declare<std::string>(
260  std::string(FKSolverTag) + "." + std::string(ParamsTag) + "." + param, "");
261  // insert params
262  std::string data = node_->get_parameter(
263  std::string(FKSolverTag) + "." + std::string(ParamsTag) + "." + param).as_string();
264  fk_params.insert(std::make_pair(param, data));
265  // Undeclare fk params
266  // to prepare when the on_configure function be called again
267  node_->undeclare_parameter(
268  std::string(FKSolverTag) + "." + std::string(ParamsTag) + "." + param);
269  RCLCPP_INFO_STREAM(node_->get_logger(), " " << param << " : " << data);
270  }
271  if (!fk_solver->set_params(fk_params)) {
272  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to set params forward kinematics solver.");
273  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
274  }
275  std::cout << std::endl;
276 
277  /* IK Solver */
278  // get ik solver name(type)
279  IKSolverModulePtr ik_solver;
280  std::string ik_solver_type =
281  node_->get_parameter(std::string(IKSolverTag) + "." + std::string(TypeTag)).as_string();
282  RCLCPP_INFO_STREAM(node_->get_logger(), "IK Solver Module : " << ik_solver_type);
283  // load ik solver
284  if (!class_loader_ik_solver_->isClassAvailable(ik_solver_type)) {
285  const std::string error_msg("Loader for IK solver '" + ik_solver_type + "' not found.");
286  RCLCPP_ERROR_STREAM(node_->get_logger(), error_msg.c_str());
287  RCLCPP_ERROR_STREAM(node_->get_logger(), "available classes:");
288  for (const auto & c : class_loader_ik_solver_->getDeclaredClasses()) {
289  RCLCPP_ERROR_STREAM(node_->get_logger(), c.c_str());
290  }
291  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
292  }
293  try {
294  ik_solver = class_loader_ik_solver_->createSharedInstance(ik_solver_type);
295 
296  // set ik solver to handler
297  if (!robot_handler_->set_ik_solver(ik_solver)) {
298  const std::string error_msg("Can not initialize IK solver '" + ik_solver_type + "'.");
299  RCLCPP_ERROR_STREAM(node_->get_logger(), error_msg.c_str());
300  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
301  }
302 
303  // init ik solver
304  if (!ik_solver->initialize()) {
305  RCLCPP_INFO_STREAM(
306  node_->get_logger(),
307  "Fail to initialize forward kinematics solver.");
308  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
309  }
310  } catch (pluginlib::LibraryLoadException & e) {
311  RCLCPP_ERROR_STREAM(
312  node_->get_logger(),
313  "Fail to load forward kinematics solver : " << e.what());
314  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
315  }
316 
317  // set ik solver params
318  std::map<std::string, std::string> ik_params;
319  for (auto param : ik_solver->get_param_list()) {
320  // Declare ik params
321  auto_declare<std::string>(
322  std::string(IKSolverTag) + "." + std::string(ParamsTag) + "." + param, "");
323  // insert params
324  std::string data = node_->get_parameter(
325  std::string(IKSolverTag) + "." + std::string(ParamsTag) + "." + param).as_string();
326  ik_params.insert(std::make_pair(param, data));
327  // Undeclare ik params
328  // to prepare when the on_configure function be called again
329  node_->undeclare_parameter(
330  std::string(IKSolverTag) + "." + std::string(ParamsTag) + "." + param);
331  RCLCPP_INFO_STREAM(node_->get_logger(), " " << param << " : " << data);
332  }
333  if (!ik_solver->set_params(ik_params)) {
334  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to set params inverse kinematics solver.");
335  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
336  }
337  std::cout << std::endl;
338 
339  /* Jacobian Solver */
340  // get jacobian solver name(type)
341  JacSolverModulePtr jacobian_solver;
342  auto jacobian_solver_type =
343  node_->get_parameter(std::string(JacSolverTag) + "." + std::string(TypeTag)).as_string();
344  RCLCPP_INFO_STREAM(node_->get_logger(), "Jacobian Solver Module : " << jacobian_solver_type);
345  // load jacobian solver
346  if (!class_loader_jacobian_solver_->isClassAvailable(jacobian_solver_type)) {
347  const std::string error_msg(
348  "Loader for Jacobian solver '" + jacobian_solver_type + "' not found.");
349  RCLCPP_ERROR_STREAM(node_->get_logger(), error_msg.c_str());
350  RCLCPP_ERROR_STREAM(node_->get_logger(), "available classes:");
351  for (const auto & c : class_loader_jacobian_solver_->getDeclaredClasses()) {
352  RCLCPP_ERROR_STREAM(node_->get_logger(), c.c_str());
353  }
354  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
355  }
356  try {
357  jacobian_solver = class_loader_jacobian_solver_->createSharedInstance(
358  jacobian_solver_type);
359  // init jacobian solver
360  if (!jacobian_solver->initialize()) {
361  RCLCPP_ERROR_STREAM(
362  node_->get_logger(),
363  "Fail to initialize forward kinematics solver.");
364  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
365  }
366  } catch (pluginlib::LibraryLoadException & e) {
367  RCLCPP_ERROR_STREAM(
368  node_->get_logger(),
369  "Fail to load forward kinematics solver : " << e.what());
370  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
371  }
372 
373  // set jacobian solver params
374  std::map<std::string, std::string> jac_params;
375  for (auto param : jacobian_solver->get_param_list()) {
376  // Declare jacobian params
377  auto_declare<std::string>(
378  std::string(JacSolverTag) + "." + std::string(ParamsTag) + "." + param, "");
379  // insert params
380  std::string data = node_->get_parameter(
381  std::string(JacSolverTag) + "." + std::string(ParamsTag) + "." + param).as_string();
382  jac_params.insert(std::make_pair(param, data));
383  // Undeclare jacobian params
384  // to prepare when the on_configure function be called again
385  node_->undeclare_parameter(
386  std::string(JacSolverTag) + "." + std::string(ParamsTag) + "." + param);
387  RCLCPP_INFO_STREAM(node_->get_logger(), " " << param << " : " << data);
388  }
389  if (!jacobian_solver->set_params(jac_params)) {
390  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to set params inverse kinematics solver.");
391  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
392  }
393  // set fk solver to handler
394  if (!robot_handler_->set_jacobian_solver(jacobian_solver)) {
395  const std::string error_msg(
396  "Can not initialize Jacobian solver '" + jacobian_solver_type + "'.");
397  RCLCPP_ERROR_STREAM(node_->get_logger(), error_msg.c_str());
398  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
399  }
400  std::cout << std::endl;
401 
402  /* Sensor List */
403  sensor_list_.clear();
404  // get sensor list
405  std::vector<std::string> sensor_list_vec = node_->get_parameter(
406  std::string(SensorListTag) + "." + std::string(NameTag)).as_string_array();
407 
408  RCLCPP_INFO_STREAM(node_->get_logger(), "Sensor List:");
409  for (auto name : sensor_list_vec) {
410  // Declare types of the sensor list
411  auto_declare<std::vector<std::string>>(
412  std::string(SensorListTag) + "." + name + "." + std::string(ValueNameTag),
413  std::vector<std::string>());
414  // Get value_name
415  std::vector<std::string> value_name_vec = node_->get_parameter(
416  std::string(SensorListTag) + "." + name + "." + std::string(ValueNameTag)).as_string_array();
417 
418  for (auto str : value_name_vec) {
419  RCLCPP_INFO_STREAM(node_->get_logger(), "[" << name << "] : " << str);
420  sensor_list_.push_back(std::make_pair(name, str));
421  robot_handler_->add_sensor(name + "/" + str);
422  }
423  std::cout << std::endl;
424  }
425 
426  /* Control Modules */
427  if (!control_modules_list_.empty()) {
428  control_modules_list_.clear();
429  }
430  int id = 0;
431 
432  // get control module names(types)
433  std::vector<std::string> control_module_names = node_->get_parameter(
434  std::string(ControlModuleTag) + "." + std::string(NameTag))
435  .as_string_array();
436 
437  RCLCPP_INFO_STREAM(node_->get_logger(), "Contol Modules:");
438  for (auto name : control_module_names) {
439  id++;
440  // Declare types of the control module
441  auto_declare<std::string>(
442  std::string(ControlModuleTag) + "." + name + "." + std::string(TypeTag), "");
443  // Get type
444  std::string type = node_->get_parameter(
445  std::string(ControlModuleTag) + "." + name + "." + std::string(TypeTag)).as_string();
446  // Undeclare types of the control module
447  // to prepare when the on_configure function be called again
448  node_->undeclare_parameter(
449  std::string(ControlModuleTag) + "." + name + "." +
450  std::string(TypeTag));
451  RCLCPP_INFO_STREAM(node_->get_logger(), "[" << id << "] : " << name << "(" << type << ")");
452  if (type == "") {
453  RCLCPP_ERROR_STREAM(
454  node_->get_logger(), "Can not find type of '" << name << "'. plaese check ros params.");
455  id--;
456  continue;
457  }
458 
459  // check if the contol module can be loaded
460  if (!class_loader_control_modules_->isClassAvailable(type)) {
461  const std::string error_msg("Loader for control module '" + name + "' not found.");
462  RCLCPP_ERROR_STREAM(node_->get_logger(), error_msg.c_str());
463  id--;
464  continue;
465 // return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
466  }
467 
468  // load contol module
469  try {
470  RCLCPP_INFO_STREAM(node_->get_logger(), "Load " << name);
471  auto control_module =
472  class_loader_control_modules_->createSharedInstance(type);
473  RCLCPP_INFO_STREAM(node_->get_logger(), "assign node and robot handler to " << name);
474  control_module->assign(name, id, node_, robot_handler_);
475  RCLCPP_INFO_STREAM(node_->get_logger(), "configure " << name);
476  if (!control_module->configure()) {
477  RCLCPP_ERROR_STREAM(
478  node_->get_logger(),
479  "Can not initialize configure control module '" << name << "'.");
480  id--;
481 // continue;
482  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
483  }
484  control_modules_list_.emplace_back(name, type, control_module);
485  } catch (pluginlib::LibraryLoadException & e) {
486  RCLCPP_ERROR_STREAM(
487  node_->get_logger(), "Can not load control module '" << name << "' : \n" << e.what());
488 // continue;
489  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::FAILURE;
490  }
491  std::cout << std::endl;
492  }
493 
494  return rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn::SUCCESS;
495 }

498 // Fill ordered_interfaces with references to the matching interfaces
499 // in the same order as in joint_names
500 template<typename T>
501 bool get_ordered_interfaces(
502  std::vector<T> & unordered_interfaces, const std::vector<std::string> & joint_names,
503  const std::string & interface_type, std::vector<std::reference_wrapper<T>> & ordered_interfaces)
504 {
505  for (const auto & joint_name : joint_names) {
506  for (auto & interface : unordered_interfaces) {
507  if ((interface.get_name() == joint_name) &&
508  (interface.get_interface_name() == interface_type))
509  {
510  ordered_interfaces.push_back(std::ref(interface));
511  }
512  }
513  }
514  return joint_names.size() == ordered_interfaces.size();
515 }

517 rclcpp_lifecycle::node_interfaces::LifecycleNodeInterface::CallbackReturn ManipulatorController::
518 on_activate(const rclcpp_lifecycle::State & previous_state)
519 {
520  // sorting interfaces
