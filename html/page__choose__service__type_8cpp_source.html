346 void PageChooseServiceType::set_available_service_type(std::string services_to_use)
347 {
348  std::vector<std::string> service_types;
349  std::istringstream iss(services_to_use);
350  std::string token;
351  while (std::getline(iss, token, '|')) {
352  service_types.push_back(token);
353  }
354 
355  // RCLCPP_INFO(ros_node_->get_logger(), "Available Service : ");
356 
357  ui->btnServiceDocent->setEnabled(false);
358  ui->btnServiceDelevery->setEnabled(false);
359  ui->btnServiceFloorInfo->setEnabled(false);
360  ui->btnServiceWebCall->setEnabled(false);
361 
362  ui->btnServiceDocent->setVisible(false);
363  ui->btnServiceDelevery->setVisible(false);
364  ui->btnServiceFloorInfo->setVisible(false);
365  ui->btnServiceWebCall->setVisible(false);
366 
367  ui->lblLeftImg->setVisible(false);
368  ui->lblRightImg->setVisible(false);
369 
370  ui->lblTextServiceLeft->setVisible(false);
371  ui->lblTextServiceRight->setVisible(false);
372 
373 
374  for (const std::string & service_type : service_types) {
375  // RCLCPP_INFO(ros_node_->get_logger(), service_type.c_str());
376 
377  if (service_type.compare("Delivery") == 0) {
378  set_btn_position(ui->btnServiceDelevery, set_btn_count_);
379  set_service_type_title(tr("배송 서비스"), set_btn_count_);
380  set_service_type_image(":/images_600p/service_type_delivery.png", set_btn_count_);
381 
382  ui->btnServiceDelevery->raise();
383  ui->btnServiceDelevery->setEnabled(true);
384  ui->btnServiceDelevery->setVisible(true);
385  } else if (service_type.compare("Docent") == 0) {
386  set_btn_position(ui->btnServiceDocent, set_btn_count_);
387  set_service_type_title(tr("전시 안내"), set_btn_count_);
388  set_service_type_image(":/images_600p/service_type_docent.png", set_btn_count_);
389 
390  ui->btnServiceDocent->raise();
391  ui->btnServiceDocent->setEnabled(true);
392  ui->btnServiceDocent->setVisible(true);
393  } else if (service_type.compare("FloorInfo") == 0) {
394  set_btn_position(ui->btnServiceFloorInfo, set_btn_count_);
395  set_service_type_title(tr("현재층 안내"), set_btn_count_);
396  set_service_type_image(":/images_600p/service_type_floor_info.png", set_btn_count_);
397 
398  ui->btnServiceFloorInfo->raise();
399  ui->btnServiceFloorInfo->setEnabled(true);
400  ui->btnServiceFloorInfo->setVisible(true);
401  } else if (service_type.compare("WebCall") == 0) {
402  set_btn_position(ui->btnServiceWebCall, set_btn_count_);
403  set_service_type_title(tr("안내직원 연결"), set_btn_count_);
404  set_service_type_image(":/images_600p/service_type_webcall.png", set_btn_count_);
405 
406  ui->btnServiceWebCall->raise();
407  ui->btnServiceWebCall->setEnabled(true);
408  ui->btnServiceWebCall->setVisible(true);
409  } else if (service_type.find("Patrol") != std::string::npos) {
410  set_btn_position(ui->btnServicePatrol, set_btn_count_);
411  set_service_type_title(tr("패트롤"), set_btn_count_);
