23 geometry_msgs::msg::Polygon LowPassFilter::get_filtering_data(
24  geometry_msgs::msg::Polygon raw_data,
25  double sampling_time)
26 {
27  if (first_input_) {
28  previous_filtering_data_ = raw_data;
29  first_input_ = false;
30  } else {
31  for (size_t i = 0; i < previous_filtering_data_.points.size(); i++) {
32  previous_filtering_data_.points.at(i).x =
33  ((static_cast<float>(tau_) * previous_filtering_data_.points.at(i).x) +
34  (static_cast<float>(sampling_time) * raw_data.points.at(i).x)) /
35  static_cast<float>(tau_ + sampling_time);
36  previous_filtering_data_.points.at(i).y =
37  ((static_cast<float>(tau_) * previous_filtering_data_.points.at(i).y) +
38  (static_cast<float>(sampling_time) * raw_data.points.at(i).y)) /
39  static_cast<float>(tau_ + sampling_time);
40  previous_filtering_data_.points.at(i).z =
41  ((static_cast<float>(tau_) * previous_filtering_data_.points.at(i).z) +
42  (static_cast<float>(sampling_time) * raw_data.points.at(i).z)) /
43  static_cast<float>(tau_ + sampling_time);
44  }
45  }
46  return previous_filtering_data_;
47 }

60 BT::NodeStatus ManipulatorKnockAction::on_tick()
61 {
62  std::string manipulator_namespace = "gc_manipulator";
63  std::string action_name = manipulator_namespace + "/manipulator_controller/knock";
64  std::chrono::milliseconds action_timeout = std::chrono::milliseconds(10000); // 10s
65 
66  if (!create_action_client(action_name, action_timeout)) {
67  return BT::NodeStatus::FAILURE;
68  }
69 
70  int knock_cnt;
71  if (!getInput<int>("knock_cnt", knock_cnt)) {
72  knock_cnt = 3;
73  }
74  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
75  RCLCPP_INFO_STREAM(
76  node_->get_logger(), "ManipulatorKnockAction: Set knock cnt : " << knock_cnt);
77  }
78 
79  double tau;
80  if (getInput<double>("lpf_tau", tau)) {
81  filter_.set_tau(tau);
82  }
83  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
84  RCLCPP_INFO_STREAM(
85  node_->get_logger(), "ManipulatorKnockAction: Set tau : " << filter_.get_tau());
86  }
87  filter_.set_first_input_flag(true);
88 
89  if (!getInput<int>("door_open_count", door_open_count_)) {
90  door_open_count_ = 3;
91  }
92  open_cnt_ = 0;
93 
94  double knock_position_rate;
95  if (!getInput<double>("knock_position_rate", knock_position_rate)) {
96  knock_position_rate = 0.8;
97  }
98  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
99  RCLCPP_INFO_STREAM(
100  node_->get_logger(),
101  "ManipulatorKnockAction: Set knock position rate : " << knock_position_rate);
102  }
103 
104  geometry_msgs::msg::Pose2D knock_direction;
105  if (!getInput<geometry_msgs::msg::Pose2D>("knock_direction", knock_direction)) {
106  knock_direction.x = 0.03;
107  knock_direction.y = -0.01;
108  knock_direction.theta = 10.0;
109  }
110 
111  // [msg subscriber]
112  std::string msg_name = "detected_door";
113  if (detected_door_sub_ == nullptr) {
114  detected_door_sub_ = node_->create_subscription<perception_interfaces::msg::DetectedDoor>(
115  msg_name, 0,
116  std::bind(&ManipulatorKnockAction::detected_door_callback, this, std::placeholders::_1));
117  }
118 
119  // [msg publisher]
120  // Log [msg]
121  try {
122  bool on_blackboard = config().blackboard->get<decltype(log_pub_)>("/aimbot_log", log_pub_);
123  if (on_blackboard == false) {
124  log_pub_ = node_->create_publisher<aimbot_log_interfaces::msg::Log>("/aimbot_log", 10);
125  config().blackboard->set<decltype(log_pub_)>("/aimbot_log", log_pub_);
126  }
127  } catch (std::exception & ex) {
128  RCLCPP_ERROR_STREAM(
129  node_->get_logger(),
130  "Failed to create \"/aimbot_log\" action : " << ex.what());
131  return BT::NodeStatus::FAILURE;
132  }
133 
134  msg_name = "detected_door_polygon_filtered";
135  detected_door_polygon_pub_ =
136  node_->create_publisher<geometry_msgs::msg::PolygonStamped>(msg_name, 10);
137 
138  msg_name = "filtered_door_polygon_filtered";
139  filtered_door_polygon_pub_ =
140  node_->create_publisher<geometry_msgs::msg::PolygonStamped>(msg_name, 10);
141 
142  msg_recive_ = false;
143 
144  // wait detected door data sub.
145  auto check_start_time = node_->get_clock()->now();
146  rclcpp::Duration timeout_dur(std::chrono::seconds(5));
147  rclcpp::WallRate loop_wall_rate(std::chrono::milliseconds(10));
148  RCLCPP_WARN_STREAM(node_->get_logger(), "Waiting for detected door data");
149  while (rclcpp::ok() && msg_recive_ == false) {
150  rclcpp::spin_some(node_);
151 
152  auto dur = node_->get_clock()->now() - check_start_time;
153  if (dur > timeout_dur) {
154  RCLCPP_ERROR_STREAM(node_->get_logger(), "timeout to subscribe msg");
155  return BT::NodeStatus::FAILURE;
156  }
157  // sleep
158  loop_wall_rate.sleep();
159  }
160 
161  set_ros_spin_loop_rate(10);
162 
163  goal_.header = recv_data_.header;
164  goal_.door_position = recv_data_.door_polygon;
165  goal_.knock_num = knock_cnt;
166  goal_.knock_position_rate = knock_position_rate;
167  goal_.knock_direction = knock_direction;
168  goal_.knock_direction.theta *= (M_PI / 180.0);
169 
170  if (RVIZ_POLYGON) {
171  geometry_msgs::msg::PolygonStamped door_polygon;
172  door_polygon.header = recv_data_.header;
173  door_polygon.polygon = goal_.door_position;
174  detected_door_polygon_pub_->publish(door_polygon);
175  }
176 
177  return BT::NodeStatus::RUNNING;
178 }

180 BT::NodeStatus ManipulatorKnockAction::on_wait_for_result()
181 {
182  using DetectedDoor = perception_interfaces::msg::DetectedDoor;
183  rclcpp::WallRate loop_rate(std::chrono::milliseconds(static_cast<int>(10)));
184  while (status() == BT::NodeStatus::RUNNING) {
185  // sleep
186  loop_rate.sleep();
187  // RCLCPP_INFO_STREAM(get_logger(), "Check is_door_open");
188  if (recv_data_.door_status == DetectedDoor::CLOSED) {
189  open_cnt_ = 0;
190  } else if (recv_data_.door_status == DetectedDoor::OPENED) {
191  open_cnt_++;
192  RCLCPP_WARN_STREAM(node_->get_logger(), "Door is OPENED!! : " + std::to_string(open_cnt_));
193  publish_log("Door is OPENED!! : " + std::to_string(open_cnt_));
194  return BT::NodeStatus::FAILURE;
195  } else if (recv_data_.door_status == DetectedDoor::NONE) {
196  open_cnt_++;
197  RCLCPP_WARN_STREAM(node_->get_logger(), "Door is NONE!! : " + std::to_string(open_cnt_));
198  publish_log("Door is NONE!! : " + std::to_string(open_cnt_));
199  return BT::NodeStatus::FAILURE;
200  } else if (recv_data_.door_status == DetectedDoor::UNKNOWN) {
201  open_cnt_++;
202  RCLCPP_WARN_STREAM(node_->get_logger(), "Door is UNKNOWN!! : " + std::to_string(open_cnt_));
203  publish_log("Door is UNKNOWN!! : " + std::to_string(open_cnt_));
204  return BT::NodeStatus::FAILURE;
205  } else if (recv_data_.door_status == DetectedDoor::OUT_OF_RANGE) {
206  open_cnt_++;
207  RCLCPP_WARN_STREAM(
208  node_->get_logger(), "Door is OUT_OF_RANGE!! : " + std::to_string(open_cnt_));
209  publish_log("Door is OUT_OF_RANGE!! : " + std::to_string(open_cnt_));
210  return BT::NodeStatus::FAILURE;
211  } else if (recv_data_.door_status == DetectedDoor::LESS_THAN_MIN_WIDTH) {
212  open_cnt_++;
213  RCLCPP_WARN_STREAM(
214  node_->get_logger(), "Door is LESS_THAN_MIN_WIDTH!! : " + std::to_string(open_cnt_));
215  publish_log("Door is LESS_THAN_MIN_WIDTH!! : " + std::to_string(open_cnt_));
216  return BT::NodeStatus::FAILURE;
217  }
218 
219  // Count
220  if (open_cnt_ > door_open_count_) {
221  RCLCPP_WARN_STREAM(node_->get_logger(), "OVER OPEN COUNT");
222  publish_log("OVER OPEN COUNT");
223  return BT::NodeStatus::FAILURE;
224  }
225  }
226  return status();
227 }

229 BT::NodeStatus ManipulatorKnockAction::on_success()
230 {
231  auto value = result_.result->result_state.val;
232  if (value == manipulator_interfaces::msg::Result::SUCCESS) {
233  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
234  RCLCPP_INFO_STREAM(node_->get_logger(), "[Knock] SUCCESS!!!");
235  }
236  publish_log("SUCCESS!!!");
237  return BT::NodeStatus::SUCCESS;
238  } else if (value == manipulator_interfaces::msg::Result::CANCELED) {
239  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
240  RCLCPP_INFO_STREAM(node_->get_logger(), "[Knock] canceled manipulator action.");
241  }
242  publish_log("CANCELED!");
243  return BT::NodeStatus::SUCCESS;
244  } else if (value == manipulator_interfaces::msg::Result::PREEMPTED) {
245  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
246  RCLCPP_INFO_STREAM(
247  node_->get_logger(), "[Knock] Accepted another manipulator action.");
248  }
249  publish_log(
250  "PREEMPTED : " + std::to_string(value),
251  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
252  aimbot_log_interfaces::msg::Log::TYPE_ERROR);
253  return BT::NodeStatus::SUCCESS;
254  } else {
255  RCLCPP_ERROR_STREAM(
256  node_->get_logger(), "[Knock] FAIL :( ERROR : " << value);
257  publish_log(
258  "ERROR : " + std::to_string(value),
259  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
260  aimbot_log_interfaces::msg::Log::TYPE_ERROR);
261  return BT::NodeStatus::FAILURE;
262  }

275 void ManipulatorKnockAction::detected_door_callback(
276  const perception_interfaces::msg::DetectedDoor::SharedPtr msg)
277 {
278  if (status() != BT::NodeStatus::RUNNING) {
279  return;
280  }
281 
282  if ((prev_detected_time_ + rclcpp::Duration(1, 0)).seconds() <= node_->now().seconds())
283  {filter_.set_first_input_flag(true);}
284 
285  prev_detected_time_ = node_->now();
286 
287  recv_data_ = *msg;
288  if (msg->door_status == perception_interfaces::msg::DetectedDoor::CLOSED) {
289  recv_data_.door_polygon = filter_.get_filtering_data(msg->door_polygon);
290  }
291 
292  // RCLCPP_INFO_STREAM(get_logger(), "Detected Door Sub");
293  if (RVIZ_POLYGON) {
294  geometry_msgs::msg::PolygonStamped door_polygon;
295  door_polygon.header = recv_data_.header;
296  door_polygon.polygon = recv_data_.door_polygon;
297  filtered_door_polygon_pub_->publish(door_polygon);
298  }
299  msg_recive_ = true;
300 }
