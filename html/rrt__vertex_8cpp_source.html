29 RRTVertex::RRTVertex(int xy_min, int xy_max)
30 {
31  std::random_device rd;
32  std::mt19937 gen(rd());
33  std::uniform_int_distribution<int> dist(xy_min, xy_max); // mm
34 // std::uniform_real_distribution<double> dist_angle(-3.14, 3.14);
35 
36  pose_.x = static_cast<double>(dist(gen)) * 0.001; // mm -> m
37  pose_.y = static_cast<double>(dist(gen)) * 0.001;
38 // pose_.theta = dist_angle(gen);
39 }

41 RRTVertex::RRTVertex(const geometry_msgs::msg::Pose2D normal, double standard_deviation)
42 {
43  std::random_device rd;
44  std::mt19937 gen(rd());
45  std::normal_distribution<> n_dist_x{normal.x, standard_deviation};
46  std::normal_distribution<> n_dist_y{normal.y, standard_deviation};
47 
48  pose_.x = static_cast<int>(n_dist_x(gen) / RESOLUTION) * RESOLUTION;
49  pose_.y = static_cast<int>(n_dist_y(gen) / RESOLUTION) * RESOLUTION;
50 }
