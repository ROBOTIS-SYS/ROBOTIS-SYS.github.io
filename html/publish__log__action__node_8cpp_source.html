77 bool PublishLogAction::publish_log(
78  const std::string level, const std::string type,
79  const std::string source, const std::string message,
80  const std::string target_name, const int pgid,
81  const bool capture)
82 {
83  aimbot_log_interfaces::msg::Log msg;
84  msg.header.stamp = node_->now();
85 
86  if (level == "high") {
87  msg.level = msg.LEVEL_HIGH;
88  } else if (level == "medium") {
89  msg.level = msg.LEVEL_MEDIUM;
90  } else if (level == "low") {
91  msg.level = msg.LEVEL_LOW;
92  } else if (level == "debug") {
93  msg.level = msg.LEVEL_DEBUG;
94  } else {
95  msg.level = msg.LEVEL_DEBUG;
96  }
97 
98  if (type == "error") {
99  msg.type = msg.TYPE_ERROR;
100  } else if (type == "warn") {
101  msg.type = msg.TYPE_WARN;
102  } else if (type == "info") {
103  msg.type = msg.TYPE_INFO;
104  } else {
105  msg.type = msg.TYPE_INFO;
106  }
107 
108  msg.source = source;
109  msg.contents = message;
110 
111  msg.target_poi = target_name;
112  if (pgid > 0) {
113  msg.target_poi = std::to_string(pgid);
114  }
115  msg.capture = capture;
116 
117  log_pub_->publish(msg);
118  RCLCPP_WARN_STREAM(get_logger(), "Publish log");
119  return true;
120 }
