175 void DialogSelectTarget::on_btnArea_clicked()
176 {
177  if (is_area_selected == false && destination_list.size() > 0) {
178  is_area_selected = true;
179  set_btn_style(ui->btnArea, BTN_CHECKED);
180  set_btn_style(ui->btnTarget, BTN_NOT_CHECKED);
181 
182  ui->btnArea->setChecked(true);
183  ui->btnTarget->setChecked(false);
184  ui->listArea->setVisible(true);
185  ui->listTarget->setVisible(false);
186 
187  QList<std::string> items;
188  for (const auto & it : destination_list) {
189  if (it.type != "DESTINATION") {
190  continue;
191  }
192 
193  bool already_added = false;
194  for (int i = 0; i < items.size(); ++i) {
195  if (items.at(i) == it.floor.c_str()) {
196  already_added = true;
197  break;
198  }
199  }
200  if (already_added == false) {
201  items.append(it.floor);
202  }
203  }
204 
205  std::sort(
206  items.begin(), items.end(), [](const std::string & a, const std::string & b) {
207  if (std::isdigit(a[0]) && !std::isdigit(b[0])) {
208  return false;
209  } else if (!std::isdigit(a[0]) && std::isdigit(b[0])) {
210  return true;
211  }
212 
213  if (!std::isdigit(a[0]) && !std::isdigit(b[0])) {
214  return std::stoi(a.substr(1)) > std::stoi(b.substr(1));
215  } else {
216  return std::stoi(a) < std::stoi(b);
217  }
218  });
219 
220  ui->listArea->clear();
221  for (int i = 0; i < items.size(); ++i) {
222  QListWidgetItem * newItem = new QListWidgetItem(items.at(i).c_str());
223  newItem->setTextAlignment(Qt::AlignCenter);
224  ui->listArea->addItem(newItem);
225  }
226 
227  QListWidgetItem * newItem = new QListWidgetItem(tr("전체"));
228  newItem->setTextAlignment(Qt::AlignCenter);
229  ui->listArea->insertItem(0, newItem);
230 
231  if (ui->btnArea->text().compare("-") == 0) {
232  ui->listArea->blockSignals(true);
233  ui->listArea->setCurrentRow(0);
234  ui->listArea->blockSignals(false);
235 
236  area_current_row_changed();
237  } else {
238  for (int i = 0; i < ui->listArea->count(); i++) {
239  if (ui->listArea->item(i)->text() == ui->btnArea->text()) {
240  ui->listArea->blockSignals(true);
241  ui->listArea->setCurrentItem(ui->listArea->item(i));
242  ui->listArea->blockSignals(false);
243 
244  area_current_row_changed();
245  break;
246  }
247  }
248  }
249  }
250 }

252 void DialogSelectTarget::on_btnTarget_clicked()
253 {
254  if (is_area_selected == true && destination_list.size() > 0) {
255  is_area_selected = false;
256  set_btn_style(ui->btnArea, BTN_NOT_CHECKED);
257  set_btn_style(ui->btnTarget, BTN_CHECKED);
258 
259  ui->btnArea->setChecked(false);
260  ui->btnTarget->setChecked(true);
261  ui->listArea->setVisible(false);
262  ui->listTarget->setVisible(true);
263 
264  ui->listTarget->clear();
265  for (const auto & it : destination_list) {
266  if (ui->btnArea->text().compare(tr("전체")) == 0 ||
267  ui->btnArea->text().compare(it.floor.c_str()) == 0)
268  {
269  if (it.type.compare("DESTINATION") == 0) {
270  QListWidgetItem * newItem = new QListWidgetItem(it.name.c_str());
271  newItem->setTextAlignment(Qt::AlignCenter);
272  ui->listTarget->addItem(newItem);
273  }
274  }
275  }
276 
277  ui->listTarget->sortItems(Qt::AscendingOrder);
278 
279  if (ui->btnTarget->text().compare("-") != 0) {
280  for (int i = 0; i < ui->listTarget->count(); i++) {
281  if (ui->listTarget->item(i)->text() == ui->btnTarget->text()) {
282  ui->listTarget->setCurrentItem(ui->listTarget->item(i));
283  break;
284  }
285  }
286  }
287  }
