75 int EvIndicatorDetector::load_network()
76 {
77  nvtxRangeId_t nvtx_load = nvtxRangeStartA("Load Network");
78  using ModelState = perception_interfaces::srv::GetModelState::Response;
79 
80  if (obj_detection_.use_count() == 0) {
81  RCLCPP_INFO(
82  get_logger(), "Object detection model loading...");
83  obj_detection_ = std::make_shared<ml_model::ObjectDetection>(dnn_model_name_);
84  int obj_det_state = obj_detection_->initialize();
85  if (obj_det_state >= PERCEPTION_STATE::SUCCESS_NO_ISSUE) {
86  RCLCPP_INFO(
87  get_logger(), "Object detection model successfully loaded");
88  } else if (obj_det_state == PERCEPTION_STATE::ERROR_MODEL_READ) {
89  RCLCPP_ERROR(
90  this->get_logger(), "No object detection model exists");
91  return ModelState::MODEL_DOES_NOT_EXIST;
92  } else {
93  RCLCPP_ERROR(
94  this->get_logger(), "Failed to load object detection model",
95  get_error_log(obj_det_state));
96  return ModelState::MODEL_LOAD_FAILED;
97  }
98  } else {
99  RCLCPP_INFO(
100  get_logger(), "Already object detection model loaded");
101  }
102 
103  this->set_model_state(ModelState::LOAD);
104  nvtxRangeEnd(nvtx_load);
105  return ModelState::SUCCESS;
106 }

484 perception_interfaces::msg::EVStatus EvIndicatorDetector::merge_label(
485  const std::vector<ObjectDetectorOutputs> & infer_outputs)
486 {
487  std::vector<DoorStatus> door_sts;
488  std::vector<IndicatorStatus> indi_sts;
489  std::vector<HallLantern> hall_sts;
490  DoorStatus best_door_sts;
491  IndicatorStatus best_indicator_sts;
492 
493  EVStatusMsg result_msg;
494  result_msg.ev_direction = 0;
495  result_msg.ev_door_status = 0;
496  result_msg.ev_floor = "";
497  result_msg.current_floor = "";
498  result_msg.in_ev = true;
499 
500  // find upper label bbox (Door, Indicator, Hall_LED)
501  find_upper_class_bbox(
502  infer_outputs,
503  door_sts,
504  indi_sts,
505  hall_sts);
506  update_pre_upper_label(indi_sts, hall_sts);
507 
508  // find lower label bbox (Sticker, Floor, Direction, Hall_LED_on, Hall_LED_off)
509  find_low_rank_bbox(
510  infer_outputs,
511  indi_sts,
512  hall_sts);
513 
514  // compare left, right door_sts value, exclude front_door_sts
515  best_door_sts = compare_door_status(door_sts);
516  best_indicator_sts = compare_indicator_status(indi_sts);
517 
518  // compare left, right, front indicator_sts
519  if (best_indicator_sts.direction.first == "up") {
520  result_msg.ev_direction = DIRECTION_STATE::UP;
521  } else if (best_indicator_sts.direction.first == "down") {
522  result_msg.ev_direction = DIRECTION_STATE::DOWN;
523  } else if (best_indicator_sts.direction.first == "None") {
524  result_msg.ev_direction = DIRECTION_STATE::NONE;
525  }
526  result_msg.current_floor = "None";
527  result_msg.ev_floor = best_indicator_sts.total_floor.first;
528  result_msg.ev_door_status = best_door_sts.status;
529  return result_msg;
530 }

549 void EvIndicatorDetector::update_direction(
550  std::vector<IndicatorStatus> & indi_sts,
551  const ObjectDetectorOutputs infer_result)
552 {
553  std::string label = dnn_model_lbs_[infer_result.cls_num];
554  cv::Point2i c_pts = cv::Point2i(
555  static_cast<int>(infer_result.x1 * 0.5 + infer_result.x2 * 0.5),
556  static_cast<int>(infer_result.y1 * 0.5 + infer_result.y2 * 0.5));
557 
558  for (size_t i = 0; i < indi_sts.size(); ++i) {
559  if (indi_sts[i].direction.first == "None") {
560  if (indi_sts[i].area_roi.contains(c_pts)) {
561  indi_sts[i].direction.first = label;
562  indi_sts[i].direction.second = infer_result.conf;
563  }
564  } else {
565  if (infer_result.conf > indi_sts[i].direction.second) {
566  indi_sts[i].direction.first = label;
567  indi_sts[i].direction.second = infer_result.conf;
568  }
569  }
570  }

573 void EvIndicatorDetector::update_floor(
574  std::vector<IndicatorStatus> & indi_sts,
575  const ObjectDetectorOutputs infer_result)
576 {
577  std::string label = dnn_model_lbs_[infer_result.cls_num];
578  cv::Point2i c_pts = cv::Point2i(
579  static_cast<int>(infer_result.x1 * 0.5 + infer_result.x2 * 0.5),
580  static_cast<int>(infer_result.y1 * 0.5 + infer_result.y2 * 0.5));
581 
582  for (size_t i = 0; i < indi_sts.size(); ++i) {
583  // 10자리 숫자 중 처음부터 채우기
584  if (indi_sts[i].digit1_floor.first == "None") {
585  if (indi_sts[i].area_roi.contains(c_pts)) {
586  indi_sts[i].digit1_floor.first = label;
587  indi_sts[i].digit1_floor.second = infer_result.conf;
588  indi_sts[i].digit1_floor_center_point = c_pts;

803 void EvIndicatorDetector::create_combine_params(
804  std::vector<PreUpperLabel> & pre_up_labels,
805  cv::Size ori_image_size,
806  cv::Size & comb_image_size,
807  float rescale_ratio,
808  float margin_ratio)
809 {
810  comb_image_size = cv::Size(
811  ori_image_size.width * rescale_ratio,
812  ori_image_size.height * rescale_ratio);
813  int side_start_x = ori_image_size.width;
814  int side_start_y = 0;
815  int bottom_start_x = 0;
816  int bottom_start_y = ori_image_size.height;
817  int total_w = 0;
818  int total_h = 0;
819  int add_width = static_cast<int>(ori_image_size.width * (rescale_ratio - 1));
820  int add_height = static_cast<int>(ori_image_size.height * (rescale_ratio - 1));
821 
822  for (auto & pre_up_label : pre_up_labels) {
823  // Ensure ROI is included in the original image
824  if ((pre_up_label.pre_roi.x + pre_up_label.pre_roi.width * 0.5) < ori_image_size.width &&
825  (pre_up_label.pre_roi.y + pre_up_label.pre_roi.height * 0.5) < ori_image_size.height &&
826  pre_up_label.pre_roi.width > 0 &&
827  pre_up_label.pre_roi.height > 0)
828  {
829  // Create margin_roi
830  pre_up_label.margin_roi = create_margin_roi(
831  pre_up_label.pre_roi,
832  ori_image_size,
833  margin_ratio);
834  }
835  }
836  for (auto & pre_up_label : pre_up_labels) {
837  if (!pre_up_label.direction) {
838  pre_up_label.rescale_ratio =
839  static_cast<float>(add_width) / static_cast<float>(pre_up_label.margin_roi.width);
840  pre_up_label.large_roi = cv::Rect(
841  side_start_x,
842  side_start_y,
843  static_cast<int>(pre_up_label.margin_roi.width * pre_up_label.rescale_ratio) - 1,
844  static_cast<int>(pre_up_label.margin_roi.height * pre_up_label.rescale_ratio) - 1);
845  total_h += pre_up_label.large_roi.height;
846  side_start_y += pre_up_label.large_roi.height;
847  } else {
848  pre_up_label.rescale_ratio =
849  static_cast<float>(add_height) / static_cast<float>(pre_up_label.margin_roi.height);
850  pre_up_label.large_roi = cv::Rect(
851  bottom_start_x,
852  bottom_start_y,
853  static_cast<int>(pre_up_label.margin_roi.width * pre_up_label.rescale_ratio) - 1,
854  static_cast<int>(pre_up_label.margin_roi.height * pre_up_label.rescale_ratio) - 1);
855  total_w += pre_up_label.large_roi.width;
856  bottom_start_x += pre_up_label.large_roi.width;
857  }
858  }
859  // Recalculate if the sum of the portrait images is greater
860  // than the height of the background image
861  if (total_h >= comb_image_size.height) {
862  side_start_y = 0;
863  for (auto & pre_up_label : pre_up_labels) {
864  if (!pre_up_label.direction) {
865  pre_up_label.rescale_ratio *=
866  (static_cast<float>(comb_image_size.height) /
867  static_cast<float>(total_h * 1.05));

869  side_start_x,
870  side_start_y,
871  static_cast<int>(pre_up_label.margin_roi.width * pre_up_label.rescale_ratio) - 2,
872  static_cast<int>(pre_up_label.margin_roi.height * pre_up_label.rescale_ratio) - 2);
873  side_start_y += pre_up_label.large_roi.height;
874  }
875  }
876  }
877  // Recalculate if the sum of horizontal images is greater than the width of the background image
878  if (total_w >= comb_image_size.width) {
879  bottom_start_x = 0;
880  for (auto & pre_up_label : pre_up_labels) {
881  if (pre_up_label.direction) {
882  pre_up_label.rescale_ratio *=
883  (static_cast<float>(comb_image_size.width) /
884  static_cast<float>(total_w * 1.05));
885  pre_up_label.large_roi = cv::Rect(
886  bottom_start_x,
887  bottom_start_y,
888  static_cast<int>(pre_up_label.margin_roi.width * pre_up_label.rescale_ratio - 2),
889  static_cast<int>(pre_up_label.margin_roi.height * pre_up_label.rescale_ratio - 2));
890  bottom_start_x += pre_up_label.large_roi.width;
891  }
892  }
893  }

1271  Eigen::Vector3f trans = geo->get_cam_ext_mat(cam_name).block<3, 1>(0, 3);

1278  trans_pts.push_back(trans_pt);

1280

1282 
1283  std::vector<cv::Point2i> comp_pts;
1284  // 절두체 내에 경계면 꼭지점이 있는지 확인
1285  for (int curr_id = 0; curr_id < frustum.other_planes.size(); curr_id++) {

1288  cv::Point2i cv_proj_pt;
1289  cv_proj_pt.x = proj_pt(0);
1290  cv_proj_pt.y = proj_pt(1);

1294

1296  for (int curr_id = 0; curr_id < frustum.other_planes.size(); curr_id++) {
1297  int next_id = (curr_id + 1) % frustum.other_planes.size();
1298 
1299  for (auto plane : frustum.other_planes) {
1300  Eigen::Vector3f intersected_pt;
1301  float t;
1302  bool intersected =
1303  Util::calculate_intersected_point_between_line_and_plane(
1304  trans_pts[curr_id], trans_pts[next_id], plane.second, plane.first, t, intersected_pt);
1305  bool inside = is_frustum_inside(intersected_pt, frustum.other_planes);
1306

1308  Eigen::Vector2f proj_pt = geo->project_to_image(proj_mat, intersected_pt);
1309  cv::Point2i cv_proj_pt;
1310  cv_proj_pt.x = proj_pt(0);
1311  cv_proj_pt.y = proj_pt(1);
1312  comp_pts.push_back(cv_proj_pt);
1313  }
1314  }
1315  }
1316 
1317  // 경계면 구하기
1318  Eigen::Vector3f boundary_plane_cen(0, 0, 0);
1319  for (auto pt : trans_pts) {
1320  std::cout << pt.transpose() << std::endl;
1321  boundary_plane_cen += (pt * 1 / trans_pts.size());
1322  }
1323 
1324  Eigen::Vector3f boundary_plane_x = (trans_pts[1] - trans_pts[0]).normalized();
1325  Eigen::Vector3f boundary_plane_y = (trans_pts[3] - trans_pts[0]).normalized();
1326  Eigen::Vector3f boundary_plane_nor = boundary_plane_x.cross(boundary_plane_y).normalized();
1327 
1328  Eigen::Vector3f n1 = boundary_plane_y * -1.0;
1329  Eigen::Vector3f c1 = (trans_pts[0] + trans_pts[1]) / 2;
1330  Eigen::Vector3f n2 = boundary_plane_x;
1331  Eigen::Vector3f c2 = (trans_pts[1] + trans_pts[2]) / 2;
1332  Eigen::Vector3f n3 = boundary_plane_y;
1333  Eigen::Vector3f c3 = (trans_pts[2] + trans_pts[3]) / 2;
1334  Eigen::Vector3f n4 = boundary_plane_x * -1.0;
1335  Eigen::Vector3f c4 = (trans_pts[3] + trans_pts[0]) / 2;
1336 
1337  std::vector<std::pair<Eigen::Vector3f, Eigen::Vector3f>> boundary;
1338  boundary.push_back(std::make_pair(c1, n1));
1339  boundary.push_back(std::make_pair(c2, n2));
1340  boundary.push_back(std::make_pair(c3, n3));
1341  boundary.push_back(std::make_pair(c4, n4));
1342

1344  for (auto edge : frustum.edges) {

1352  Util::calculate_intersected_point_between_line_and_plane(
1353  line_pt1, line_pt2, boundary_plane_nor, boundary_plane_cen, t, intersected_pt);
1354

1356  bool inside = is_frustum_inside(intersected_pt, boundary);
1357

1359  Eigen::Vector2f proj_pt = geo->project_to_image(proj_mat, intersected_pt);
1360  cv::Point2i cv_proj_pt;
1361  cv_proj_pt.x = proj_pt(0);
1362  cv_proj_pt.y = proj_pt(1);
1363  comp_pts.push_back(cv_proj_pt);
