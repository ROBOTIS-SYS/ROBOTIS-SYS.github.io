65 BT::NodeStatus WaitEVResultAction::tick()
66 {
67  std::cout << std::endl;
68  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
69  RCLCPP_INFO_STREAM(get_logger(), "<TICK> WaitEVResultAction");
70  }
71 
72 // _halt_requested.store(false);
73  has_ev_result_ = false;
74 
75  if (!getInput<int>("request_id", request_id_)) {
76  RCLCPP_ERROR(get_logger(), "There is no request id");
77  return BT::NodeStatus::FAILURE;
78  }
79 
80  int duration;
81  if (!getInput<int>("duration", duration)) {
82  duration = 60;
83  }
84 
85  std::stringstream ss;
86  ss << "Wait EV Status for " << duration << "s , req_id : " << request_id_;
87 // RCLCPP_INFO_STREAM(
88 // get_logger(),
89 // "[WaitEVResultAction] Wait EV Status for " << duration << "s , req_id : " << request_id_);
90 
91  // s -> ms
92  double check_time = static_cast<double>(duration) * 1000.0;
93  double loop_rate = 100.0; // ms
94  int loop_count = 0;
95 
96  rclcpp::WallRate loopRate(std::chrono::milliseconds(100));
97 
98  publish_tree_status(name(), ss.str());
99 
100 // setStatus(BT::NodeStatus::RUNNING);
101 
102  while (rclcpp::ok() && has_ev_result_ == false && isHaltRequested() == false) {
103  // sleep
104  loopRate.sleep();
105 
106  if ((loop_rate * ++loop_count) >= check_time) {
107  break;
108  }
109  }
110 
111  if (isHaltRequested()) {
112  publish_tree_status(name(), "엘리베이터 상태 체크 노드를 정지했습니다.");
113  return BT::NodeStatus::IDLE;
114  }
115 
116  if (has_ev_result_ == false) {
117  publish_tree_status(name(), "엘리베이터 상태 정보를 얻어오지 못했습니다.");
118  }
119 
120  return has_ev_result_ ? BT::NodeStatus::SUCCESS : BT::NodeStatus::FAILURE;
121 }

128 void WaitEVResultAction::ev_result_callback(
129  const elevator_interfaces::msg::ResultCheckElevatorStatus::SharedPtr msg)
130 {
131  if (status() != BT::NodeStatus::RUNNING) {
132  return;
133  }
134 
135  if (has_ev_result_ == true) {
136  return;
137  }
138 
139  // check request id
140  if (request_id_ != msg->req_id) {
141  return;
142  }
143 
144  std::lock_guard<std::mutex> lock(mutex_);
145 
146  // get result from msg
147  int arrived_ev_num = msg->ev_num;
148  bool ev_result = msg->result;
149 
150  // set result to blackboard
151  setOutput<int>("ev_num", arrived_ev_num);
152  setOutput<bool>("result", ev_result);
153 
154  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
155  RCLCPP_INFO_STREAM(
156  get_logger(),
157  "[WaitEVResultAction] Ev Status Result - ev_num : " << arrived_ev_num << ", result : " <<
158  (ev_result ? "True" : "False") <<
159  ", req_id : " << request_id_);
160  }
161 
162  std::stringstream ss;
163  ss << "Ev Status 결과 - ev_num : " << arrived_ev_num <<
164  ", result : " << (ev_result ? "True" : "False");
165  publish_tree_status(name(), ss.str());
166 
167  has_ev_result_ = true;
168 }
