149

163  RCLCPP_INFO_STREAM(this->get_logger(), "executable_sensor_node : " << str);

175  for (std::string detector_str : exe_detector_node) {
176  RCLCPP_INFO_STREAM(this->get_logger(), param_name << " : " << detector_str);
177  }
178  }
179 
180  // streaming info
181  for (std::string str : exe_sensor_node) {
182  // channel name
183  std::string param_name = "streaming_info." + str + ".channel_name";
184  this->declare_parameter<std::string>(param_name, std::string());
185  node_param_map_.at(str).streaming_channel = this->get_parameter(param_name).as_string();
186  RCLCPP_INFO_STREAM(

193  RCLCPP_INFO_STREAM(
194  this->get_logger(), param_name << " : " << node_param_map_.at(str).ros_sub_msg_name);

285 void PerceptionManager::add_detector(DetectorBasePtr detector_ptr)
286 {
287  std::string name = detector_ptr->get_name();
288  for (auto & iter : node_param_map_) {
289  for (size_t i = 0; i < iter.second.detector_node_name_vec.size(); i++) {
290  if (iter.second.detector_node_name_vec.at(i) == name) {
291  iter.second.detector_node_ptr_vec.at(i) = detector_ptr;
292  iter.second.detector_node_ptr_vec.at(i)->on_deactivate();
293  iter.second.detector_node_ptr_vec.at(i)->set_node_state(false);
294  RCLCPP_INFO_STREAM(
295  this->get_logger(),
296  "add detector [" << iter.second.sensor_node_name << "] : " <<
297  iter.second.detector_node_ptr_vec.at(i)->get_name());
298  return;
299  }
300  }

304 void PerceptionManager::set_node_state_srv_callback(
305  const std::shared_ptr<perception_interfaces::srv::SetNodeState::Request> req,
306  std::shared_ptr<perception_interfaces::srv::SetNodeState::Response> res)
307 {
308  using REQUEST = perception_interfaces::srv::SetNodeState::Request;
309 
310  int result = -1;
311  std::string detector_name = req->detector_name;
312 
313  // get sensor_name
314  std::string sensor_name;
315  for (auto & iter : node_param_map_) {
316  for (auto iter_detec : iter.second.detector_node_ptr_vec) {
317  if (iter_detec->get_name() == detector_name) {
318  sensor_name = iter.second.sensor_node_name;
319  }
320  }
321  }
322 
323  if (node_param_map_.find(sensor_name) != node_param_map_.end()) {
324  for (auto iter_detec : node_param_map_[sensor_name].detector_node_ptr_vec) {
325  if (iter_detec->get_name() == detector_name) {
326  if (req->node_state == REQUEST::NODE_RUNNING) {
327  result = iter_detec->on_activate();
328  RCLCPP_INFO_STREAM(get_logger(), iter_detec->get_name() << " -> on activate");
329  if (result == 1) {
330  iter_detec->set_node_state(true);
331  iter_detec->set_streaming_state(true);
332 
333  node_param_map_[sensor_name].sensor_node_ptr->set_streaming_state(false);
334 
335  if (node_param_map_[sensor_name].grab_control) {
336  node_param_map_[sensor_name].sensor_node_ptr->on_activate();
337  }
338 
339  RCLCPP_INFO_STREAM(
340  get_logger(),
341  iter_detec->get_name() << " node is activated");
342  }
343  } else if (req->node_state == REQUEST::NODE_IDLE) {
344  result = iter_detec->on_deactivate();
345  if (result == 1) {
346  iter_detec->set_node_state(false);
347  iter_detec->set_streaming_state(false);
348 
349  // check all detectors are deactivated
350  bool detector_flag = false;
351  for (auto iter_detec2 : node_param_map_[sensor_name].detector_node_ptr_vec) {
352  if (iter_detec2->get_node_state() == true) { // running
353  detector_flag = true;
354  }
355  }
356  if (detector_flag == false) { // all detectors are deactivated
357  RCLCPP_INFO_STREAM(
358  this->get_logger(),
359  "All Detector nodes are deactivated. change the streaming channel");
360  node_param_map_[sensor_name].sensor_node_ptr->set_streaming_state(true);
361  }
362 
363  if (node_param_map_[sensor_name].grab_control) {
364  node_param_map_[sensor_name].sensor_node_ptr->on_deactivate();
365  }
366  }
367  }
368  } else {
369  iter_detec->on_deactivate();
370  iter_detec->set_node_state(false);
371  RCLCPP_INFO_STREAM(get_logger(), iter_detec->get_name() << " -> on deactivate");
372  iter_detec->set_streaming_state(false);
373  }
374  }
375  }
376 
377  using RESPONSE = perception_interfaces::srv::SetNodeState::Response;
378  switch (result) {
379  case 1:
380  res->result = RESPONSE::SUCCESS;
381  break;
382  case 0:
383  res->result = RESPONSE::DETECTOR_ACTIVATION_FAILED;
384  break;
385  case -1:
386  res->result = RESPONSE::DETECTOR_DOES_NOT_EXIST;
387  break;
388  default:
389  res->result = RESPONSE::DETECTOR_DOES_NOT_EXIST;
390  break;
391  }
392 }

394 void PerceptionManager::get_node_state_srv_callback(
395  const std::shared_ptr<perception_interfaces::srv::GetNodeState::Request> req,
396  std::shared_ptr<perception_interfaces::srv::GetNodeState::Response> res)

409 
410  using RESPONSE = perception_interfaces::srv::GetNodeState::Response;
411  switch (result) {

413  res->result = RESPONSE::SUCCESS;
414  break;
415  case -1:
416  res->result = RESPONSE::DETECTOR_DOES_NOT_EXIST;
417  break;
418  default:
419  res->result = RESPONSE::DETECTOR_DOES_NOT_EXIST;
420  break;
421  }

424 void PerceptionManager::set_model_state_srv_callback(
425  const std::shared_ptr<perception_interfaces::srv::SetModelState::Request> req,
426  std::shared_ptr<perception_interfaces::srv::SetModelState::Response> res)
427 {
428  using REQUEST = perception_interfaces::srv::SetModelState::Request;
429  using RESPONSE = perception_interfaces::srv::SetModelState::Response;
430 
431  int result = RESPONSE::DETECTOR_DOES_NOT_EXIST;
432  std::vector<std::string> detector_names = req->detector_name;
433  int load_state = req->load_state;
434 
435  for (auto & detector_name : detector_names) {
436  for (auto & iter : node_param_map_) {
437  for (auto iter_detec : iter.second.detector_node_ptr_vec) {
438  if (iter_detec->get_name() == detector_name) {
439  if (load_state == REQUEST::UNLOAD) {
440  result = iter_detec->unload_network();
441  RCLCPP_INFO_STREAM(get_logger(), iter_detec->get_name() << " -> unload network");
442  } else if (load_state == REQUEST::LOAD) {
443  result = iter_detec->load_network();
444  RCLCPP_INFO_STREAM(get_logger(), iter_detec->get_name() << " -> load network");
445  }
446  }
447  }
448  }
449  }
450 
451 
452  switch (result) {
453  case 1:
454  res->result = RESPONSE::SUCCESS;
455  break;
456  case -1:
457  res->result = RESPONSE::DETECTOR_DOES_NOT_EXIST;
458  break;
459  case -2:
460  res->result = RESPONSE::MODEL_DOES_NOT_EXIST;
461  break;
462  case -3:
463  res->result = RESPONSE::MODEL_LOAD_FAILED;
464  break;
465  case -4:
466  res->result = RESPONSE::MODEL_UNLOAD_FAILED;
467  break;
468  default:
469  res->result = RESPONSE::UNKNOWN_ERROR;
470  break;
471  }
472 }

515 void PerceptionManager::register_streaming()
516 {
517  // set streaming
518  RCLCPP_INFO_STREAM(this->get_logger(), "-- set streaming pipe --");
519 
520  for (auto & iter : node_param_map_) {
521  auto sub_opt_sub_cbgroup = rclcpp::SubscriptionOptions();
522  sub_opt_sub_cbgroup.callback_group =
523  create_callback_group(rclcpp::CallbackGroupType::MutuallyExclusive);
524  iter.second.writer = oasis::pipeWriterOpen(iter.second.streaming_channel.c_str());
525  RCLCPP_INFO_STREAM(this->get_logger(), "pipeWriterOpen : " << iter.second.streaming_channel);
526 
527  iter.second.st_sub = this->create_subscription<sensor_msgs::msg::Image>(
528  iter.second.ros_sub_msg_name, perception_tools::default_qos,
529  [iter, this](sensor_msgs::msg::Image::ConstSharedPtr msg) {
530  sensor_msgs::msg::Image img = *msg;
531 
532  cv::Mat pipe_img = perception_tools::ros_to_cvmat(img);
533  if (pipe_img.type() != CV_8UC4) {
534  cv::cvtColor(pipe_img, pipe_img, cv::COLOR_BGR2BGRA);
535  }
536 
537  if (pipe_img.size().width == 0 || pipe_img.size().height == 0) {
538  RCLCPP_ERROR_STREAM(
539  this->get_logger(),
540  "NO IMAGE DATA : " << iter.second.ros_sub_msg_name);
541  return;
542  }
543 
544  ssize_t wbytes = oasis::pipeWriterWrite(
545  iter.second.writer,
546  pipe_img.data,
547  pipe_img.total() * pipe_img.elemSize(),
548  pipe_img.size().width,
549  pipe_img.size().height,
550  pipe_img.size().width * pipe_img.channels(),
551  oasis::kOasisPipeDataTypeVideoBGRA,
552  0);
553 
554  if (enable_streaming_win_) {
555  cv::imshow(iter.second.streaming_channel, pipe_img);
556  cv::waitKey(1);
557  }
558  }, sub_opt_sub_cbgroup);
559  RCLCPP_INFO_STREAM(
560  this->get_logger(),
561  "create_subscription : " << iter.second.ros_sub_msg_name);
562  }
563 
564  RCLCPP_INFO_STREAM(this->get_logger(), "-- set streaming state--");
565 
566  for (auto & iter : node_param_map_) {
567  iter.second.sensor_node_ptr->set_streaming_state(true);
568  for (auto iter_detec : iter.second.detector_node_ptr_vec) {
569  iter_detec->set_streaming_state(false);
570  }
571  }
