162 bool PortHandler::setup_port(/*int cflag_baud*/)
163 {
164  static const int CONNECT_RETRY_MAX_CNT = 10;
165  int retry_cnt = 0;
166 
167  stop_connect_ = false;
168 
169  serv_addr_.sin_family = AF_INET;
170  serv_addr_.sin_addr.s_addr = inet_addr(/*ip_addr_.c_str()*/ port_name_);
171  serv_addr_.sin_port = htons(port_);
172 
173  while (retry_cnt < CONNECT_RETRY_MAX_CNT && rclcpp::ok()) {
174  if (stop_connect_) {
175  break;
176  }
177 
178  socket_fd_ = socket(AF_INET, SOCK_STREAM, 0);
179  if (socket_fd_ < 0) {
180  std::cout << RED << "socket() error" << COLOR_END << std::endl;
181  retry_cnt++;
182  usleep(1000000);
183  continue;
184  }
185 
186 
187  /*For Non blocking socket creation*/
188 
189  int flag = fcntl(socket_fd_, F_GETFL, 0);
190  flag |= O_NONBLOCK;
191  fcntl(socket_fd_, F_SETFL, flag);
192 
193 
194  // KEEP Alive
195  int optval = 1; // KEEPALIVE ON
196  if (setsockopt(socket_fd_, SOL_SOCKET, SO_KEEPALIVE, &optval, sizeof(optval)) < 0) {
197  std::cout << RED << "ERROR conn() : setsockopt(SO_KEEPALIVE) failed" << COLOR_END <<
198  std::endl;
199  retry_cnt++;
200  usleep(1000000);
201  continue;
202  }
203 
204  optval = 1200; // KEEPIDLE (20 min)
205  if (setsockopt(socket_fd_, IPPROTO_TCP, TCP_KEEPIDLE, &optval, sizeof(optval)) < 0) {
206  std::cout << RED << "ERROR conn() : setsockopt(TCP_KEEPIDLE) failed" << COLOR_END <<
207  std::endl;
208  retry_cnt++;
209  usleep(1000000);
210  continue;
211  }
212 
213  optval = 60; // KEEPINTVL (1 min)
214  if (setsockopt(socket_fd_, IPPROTO_TCP, TCP_KEEPINTVL, &optval, sizeof(optval)) < 0) {
215  std::cout << RED << "ERROR conn() : setsockopt(TCP_KEEPINTVL) failed" << COLOR_END <<
216  std::endl;
217  retry_cnt++;
218  usleep(1000000);
219  continue;
220  }
221 
222  optval = 10; // KEEPCNT
223  if (setsockopt(socket_fd_, IPPROTO_TCP, TCP_KEEPCNT, &optval, sizeof(optval)) < 0) {
224  std::cout << RED << "ERROR conn() : setsockopt(TCP_KEEPCNT) failed" << COLOR_END << std::endl;
225  retry_cnt++;
226  usleep(1000000);
227  continue;
228  }
229 
230  socklen_t optlen = sizeof(optval);
231  if (getsockopt(socket_fd_, SOL_SOCKET, SO_KEEPALIVE, &optval, &optlen) < 0) {
232  std::cout << RED << "ERROR conn() : getsockopt(SO_KEEPALIVE) failed" << COLOR_END <<
233  std::endl;
234  retry_cnt++;
235  usleep(1000000);
236  continue;
237  } else {
238  // std::cout <<"INFO conn() : SO_KEEPALIVE is "<< (optval ? "ON" : "OFF") <<std::endl;
239  }
240 
241 
242  struct timeval timeout;
243  timeout.tv_sec = SOCKET_READ_TIMEOUT_SEC;
244  timeout.tv_usec = 100; // SOCKET_READ_TIMEOUT_USEC;
245  if (setsockopt(socket_fd_, SOL_SOCKET, SO_RCVTIMEO, &timeout, sizeof(timeout)) < 0) {
246  std::cout << "socket set timeout error" << std::endl;
247  }
248 
249 
250  if (connect(socket_fd_, (struct sockaddr *)&serv_addr_, sizeof(serv_addr_)) == 0) {
251  std::cout << GREEN << "connect() Success!" << COLOR_END << std::endl;
252 
253  return true;
254  } else if (EINPROGRESS == errno) { // * For Non bloking connection
255  std::cout << YELLOW << "connecting..." << COLOR_END << std::endl;
256  int error = 0;
257  socklen_t err_len = sizeof(error);
258  if (getsockopt(socket_fd_, SOL_SOCKET, SO_ERROR, (void *) &error, &err_len) < 0) { // NOLINT
259  // std::cout << RED << "connect() ERROR!" << COLOR_END << std::endl;
260  retry_cnt++;
261  usleep(1000000);
262  continue;
263  }
264  if (0 != error) {
265  if (ECONNREFUSED == error) {
266  std::cout << RED << "connect() ERROR2!" << COLOR_END << std::endl;
267  retry_cnt++;
268  usleep(1000000);
269  continue;
270  } else if (ETIMEDOUT == error) {
271  std::cout << RED << "connect() ERROR3!" << COLOR_END << std::endl;
272  retry_cnt++;
273  usleep(1000000);
274  continue;
275  }
276  }
277  std::cout << GREEN << "connect() Success!" << COLOR_END << std::endl;
278  return true;
279  } else {
280  std::cout << RED << "connect() Failed!" << COLOR_END << std::endl;
281  retry_cnt++;
282  usleep(1000000);
283  continue;
284  }
285  }
286  std::cout << RED << "ERROR connect() Failed!" << COLOR_END << std::endl;
287  return false;
288 }
