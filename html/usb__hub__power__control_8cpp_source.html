32 int UsbHubPowerControl::usb_hub_get_power(UsbHubPortName_t port_name, UsbHubPortStatus_t & status)
33 {
34  int ret = 0;
35  libusb_device_handle * hub_devh;
36  libusb_context * context;
37 
38  ret = libusb_init(&context);
39 
40  if (ret != 0) {
41  printf("[error] libusb init()\n");
42  return ret;
43  }
44 
45  do {
46  hub_devh = libusb_open_device_with_vid_pid(context, vid, pid);
47 
48  if (!hub_devh) {
49  printf("[error] libusb_open_device_with_vid_pid()\n");
50  ret = -1;
51  break;
52  }
53 
54  libusb_device * dev;
55  dev = libusb_get_device(hub_devh);
56 
57  if (dev == NULL) {
58  printf("[error] libusb_get_device()\n");
59  ret = -2;
60  break;
61  }
62 
63  // uint32_t dev_rev = 0;
64  uint32_t dev_reg;
65  dev_reg = 0;
66  ret = usb_hub_read_reg(hub_devh, 0xBF803C00 + 4 * static_cast<int>(port_name), &dev_reg, 4);
67  dev_reg &= (0x3 << 0);
68 
69  if (dev_reg > 0) {
70  status = POWER_ON;
71  } else {
72  status = POWER_OFF;
73  }
74  } while (0);
75 
76  libusb_close(hub_devh);
77 
78  return ret;
79 }

133  uint32_t reg_addr, void * p_data,
134  uint32_t length)
135 {
136  int ret = 0;
137  uint16_t addr_lo;
138  uint16_t addr_hi;
139 
140  addr_lo = (reg_addr >> 0) & 0xFFFF;
141  addr_hi = (reg_addr >> 16) & 0xFFFF;
142  ret = libusb_control_transfer(
143  hub_devh,
144  0xC0,
145  0x04,
146  addr_lo,
147  addr_hi,
148  static_cast<uint8_t *>(p_data),
149  length,
150  100000);
151 
152  if (ret < 0) {
153  printf("[NG] libusb_control_transfer()\n");
154  }
155
