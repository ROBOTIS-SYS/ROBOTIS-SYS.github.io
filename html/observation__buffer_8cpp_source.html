69 bool ObservationBuffer::setGlobalFrame(const std::string new_global_frame)
70 {
71  rclcpp::Time transform_time = nh_->now();
72  std::string tf_error;
73 
74  geometry_msgs::msg::TransformStamped transformStamped;
75  if (!tf2_buffer_.canTransform(
76  new_global_frame, global_frame_, tf2_ros::fromMsg(transform_time),
77  tf2::durationFromSec(tf_tolerance_), &tf_error))
78  {
79  RCLCPP_ERROR(
80  rclcpp::get_logger(
81  "antis_nav_costmap_2d"), "Transform between %s and %s with tolerance %.2f failed: %s.",
82  new_global_frame.c_str(),
83  global_frame_.c_str(), tf_tolerance_, tf_error.c_str());
84  return false;
85  }
86 
87  std::list<Observation>::iterator obs_it;
88  for (obs_it = observation_list_.begin(); obs_it != observation_list_.end(); ++obs_it) {
89  try {
90  Observation & obs = *obs_it;
91 
92  geometry_msgs::msg::PointStamped origin;
93  origin.header.frame_id = global_frame_;
94  origin.header.stamp = transform_time;
95  origin.point = obs.origin_;
96 
97  // we need to transform the origin of the observation to the new global frame
98  tf2_buffer_.transform(origin, origin, new_global_frame, tf2::durationFromSec(tf_tolerance_));
99  obs.origin_ = origin.point;
100 
101  // we also need to transform the cloud of the observation to the new global frame
102  tf2_buffer_.transform(
103  *(obs.cloud_), *(obs.cloud_), new_global_frame, tf2::durationFromSec(tf_tolerance_));
104  } catch (tf2::TransformException & ex) {
105  RCLCPP_ERROR(
106  rclcpp::get_logger("antis_nav_costmap_2d"),
107  "TF Error attempting to transform an observation from %s to %s: %s",
108  global_frame_.c_str(),
109  new_global_frame.c_str(), ex.what());
110  return false;
111  }
112  }
113 
114  // now we need to update our global_frame member
115  global_frame_ = new_global_frame;
116  return true;
117 }

119 void ObservationBuffer::bufferCloud(const sensor_msgs::msg::PointCloud2 & cloud)
120 {
121  geometry_msgs::msg::PointStamped global_origin;
122 
123  // create a new observation on the list to be populated
124  observation_list_.push_front(Observation());
125 
126  // check whether the origin frame has been set explicitly
127  // or whether we should get it from the cloud
128  std::string origin_frame = sensor_frame_ == "" ? cloud.header.frame_id : sensor_frame_;
129 
130  try {
131  // given these observations come from sensors...
132  // we'll need to store the origin pt of the sensor
133  geometry_msgs::msg::PointStamped local_origin;
134  local_origin.header.stamp = cloud.header.stamp;
135  local_origin.header.frame_id = origin_frame;
136  local_origin.point.x = 0;
137  local_origin.point.y = 0;
138  local_origin.point.z = 0;
139  tf2_buffer_.transform(local_origin, global_origin, global_frame_);
140  tf2::convert(global_origin.point, observation_list_.front().origin_);
141 
142  // make sure to pass on the raytrace/obstacle range
143  // of the observation buffer to the observations
144  observation_list_.front().raytrace_range_ = raytrace_range_;
145  observation_list_.front().obstacle_range_ = obstacle_range_;
146 
147  sensor_msgs::msg::PointCloud2 global_frame_cloud;
148 
149  // transform the point cloud
150  tf2_buffer_.transform(cloud, global_frame_cloud, global_frame_);
151  global_frame_cloud.header.stamp = cloud.header.stamp;
152 
153  // now we need to remove observations from the cloud that are below
154  // or above our height thresholds
155  sensor_msgs::msg::PointCloud2 & observation_cloud = *(observation_list_.front().cloud_);
156  observation_cloud.height = global_frame_cloud.height;
157  observation_cloud.width = global_frame_cloud.width;
158  observation_cloud.fields = global_frame_cloud.fields;
159  observation_cloud.is_bigendian = global_frame_cloud.is_bigendian;
160  observation_cloud.point_step = global_frame_cloud.point_step;
161  observation_cloud.row_step = global_frame_cloud.row_step;
162  observation_cloud.is_dense = global_frame_cloud.is_dense;
163 
164  unsigned int cloud_size = global_frame_cloud.height * global_frame_cloud.width;
165  sensor_msgs::PointCloud2Modifier modifier(observation_cloud);
166  modifier.resize(cloud_size);
167  unsigned int point_count = 0;
168 
169  // copy over the points that are within our height bounds
170  sensor_msgs::PointCloud2Iterator<float> iter_z(global_frame_cloud, "z");
171  std::vector<unsigned char>::const_iterator iter_global = global_frame_cloud.data.begin(),
172  iter_global_end = global_frame_cloud.data.end();
173  std::vector<unsigned char>::iterator iter_obs = observation_cloud.data.begin();
174  for (; iter_global != iter_global_end; ++iter_z, iter_global +=
175  global_frame_cloud.point_step)
176  {
177  if ((*iter_z) <= max_obstacle_height_ &&
178  (*iter_z) >= min_obstacle_height_)
179  {
180  std::copy(iter_global, iter_global + global_frame_cloud.point_step, iter_obs);
181  iter_obs += global_frame_cloud.point_step;
182  ++point_count;
183  }
184  }
185 
186  // resize the cloud for the number of legal points
187  modifier.resize(point_count);
188  observation_cloud.header.stamp = cloud.header.stamp;
189  observation_cloud.header.frame_id = global_frame_cloud.header.frame_id;
190  } catch (tf2::TransformException & ex) {
191  // if an exception occurs, we need to remove the empty observation from the list
192  observation_list_.pop_front();
193  RCLCPP_ERROR(
194  rclcpp::get_logger(
195  "antis_nav_costmap_2d"),
196  "TF Exception that should never happen for sensor frame: %s, cloud frame: %s, %s",
197  sensor_frame_.c_str(),
198  cloud.header.frame_id.c_str(), ex.what());
199  return;
200  }
201 
202  // if the update was successful, we want to update the last updated time
203  last_updated_ = nh_->now();
204 
205  // we'll also remove any stale observations from the list
206  purgeStaleObservations();
207 }

210 void ObservationBuffer::getObservations(std::vector<Observation> & observations)
211 {
212  // first... let's make sure that we don't have any stale observations
213  purgeStaleObservations();
214 
215  // now we'll just copy the observations for the caller
216  std::list<Observation>::iterator obs_it;
217  for (obs_it = observation_list_.begin(); obs_it != observation_list_.end(); ++obs_it) {
218  observations.push_back(*obs_it);
219  }

222 void ObservationBuffer::purgeStaleObservations()
223 {
224  if (!observation_list_.empty()) {
225  std::list<Observation>::iterator obs_it = observation_list_.begin();
226  // if we're keeping observations for no time... then we'll only keep one observation
227  if (observation_keep_time_ == rclcpp::Duration(0.0)) {
228  observation_list_.erase(++obs_it, observation_list_.end());
229  return;
230  }
231 
232  // otherwise... we'll have to loop through the observations to see which ones are stale
233  for (obs_it = observation_list_.begin(); obs_it != observation_list_.end(); ++obs_it) {
234  Observation & obs = *obs_it;
235  // check if the observation is out of date... and if it is,
236  // remove it and those that follow from the list
237  if ((last_updated_ - obs.cloud_->header.stamp) > observation_keep_time_) {
238  observation_list_.erase(obs_it, observation_list_.end());
239  return;
240  }
241  }
242  }

245 bool ObservationBuffer::isCurrent() const
246 {
247  if (expected_update_rate_ == rclcpp::Duration(0.0)) {
248  return true;
249  }
250 
251  bool current = (nh_->now() - last_updated_) <= expected_update_rate_;
252  if (!current) {
253  RCLCPP_WARN(
254  rclcpp::get_logger(
255  "antis_nav_costmap_2d"),
256  "The %s observation buffer has not been updated for %.2f seconds, and it should be updated every %.2f seconds.", //NOLINT
257  topic_name_.c_str(),
258  (nh_->now() - last_updated_).seconds(), expected_update_rate_.seconds());
259  }
260  return current;
261 }
