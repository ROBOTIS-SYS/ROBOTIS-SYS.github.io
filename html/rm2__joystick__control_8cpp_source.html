32 bool JoystickControl::configure()

35  declare_parmas();

61 void JoystickControl::update()
62 {
63  using Rotation = rm2_common::Rotation;
64  using Pose = rm2_common::Pose;
65  using TaskValues = rm2_common::TaskValues;
66 
67  if (control_state_ == 1) { // Joystick
68  // Timer count
69  if (robot_handler_->get_tick_time() > sleep_time_) {
70  RCLCPP_INFO_STREAM(node_->get_logger(), "Stop Joystick Control.");
71  this->stop_update();
72  }
73  TaskValues present_value = robot_handler_->get_previous_goal_task_value(end_effector_name_);
74  Pose delta_pose_on_base(
75  control_period_ * positional_coefficient_ * cmd_twist_.linear,
76  Rotation::RPY(
77  control_period_ * rotational_coefficient_ * cmd_twist_.angular[0],
78  control_period_ * rotational_coefficient_ * cmd_twist_.angular[1],
79  control_period_ * rotational_coefficient_ * cmd_twist_.angular[2])
80  );
81  TaskValues goal_value(
82  Pose(
83  present_value.pose.position + delta_pose_on_base.position, // position
84  delta_pose_on_base.orientation * present_value.pose.orientation), // orientation
85  cmd_twist_);
86  robot_handler_->set_goal_task_value(end_effector_name_, goal_value);
87  control_state_ = 0;
88  } else if (control_state_ == 2) { // Traj
89  bool trajectory_finish = false;
90  rm2_common::JointWaypoint goal = joint_traj_.get(
91  robot_handler_->get_tick_time(), trajectory_finish);
92  if (trajectory_finish) {
93  control_state_ = 0;
94  RCLCPP_INFO_STREAM(node_->get_logger(), "Stop Trajectory Control.");
95  stop_update();
96  }
97  robot_handler_->set_goal_joint_value(goal);
98  } else if (control_state_ == 3) {
99  bool trajectory_finish = false;
100  rm2_common::JointWaypoint goal = joint_traj_.get(
101  robot_handler_->get_tick_time(), trajectory_finish);
102  if (trajectory_finish) {
103  control_state_ = 0;
104  RCLCPP_INFO_STREAM(node_->get_logger(), "Stop unlimited Trajectory Control.");
105  stop_update();
106  }
107  robot_handler_->set_goal_joint_value_unlimited(goal);
108  }
109 }

320 void JoystickControl::joy_callback(const std::shared_ptr<JoystickControl::JoyMsg> cmd)
321 {
322  if (!subscriber_is_active_) {
323  RCLCPP_WARN_STREAM(
324  node_->get_logger(),
325  "Can't accept new commands. subscriber is inactive");
326  return;
327  }
328 
329  bool holded = false;
330  if (0 <= hold_btn_ && hold_btn_ < static_cast<int64_t>(cmd->buttons.size())) {
331  if (cmd->buttons.at(static_cast<uint>(hold_btn_))) {
332  holded = true;
333  }
334  } else {
335  holded = true;
336  }
337 
338  if (holded == true) {
339  // UP Positional coefficient
340  if (0 <= positional_coeff_up_btn_ &&
341  positional_coeff_up_btn_ < static_cast<int64_t>(cmd->buttons.size()))
342  {
343  static int prev_positional_coeff_up_btn;
344  if (cmd->buttons.at(static_cast<uint>(positional_coeff_up_btn_)) == 1 &&
345  cmd->buttons.at(static_cast<uint>(positional_coeff_up_btn_)) !=
346  prev_positional_coeff_up_btn)
347  {
348  positional_coefficient_ += delta_positional_coef_;
349  if (positional_coefficient_ > max_positional_coef_) {
350  positional_coefficient_ = max_positional_coef_;
351  }
352  RCLCPP_INFO_STREAM(
353  node_->get_logger(), "Positional coef up to '" << positional_coefficient_ << "'.");
354  }
355  prev_positional_coeff_up_btn = cmd->buttons.at(static_cast<uint>(positional_coeff_up_btn_));
356  }
357 
358  // Down Positional coefficient
359  if (0 <= positional_coeff_down_btn_ &&
360  positional_coeff_down_btn_ < static_cast<int64_t>(cmd->buttons.size()))
361  {
362  static int prev_positional_coeff_down_btn;
363  if (
364  cmd->buttons.at(static_cast<uint>(positional_coeff_down_btn_)) == 1 &&
365  cmd->buttons.at(static_cast<uint>(positional_coeff_down_btn_)) !=
366  prev_positional_coeff_down_btn)
367  {
368  positional_coefficient_ -= delta_positional_coef_;
369  if (positional_coefficient_ < 0.0) {
370  positional_coefficient_ = 0.0;
371  }
372  RCLCPP_INFO_STREAM(
373  node_->get_logger(), "Positional coef up to '" << positional_coefficient_ << "'.");
374  }
375  prev_positional_coeff_down_btn =
376  cmd->buttons.at(static_cast<uint>(positional_coeff_down_btn_));
377  }
378 
379  // UP Rotational coefficient
380  if (0 <= rotational_coeff_up_btn_ &&
381  rotational_coeff_up_btn_ < static_cast<int64_t>(cmd->buttons.size()))
382  {
383  static int prev_rotational_coeff_up_btn;
384  if (
385  cmd->buttons.at(static_cast<uint>(rotational_coeff_up_btn_)) == 1 &&
386  cmd->buttons.at(static_cast<uint>(rotational_coeff_up_btn_)) !=
387  prev_rotational_coeff_up_btn)
388  {
389  rotational_coefficient_ += delta_rotational_coef_;
390  if (rotational_coefficient_ > max_rotational_coef_) {
391  rotational_coefficient_ = max_rotational_coef_;
392  }
393  RCLCPP_INFO_STREAM(
394  node_->get_logger(), "Rotational coef up to '" << rotational_coefficient_ << "'.");
395  }
396  prev_rotational_coeff_up_btn =
397  cmd->buttons.at(static_cast<uint>(rotational_coeff_up_btn_));
398  }
399 
400  // Down Rotational coefficient
401  if (0 <= rotational_coeff_down_btn_ &&
402  rotational_coeff_down_btn_ < static_cast<int64_t>(cmd->buttons.size()))
403  {
404  static int prev_rotational_coeff_down_btn;
405  if (
406  cmd->buttons.at(static_cast<uint>(rotational_coeff_down_btn_)) == 1 &&
407  cmd->buttons.at(static_cast<uint>(rotational_coeff_down_btn_)) !=
408  prev_rotational_coeff_down_btn)
409  {
410  rotational_coefficient_ -= delta_rotational_coef_;
411  if (rotational_coefficient_ < 0.0) {
412  rotational_coefficient_ = 0.0;
413  }
414  RCLCPP_INFO_STREAM(
415  node_->get_logger(), "Rotational coef up to '" << rotational_coefficient_ << "'.");
416  }
417  prev_rotational_coeff_down_btn =
418  cmd->buttons.at(static_cast<uint>(rotational_coeff_down_btn_));
419  }
420 
421  // Joystick (Axis)
422  bool axes_input = false;
423  rm2_common::Twist cmd_twist;
424  // X
425  if (0 <= x_axis_ && x_axis_ < static_cast<int64_t>(cmd->axes.size())) {
426  if (fabs(
427  positional_coefficient_ *
428  static_cast<double>(cmd->axes.at(static_cast<uint>(x_axis_)))) > 0.001)
429  {
430  cmd_twist.linear.x(static_cast<double>(cmd->axes.at(static_cast<uint>(x_axis_))));
431  axes_input = true;
432  }
433  }
434  // Y
435  if (0 <= y_axis_ && y_axis_ < static_cast<int64_t>(cmd->axes.size())) {
436  if (fabs(
437  positional_coefficient_ *
438  static_cast<double>(cmd->axes.at(static_cast<uint>(y_axis_)))) > 0.001)
439  {
440  cmd_twist.linear.y(static_cast<double>(cmd->axes.at(static_cast<uint>(y_axis_))));
441  axes_input = true;
442  }
443  }
444  // Z
445  if (0 <= z_axis_ && z_axis_ < static_cast<int64_t>(cmd->axes.size())) {
446  if (fabs(
447  positional_coefficient_ *
448  static_cast<double>(cmd->axes.at(static_cast<uint>(z_axis_)))) > 0.001)
449  {
450  cmd_twist.linear.z(static_cast<double>(cmd->axes.at(static_cast<uint>(z_axis_))));
451  axes_input = true;
452  }
453  }
454  // ROLL
455  if (0 <= roll_axis_ && roll_axis_ < static_cast<int64_t>(cmd->axes.size())) {
456  if (fabs(
457  rotational_coefficient_ *
458  static_cast<double>(cmd->axes.at(static_cast<uint>(roll_axis_)))) > 0.001)
459  {
460  cmd_twist.angular.x(static_cast<double>(cmd->axes.at(static_cast<uint>(roll_axis_))));
461  axes_input = true;
462  }
463  }
464  // PITCH
465  if (0 <= pitch_axis_ && pitch_axis_ < static_cast<int64_t>(cmd->axes.size())) {
466  if (fabs(
467  rotational_coefficient_ *
468  static_cast<double>(cmd->axes.at(static_cast<uint>(pitch_axis_)))) > 0.001)
469  {
470  cmd_twist.angular.y(static_cast<double>(cmd->axes.at(static_cast<uint>(pitch_axis_))));
471  axes_input = true;
472  }
473  }
474  // YAW
475  if (0 <= yaw_axis_ && yaw_axis_ < static_cast<int64_t>(cmd->axes.size())) {
476  if (fabs(
477  rotational_coefficient_ *
478  static_cast<double>(cmd->axes.at(static_cast<uint>(yaw_axis_)))) > 0.001)
479  {
480  cmd_twist.angular.z(static_cast<double>(cmd->axes.at(static_cast<uint>(yaw_axis_))));
481  axes_input = true;
482  }
483  }
484 
485  // AXES
486  if (axes_input) {
487  cmd_twist_ = cmd_twist;
488  control_state_ = 1;
