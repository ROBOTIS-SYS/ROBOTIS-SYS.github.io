99 bool SimpleMobilePathPlanner::send_path()
100 {
101  std::lock_guard<std::mutex> lock(mutex_);
102 
103  if (has_path_ == false) {
104  return false;
105  }
106 
107  if (path_index_ == 0) {
108  RCLCPP_INFO_STREAM(this->get_logger(), "Started : Send Vel Path");
109  is_moving_ = true;
110  }
111 
112  if (path_index_ >= control_vels_.size()) {
113  is_moving_ = false;
114  return false;
115  }
116 
117  cmd_vel_pub_->publish(control_vels_.at(path_index_));
118  if (control_vels_.at(path_index_).linear.x > max_linear_vel_ * 1.05 ||
119  control_vels_.at(path_index_).angular.z > max_angular_vel_ * 1.05)
120  {
121  RCLCPP_WARN_STREAM(
122  this->get_logger(), "Cmd_vel(over max 105%) : " << control_vels_.at(
123  path_index_).linear.x << ", " <<
124  control_vels_.at(path_index_).angular.z);
125  }
126  path_index_++;
127 
128  // feedback
129  if (goal_handle_ != nullptr) {
130  auto feedback = std::make_shared<SimpleMobilePath::Feedback>();
131  feedback->remain_time =
132  1.0f / control_frequency_ *
133  static_cast<float>(control_vels_.size() - path_index_);
134  goal_handle_->publish_feedback(feedback);
135  }
136 
137  if (path_index_ >= control_vels_.size()) {
138  RCLCPP_INFO_STREAM(this->get_logger(), "Finished : Send Vel Path");
139  has_path_ = false;
140  path_index_ = 0;
141  is_moving_ = false;
142 
143  // result
144  if (goal_handle_ != nullptr) {
145  auto result = std::make_shared<SimpleMobilePath::Result>();
146  result->result = true;
147 
148  goal_handle_->succeed(result);
149  RCLCPP_INFO_STREAM(this->get_logger(), "Goal Succeeded");
150 
151  goal_handle_.reset();
152  }
153  }
154 
155  return true;
156 }

177 void SimpleMobilePathPlanner::simple_mobile_path_callback(
178  const std::shared_ptr<SimpleMobilePathService::Request> req,
179  std::shared_ptr<SimpleMobilePathService::Response> res)
180 {
181  // 동작 중 확인
182  RCLCPP_INFO_STREAM(
183  this->get_logger(), "Received goal request with Target " <<
184  req->goal.x << ", " << req->goal.y << ", " << (req->goal.theta * 180 / M_PI) << ".");
185 
186  std::lock_guard<std::mutex> lock(mutex_);
187 
188  if (is_moving_ == true) {
189  RCLCPP_WARN_STREAM(this->get_logger(), "It's already running.");
190  res->result = false;
191  res->moving_time = 0.0;
192  return;
193  }
194 
195  // 패스 및 이동시간 계산
196  geometry_msgs::msg::Pose2D from;
197  auto goal = req->goal;
198  bool movable_back = req->movable_back;
199  has_path_ = plan_mobile_path(from, goal, max_linear_vel_, max_angular_vel_, movable_back);
200 
201  // 응답
202  if (has_path_ == true) {
203  path_index_ = 0;
204  res->result = true;
205  res->moving_time = static_cast<float>(control_vels_.size()) /
206  static_cast<float>(control_frequency_);
207  RCLCPP_INFO_STREAM(
208  this->get_logger(), "Success to plan the path, moving_time : " << res->moving_time);
209  } else {
210  RCLCPP_WARN_STREAM(this->get_logger(), "It's already running.");
211  res->result = false;
212  res->moving_time = 0.0;
213  }
214 }

216 bool SimpleMobilePathPlanner::plan_mobile_path(
217  const geometry_msgs::msg::Pose2D & from,
218  const geometry_msgs::msg::Pose2D & to,
219  double max_linear_vel, double max_anglur_vel,
220  bool movable_back)
221 {
222  geometry_msgs::msg::Twist cmd_vel;
223  control_vels_.clear();
224 
225  // calc path
226  geometry_msgs::msg::Pose2D target;
227  target.x = to.x - from.x;
228  target.y = to.y - from.y;
229  target.theta = to.theta - from.theta;
230 
231  double distance = sqrt(target.x * target.x + target.y * target.y);
232  double yaw_1 = atan2(target.y, target.x);
233  if (movable_back && target.x < 0) {
234  yaw_1 = (yaw_1 > 0) ? yaw_1 - M_PI : yaw_1 + M_PI;
235  distance = -distance;
236  }
237  double yaw_2 = target.theta - yaw_1;
238  yaw_2 = (yaw_2 > M_PI) ? yaw_2 - 2 * M_PI :
239  (yaw_2 < -M_PI) ? yaw_2 + 2 * M_PI : yaw_2;
240 
241  double accel_time = 1.0;
242  double de_accel_time = 1.0;
243 
244  // turn to target
245  if (yaw_1 != 0.0) {
246  cmd_vel = geometry_msgs::msg::Twist();
247 
248  std::vector<double> vel_list;
249  calc_vel(yaw_1, accel_time, de_accel_time, max_anglur_vel, vel_list);
250 
251  for (const auto & vel : vel_list) {
252  cmd_vel.angular.z = vel;
253  control_vels_.push_back(cmd_vel);
254  }
255  }
256 
257  // move to target
258  if (distance != 0.0) {
259  cmd_vel = geometry_msgs::msg::Twist();
260 
261  std::vector<double> vel_list;
262  calc_vel(distance, accel_time, de_accel_time, max_linear_vel, vel_list);
263 
264  for (const auto & vel : vel_list) {
265  cmd_vel.linear.x = vel;
266  control_vels_.push_back(cmd_vel);
267  }
268  }
269 
270  // align rotation
271  if (yaw_2 != 0.0) {
272  cmd_vel = geometry_msgs::msg::Twist();
273 
274  std::vector<double> vel_list;
275  calc_vel(yaw_2, accel_time, de_accel_time, max_anglur_vel, vel_list);
276 
277  for (const auto & vel : vel_list) {
278  cmd_vel.angular.z = vel;
279  control_vels_.push_back(cmd_vel);
280  }
281  }
282 
283  if (control_vels_.size() == 0) {
284  return false;
285  }
286 
287  return true;
288 }

290 void SimpleMobilePathPlanner::calc_vel(
291  double value, double accel_time, double de_accel_time, double max_vel,
292  std::vector<double> & vel_list_)
293 {
294  double direction = (value > 0) ? 1.0 : -1.0;
295  double vel = 0.0;
296  vel_list_.clear();
297 
298  // no constant velocity section
299  double acc_deacc_section =
300  accel_time * max_vel * 0.5 + de_accel_time * max_vel * 0.5;
301  if (acc_deacc_section > fabs(value)) {
302  double scale_ratio = sqrt(fabs(value) / acc_deacc_section);
303  // max_vel = fabs(yaw_1) / (accel_time * 0.5 + de_accel_time * 0.5);
304 
305  // acceleration section
306  int step_size =
307  static_cast<int>(round(accel_time * scale_ratio * control_frequency_));
308  double step_vel = max_vel / round(accel_time * control_frequency_);
309  // double step_vel = max_vel / static_cast<double>(step_size);
310  for (int ix = 0; ix < step_size; ix++) {
311  vel = (step_vel * (ix + 1)) * direction;
312  vel_list_.push_back(vel);
313  }
314 
315  // de acceleration section
316  step_size =
317  static_cast<int>(round(de_accel_time * scale_ratio * control_frequency_));
318  step_vel = max_vel / round(de_accel_time * control_frequency_);
319  // step_vel = max_vel / static_cast<double>(step_size);
320  for (int ix = 0; ix < step_size; ix++) {
321  vel = (step_vel * (step_size - (ix + 1))) * direction;
322  vel_list_.push_back(vel);
323  }
324  } else {
325  // max_vel = MAX_ANGULAR_VEL;
326 
327  // acceleration section
328  int step_size = static_cast<int>(round(accel_time * control_frequency_));
329  double step_vel = max_vel / static_cast<double>(step_size);
330  for (int ix = 0; ix < step_size; ix++) {
331  vel = (step_vel * (ix + 1)) * direction;
332  vel_list_.push_back(vel);
333  }
334 
335  // constant velocity section
336  step_size = static_cast<int>(round(
337  ((fabs(value) - max_vel * accel_time * 0.5 -
338  max_vel * de_accel_time * 0.5) /
339  max_vel) *
340  control_frequency_));
341  for (int ix = 0; ix < step_size; ix++) {
342  vel = max_vel * direction;
343  vel_list_.push_back(vel);
344  }
345 
346  // de acceleration section
347  step_size = static_cast<int>(round(de_accel_time * control_frequency_));
348  for (int ix = 0; ix < step_size; ix++) {
349  vel = (step_vel * (step_size - (ix + 1))) * direction;
350  vel_list_.push_back(vel);
351  }
352  }
353 }
