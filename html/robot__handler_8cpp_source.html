79 bool RobotHandler::start_update(
80  int id,
81  std::function<void()> update,
82  std::function<void(CancelResult)> update_cancel)
83 {
84  if (current_control_module_id_ == id) {
85  return false;
86  } else {
87  int loop_rate = 10; // ms
88  int time_out = 1000; // ms
89  for (int t = 0; current_control_module_id_ < 0; ++t) {
90  std::this_thread::sleep_for(std::chrono::milliseconds(loop_rate));
91  if (t > time_out / loop_rate) {
92  return false;
93  }
94  }
95  mutex_.try_lock();
96  current_control_module_id_ = id;
97  update_cancel_(CancelResult::PREEMPTED);
98  update_ = update;
99  update_cancel_ = update_cancel;
100  mutex_.unlock();
101 
102  return true;
103  }
104 }

106 bool RobotHandler::stop_update(int id)
107 {
108  if (current_control_module_id_ == id) {
109  current_control_module_id_ = -1;
110  return true;
111  } else {
112  return false;
113  }
114 }

116 // manipulator_controller의 update 함수에서 사용하는 함수
117 // update_ 밖에서 사용되며, thread loop는 동일하기 때문에 mutex lock을 걸지 않음
118 // update_와 update_cancel_을 즉시 바꿔주는 역할
119 void RobotHandler::update_cancel(CancelResult result)
120 {
121  mutex_.try_lock();
122  update_cancel_(result);
123  current_control_module_id_ = 0;
124  update_ = []() -> void {};
125  update_cancel_ = [](CancelResult r) -> void {(void)r;};
126  mutex_.unlock();
127 }
128 
129 // Robot Handler 내부에서만 사용하는 함수 (외부에서는 사용하지 않음)
130 // update_ 함수내에서 사용되기 때문에 바로 update()함수를 빼버리면 안됨
131 // 그래서 current_control_module_id_를 -1로 바꿔주고 update함수내에서
132 // 다음 루프에서 update_를 빈 함수로 바꿔줌
133 // control moudle에서는 알지못하는 변수로 인해 발생하는 cancel에 사용된다.
134 // Action cancel 명령으로 취소가 발생한다면 stop_update를 사용한다.
135 void RobotHandler::cancel_update(CancelResult result)
136 {
137  mutex_.try_lock();
138  update_cancel_(result);
139  current_control_module_id_ = -1;
140  mutex_.unlock();
141 }

143 void RobotHandler::update()
144 {
145  // Activated stop_update function
146  // Reset update Function to empty
147  if (current_control_module_id_ < 0) {
148  current_control_module_id_ = 0;
149  mutex_.try_lock();
150  update_ = []() -> void {};
151  update_cancel_ = [](CancelResult r) -> void {(void)r;};
152  mutex_.unlock();
153  }
154 
155  // sync present values and goal values
156  sync_joint_value();
157 
158  // Control
159  mutex_.try_lock();
160  update_();
161  mutex_.unlock();
162 }

174 bool RobotHandler::set_goal_joint_value(JointValues goal_value)
175 {
176  mutex_.try_lock();
177  std::vector<std::string> joint_over_limit;
178  if (!robot_model_.check_joint_limit(goal_value, &joint_over_limit)) {
179  fprintf(stderr, "[RobotHandler] Limit over the joint num : ");
180  for (std::string joint_name : joint_over_limit) {
181  fprintf(stderr, "%s, ", joint_name.c_str());
182  }
183  fprintf(stderr, "\nStop control.\n");
184  goal_joint_values_ = goal_joint_values_.stoped();
185  cancel_update(CancelResult::JOINT_LIMIT);
186  mutex_.unlock();
187  return false;
188  }
189 
190  bool fail = false;
191 
192  uint num = 0;
193  while (num < goal_value.size()) {
194  if (fabs(goal_joint_values_.position(num) - goal_value.position(num)) > 0.50) {
195  fprintf(
196  stderr,
197  "[set_goal_joint_value] to fast move value at the joint num %d : %lf rad / one tick\n",
198  num, goal_joint_values_.position(num) - goal_value.position(num));
199  fail = true;
200  }
201  num++;
202  }
203  if (fail) {
204  cancel_update(CancelResult::TOO_FAST_MOVE);
205  mutex_.unlock();
206  return false;
207  }
208 
209  num = 0;
210  while (num < goal_value.size()) {
211  if (std::isnan(goal_value.position(num)) || std::isinf(goal_value.position(num))) {
212  fprintf(
213  stderr, "[set_goal_joint_value] nan or inf value at the joint num %d : %lf\n",
214  num, goal_value.position(num));
215  fail = true;
216  }
217  num++;
218  }
219  if (fail) {
220  cancel_update(CancelResult::INVALID_GOAL_VALUES);
221  mutex_.unlock();
222  return false;
223  }
224 
225  goal_joint_values_ = goal_value;
226  mutex_.unlock();
227  return true;
228 }

273 bool RobotHandler::set_goal_task_value(
274  std::string component_name,
275  TaskValues goal_value)
276 {
277  mutex_.try_lock();
278  // Solve IK
279  JointValues next_goal;
280  if (ik_solver_->solve(
281  component_name, goal_value, &next_goal,
282  goal_joint_values_))
283  {
284  // Check the joint limit
285  std::vector<std::string> joint_over_limit;
286  if (!robot_model_.check_joint_limit(next_goal, &joint_over_limit)) {
287  fprintf(stderr, "[RobotHandler] Limit over the joint num : ");
288  for (std::string joint_name : joint_over_limit) {
289  fprintf(stderr, "%s, ", joint_name.c_str());
290  }
291  fprintf(stderr, "\nStop control.\n");
292  goal_joint_values_ = goal_joint_values_.stoped();
293  cancel_update(CancelResult::JOINT_LIMIT);
294  mutex_.unlock();
295  return false;
296  }
297 
298  bool fail = false;
299 
300  uint num = 0;
301  while (num < next_goal.size()) {
302  if (fabs(goal_joint_values_.position(num) - next_goal.position(num)) > 0.50) {
303  fprintf(
304  stderr,
305  "[set_goal_task_value] to fast move value at the joint num %d : %lf rad/tick\n",
306  num, goal_joint_values_.position(num) - next_goal.position(num));
307  fail = true;
308  }
309  num++;
310  }
311  if (fail) {
312  cancel_update(CancelResult::TOO_FAST_MOVE);
313  mutex_.unlock();
314  return false;
315  }
316 
317  num = 0;
318  while (num < next_goal.size()) {
319  if (std::isnan(next_goal.position(num)) || std::isinf(next_goal.position(num))) {
320  fprintf(
321  stderr, "[set_goal_task_value] nan or inf value at the joint num %d : %lf\n",
322  num, next_goal.position(num));
323  fail = true;
324  }
325  num++;
326  }
327  if (fail) {
328  cancel_update(CancelResult::INVALID_GOAL_VALUES);
329  mutex_.unlock();

331  }
332  goal_joint_values_ = next_goal;
333  mutex_.unlock();
334  return true;
335  } else {
336  goal_joint_values_ = goal_joint_values_.stoped();
337  fprintf(
338  stderr,
339  "[RobotHandler] Can not solve the inverse kinematics about goal task value.\n");
340  cancel_update(CancelResult::IK_FAIL);

343  }
