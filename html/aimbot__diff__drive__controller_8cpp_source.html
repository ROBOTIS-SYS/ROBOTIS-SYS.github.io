156 controller_interface::return_type AimbotDiffDriveController::update()
157 {
158  auto logger = node_->get_logger();
159  if (get_current_state().id() == State::PRIMARY_STATE_INACTIVE) {
160  if (!is_halted) {
161  halt();
162  is_halted = true;
163  }
164  return controller_interface::return_type::OK;
165  }
166 
167  const auto current_time = node_->get_clock()->now();
168 
169  std::shared_ptr<Twist> last_msg;
170  received_velocity_msg_ptr_.get(last_msg);
171 
172  if (last_msg == nullptr) {
173  RCLCPP_WARN(logger, "Velocity message received was a nullptr.");
174  return controller_interface::return_type::ERROR;
175  }
176 
177  const auto dt = current_time - last_msg->header.stamp;
178  // Brake if cmd_vel has timeout, override the stored command
179  if (dt > cmd_vel_timeout_) {
180  last_msg->twist.linear.x = 0.0;
181  last_msg->twist.angular.z = 0.0;
182  }
183 
184  Twist command = *last_msg;
185  double & linear_command = command.twist.linear.x;
186  double & angular_command = command.twist.angular.z;
187 
188  // Apply (possibly new) multipliers:
189  const auto wheels = wheel_params_;
190  const double wheel_separation = wheels.separation_multiplier * wheels.separation;
191  const double left_wheel_radius = wheels.left_radius_multiplier * wheels.radius;
192  const double right_wheel_radius = wheels.right_radius_multiplier * wheels.radius;
193 
194  if (odom_params_.open_loop) {
195  odometry_.updateOpenLoop(linear_command, angular_command, current_time);
196  } else {
197  double left_position_mean = 0.0;
198  double right_position_mean = 0.0;
199  for (size_t index = 0; index < wheels.wheels_per_side; ++index) {
200  const double left_position =
201  registered_left_wheel_handles_[index].position.get().get_value();
202  const double right_position =
203  registered_right_wheel_handles_[index].position.get().get_value();
204 
205  if (std::isnan(left_position) || std::isnan(right_position)) {
206  RCLCPP_ERROR(
207  logger, "Either the left or right wheel position is invalid for index [%zu]", index);
208  return controller_interface::return_type::ERROR;
209  }
210 
211  left_position_mean += left_position;
212  right_position_mean += right_position;
213  }
214  left_position_mean /= wheels.wheels_per_side;
215  right_position_mean /= wheels.wheels_per_side;
216 
217  odometry_.update(left_position_mean, right_position_mean, current_time);
218  }
219 
220  tf2::Quaternion orientation;
221  orientation.setRPY(0.0, 0.0, odometry_.getHeading());
222 
223  if (previous_publish_timestamp_ + publish_period_ < current_time) {
224  previous_publish_timestamp_ += publish_period_;
225 
226  if (realtime_odometry_publisher_->trylock()) {
227  auto & odometry_message = realtime_odometry_publisher_->msg_;
228  odometry_message.header.stamp = current_time;
229  odometry_message.pose.pose.position.x = odometry_.getX();
230  odometry_message.pose.pose.position.y = odometry_.getY();
231  odometry_message.pose.pose.orientation.x = orientation.x();
232  odometry_message.pose.pose.orientation.y = orientation.y();
233  odometry_message.pose.pose.orientation.z = orientation.z();
234  odometry_message.pose.pose.orientation.w = orientation.w();
235  odometry_message.twist.twist.linear.x = odometry_.getLinear();
236  odometry_message.twist.twist.angular.z = odometry_.getAngular();
237  realtime_odometry_publisher_->unlockAndPublish();
238  }
239 
240  if (odom_params_.enable_odom_tf && realtime_odometry_transform_publisher_->trylock()) {
241  auto & transform = realtime_odometry_transform_publisher_->msg_.transforms.front();
242  transform.header.stamp = current_time;
243  transform.transform.translation.x = odometry_.getX();
244  transform.transform.translation.y = odometry_.getY();
245  transform.transform.rotation.x = orientation.x();
246  transform.transform.rotation.y = orientation.y();
247  transform.transform.rotation.z = orientation.z();
248  transform.transform.rotation.w = orientation.w();
249  realtime_odometry_transform_publisher_->unlockAndPublish();
250  }
251  }
252 
253  const auto update_dt = current_time - previous_update_timestamp_;
254  previous_update_timestamp_ = current_time;
255 
256  auto & last_command = previous_commands_.back().twist;
257  auto & second_to_last_command = previous_commands_.front().twist;
258  limiter_linear_.limit(
259  linear_command, last_command.linear.x, second_to_last_command.linear.x, update_dt.seconds());
260  limiter_angular_.limit(
261  angular_command, last_command.angular.z, second_to_last_command.angular.z, update_dt.seconds());
262 
263  previous_commands_.pop();
264  previous_commands_.emplace(command);
265 
266  // Publish limited velocity
267  if (publish_limited_velocity_ && realtime_limited_velocity_publisher_->trylock()) {
268  auto & limited_velocity_command = realtime_limited_velocity_publisher_->msg_;
269  limited_velocity_command.header.stamp = current_time;
270  limited_velocity_command.twist = command.twist;
271  realtime_limited_velocity_publisher_->unlockAndPublish();
272  }
273 
274 
276  SafetyFilter::UltrasonicRawData ultrasonic_front;
277  ultrasonic_front.range =
278  ultrasonic_handles_.at("ultrasonic_front").range.get().get_value();
279  ultrasonic_front.field_of_view =
280  ultrasonic_handles_.at("ultrasonic_front").fov.get().get_value();
281  SafetyFilter::UltrasonicRawData ultrasonic_front_right;
282  ultrasonic_front_right.range =
283  ultrasonic_handles_.at("ultrasonic_front_right").range.get().get_value();
284  ultrasonic_front_right.field_of_view =
285  ultrasonic_handles_.at("ultrasonic_front_right").fov.get().get_value();
286  SafetyFilter::UltrasonicRawData ultrasonic_front_left;
287  ultrasonic_front_left.range =
288  ultrasonic_handles_.at("ultrasonic_front_left").range.get().get_value();
289  ultrasonic_front_left.field_of_view =
290  ultrasonic_handles_.at("ultrasonic_front_left").fov.get().get_value();
291  SafetyFilter::UltrasonicRawData ultrasonic_back;
292  ultrasonic_back.range =
293  ultrasonic_handles_.at("ultrasonic_back").range.get().get_value();
294  ultrasonic_back.field_of_view =
295  ultrasonic_handles_.at("ultrasonic_back").fov.get().get_value();
296  SafetyFilter::UltrasonicRawData ultrasonic_back_right;
297  ultrasonic_back_right.range =
298  ultrasonic_handles_.at("ultrasonic_back_right").range.get().get_value();
299  ultrasonic_back_right.field_of_view =
300  ultrasonic_handles_.at("ultrasonic_back_right").fov.get().get_value();
301  SafetyFilter::UltrasonicRawData ultrasonic_back_left;
302  ultrasonic_back_left.range =
303  ultrasonic_handles_.at("ultrasonic_back_left").range.get().get_value();
304  ultrasonic_back_left.field_of_view =
305  ultrasonic_handles_.at("ultrasonic_back_left").fov.get().get_value();
306 
307  safety_filter_.update_ultrasonic_data(
308  ultrasonic_front,
309  ultrasonic_front_right,
310  ultrasonic_front_left,
311  ultrasonic_back,
312  ultrasonic_back_right,
313  ultrasonic_back_left
314  );
315 
316  // command may be limited further by SpeedLimit,
317  // without affecting the stored twist command
318  SafetyFilter::FilterResult filter_result = safety_filter_.filtering_cmd_vel(command);
319  command = filter_result.cmd_vel;
320 
322  static std::stringstream log;
323  static double log_most_ct = 100.0;
324  static int log_cnt = 0;
325  static bool detected = false;
326  if (filter_result.decelration) {
327  if (log_most_ct > filter_result.clash_time) {
328  log.str("");
329  log.setf(std::ios::fixed);
330  log.precision(1);
331  log << "Filterd Vel " << (filter_result.deceleration_rate * 100.0) << "%";
332  log << "(clash time is " << filter_result.clash_time << "s";
333  log.precision(2);
334  for (auto detected : filter_result.detected_sensor_data) {
335  log << " /" << detected.first << ":" << detected.second << "m";
336  }
337  log << ")";
338  log_most_ct = filter_result.clash_time;
339  }
340  detected = true;
341  }
342  log_cnt++;
343  if (log_cnt > 50 && detected) {
344  publish_log(
345  log.str(),
346  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
347  aimbot_log_interfaces::msg::Log::TYPE_WARN);
348  log_cnt = 0;
349  detected = false;
350  }
352 
353  // Compute wheels velocities:
354  const double velocity_left =
355  (linear_command - angular_command * wheel_separation / 2.0) / left_wheel_radius;
356  const double velocity_right =
357  (linear_command + angular_command * wheel_separation / 2.0) / right_wheel_radius;
358 
359  // Set wheels velocities:
360  for (size_t index = 0; index < wheels.wheels_per_side; ++index) {
361  registered_left_wheel_handles_[index].velocity.get().set_value(velocity_left);
362  registered_right_wheel_handles_[index].velocity.get().set_value(velocity_right);
363  }
364 
365  return controller_interface::return_type::OK;
366 }

368 CallbackReturn AimbotDiffDriveController::on_configure(const rclcpp_lifecycle::State &)
369 {
370  auto logger = node_->get_logger();
371 
372  // update parameters
373  left_wheel_names_ = node_->get_parameter("left_wheel_names").as_string_array();
374  right_wheel_names_ = node_->get_parameter("right_wheel_names").as_string_array();
375 
376  if (left_wheel_names_.size() != right_wheel_names_.size()) {
377  RCLCPP_ERROR(
378  logger, "The number of left wheels [%zu] and the number of right wheels [%zu] are different",
379  left_wheel_names_.size(), right_wheel_names_.size());
380  return CallbackReturn::ERROR;
381  }
382 
383  if (left_wheel_names_.empty()) {
384  RCLCPP_ERROR(logger, "Wheel names parameters are empty!");
385  return CallbackReturn::ERROR;
386  }
387 
388  wheel_params_.separation = node_->get_parameter("wheel_separation").as_double();
389  wheel_params_.wheels_per_side =
390  static_cast<size_t>(node_->get_parameter("wheels_per_side").as_int());
391  wheel_params_.radius = node_->get_parameter("wheel_radius").as_double();
392  wheel_params_.separation_multiplier =
393  node_->get_parameter("wheel_separation_multiplier").as_double();
394  wheel_params_.left_radius_multiplier =
395  node_->get_parameter("left_wheel_radius_multiplier").as_double();
396  wheel_params_.right_radius_multiplier =
397  node_->get_parameter("right_wheel_radius_multiplier").as_double();
398 
399  const auto wheels = wheel_params_;
400 
401  const double wheel_separation = wheels.separation_multiplier * wheels.separation;
402  const double left_wheel_radius = wheels.left_radius_multiplier * wheels.radius;
403  const double right_wheel_radius = wheels.right_radius_multiplier * wheels.radius;
404 
405  odometry_.setWheelParams(wheel_separation, left_wheel_radius, right_wheel_radius);
406  odometry_.setVelocityRollingWindowSize(
407  node_->get_parameter("velocity_rolling_window_size").as_int());
408 
409  odom_params_.odom_frame_id = node_->get_parameter("odom_frame_id").as_string();
410  odom_params_.base_frame_id = node_->get_parameter("base_frame_id").as_string();
411 
412  auto pose_diagonal = node_->get_parameter("pose_covariance_diagonal").as_double_array();
413  std::copy(
414  pose_diagonal.begin(), pose_diagonal.end(), odom_params_.pose_covariance_diagonal.begin());
415 
416  auto twist_diagonal = node_->get_parameter("twist_covariance_diagonal").as_double_array();
417  std::copy(
418  twist_diagonal.begin(), twist_diagonal.end(), odom_params_.twist_covariance_diagonal.begin());
419 
420  odom_params_.open_loop = node_->get_parameter("open_loop").as_bool();
421  odom_params_.enable_odom_tf = node_->get_parameter("enable_odom_tf").as_bool();
422 
423  cmd_vel_timeout_ = std::chrono::milliseconds{
424  static_cast<int>(node_->get_parameter("cmd_vel_timeout").as_double() * 1000.0)};
425  publish_limited_velocity_ = node_->get_parameter("publish_limited_velocity").as_bool();
426  use_stamped_vel_ = node_->get_parameter("use_stamped_vel").as_bool();
427 
428  try {
429  limiter_linear_ = SpeedLimiter(
430  node_->get_parameter("linear.x.has_velocity_limits").as_bool(),
431  node_->get_parameter("linear.x.has_acceleration_limits").as_bool(),
432  node_->get_parameter("linear.x.has_jerk_limits").as_bool(),
433  node_->get_parameter("linear.x.min_velocity").as_double(),
434  node_->get_parameter("linear.x.max_velocity").as_double(),
435  node_->get_parameter("linear.x.min_acceleration").as_double(),
436  node_->get_parameter("linear.x.max_acceleration").as_double(),
437  node_->get_parameter("linear.x.min_jerk").as_double(),
438  node_->get_parameter("linear.x.max_jerk").as_double());
439  } catch (const std::runtime_error & e) {
440  RCLCPP_ERROR(node_->get_logger(), "Error configuring linear speed limiter: %s", e.what());
441  }
442 
443  try {
444  limiter_angular_ = SpeedLimiter(
445  node_->get_parameter("angular.z.has_velocity_limits").as_bool(),
446  node_->get_parameter("angular.z.has_acceleration_limits").as_bool(),
447  node_->get_parameter("angular.z.has_jerk_limits").as_bool(),
448  node_->get_parameter("angular.z.min_velocity").as_double(),
449  node_->get_parameter("angular.z.max_velocity").as_double(),
450  node_->get_parameter("angular.z.min_acceleration").as_double(),
451  node_->get_parameter("angular.z.max_acceleration").as_double(),
452  node_->get_parameter("angular.z.min_jerk").as_double(),
453  node_->get_parameter("angular.z.max_jerk").as_double());
454  } catch (const std::runtime_error & e) {
455  RCLCPP_ERROR(node_->get_logger(), "Error configuring angular speed limiter: %s", e.what());
456  }
457 
458  safety_filter_params_.check_time = node_->get_parameter("check_time").as_double();
459  RCLCPP_INFO_STREAM(
460  node_->get_logger(), "Check time is " << safety_filter_params_.check_time << "s.");
461  safety_filter_params_.stop_time = node_->get_parameter("stop_time").as_double();
462  RCLCPP_INFO_STREAM(
463  node_->get_logger(),
464  "Stop time is " << safety_filter_params_.stop_time << "s.");
465  safety_filter_params_.base_radius = node_->get_parameter("base_radius").as_double();
466  RCLCPP_INFO_STREAM(
467  node_->get_logger(), "Base radius is " << safety_filter_params_.base_radius << "m.");
468 
469  ultrasonic_names_ = node_->get_parameter("ultrasonic_names").as_string_array();
470 
471  safety_filter_.set_parameters(safety_filter_params_);
472 
473  if (!reset()) {
474  return CallbackReturn::ERROR;
475  }
476 
477  // Safety Filter enable state serice server
478  safety_filter_.set_safety_filter(true);
479  safety_filter_enable_srv_ = node_->create_service<std_srvs::srv::SetBool>(
480  DEFAULT_SAFETY_FILTER_SERVICE, [this](
481  const std::shared_ptr<std_srvs::srv::SetBool::Request> request,
482  std::shared_ptr<std_srvs::srv::SetBool::Response> response) {
483  response->message = this->safety_filter_.set_safety_filter(request->data);
484  response->success = true;
485  });
486 
487  // Debuging Path publisher
488  debug_path_pub_ = node_->create_publisher<nav_msgs::msg::Path>(
489  DEFAULT_DEBUG_PATH_TOPIC, rclcpp::SystemDefaultsQoS());
490 
491  // LOG publisher
492  log_pub_ = node_->create_publisher<aimbot_log_interfaces::msg::Log>("/aimbot_log", 10);
493 
494  // left and right sides are both equal at this point
495  wheel_params_.wheels_per_side = left_wheel_names_.size();
496 
497  if (publish_limited_velocity_) {
498  limited_velocity_publisher_ =
499  node_->create_publisher<Twist>(DEFAULT_COMMAND_OUT_TOPIC, rclcpp::SystemDefaultsQoS());
500  realtime_limited_velocity_publisher_ =
501  std::make_shared<realtime_tools::RealtimePublisher<Twist>>(limited_velocity_publisher_);
502  }
503 
504  const Twist empty_twist;
505  received_velocity_msg_ptr_.set(std::make_shared<Twist>(empty_twist));
506 
507  // Fill last two commands with default constructed commands
508  previous_commands_.emplace(empty_twist);
509  previous_commands_.emplace(empty_twist);
510 
511  // initialize command subscriber
512  if (use_stamped_vel_) {
513  velocity_command_subscriber_ = node_->create_subscription<Twist>(
514  DEFAULT_COMMAND_TOPIC, rclcpp::SystemDefaultsQoS(),
515  [this](const std::shared_ptr<Twist> msg) -> void {
516  if (!subscriber_is_active_) {
517  RCLCPP_WARN(node_->get_logger(), "Can't accept new commands. subscriber is inactive");
518  return;
519  }
520  if ((msg->header.stamp.sec == 0) && (msg->header.stamp.nanosec == 0)) {
521  RCLCPP_WARN_ONCE(
522  node_->get_logger(),
523  "Received TwistStamped with zero timestamp, setting it to current "
524  "time, this message will only be shown once");
525  msg->header.stamp = node_->get_clock()->now();
526  }
527  received_velocity_msg_ptr_.set(std::move(msg));
528  });
529  } else {
530  velocity_command_unstamped_subscriber_ = node_->create_subscription<geometry_msgs::msg::Twist>(
531  DEFAULT_COMMAND_UNSTAMPED_TOPIC, rclcpp::SystemDefaultsQoS(),
532  [this](const std::shared_ptr<geometry_msgs::msg::Twist> msg) -> void {
533  if (!subscriber_is_active_) {
534  RCLCPP_WARN(node_->get_logger(), "Can't accept new commands. subscriber is inactive");
535  return;
536  }
537 
538  // Write fake header in the stored stamped command
539  std::shared_ptr<Twist> twist_stamped;
540  received_velocity_msg_ptr_.get(twist_stamped);
541  twist_stamped->twist = *msg;
542  twist_stamped->header.stamp = node_->get_clock()->now();
543  });
544  }
545 
546  // initialize odometry publisher and messasge
547  odometry_publisher_ = node_->create_publisher<nav_msgs::msg::Odometry>(
548  DEFAULT_ODOMETRY_TOPIC, rclcpp::SystemDefaultsQoS());
549  realtime_odometry_publisher_ =
550  std::make_shared<realtime_tools::RealtimePublisher<nav_msgs::msg::Odometry>>(
551  odometry_publisher_);
552 
553  auto & odometry_message = realtime_odometry_publisher_->msg_;
554  odometry_message.header.frame_id = odom_params_.odom_frame_id;
555  odometry_message.child_frame_id = odom_params_.base_frame_id;
556 
557  // limit the publication on the topics /odom and /tf
558  publish_rate_ = node_->get_parameter("publish_rate").as_double();
559  publish_period_ = rclcpp::Duration::from_seconds(1.0 / publish_rate_);
560  previous_publish_timestamp_ = node_->get_clock()->now();
561 
562  // initialize odom values zeros
563  odometry_message.twist =
564  geometry_msgs::msg::TwistWithCovariance(rosidl_runtime_cpp::MessageInitialization::ALL);
565 
566  constexpr size_t NUM_DIMENSIONS = 6;
567  for (size_t index = 0; index < 6; ++index) {
568  // 0, 7, 14, 21, 28, 35
569  const size_t diagonal_index = NUM_DIMENSIONS * index + index;
570  odometry_message.pose.covariance[diagonal_index] = odom_params_.pose_covariance_diagonal[index];
571  odometry_message.twist.covariance[diagonal_index] =
572  odom_params_.twist_covariance_diagonal[index];
573  }
574 
575  // initialize transform publisher and message
576  odometry_transform_publisher_ = node_->create_publisher<tf2_msgs::msg::TFMessage>(
577  DEFAULT_TRANSFORM_TOPIC, rclcpp::SystemDefaultsQoS());
578  realtime_odometry_transform_publisher_ =
579  std::make_shared<realtime_tools::RealtimePublisher<tf2_msgs::msg::TFMessage>>(
580  odometry_transform_publisher_);
581 
582  // keeping track of odom and base_link transforms only
583  auto & odometry_transform_message = realtime_odometry_transform_publisher_->msg_;
584  odometry_transform_message.transforms.resize(1);
585  odometry_transform_message.transforms.front().header.frame_id = odom_params_.odom_frame_id;
586  odometry_transform_message.transforms.front().child_frame_id = odom_params_.base_frame_id;
587 
588  previous_update_timestamp_ = node_->get_clock()->now();
589  return CallbackReturn::SUCCESS;

683 CallbackReturn AimbotDiffDriveController::configure_side(
684  const std::string & side, const std::vector<std::string> & wheel_names,
685  std::vector<WheelHandle> & registered_handles)
686 {
687  auto logger = node_->get_logger();
688 
689  if (wheel_names.empty()) {
690  RCLCPP_ERROR(logger, "No '%s' wheel names specified", side.c_str());
691  return CallbackReturn::ERROR;
692  }
693 
694  // register handles
695  registered_handles.reserve(wheel_names.size());
696  for (const auto & wheel_name : wheel_names) {
697  const auto state_handle = std::find_if(
698  state_interfaces_.cbegin(), state_interfaces_.cend(), [&wheel_name](const auto & interface) {
699  return interface.get_name() == wheel_name &&
700  interface.get_interface_name() == HW_IF_POSITION;
701  });
702 
703  if (state_handle == state_interfaces_.cend()) {
704  RCLCPP_ERROR(logger, "Unable to obtain joint state handle for %s", wheel_name.c_str());
705  return CallbackReturn::ERROR;
706  }
707 
708  const auto command_handle = std::find_if(
709  command_interfaces_.begin(), command_interfaces_.end(),
710  [&wheel_name](const auto & interface) {
711  return interface.get_name() == wheel_name &&
712  interface.get_interface_name() == HW_IF_VELOCITY;
713  });
714 
715  if (command_handle == command_interfaces_.end()) {
716  RCLCPP_ERROR(logger, "Unable to obtain joint command handle for %s", wheel_name.c_str());
717  return CallbackReturn::ERROR;
718  }
719 
720  registered_handles.emplace_back(
721  WheelHandle{std::ref(*state_handle), std::ref(*command_handle)});
722  }
723 
724  return CallbackReturn::SUCCESS;
725 }

727 CallbackReturn AimbotDiffDriveController::configure_ultrasonic_handles(
728  std::vector<std::string> sensor_names)
729 {
730  if (sensor_names.empty()) {
731  RCLCPP_ERROR(node_->get_logger(), "No sensor names specified");
732  return CallbackReturn::ERROR;
733  }
734 
735  // register handles
736  ultrasonic_handles_.clear();
737  for (const auto & sensor_name : sensor_names) {
738  // find range
739  const auto range_handle = std::find_if(
740  state_interfaces_.cbegin(),
741  state_interfaces_.cend(),
742  [&sensor_name](const auto & interface) {
743  return interface.get_name() == sensor_name &&
744  interface.get_interface_name() == "range";
745  });
746  if (range_handle == state_interfaces_.cend()) {
747  RCLCPP_ERROR(
748  node_->get_logger(), "Unable to obtain Range state handle for %s", sensor_name.c_str());
749  return CallbackReturn::ERROR;
750  }
751 
752  // find fov
753  const auto fov_handle = std::find_if(
754  state_interfaces_.cbegin(),
755  state_interfaces_.cend(),
756  [&sensor_name](const auto & interface) {
757  return interface.get_name() == sensor_name &&
758  interface.get_interface_name() == "field_of_view";
759  });
760  if (fov_handle == state_interfaces_.cend()) {
761  RCLCPP_ERROR(
762  node_->get_logger(), "Unable to obtain fov state handle for %s", sensor_name.c_str());
763  return CallbackReturn::ERROR;
764  }
765 
766  ultrasonic_handles_.insert(
767  std::make_pair(
768  sensor_name,
769  UltrasonicHandle{std::ref(*range_handle), std::ref(*fov_handle)}));
770  }
771 
772  return CallbackReturn::SUCCESS;

775 void AimbotDiffDriveController::pub_debug_path(geometry_msgs::msg::Twist cmd_vel)
776 {
777  nav_msgs::msg::Path path;
778  path.header.stamp = node_->now();
779  path.header.frame_id = "base_link";
780  for (double check_time = 0.0;
781  check_time < safety_filter_params_.check_time;
782  check_time += 0.1)
783  {
784  geometry_msgs::msg::PoseStamped single_pose;
785 
786  double v = cmd_vel.linear.x;
787  double w = cmd_vel.angular.z;
788 
789  double delta_x = 0.0;
790  double delta_y = 0.0;
791  double delta_theta = 0.0;
792  if (w == 0.0) {
793  delta_x = v * check_time;
794  delta_y = 0.0;
795  } else {
796  double r = v / w;
797  delta_theta = w * check_time;
798  delta_x = r * sin(delta_theta);
799  delta_y = r * (1 - cos(delta_theta));
800  }
801 
802  single_pose.header.stamp = node_->now();
803  single_pose.header.frame_id = "base_link";
804  single_pose.pose.position.x = delta_x;
805  single_pose.pose.position.y = delta_y;
806  single_pose.pose.position.z = 0.0;
807  single_pose.pose.orientation.w = cos(delta_theta / 2);
808  single_pose.pose.orientation.x = 0.0;
809  single_pose.pose.orientation.y = 0.0;
810  single_pose.pose.orientation.z = sin(delta_theta / 2);
811  path.poses.push_back(single_pose);
812  }
813 
814  debug_path_pub_->publish(path);
815 }
