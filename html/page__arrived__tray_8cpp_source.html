87 void PageArrivedTray::set_title_text(std::string option)
88 {
89  for (int i = 0; i < 3; i++) {
90  set_btn_selected(i + 1, false);
91  }
92 
93  QString qstr_tray = "";
94  QStringList list = QString(option.c_str()).split("|");
95  for (const QString & str : list) {
96  bool ok;
97  int tray_num = str.toInt(&ok);
98  if (tray_num == 0) {
99  set_btn_selected(1, true);
100  set_btn_selected(2, true);
101  set_btn_selected(3, true);
102  qstr_tray += " ① / ② / ③ /";
103  } else if (ok && tray_num > 0 && tray_num < 4) {
104  set_btn_selected(tray_num, true);
105  if (tray_num == 1) {
106  qstr_tray += " ① /";
107  } else if (tray_num == 2) {
108  qstr_tray += " ② /";
109  } else if (tray_num == 3) {
110  qstr_tray += " ③ /";
111  }
112  }
113  }
114  qstr_tray.chop(1);
115  ui->lblTitle->setText(tr("<b>%1번 트레이</b>에서 물품을 찾아가세요.").arg(qstr_tray));
116 }
