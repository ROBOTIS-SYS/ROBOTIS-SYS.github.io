47 template<typename CostmapT>
48 double FootprintCollisionChecker<CostmapT>::footprintCost(const Footprint footprint)
49 {
50  // now we really have to lay down the footprint in the costmap_ grid
51  unsigned int x0, x1, y0, y1;
52  double footprint_cost = 0.0;
53 
54  // we need to rasterize each line in the footprint
55  for (unsigned int i = 0; i < footprint.size() - 1; ++i) {
56  // get the cell coord of the first point
57  if (!worldToMap(footprint[i].x, footprint[i].y, x0, y0)) {
58  return static_cast<double>(LETHAL_OBSTACLE);
59  }
60 
61  // get the cell coord of the second point
62  if (!worldToMap(footprint[i + 1].x, footprint[i + 1].y, x1, y1)) {
63  return static_cast<double>(LETHAL_OBSTACLE);
64  }
65 
66  footprint_cost = std::max(lineCost(x0, x1, y0, y1), footprint_cost);
67  }
68 
69  // we also need to connect the first point in the footprint to the last point
70  // get the cell coord of the last point
71  if (!worldToMap(footprint.back().x, footprint.back().y, x0, y0)) {
72  return static_cast<double>(LETHAL_OBSTACLE);
73  }
74 
75  // get the cell coord of the first point
76  if (!worldToMap(footprint.front().x, footprint.front().y, x1, y1)) {
77  return static_cast<double>(LETHAL_OBSTACLE);
78  }
79 
80  footprint_cost = std::max(lineCost(x0, x1, y0, y1), footprint_cost);
81 
82  // if all line costs are legal... then we can return that the footprint is legal
83  return footprint_cost;
84 }

86 template<typename CostmapT>
87 double FootprintCollisionChecker<CostmapT>::lineCost(int x0, int x1, int y0, int y1) const
88 {
89  double line_cost = 0.0;
90  double point_cost = -1.0;
91 
92  for (antis_nav_util::LineIterator line(x0, y0, x1, y1); line.isValid(); line.advance()) {
93  point_cost = pointCost(line.getX(), line.getY()); // Score the current point
94 
95  if (line_cost < point_cost) {
96  line_cost = point_cost;
97  }
98  }
99 
100  return line_cost;
101 }

122 template<typename CostmapT>
123 double FootprintCollisionChecker<CostmapT>::footprintCostAtPose(
124  double x, double y, double theta, const Footprint footprint)
125 {
126  double cos_th = cos(theta);
127  double sin_th = sin(theta);
128  Footprint oriented_footprint;
129  for (unsigned int i = 0; i < footprint.size(); ++i) {
130  geometry_msgs::msg::Point new_pt;
131  new_pt.x = x + (footprint[i].x * cos_th - footprint[i].y * sin_th);
132  new_pt.y = y + (footprint[i].x * sin_th + footprint[i].y * cos_th);
133  oriented_footprint.push_back(new_pt);
134  }
135 
136  return footprintCost(oriented_footprint);
137 }
