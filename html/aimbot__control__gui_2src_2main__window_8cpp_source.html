207 void MainWindow::update_image(const sensor_msgs::msg::CompressedImage & image)
208 {
209  if (!image.data.size()) {return;}
210 
211  QPixmap pix;
212 
213  int label_w = ui_->label_cam_image->geometry().width();
214  int label_h = ui_->label_cam_image->geometry().height();
215 
216  try {
217  pix.loadFromData(&(image.data[0]), static_cast<unsigned int>(image.data.size()), "JPEG");
218  } catch (int expn) {
219  fprintf(stderr, " error pix.loadFromData\n");
220  }
221 
222  try {
223  ui_->label_cam_image->setPixmap(pix.scaled(label_w, label_h));
224  } catch (int expn) {
225  fprintf(stderr, "errer pix.scaled\n");
226  }
227 }
