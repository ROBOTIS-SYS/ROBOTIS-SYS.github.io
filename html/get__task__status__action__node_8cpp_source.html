111 bool GetTaskStatusAction::get_task_status()
112 {
113  // Call Service
114  auto request = std::make_shared<task_manager_interfaces::srv::TaskStatus::Request>();
115  auto future = get_task_status_client_->async_send_request(request);
116 // auto future_result =
117 // future.wait_for(std::chrono::milliseconds(static_cast<int>(wait_time_for_service_ * 1000)));
118  double check_time = 1000.0 * wait_time_for_service_;
119  double loop_rate = 100.0; // ms
120  int loop_count = 0;
121 
122  rclcpp::WallRate loopRate(std::chrono::milliseconds(100));
123 
124  std::future_status future_result = std::future_status::deferred;
125  while (rclcpp::ok() && isHaltRequested() == false) {
126  future_result = future.wait_for(std::chrono::milliseconds(0));
127  if (future_result == std::future_status::ready) {
128  break;
129  }
130 
131  // sleep
132  loopRate.sleep();
133 
134  if ((loop_rate * ++loop_count) > check_time) {
135  break;
136  }
137  }
138 
139  if (isHaltRequested()) {
140  return false;
141  }
142 
143  switch (future_result) {
144  case std::future_status::ready:
145  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
146  RCLCPP_INFO(get_logger(), "[GetTaskStatusAction] get task list service future : ready");
147  }
148  break;
149 
150  case std::future_status::timeout:
151  RCLCPP_ERROR(
152  get_logger(), "[GetTaskStatusAction] get task list service future : timeout");
153  return false;
154 
155  default:
156  return false;
157  }
158 
159  auto task_status_result = future.get()->status;
160 
161  std::string task_status = "IDLE";
162  using TASKSTATUS = task_manager_interfaces::srv::TaskStatus::Response;
163  switch (task_status_result) {
164  case TASKSTATUS::IDLE:
165  task_status = "IDLE";
166  break;
167  case TASKSTATUS::RUNNING:
168  task_status = "RUNNING";
169  break;
170  case TASKSTATUS::FINISHED:
171  task_status = "FINISHED";
172  break;
173  case TASKSTATUS::FAILED:
174  task_status = "FAILED";
175  break;
176  case TASKSTATUS::CANCELED:
177  task_status = "CANCELED";
178  break;
179  default:
180  return false;
181  }
182 
183  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
184  RCLCPP_INFO_STREAM(get_logger(), "Task Status : " << task_status);
185  }
186 
187  setOutput<std::string>("task_status", task_status);
188 
189  return true;
190 }
