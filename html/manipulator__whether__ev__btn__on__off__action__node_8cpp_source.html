44 BT::NodeStatus ManipulatorWhetherEvBtnOnOffAction::tick()
45 {
46  std::cout << std::endl;
47 
48  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
49  RCLCPP_INFO_STREAM(get_logger(), "<TICK> ManipulatorWhetherEvBtnOnOffAction");
50  }
51 
52  ev_button_state_sub_ = node_->create_subscription<BtnMsg>(
53  "/gc_manipulator/detection_manager/ev_detected_btn_arr", 0,
54  std::bind(
55  &ManipulatorWhetherEvBtnOnOffAction::ev_btn_state_callback, this, std::placeholders::_1));
56 
57  if (!getInput<std::string>("target_ev_button", target_ev_btn_)) {
58  RCLCPP_ERROR_STREAM(get_logger(), "target_ev_button is empty!!");
59  return BT::NodeStatus::FAILURE;
60  }
61 
62  bool target_state;
63  if (!getInput<bool>("target_state", target_state)) {
64  RCLCPP_ERROR_STREAM(get_logger(), "target_state is empty!!");
65  return BT::NodeStatus::FAILURE;
66  }
67 
68  double undetected_timeout_sec;
69  if (!getInput<double>("undetected_timeout_sec", undetected_timeout_sec)) {
70  undetected_timeout_sec = 1.0;
71  }
72  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
73  RCLCPP_INFO_STREAM(
74  get_logger(), "undetected_timeout_sec is " << undetected_timeout_sec << "s!!");
75  }
76 
77  double unmatched_timeout_sec;
78  if (!getInput<double>("unmatched_timeout_sec", unmatched_timeout_sec)) {
79  unmatched_timeout_sec = 10.0;
80  }
81  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
82  RCLCPP_INFO_STREAM(get_logger(), "unmatched_timeout is " << unmatched_timeout_sec << "s!!");
83  }
84 
85  double detection_time_sec;
86  if (!getInput<double>("detection_time_sec", detection_time_sec)) {
87  detection_time_sec = 0.5;
88  }
89  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
90  RCLCPP_INFO_STREAM(get_logger(), "detection time is " << detection_time_sec << "s!!");
91  }
92 
93  last_detected_time_ = node_->now();
94  start_check_ = true;
95  bool prev_mathched_state = false;
96  btn_detected_ = false;
97 
98  rclcpp::Time matched_time = node_->now();
99  rclcpp::Time unmatched_time = node_->now();
100  rclcpp::WallRate loopRate(std::chrono::milliseconds(100));
101  while (node_->now() - last_detected_time_ <
102  rclcpp::Duration::from_seconds(undetected_timeout_sec)) // undetected timeout
103  {
104  loopRate.sleep(); // sleep
105 
106  if (isHaltRequested()) {
107  btn_detected_ = false;
108  start_check_ = false;
109 
110  ev_button_state_sub_.reset();
111 
112  return BT::NodeStatus::IDLE;
113  }
114 
115  if (btn_detected_ == false) {
116  if (
117  config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true)
118  {
119  RCLCPP_INFO_STREAM(get_logger(), "Button is not detected.");
120  }
121  matched_time = node_->now();
122  unmatched_time = node_->now();
123  continue;
124  }
125 
126  last_detected_time_ = node_->now();
127 
128  if (btn_state_ == target_state) { // state matched
129  RCLCPP_DEBUG_STREAM(
130  get_logger(),
131  "[Match] match_time : " << (node_->now() - matched_time).seconds() << ", Duration : " <<
132  rclcpp::Duration::from_seconds(detection_time_sec).seconds() <<
133  " button : " << btn_state_ << ", target : " << target_state);
134 
135  if (!prev_mathched_state) {
136  matched_time = last_detected_time_;
137  }
138 
139  if (node_->now() - matched_time > rclcpp::Duration::from_seconds(detection_time_sec)) {
140  btn_detected_ = false;
141  start_check_ = false;
142 
143  if (
144  config().blackboard->get("enable_log", enable_log_) == false ||
145  enable_log_ == true)
146  {
147  RCLCPP_INFO_STREAM(get_logger(), "Success.");
148  }
149 
150  ev_button_state_sub_.reset();
151  return BT::NodeStatus::SUCCESS; // Match Success
152  }
153  prev_mathched_state = true;
154  } else {
155  RCLCPP_DEBUG_STREAM(
156  get_logger(),
157  "[Unmatch] unmatch_time : " <<
158  (node_->now() - unmatched_time).seconds() <<
159  ", Duration : " << rclcpp::Duration::from_seconds(unmatched_timeout_sec).seconds() <<
160  " button : " << btn_state_ << ", target : " << target_state);
161 
162  if (prev_mathched_state) {
163  unmatched_time = last_detected_time_;
164  }
165 
166  if (
167  node_->now() - unmatched_time > rclcpp::Duration::from_seconds(unmatched_timeout_sec))
168  {
169  btn_detected_ = false;
170  start_check_ = false;
171 
172  if (
173  config().blackboard->get("enable_log", enable_log_) == false ||
174  enable_log_ == true)
175  {
176  RCLCPP_INFO_STREAM(get_logger(), "Unmatched Failure.");
177  }
178 
179  ev_button_state_sub_.reset();
180  return BT::NodeStatus::FAILURE; // Unmatched Failure
181  }
182  prev_mathched_state = false;
183  }
184  }
185 
186  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
187  RCLCPP_INFO_STREAM(get_logger(), "Undetected Failure.");
188  }
189  btn_detected_ = false;
190  start_check_ = false;
191 
192  ev_button_state_sub_.reset();
193  return BT::NodeStatus::FAILURE; // Undetected Failure
194 }
