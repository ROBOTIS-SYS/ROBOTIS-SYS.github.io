95 // previous_joint_position_ = joint_position;

98  } else if (type_ == TransAxis) {
99 // previous_joint_position_ = joint_position;
100  joint_pose_from_parent_.position = origin_from_parent_.position +

103  joint_pose_from_parent_ = origin_from_parent_;
