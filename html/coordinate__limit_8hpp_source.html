93  bool is_flat(double x, double y, double z)
94  {
95  bool is_flat = (pow((x / x_limit_), 2) + pow((y / y_limit_), 2) + pow((z / z_limit_), 2)) < 1;
96  if (!is_flat) {
97  std::cout << "spike limit exceeded!" << std::endl;
98  std::cout << "x: " << x << "m, y: " << y << "m, z:" << z << "m \n";
99  }
100  return is_flat;
101  }

124  bool is_flat(
125  const Eigen::Isometry3d & pose1,
126  const Eigen::Isometry3d & pose2)
127  { // relative point
128  double x_diff = std::fabs(pose1.translation()(0) - pose2.translation()(0));
129  double y_diff = std::fabs(pose1.translation()(1) - pose2.translation()(1));

135  bool is_flat(

138  { // relative speed
139  double x_vel = std::fabs(pose1.pose.pose.position.x - pose2.pose.pose.position.x);
140  double y_vel = std::fabs(pose1.pose.pose.position.y - pose2.pose.pose.position.y);
141  double z_vel = std::fabs(pose1.pose.pose.position.z - pose2.pose.pose.position.z);
142

146  bool is_speed_flat(double dx, double dy, double dz, rclcpp::Duration due)
147  {
148  double sample_time_d = std::fabs(due.seconds());
149  double vx = std::fabs(dx) / sample_time_d;
150  double vy = std::fabs(dy) / sample_time_d;
151  double vz = std::fabs(dz) / sample_time_d;
152  bool is_flat =
153  (pow((vx / vx_limit_), 2) + pow((vy / vy_limit_), 2) + pow((vz / vz_limit_), 2)) < 1;
154  if (!is_flat) {
155  std::cout << "Speed limit exceeded with value:\n";
156  std::cout << "vx: " << vx << "m/s, vy: " << vy << "m/s, vz:" << vz << "m/s \n";
157  }
158  return is_flat;
159  }

161  bool is_speed_flat(

163  const nav_msgs::msg::Odometry & pose2)
164  { // relative speed
165  double x_vel = std::fabs(pose1.pose.pose.position.x - pose2.pose.pose.position.x);
166  double y_vel = std::fabs(pose1.pose.pose.position.y - pose2.pose.pose.position.y);
167  double z_vel = std::fabs(pose1.pose.pose.position.z - pose2.pose.pose.position.z);
168  rclcpp::Time timepoint1(pose1.header.stamp);
169  rclcpp::Time timepoint2(pose2.header.stamp);
170  return is_speed_flat(x_vel, y_vel, z_vel, timepoint1 - timepoint2);
171  }
