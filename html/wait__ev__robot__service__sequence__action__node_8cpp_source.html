75  // _halt_requested.store(false);

78  // std::string sound_file_path;

85  int duration;
86  if (!getInput<int>("duration", duration)) {
87  duration = 60;
88  }
89  if (!getInput<std::vector<std::string>>("wait_robot_status", wait_status_)) {
90  RCLCPP_ERROR_STREAM(get_logger(), "No Input : [wait_robot_status]");

92  }
93 
94  if (wait_status_.size() == 0) {
95  RCLCPP_ERROR_STREAM(get_logger(), "[wait_robot_status] is empty");
96  return BT::NodeStatus::FAILURE;
97  }
98 
99  std::stringstream ss;
100  ss << "Wait EV Robot Service Sequence for " << duration << "s, ";
101  for (auto status : wait_status_) {
102  ss << status;
103  ss << "|";
104  }
105 
106  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
107  RCLCPP_INFO_STREAM(get_logger(), ss.str());
108  }
109 
110  // s -> ms
111  double check_time = static_cast<double>(duration) * 1000.0;
112  double loop_rate = 100.0; // ms
113  int loop_count = 0;
114 
115  rclcpp::WallRate loopRate(std::chrono::milliseconds(100));
116 
117  publish_tree_status(name(), ss.str());
118 
119  setStatus(BT::NodeStatus::RUNNING);
120 
121  while (rclcpp::ok() && has_ev_result_ == false && isHaltRequested() == false) {
122  // sleep
123  loopRate.sleep();
124 
125  if ((check_time > 0) && ((loop_rate * ++loop_count) >= check_time)) {
126  break;
127  }
128  }
129 
130  if (isHaltRequested() == true) {
131  publish_tree_status(name(), "엘리베이터 상태 체크 노드를 정지했습니다.");
132  return BT::NodeStatus::IDLE;
133  }
134 
135  if (has_ev_result_ == false) {
136  publish_tree_status(name(), "엘리베이터 서비스의 로봇 시퀀스 정보를 얻어오지 못했습니다.");
137  }
138 
139  return has_ev_result_ ? BT::NodeStatus::SUCCESS : BT::NodeStatus::FAILURE;
140 }
