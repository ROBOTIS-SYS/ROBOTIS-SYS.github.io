176 inline void Utf16Writer::WriteCharacter(
177  StringType & stringDestination, char32_t codePoint)
178 {
179  if ((codePoint <= 0xD7FF) || (codePoint >= 0xE000 && codePoint <= 0xFFFF)) {
180  stringDestination.push_back(static_cast<StringType::value_type>(codePoint));
181  } else if (codePoint >= 0x10000 && codePoint <= 0x10FFFF) {
182  // We have surrogate pair.
183  codePoint -= 0x10000;
184  char32_t surr = (codePoint >> 10) + 0xD800; // Lead surrogate.
185  stringDestination.push_back(static_cast<StringType::value_type>(surr));
186  surr = (codePoint & 0x3FF) + 0xDC00; // Trail surrogate.
187  stringDestination.push_back(static_cast<StringType::value_type>(surr));
188  }
189 }
