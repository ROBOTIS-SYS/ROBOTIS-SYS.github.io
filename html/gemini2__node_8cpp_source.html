187 template<typename T>
188 void Gemini2Node::set_and_get_node_parameter(
189  std::string param_name,
190  T & out_value,
191  T default_value,
192  bool dynamic)
193 {
194  rcl_interfaces::msg::ParameterDescriptor descriptor;
195  descriptor.read_only = !dynamic;
196 
197  node_->declare_parameter(param_name, rclcpp::ParameterValue(default_value), descriptor);
198 
199  if (!(node_->get_parameter(param_name, out_value))) {
200  RCLCPP_WARN_STREAM(
201  node_->get_logger(), "The parameter '" <<
202  param_name <<
203  "' is not available or is not valid, using the default value: " <<
204  default_value);
205  } else {
206  RCLCPP_INFO_STREAM(
207  node_->get_logger(), "The parameter '" << param_name << "' - " << out_value);
208  }
209 }

211 void Gemini2Node::setup_devices()
212 {
213  auto sensor_list = device_->getSensorList();
214  for (size_t i = 0; i < sensor_list->count(); i++) {
215  auto sensor = sensor_list->getSensor(i);
216  auto profiles = sensor->getStreamProfileList();
217  for (size_t j = 0; j < profiles->count(); j++) {
218  auto profile = profiles->getProfile(j);
219  stream_index_pair sip{profile->type(), 0};
220  if (sensors_.find(sip) != sensors_.end()) {
221  continue;
222  }
223  sensors_[sip] = sensor;
224  }
225  }
226 
227  auto info = device_->getDeviceInfo();
228 
229  try {
230  if (!depth_work_mode_.empty()) {
231  device_->switchDepthWorkMode(depth_work_mode_.c_str());
232  }
233  if (sync_mode_ != OB_SYNC_MODE_CLOSE) {
234  OBDeviceSyncConfig sync_config;
235  sync_config.syncMode = sync_mode_;
236  sync_config.rgbTriggerSignalInDelay = rgb_trigger_signal_in_delay_;
237  sync_config.deviceTriggerSignalOutDelay = device_trigger_signal_out_delay_;
238  device_->setSyncConfig(sync_config);
239  }
240  if (info->pid() == GEMINI2_PID) {
241  auto default_precision_level = device_->getIntProperty(OB_PROP_DEPTH_PRECISION_LEVEL_INT);
242  if (default_precision_level != depth_precision_) {
243  device_->setIntProperty(OB_PROP_DEPTH_PRECISION_LEVEL_INT, depth_precision_);
244  }
245  }
246 
247  device_->setBoolProperty(OB_PROP_DEPTH_SOFT_FILTER_BOOL, enable_soft_filter_);
248  device_->setBoolProperty(OB_PROP_COLOR_AUTO_EXPOSURE_BOOL, enable_color_auto_exposure_);
249  auto default_soft_filter_max_diff = device_->getIntProperty(OB_PROP_DEPTH_MAX_DIFF_INT);
250  if (soft_filter_max_diff_ != -1 && default_soft_filter_max_diff != soft_filter_max_diff_) {
251  device_->setIntProperty(OB_PROP_DEPTH_MAX_DIFF_INT, soft_filter_max_diff_);
252  }
253  auto default_soft_filter_speckle_size =
254  device_->getIntProperty(OB_PROP_DEPTH_MAX_SPECKLE_SIZE_INT);
255  if (soft_filter_speckle_size_ != -1 &&
256  default_soft_filter_speckle_size != soft_filter_speckle_size_)
257  {
258  device_->setIntProperty(OB_PROP_DEPTH_MAX_SPECKLE_SIZE_INT, soft_filter_speckle_size_);
259  }
260  } catch (const ob::Error & e) {
261  RCLCPP_ERROR_STREAM(logger_, "Failed to setup devices: " << e.getMessage());
262  } catch (const std::exception & e) {
263  RCLCPP_ERROR_STREAM(logger_, "Failed to setup devices: " << e.what());
264  }
265 }

267 void Gemini2Node::setup_profiles()
268 {
269  for (const auto & elem : IMAGE_STREAMS) {
270  if (enable_stream_[elem]) {
271  const auto & sensor = sensors_[elem];
272  CHECK_NOTNULL(sensor.get());
273  auto profiles = sensor->getStreamProfileList();
274  CHECK_NOTNULL(profiles.get());
275  // CHECK(profiles->count() > 0);
276  CHECK_GT(profiles->count(), 0);
277 
278  for (size_t i = 0; i < profiles->count(); i++) {
279  auto profile = profiles->getProfile(i)->as<ob::VideoStreamProfile>();
280  RCLCPP_DEBUG_STREAM(
281  logger_, "Sensor profile: " <<
282  "stream_type: " << magic_enum::enum_name(profile->type()) <<
283  "Format: " << profile->format() << ", Width: " << profile->width() <<
284  ", Height: " << profile->height() << ", FPS: " << profile->fps());
285  supported_profiles_[elem].emplace_back(profile);
286  }
287  std::shared_ptr<ob::VideoStreamProfile> selected_profile;
288  std::shared_ptr<ob::VideoStreamProfile> default_profile;
289  try {
290  selected_profile =
291  profiles->getVideoStreamProfile(width_[elem], height_[elem], format_[elem], fps_[elem]);
292  default_profile =
293  profiles->getVideoStreamProfile(width_[elem], height_[elem], format_[elem]);
294  } catch (const ob::Error & ex) {
295  RCLCPP_ERROR_STREAM(logger_, "Failed to get profile: " << ex.getMessage());
296  RCLCPP_WARN_STREAM(
297  logger_, "Given stream configuration is not supported by the device! " <<
298  " Stream: " << magic_enum::enum_name(elem.first) <<
299  ", Stream Index: " << elem.second <<
300  ", Width: " << width_[elem] <<
301  ", Height: " << height_[elem] << ", FPS: " << fps_[elem] <<
302  ", Format: " << magic_enum::enum_name(format_[elem]));
303  throw;
304  }
305 
306  if (!selected_profile) {
307  RCLCPP_WARN_STREAM(
308  logger_, "Given stream configuration is not supported by the device! " <<
309  " Stream: " << magic_enum::enum_name(elem.first) <<
310  ", Stream Index: " << elem.second <<
311  ", Width: " << width_[elem] <<
312  ", Height: " << height_[elem] << ", FPS: " << fps_[elem] <<
313  ", Format: " << magic_enum::enum_name(format_[elem]));
314  if (default_profile) {
315  RCLCPP_WARN_STREAM(logger_, "Using default profile instead.");
316  RCLCPP_WARN_STREAM(logger_, "default FPS " << default_profile->fps());
317  selected_profile = default_profile;
318  } else {
319  RCLCPP_ERROR_STREAM(
320  logger_, " NO default_profile found , Stream: " << magic_enum::enum_name(elem.first) <<
321  " will be disable");
322  enable_stream_[elem] = false;
323  continue;
324  }
325  }
326  CHECK_NOTNULL(selected_profile);
327  stream_profile_[elem] = selected_profile;
328  images_[elem] =
329  cv::Mat(height_[elem], width_[elem], image_format_[elem], cv::Scalar(0, 0, 0));
330  RCLCPP_INFO_STREAM(
331  logger_, " stream " << stream_name_[elem] << " is enabled - width: " << width_[elem] <<
332  ", height: " << height_[elem] << ", fps: " << fps_[elem] << ", " <<
333  "Format: " << magic_enum::enum_name(selected_profile->format()));
334  }
335  }
336 }

423  sensor_msgs::msg::CameraInfo color_cam_info;
424  std::pair<cv::Mat, sensor_msgs::msg::CameraInfo> color_frame_data;
425

434  std::pair<cv::Mat, sensor_msgs::msg::CameraInfo> depth_frame_data;
435 
436  auto depth_frame = std::dynamic_pointer_cast<ob::Frame>(frame_set->depthFrame());
437  request_img_data_from_frameset(depth_frame, DEPTH, output_depth_img, depth_cam_info);
438  depth_frame_data = std::make_pair(output_depth_img, depth_cam_info);
439 
440  pair_depth_img = std::move(depth_frame_data);
441 }

507 void Gemini2Node::request_img_data_from_frameset(
508  const std::shared_ptr<ob::Frame> & frame,
509  const stream_index_pair & stream_index,
510  cv::Mat & img,
511  sensor_msgs::msg::CameraInfo & cam_info)
512 {
513  if (frame == nullptr) {
514  return;
515  }
516 
517  std::shared_ptr<ob::VideoFrame> video_frame;
518  if (frame->type() == OB_FRAME_COLOR && frame->format() != OB_FORMAT_RGB888) {
519  if (!setup_format_convert_type(frame->format())) {
520  RCLCPP_ERROR(logger_, "Unsupported color format: %d", frame->format());
521  return;
522  }
523  auto color_frame = format_convert_filter_.process(frame);
524  if (color_frame == nullptr) {
525  RCLCPP_ERROR(logger_, "Failed to convert color frame");
526  return;
527  }
528  video_frame = color_frame->as<ob::ColorFrame>();
529  } else if (frame->type() == OB_FRAME_COLOR) {
530  video_frame = frame->as<ob::ColorFrame>();
531  } else if (frame->type() == OB_FRAME_DEPTH) {
532  video_frame = frame->as<ob::DepthFrame>();
533  } else if (frame->type() == OB_FRAME_IR) {
534  video_frame = frame->as<ob::IRFrame>();
535  } else {
536  RCLCPP_ERROR(logger_, "Unsupported frame type: %d", frame->type());
537  return;
538  }
539  if (!video_frame) {
540  RCLCPP_ERROR(logger_, "Failed to convert frame to video frame");
541  return;
542  }
543  int width = static_cast<int>(video_frame->width());
544  int height = static_cast<int>(video_frame->height());
545  auto & image = images_[stream_index];
546  if (image.empty() || image.cols != width || image.rows != height) {
547  image.create(height, width, image_format_[stream_index]);
548  }
549  // image.data = (uchar*)video_frame->data();
550  image.data = reinterpret_cast<uchar *>(video_frame->data());
551 
552  if (stream_index == DEPTH) {
553  auto depth_scale = video_frame->as<ob::DepthFrame>()->getValueScale();
554  image = image * depth_scale;
555  }
556 
557  auto timestamp = frameTimeStampToROSTime(video_frame->systemTimeStamp());
558  if (!camera_param_ && depth_registration_) {
559  camera_param_ = pipeline_->getCameraParam();
560  } else if (!camera_param_ && stream_index == COLOR) {
561  camera_param_ = get_color_camera_param();
562  } else if (!camera_param_ && (stream_index == DEPTH)) {
563  camera_param_ = get_depth_camera_param();
564  }
565  auto & intrinsic =
566  stream_index == COLOR ? camera_param_->rgbIntrinsic : camera_param_->depthIntrinsic;
567  auto & distortion =
568  stream_index == COLOR ? camera_param_->rgbDistortion : camera_param_->depthDistortion;
569  auto camera_info = convertToCameraInfo(intrinsic, distortion, width);
570  img = std::move(image);
571  cam_info = std::move(camera_info);
572 }

602 void Gemini2Node::grab_depth_pointcloud(
603  const std::shared_ptr<ob::FrameSet> & frame_set,
604  sensor_msgs::msg::PointCloud2 & point_cloud_msg)
605 {
606  if (!enable_point_cloud_) {
607  return;
608  }
609  if (!camera_param_ && depth_registration_) {
610  camera_param_ = pipeline_->getCameraParam();
611  } else if (!camera_param_) {
612  camera_param_ = get_depth_camera_param();
613  }
614 
615 
616  point_cloud_filter_.setCameraParam(*camera_param_);
617  point_cloud_filter_.setCreatePointFormat(OB_FORMAT_POINT);
618  auto depth_frame = frame_set->depthFrame();
619  if (!depth_frame) {
620  return;
621  }
622  auto frame = point_cloud_filter_.process(frame_set);
623  if (!frame) {
624  RCLCPP_ERROR_STREAM(logger_, "point cloud filter process failed");
625  return;
626  }
627  size_t point_size = frame->dataSize() / sizeof(OBPoint);
628  // auto* points = (OBPoint*)frame->data();
629  auto * points = reinterpret_cast<OBPoint *>(frame->data());
630 
631  if (!points) {
632  RCLCPP_ERROR_STREAM(logger_, "point cloud data is null");
633  return;
634  }
635  CHECK_NOTNULL(points);
636  sensor_msgs::PointCloud2Modifier modifier(point_cloud_msg);
637  modifier.setPointCloud2FieldsByString(1, "xyz");
638  modifier.resize(point_size);
639  point_cloud_msg.width = depth_frame->width();
640  point_cloud_msg.height = depth_frame->height();
641  point_cloud_msg.row_step = point_cloud_msg.width * point_cloud_msg.point_step;
642  point_cloud_msg.data.resize(point_cloud_msg.height * point_cloud_msg.row_step);
643  sensor_msgs::PointCloud2Iterator<float> iter_x(point_cloud_msg, "x");
644  sensor_msgs::PointCloud2Iterator<float> iter_y(point_cloud_msg, "y");
645  sensor_msgs::PointCloud2Iterator<float> iter_z(point_cloud_msg, "z");
646  size_t valid_count = 0;
647  auto scale = depth_frame->getValueScale();
648  for (size_t point_idx = 0; point_idx < point_size; point_idx++, points++) {
649  bool valid_pixel(points->z > 0);
650  if (valid_pixel) {
651  *iter_x = static_cast<float>((points->x * scale) / 1000.0);
652  *iter_y = -static_cast<float>((points->y * scale) / 1000.0);
653  *iter_z = static_cast<float>((points->z * scale) / 1000.0);
654  ++iter_x;
655  ++iter_y;
656  ++iter_z;
657  ++valid_count;
658  }
659  }
660 
661  point_cloud_msg_.is_dense = true;
662  point_cloud_msg_.width = valid_count;
663  point_cloud_msg_.height = 1;
664  modifier.resize(valid_count);
665 }
