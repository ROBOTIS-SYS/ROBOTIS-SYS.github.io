281 bool PageFloorInfo::is_end_of_action(int action_index)
282 {
283  YAML::Node action_plan_doc = doc_["action_config"];
284  for (const auto & iter : action_plan_doc[action_index]["action"]) {
285  // std::cout << iter.first.as<std::string>() << std::endl;
286  if (iter.first.as<std::string>().compare("end_of_action") == 0) {
287  return true;
288  }
289  }
290  if (static_cast<bool>(action_plan_doc[action_index + 1]) == false) {
291  RCLCPP_ERROR(
292  ros_node_->get_logger(),
293  "index(%d) is not E.O.A. but it is the last action", action_index);
294  return true;
295  }
296  return false;

299 double PageFloorInfo::get_action_time(int action_index)
300 {
301  YAML::Node action_plan_doc = doc_["action_config"];
302 
303  if (action_plan_doc[action_index]) {
304  if (is_end_of_action(action_index)) {
305  RCLCPP_ERROR(ros_node_->get_logger(), "index(%d) is end_of_action", action_index);
306  return 0.0;
307  } else if (static_cast<bool>(action_plan_doc[action_index + 1]) == false) {
308  RCLCPP_ERROR(
309  ros_node_->get_logger(),
310  "next index(%d)(= %d + 1) not exists", action_index + 1, action_index);
311  return 0.0;
312  } else {
313  return action_plan_doc[action_index + 1]["time_sec"].as<double>() -
314  action_plan_doc[action_index]["time_sec"].as<double>();
315  }
316  }
317 }

319 void PageFloorInfo::do_actions(int action_index)
320 {
321  YAML::Node action_plan_doc = doc_["action_config"];
322 
323  for (const auto & it : action_plan_doc[action_index]["action"]) {
324  std::string action_name = it.first.as<std::string>();
325  if (action_name.compare("set_image") == 0) {
326  std::string img_file_name = it.second[ros_node_->curr_language_.c_str()].as<std::string>();
327  ui->lblBG->setPixmap(QPixmap((image_folder_path_ + "/" + img_file_name).c_str()));
328  } else if (action_name.compare("start_mp3") == 0) {
329  std::string mp3_file_name = it.second.as<std::string>();
330  emit playSound(mp3_file_name.c_str());
331  } else if (action_name.compare("print_term") == 0) {
332  std::string contents = it.second.as<std::string>();
333  RCLCPP_INFO(ros_node_->get_logger(), contents);
334  } else if (action_name.compare("do_nothing") == 0) {
335  // RCLCPP_INFO(ros_node_->get_logger, "do nothing action");
336  } else if (action_name.compare("end_of_action") == 0) {
337  // RCLCPP_INFO(ros_node_->get_logger(), "Docent E.O.A");
338  } else if (action_name.compare("sleep") == 0) {
339  rclcpp::sleep_for(
340  std::chrono::milliseconds(static_cast<int>(round(it.second.as<double>() * 1000.0))));
341  } else if (action_name.compare("ask_finish_info_service") == 0) {
342  ask_finish_info_action();
343  } else if (action_name.compare("dialog_with_btn") == 0) {
344  // Get Dialog Name
345  std::string dialog_name = it.second["dialog_option"].as<std::string>();
346  // Get Timeout Sec
347  double timeout_sec = it.second["timeout_sec"].as<double>();
348  // Set Dialog Timeout
349  ros_node_->set_dialog_timeout(timeout_sec * 1000);
350  // Show Dialog
351  emit showDialog(dialog_name);
352  ros_node_->dialog_timeout_check_done();
353  // Dialog Result
354  if (ros_node_->dialog_timeout_result_ == RosNode::DialogTimeoutCode::TIMEOUT) {
355  do_dialog_action(it.second["timed_out"]);
356  }
357  if (ros_node_->dialog_result_ == QDialog::Accepted) {
358  do_dialog_action(it.second["accepted"]);
359  } else if (ros_node_->dialog_result_ == QDialog::Rejected) {
360  do_dialog_action(it.second["rejected"]);
361  }
362  }
363  }
364 }
