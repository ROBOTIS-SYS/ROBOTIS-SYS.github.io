80 // trim from start (in place)
81 static inline void ltrim(std::string & s)
82 {
83  s.erase(s.begin(), std::find_if(s.begin(), s.end(), [](int ch) {return !std::isspace(ch);}));
84 }
85 
86 // trim from end (in place)
87 static inline void rtrim(std::string & s)
88 {
89  s.erase(
90  std::find_if(s.rbegin(), s.rend(), [](int ch) {return !std::isspace(ch);}).base(), s.end());
91 }
92 
93 // trim from both ends (in place)
94 static inline void trim(std::string & s)
95 {
96  ltrim(s);
97  rtrim(s);
98 }
