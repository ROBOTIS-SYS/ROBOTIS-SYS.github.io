168  match_t find_longest_match(size_t a_low, size_t a_high, size_t b_low, size_t b_high) {
169  using std::begin;
170  using std::end;
171 
172 
173  size_t best_i = a_low;
174  size_t best_j = b_low;
175  size_t best_size = 0;
176 
177  // Find longest junk free match
178  {
179  j2_values_to_erase_.clear();
180  for(size_t i = a_low; i < a_high; ++i) {
181  j2_values_to_affect_.clear();
182 
183  for(size_t j : b2j_[a_[i]]) {
184  if (j < b_low) continue;
185  if (j >= b_high) break;
186  size_t k = j2len_[j] + 1;
187  j2_values_to_affect_.emplace_back(j + 1, k);
188  if (k > best_size) {
189  best_i = i-k+1;
190  best_j = j-k+1;
191  best_size = k;
192  }
193  }
194 
195  for(auto const& elem : j2_values_to_erase_) {
196  j2len_[elem.first] = 0;
197  }
198  for(auto const& elem : j2_values_to_affect_) {
199  j2len_[elem.first] = elem.second;
200  }
201  std::swap(j2_values_to_erase_, j2_values_to_affect_);
202  }
203  for(auto const& elem : j2_values_to_erase_) {
204  j2len_[elem.first] = 0;
205  }
206  }
207 
208  // Utility lambdas for factoring
209  auto low_bound_expand = [&best_i, &best_j, a_low, b_low, &best_size, this] (bool isjunk) {
210  while (
211  best_i > a_low
212  && best_j > b_low
213  && this->a_[best_i-1] == this->b_[best_j-1]
214  && isjunk == b2j_.count(b_[best_j-1])
215  ) {
216  --best_i; --best_j; ++best_size;
217  }
218  };
219 
220  // References on best_i best_k are needed here even if modified inside the code
221  // because modified betweent the calls
222  auto high_bound_expand = [&best_i, &best_j, a_high, b_high, &best_size, this] (bool isjunk) {
223  while (
224  (best_i+best_size) < a_high
225  && (best_j+best_size) < b_high
226  && this->a_[best_i+best_size] == this->b_[best_j+best_size]
227  && isjunk == b2j_.count(b_[best_j + best_size])
228  ) {
229  ++best_size;
230  }
231  };
232 
233  // Extend the best by non-junk elements on each end
234  low_bound_expand(false);
235  high_bound_expand(false);
236 
237  // Adds matching junks at each end
238  low_bound_expand(true);
239  high_bound_expand(true);
240 
241  return make_tuple(best_i, best_j, best_size);
242  }

244  match_list_t get_matching_blocks() {
245  // The following are tuple extracting aliases
246  using std::get;
247 
248  if (matching_blocks_)
249  return *matching_blocks_;
250 
251  vector<tuple<size_t, size_t, size_t, size_t>> queue;
252  vector<match_t> matching_blocks_pass1;
253 
254  std::size_t queue_head = 0;
255  queue.reserve(std::min(a_.size(), b_.size()));
256  queue.emplace_back(0, a_.size(), 0, b_.size());
257 
258  while(queue_head < queue.size()) {
259  size_t a_low, a_high, b_low, b_high;
260  tie(a_low, a_high, b_low, b_high) = queue[queue_head++];
261  match_t m = find_longest_match(a_low, a_high, b_low, b_high);
262  if (get<2>(m)) {
263  if (a_low < get<0>(m) && b_low < get<1>(m)) {
264  queue.emplace_back(a_low, get<0>(m), b_low, get<1>(m));
265  }
266  if ((get<0>(m)+get<2>(m)) < a_high && (get<1>(m)+get<2>(m)) < b_high) {
267  queue.emplace_back(get<0>(m)+get<2>(m), a_high, get<1>(m)+get<2>(m), b_high);
268  }
269  matching_blocks_pass1.push_back(std::move(m));
270  }
271  }
272  std::sort(std::begin(matching_blocks_pass1), end(matching_blocks_pass1));
273 
274  matching_blocks_.reset(new match_list_t);
275  matching_blocks_->reserve(matching_blocks_pass1.size());
276 
277  size_t i1, j1, k1;
278  i1 = j1 = k1 = 0;
279 
280  for(match_t const& m : matching_blocks_pass1) {
281  if (i1 + k1 == get<0>(m) && j1 + k1 == get<1>(m)) {
282  k1 += get<2>(m);
283  } else {
284  if (k1) {
285  matching_blocks_->emplace_back(i1, j1, k1);
286  }
287  std::tie(i1, j1, k1) = m;
288  }
289  }
290  if (k1) {
291  matching_blocks_->emplace_back(i1, j1, k1);
292  }
293  matching_blocks_->emplace_back(a_.size(), b_.size(), 0);
294 
295  return *matching_blocks_;
296  }

298  chunk_list_t get_opcodes() {
299  using std::get;
300 
301  if (opcodes_)
302  return *opcodes_;
303 
304  size_t i = 0;
305  size_t j = 0;
306  opcodes_.reset(new chunk_list_t);
307  for (match_t const & m : get_matching_blocks()) {
308  size_t ai, bj, size;
309  std::tie(ai, bj, size) = m;
310 
311  // invariant: we've pumped out correct diffs to change
312  // a[:i] into b[:j], and the next matching block is
313  // a[ai:ai+size] == b[bj:bj+size]. So we need to pump
314  // out a diff to change a[i:ai] into b[j:bj], pump out
315  // the matching block, and move (i,j) beyond the match
316  std::string tag;
317  if (i < ai && j < bj) {
318  tag = "replace";
319  } else if (i < ai) {
320  tag = "delete";
321  } else if (j < bj) {
322  tag = "insert";
323  }
324 
325  if (!tag.empty()) {
326  opcodes_->emplace_back(tag, i, ai, j, bj);
327  }
328 
329  i = ai+size;
330  j = bj+size;
331 
332  // the list of matching blocks is terminated by a
333  // sentinel with size 0
334  if (size) {
335  opcodes_->emplace_back("equal", ai, i, bj, j);
336  }
337  }
338 
339  return *opcodes_;
340  }

353  void chain_b() {
354  size_t index = 0;
355 
356  // Counting occurences
357  b2j_.clear();
358  for (hashable_type const& elem : b_) {
359  b2j_[elem].push_back(index++);
360  }
361 
362  // Purge junk elements
363  junk_set_.clear();
364  if (is_junk_) {
365  for (auto it = b2j_.begin(); it != b2j_.end();) {
366  if (is_junk_(it->first)) {
367  junk_set_.insert(it->first);
368  it = b2j_.erase(it);
369  } else {
370  ++it;
371  }
372  }
373  }
374 
375  // Purge popular elements that are not junk
376  popular_set_.clear();
377  if (auto_junk_ && auto_junk_minsize_ <= b_.size()) {
378  size_t ntest = b_.size() / 100 + 1;
379  for (auto it = b2j_.begin(); it != b2j_.end();) {
380  if (ntest < it->second.size()) {
381  popular_set_.insert(it->first);
382  it = b2j_.erase(it);
383  } else {
384  ++it;
385  }
386  }
387  }
388  }
389 
390  bool auto_junk_ = true;
391  std::size_t auto_junk_minsize_ = 200u;
392  b2j_t b2j_;
393  junk_set_t junk_set_;
394  junk_set_t popular_set_;
395 
396  // Cache to avoid reallocations
397  std::vector<size_t> j2len_;
398  std::vector<std::pair<size_t, size_t>> j2_values_to_affect_;
399  std::vector<std::pair<size_t, size_t>> j2_values_to_erase_;
400 };
