101 void ClearCostmapService::clearExceptRegion(const double reset_distance)
102 {
103  double x, y;
104 
105  if (!getPosition(x, y)) {
106  RCLCPP_ERROR(node_->get_logger(), "Cannot clear map because robot pose cannot be retrieved.");
107  return;
108  }
109 
110  auto layers = costmap_.getLayeredCostmap()->getPlugins();
111 
112  for (auto & layer : *layers) {
113  if (isClearable(getLayerName(*layer))) {
114  auto costmap_layer = std::static_pointer_cast<CostmapLayer>(layer);
115  clearLayerExceptRegion(costmap_layer, x, y, reset_distance);
116  }
117  }
118 }

120 void ClearCostmapService::clearAroundRobot(double window_size_x, double window_size_y)
121 {
122  double pose_x, pose_y;
123 
124  if (!getPosition(pose_x, pose_y)) {
125  RCLCPP_ERROR(node_->get_logger(), "Cannot clear map because robot pose cannot be retrieved.");
126  return;
127  }
128 
129  std::vector<geometry_msgs::msg::Point> clear_poly;
130  geometry_msgs::msg::Point pt;
131 
132  pt.x = pose_x - window_size_x / 2;
133  pt.y = pose_y - window_size_y / 2;
134  clear_poly.push_back(pt);
135 
136  pt.x = pose_x + window_size_x / 2;
137  pt.y = pose_y - window_size_y / 2;
138  clear_poly.push_back(pt);
139 
140  pt.x = pose_x + window_size_x / 2;
141  pt.y = pose_y + window_size_y / 2;
142  clear_poly.push_back(pt);
143 
144  pt.x = pose_x - window_size_x / 2;
145  pt.y = pose_y + window_size_y / 2;
146  clear_poly.push_back(pt);
147 
148  costmap_.getCostmap()->setConvexPolygonCost(clear_poly, reset_value_);
149 }

162 void ClearCostmapService::clearLayerExceptRegion(
163  shared_ptr<CostmapLayer> & costmap, double pose_x, double pose_y, double reset_distance)
164 {
165  std::unique_lock<Costmap2D::mutex_t> lock(*(costmap->getMutex()));
166 
167  double start_point_x = pose_x - reset_distance / 2;
168  double start_point_y = pose_y - reset_distance / 2;
169  double end_point_x = start_point_x + reset_distance;
170  double end_point_y = start_point_y + reset_distance;
171 
172  int start_x, start_y, end_x, end_y;
173  costmap->worldToMapNoBounds(start_point_x, start_point_y, start_x, start_y);
174  costmap->worldToMapNoBounds(end_point_x, end_point_y, end_x, end_y);
175 
176  unsigned int size_x = costmap->getSizeInCellsX();
177  unsigned int size_y = costmap->getSizeInCellsY();
178 
179  // Clearing the four rectangular regions around the one we want to keep
180  // top region
181  costmap->resetMapToValue(0, 0, size_x, start_y, reset_value_);
182  // left region
183  costmap->resetMapToValue(0, start_y, start_x, end_y, reset_value_);
184  // right region
185  costmap->resetMapToValue(end_x, start_y, size_x, end_y, reset_value_);
186  // bottom region
187  costmap->resetMapToValue(0, end_y, size_x, size_y, reset_value_);
188 
189  double ox = costmap->getOriginX(), oy = costmap->getOriginY();
190  double width = costmap->getSizeInMetersX(), height = costmap->getSizeInMetersY();
191  costmap->addExtraBounds(ox, oy, ox + width, oy + height);
192 }

194 bool ClearCostmapService::getPosition(double & x, double & y) const
195 {
196  geometry_msgs::msg::PoseStamped pose;
197  if (!costmap_.getRobotPose(pose)) {
198  return false;
199  }
200 
201  x = pose.pose.position.x;
202  y = pose.pose.position.y;
203 
204  return true;
205 }
