64 void Gemini2::init_device_params()
65 {
66  auto log_level_str = declare_parameter<std::string>("log_level", "none");
67  auto log_level = log_serverity_from_string(log_level_str);
68  ob::Context::setLoggerSeverity(log_level);
69  is_alive_.store(true);
70  parameters_ = std::make_shared<Parameters>(this);
71 
72  node_name_ = this->get_name();
73 
74  std::string charater_device =
75  std::string("/sys/devices/platform/3610000.xhci/usb2/2-3/");
76  charater_device.append(camera_uid_);
77 
78  std::cout << " character device : " << charater_device << std::endl;
79  camera_charater_device_ = charater_device;
80 
81  uint32_t port_num;
82 
83  if (node_name_ == "cliff_gemini2") {
84  port_num = GEMINI::CLIFF_CAM;
85  } else if (node_name_ == "front_gemini2") {
86  port_num = GEMINI::FRONT_CAM;
87  }
88 
89  while (true) {
90  // std::cout << " file check " << std::filesystem::exists(charater_device) << std::endl;
91  if (std::filesystem::exists(charater_device)) {
92  charater_device.append("/serial");
93  std::ifstream file(charater_device);
94 
95  if (file.is_open()) {
96  std::getline(file, serial_number_);
97  }
98  file.close();
99 
100  RCLCPP_INFO_STREAM(
101  this->get_logger(),
102  "UID : " << camera_uid_ << ", " << "Serial : " << serial_number_);
103  break;
104  } else {
105  RCLCPP_ERROR_STREAM(this->get_logger(), " " << charater_device << " file not found. ");
106 
107  UsbHubPortName_t port_name;
108 
109  port_name = (UsbHubPortName_t)port_num;
110 
111  int ret = UsbHubPowerControl::get_instance()->usb_hub_set_power(port_name, POWER_OFF);
112 
113  if (ret >= 0) {
114  RCLCPP_INFO_STREAM(this->get_logger(), "Usb Hub Power Off Success");
115  } else {
116  RCLCPP_ERROR_STREAM(this->get_logger(), "Usb Hub Power Off Fail");
117  }
118 
119  rclcpp::sleep_for(std::chrono::milliseconds(500));
120 
121  ret = UsbHubPowerControl::get_instance()->usb_hub_set_power(port_name, POWER_ON);
122 
123  if (ret >= 0) {
124  RCLCPP_INFO_STREAM(this->get_logger(), "Usb Hub Power On Success");
125  } else {
126  RCLCPP_ERROR_STREAM(this->get_logger(), "Usb Hub Power On Fail");
127  }
128 
129  rclcpp::sleep_for(std::chrono::milliseconds(3000));
130  RCLCPP_ERROR_STREAM(this->get_logger(), " Usb Port Off and On. ");
131  }
132  }
133 
134  get_param(
135  "gem_cam_params.device_num",
136  device_num_, 0);
137 
138  check_connect_state_timer_ =
139  this->create_wall_timer(
140  std::chrono::milliseconds(1000),
141  [this]() {check_connect_state_timer();});
142  CHECK_NOTNULL(check_connect_state_timer_);
143 
144  query_thread_ = std::make_shared<std::thread>([this]() {query_device();});
145  device_count_update_thread_ = std::make_shared<std::thread>([this]() {device_count_update();});
146  CHECK_NOTNULL(device_count_update_thread_);
147 }

239  RCLCPP_INFO_STREAM_THROTTLE(logger_, *get_clock(), 1000, "semaphore value: " << sem_value);

247 void Gemini2::update_connected_device_count(
248  int & num_devices_connected,
249  DeviceConnectionEvent connection_event)
250 {
251  // write connected device count to file
252  int shm_id = shmget(DEFAULT_SEM_KEY, 1, 0666 | IPC_CREAT);
253  if (shm_id == -1) {
254  RCLCPP_INFO_STREAM(logger_, "Failed to create shared memory " << strerror(errno));
255  return;
256  }
257  // auto shm_ptr = (int *)shmat(shm_id, nullptr, 0);
258  auto shm_ptr = reinterpret_cast<int *>(shmat(shm_id, nullptr, 0));
259  if (reinterpret_cast<std::intptr_t>(shm_ptr) == -1) {
260  RCLCPP_INFO_STREAM(logger_, "Failed to attach shared memory " << strerror(errno));
261  return;
262  }
263  if (connection_event == DeviceConnectionEvent::kDeviceConnected) {
264  num_devices_connected = *shm_ptr + 1;
265  } else if (connection_event == DeviceConnectionEvent::kDeviceDisconnected && *shm_ptr > 0) {
266  num_devices_connected = *shm_ptr - 1;
267  } else {
268  num_devices_connected = *shm_ptr;
269  }
270  RCLCPP_DEBUG_STREAM_THROTTLE(
271  logger_, *get_clock(), 5000,
272  "Current connected device " << num_devices_connected);
273  *shm_ptr = static_cast<int>(num_devices_connected);
274  shmdt(shm_ptr);
275  if (connection_event == DeviceConnectionEvent::kDeviceDisconnected &&
276  num_devices_connected == 0)
277  {
278  shmctl(shm_id, IPC_RMID, nullptr);
279  sem_unlink(DEFAULT_SEM_NAME);
280  }
281 }

283 std::shared_ptr<ob::Device> Gemini2::select_device(
284  const std::shared_ptr<ob::DeviceList> & list)
285 {
286  if (device_num_ == 1) {
287  RCLCPP_INFO_STREAM(logger_, "Connecting to the default device");
288  return list->getDevice(0);
289  }
290  sem_t * device_sem = sem_open(DEFAULT_SEM_NAME, O_CREAT, 0644, 1);
291  if (device_sem == SEM_FAILED) {
292  RCLCPP_INFO_STREAM(logger_, "Failed to open semaphore");
293  return nullptr;
294  }
295  RCLCPP_INFO_STREAM_THROTTLE(
296  logger_, *get_clock(), 1000,
297  "Connecting to device with serial number: " << serial_number_);
298  int sem_value = 0;
299  sem_getvalue(device_sem, &sem_value);
300  RCLCPP_INFO_STREAM_THROTTLE(logger_, *get_clock(), 1000, "semaphore value: " << sem_value);
301  int ret = sem_wait(device_sem);
302  if (ret != 0) {
303  RCLCPP_ERROR_STREAM(logger_, "Failed to wait semaphore " << strerror(errno));
304  release_device_semaphore(device_sem, num_devices_connected_);
305  return nullptr;
306  }
307  auto device = select_device_by_serial_number(list, serial_number_);
308  std::shared_ptr<int> sem_guard(nullptr, [&, device](int const *) {
309  auto connect_event = device != nullptr ? DeviceConnectionEvent::kDeviceConnected :
310  DeviceConnectionEvent::kOtherDeviceConnected;
311  update_connected_device_count(num_devices_connected_, connect_event);
312 
313  release_device_semaphore(device_sem, num_devices_connected_);
314  });
315  if (device == nullptr) {
316  RCLCPP_WARN_THROTTLE(
317  logger_, *get_clock(), 1000, "Device with serial number %s not found",
318  serial_number_.c_str());
319  device_connected_ = false;
320  return nullptr;
321  }
322  return device;
323 }

325 std::shared_ptr<ob::Device> Gemini2::select_device_by_serial_number(
326  const std::shared_ptr<ob::DeviceList> & list, const std::string & serial_number)
327 {
328  std::string lower_sn;
329  std::transform(
330  serial_number.begin(), serial_number.end(), std::back_inserter(lower_sn),
331  [](auto ch) {return isalpha(ch) ? tolower(ch) : static_cast<int>(ch);});
332 
333  for (size_t i = 0; i < list->deviceCount(); i++) {
334  try {
335  auto pid = list->pid(i);
336  if (isOpenNIDevice(pid)) {
337  // openNI device
338  auto device = list->getDevice(i);
339  auto device_info = device->getDeviceInfo();
340  if (device_info->serialNumber() == serial_number) {
341  RCLCPP_INFO_STREAM(
342  logger_,
343  "Device serial number " << device_info->serialNumber() << " matched");
344  return device;
345  }
346  } else {
347  std::string sn = list->serialNumber(i);
348  RCLCPP_INFO_STREAM_THROTTLE(logger_, *get_clock(), 1000, "Device serial number: " << sn);
349  if (sn == serial_number) {
350  RCLCPP_INFO_STREAM(logger_, "Device serial number " << sn << " matched");
351  return list->getDevice(i);
352  }
353  }
354  } catch (ob::Error & e) {
355  RCLCPP_ERROR_STREAM_THROTTLE(
356  logger_, *get_clock(), 1000,
357  "Failed to get device info " << e.getMessage());
358  } catch (std::exception & e) {
359  RCLCPP_ERROR_STREAM(logger_, "Failed to get device info " << e.what());
360  } catch (...) {
361  RCLCPP_ERROR_STREAM(logger_, "Failed to get device info");
362  }
363  }
364  return nullptr;
365 }

383  RCLCPP_INFO_STREAM(logger_, "Hardware version: " << device_info_->hardwareVersion());
384  RCLCPP_INFO_STREAM(logger_, "device type: " << ObDeviceTypeToString(device_info_->deviceType()));

389 void Gemini2::start_device(const std::shared_ptr<ob::DeviceList> & list)
390 {
391  std::scoped_lock<decltype(device_lock_)> lock(device_lock_);
392  if (device_connected_) {
393  return;
394  }
395  if (list->deviceCount() == 0) {
396  RCLCPP_WARN(logger_, "No device found");
397  return;
398  }
399  if (device_) {
400  device_.reset();
401  }
402  auto device = select_device(list);
403  if (device == nullptr) {
404  this->fms_notifyer();
405  device_connected_ = false;
406  return;
407  }
408  initialize_device(device);
409 }
