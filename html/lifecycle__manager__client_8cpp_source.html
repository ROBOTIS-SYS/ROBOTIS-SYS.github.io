72 SystemStatus
73 LifecycleManagerClient::is_active(const std::chrono::nanoseconds timeout)
74 {
75  auto request = std::make_shared<std_srvs::srv::Trigger::Request>();
76 
77  RCLCPP_INFO(
78  node_->get_logger(), "Waiting for the %s service...",
79  active_service_name_.c_str());
80 
81  if (!is_active_client_->wait_for_service(timeout)) {
82  return SystemStatus::TIMEOUT;
83  }
84 
85  RCLCPP_INFO(
86  node_->get_logger(), "Sending %s request",
87  active_service_name_.c_str());
88  auto future_result = is_active_client_->async_send_request(request);
89 
90  if (rclcpp::spin_until_future_complete(node_, future_result, timeout) !=
91  rclcpp::FutureReturnCode::SUCCESS)
92  {
93  return SystemStatus::TIMEOUT;
94  }
95 
96  if (future_result.get()->success) {
97  return SystemStatus::ACTIVE;
98  } else {
99  return SystemStatus::INACTIVE;
100  }
101 }

103 bool
104 LifecycleManagerClient::callService(uint8_t command, const std::chrono::nanoseconds timeout)
105 {
106  auto request = std::make_shared<ManageLifecycleNodes::Request>();
107  request->command = command;
108 
109  RCLCPP_INFO(
110  node_->get_logger(), "Waiting for the %s service...",
111  manage_service_name_.c_str());
112 
113  while (!manager_client_->wait_for_service(std::chrono::seconds(1))) {
114  if (!rclcpp::ok()) {
115  RCLCPP_ERROR(node_->get_logger(), "Client interrupted while waiting for service to appear");
116  return false;
117  }
118  RCLCPP_INFO(node_->get_logger(), "Waiting for service to appear...");
119  }
120 
121  RCLCPP_INFO(
122  node_->get_logger(), "Sending %s request",
123  manage_service_name_.c_str());
124  auto future_result = manager_client_->async_send_request(request);
125 
126  if (rclcpp::spin_until_future_complete(node_, future_result, timeout) !=
127  rclcpp::FutureReturnCode::SUCCESS)
128  {
129  return false;
130  }
131 
132  return future_result.get()->success;
133 }
