107 bool GetDiffPose2DAction::get_diff_pose(
108  const geometry_msgs::msg::Pose2D & target, const std::string & map_frame_id,
109  const std::string & robot_frame_id)
110 {
111  try {
112  geometry_msgs::msg::TransformStamped base_link_transform;
113 
114  base_link_transform = tf_buffer_->lookupTransform(
115  robot_frame_id, map_frame_id, rclcpp::Time(0), rclcpp::Duration::from_seconds(0.5));
116 
117  auto dur = node_->get_clock()->now() - base_link_transform.header.stamp;
118  RCLCPP_WARN_STREAM(
119  get_logger(), "[" << name() << "] Transform diff : " << dur.seconds() << "s");
120 
121  if (isHaltRequested()) {
122  return false;
123  }
124 
125  geometry_msgs::msg::Pose current_pose;
126  current_pose.position.x = base_link_transform.transform.translation.x;
127  current_pose.position.y = base_link_transform.transform.translation.y;
128  current_pose.position.z = base_link_transform.transform.translation.z;
129  current_pose.orientation = base_link_transform.transform.rotation;
130 
131  std::cout << "----------------" << std::endl;
132  std::cout << "current x : " << current_pose.position.x <<
133  ", y : " << current_pose.position.y <<
134  ", theta : " << (tf2::getYaw(current_pose.orientation) * 180.0 / M_PI) <<
135  std::endl;
136 
137  // set before pose
138  geometry_msgs::msg::PoseStamped before_pose, after_pose;
139  before_pose.pose.position.x = target.x;
140  before_pose.pose.position.y = target.y;
141  // orientation
142  tf2::Quaternion before_q;
143  before_q.setRPY(0.0, 0.0, target.theta);
144  tf2::convert(before_q, before_pose.pose.orientation);
145 
146  // transform
147  tf2::doTransform(before_pose, after_pose, base_link_transform);
148 
149  // set result to pose2d
150  double diff_x = after_pose.pose.position.x;
151  double diff_y = after_pose.pose.position.y;
152  double diff_theta = tf2::getYaw(after_pose.pose.orientation);
153 
154  setOutput<double>("diff_x", diff_x);
155  setOutput<double>("diff_y", diff_y);
156  setOutput<double>("diff_theta", diff_theta);
157  } catch (tf2::TransformException & ex) {
158  RCLCPP_ERROR(get_logger(), "Failed to get the current pose or to calc diff");
159 
160  return false;
161  }
162 
163  return true;
164 }
