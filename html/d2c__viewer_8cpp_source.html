52 void D2CViewer::messageCallback(
53  const sensor_msgs::msg::Image::ConstSharedPtr & rgb_msg,
54  const sensor_msgs::msg::Image::ConstSharedPtr & depth_msg)
55 {
56  if (rgb_msg->width != depth_msg->width || rgb_msg->height != depth_msg->height) {
57  RCLCPP_ERROR(
58  logger_, "rgb and depth image size not match(%d, %d) vs (%d, %d)",
59  rgb_msg->width, rgb_msg->height, depth_msg->width, depth_msg->height);
60  return;
61  }
62  auto rgb_img_ptr = cv_bridge::toCvCopy(rgb_msg, sensor_msgs::image_encodings::RGB8);
63  auto depth_img_ptr = cv_bridge::toCvCopy(depth_msg, sensor_msgs::image_encodings::TYPE_16UC1);
64  cv::Mat gray_depth, depth_img, d2c_img;
65  depth_img_ptr->image.convertTo(gray_depth, CV_8UC1);
66  cv::cvtColor(gray_depth, depth_img, cv::COLOR_GRAY2RGB);
67  depth_img.setTo(cv::Scalar(255, 255, 0), depth_img);
68  cv::bitwise_or(rgb_img_ptr->image, depth_img, d2c_img);
69  sensor_msgs::msg::Image::SharedPtr d2c_msg =
70  cv_bridge::CvImage(
71  std_msgs::msg::Header(),
72  sensor_msgs::image_encodings::RGB8, d2c_img).toImageMsg();
73 
74  d2c_msg->header = rgb_msg->header;
75  d2c_viewer_pub_->publish(*d2c_msg);
76 }
