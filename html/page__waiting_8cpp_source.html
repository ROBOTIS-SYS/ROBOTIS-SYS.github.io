178 void PageWaiting::setOption(std::string option)
179 {
180  option = QString(option.c_str()).split("|")[0].toStdString();
181  if (option.compare("Unreceived") == 0 || option.compare("unreceived") == 0) {
182  emit showDialog("Unreceived");
183 
184  if (ros_node_->dialog_result_ == QDialog::Accepted) {
185  emit pageChanged("InputPassNumber", "Waiting|Arrived_0|Unreceived");
186  movie->stop();
187  }
188 
189  return;
190  }
191 
192  if (option.compare("recovery") != 0) {
193  return;
194  }
195 
196  emit showDialog("ResumeService");
197 
198  if (ros_node_->dialog_result_ == QDialog::Accepted) {
199  // 이어 하기 RESUME
200  std_msgs::msg::Int8 msg;
201  msg.data = task_manager_interfaces::msg::Task::RESUME;
202  ros_node_->command_task_pub_->publish(msg);
203  } else if (ros_node_->dialog_result_ == QDialog::Rejected) {
204  // 모두 취소 STOP_ALL && START
205  std_msgs::msg::Int8 msg;
206  msg.data = task_manager_interfaces::msg::Task::STOP_ALL;
207  ros_node_->command_task_pub_->publish(msg);
208 
209  rclcpp::sleep_for(std::chrono::milliseconds(100));
210 
211  msg.data = task_manager_interfaces::msg::Task::START;
212  ros_node_->command_task_pub_->publish(msg);
213  }
214 }
