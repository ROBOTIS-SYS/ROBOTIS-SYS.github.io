34 LifecycleManager::LifecycleManager()
35 : Node("antis_nav_lifecycle_manager")
36 {
37  RCLCPP_INFO(get_logger(), "Creating");
38 
39  // The list of names is parameterized, allowing this module to be used with a different set
40  // of nodes
41  declare_parameter("node_names");
42  declare_parameter("autostart", rclcpp::ParameterValue(false));
43 
44  node_names_ = get_parameter("node_names").as_string_array();
45  get_parameter("autostart", autostart_);
46 
47  manager_srv_ = create_service<ManageLifecycleNodes>(
48  get_name() + std::string("/manage_nodes"),
49  std::bind(&LifecycleManager::managerCallback, this, _1, _2, _3));
50 
51  is_active_srv_ = create_service<std_srvs::srv::Trigger>(
52  get_name() + std::string("/is_active"),
53  std::bind(&LifecycleManager::isActiveCallback, this, _1, _2, _3));
54 
55  auto options = rclcpp::NodeOptions().arguments(
56  {"--ros-args", "-r", std::string("__node:=") + get_name() + "_service_client", "--"});
57  service_client_node_ = std::make_shared<rclcpp::Node>("_", options);
58 
59  transition_state_map_[Transition::TRANSITION_CONFIGURE] = State::PRIMARY_STATE_INACTIVE;
60  transition_state_map_[Transition::TRANSITION_CLEANUP] = State::PRIMARY_STATE_UNCONFIGURED;
61  transition_state_map_[Transition::TRANSITION_ACTIVATE] = State::PRIMARY_STATE_ACTIVE;
62  transition_state_map_[Transition::TRANSITION_DEACTIVATE] = State::PRIMARY_STATE_INACTIVE;
63  transition_state_map_[Transition::TRANSITION_UNCONFIGURED_SHUTDOWN] =
64  State::PRIMARY_STATE_FINALIZED;
65 
66  transition_label_map_[Transition::TRANSITION_CONFIGURE] = std::string("Configuring ");
67  transition_label_map_[Transition::TRANSITION_CLEANUP] = std::string("Cleaning up ");
68  transition_label_map_[Transition::TRANSITION_ACTIVATE] = std::string("Activating ");
69  transition_label_map_[Transition::TRANSITION_DEACTIVATE] = std::string("Deactivating ");
70  transition_label_map_[Transition::TRANSITION_UNCONFIGURED_SHUTDOWN] =
71  std::string("Shutting down ");
72 
73  createLifecycleServiceClients();
74 
75  if (autostart_) {
76  startup();
77  }
78 }

85 void
86 LifecycleManager::managerCallback(

88  const std::shared_ptr<ManageLifecycleNodes::Request> request,
89  std::shared_ptr<ManageLifecycleNodes::Response> response)

91  switch (request->command) {

101  case ManageLifecycleNodes::Request::PAUSE:
102  response->success = pause();
103  break;
104  case ManageLifecycleNodes::Request::RESUME:
105  response->success = resume();
106  break;
107  }

138 bool
139 LifecycleManager::changeStateForNode(const std::string & node_name, std::uint8_t transition)
140 {
141  message(transition_label_map_[transition] + node_name);
142  if (!node_map_[node_name]->change_state(transition) ||
143  !(node_map_[node_name]->get_state() == transition_state_map_[transition]))
144  {
145  RCLCPP_ERROR(get_logger(), "Failed to change state for node: %s", node_name.c_str());
146  return false;
147  }
148 
149  return true;
150 }

152 bool
153 LifecycleManager::changeStateForAllNodes(std::uint8_t transition)
154 {
155  if (transition == Transition::TRANSITION_CONFIGURE ||
156  transition == Transition::TRANSITION_ACTIVATE)
157  {
158  for (auto & node_name : node_names_) {
159  if (!changeStateForNode(node_name, transition)) {
160  return false;
161  }
162  }
163  } else {
164  std::vector<std::string>::reverse_iterator rit;
165  for (rit = node_names_.rbegin(); rit != node_names_.rend(); ++rit) {
166  if (!changeStateForNode(*rit, transition)) {
167  return false;
168  }
169  }
170  }
171  return true;
172 }

174 void
175 LifecycleManager::shutdownAllNodes()

183 bool
184 LifecycleManager::startup()
185 {
186  message("Starting managed nodes bringup...");
187  if (!changeStateForAllNodes(Transition::TRANSITION_CONFIGURE) ||
188  !changeStateForAllNodes(Transition::TRANSITION_ACTIVATE))
189  {
190  RCLCPP_ERROR(get_logger(), "Failed to bring up all requested nodes. Aborting bringup.");
191  return false;
192  }
193  message("Managed nodes are active");
194  system_active_ = true;
195  return true;
196 }

209 bool
210 LifecycleManager::reset()
211 {
212  message("Resetting managed nodes...");
213  // Should transition in reverse order
214  if (!changeStateForAllNodes(Transition::TRANSITION_DEACTIVATE) ||
215  !changeStateForAllNodes(Transition::TRANSITION_CLEANUP))
216  {
217  RCLCPP_ERROR(get_logger(), "Failed to reset nodes: aborting reset");
218  return false;
219  }
220  message("Managed nodes have been reset");
221  system_active_ = false;
222  return true;
223 }

225 bool
226 LifecycleManager::pause()
227 {
228  message("Pausing managed nodes...");
229  if (!changeStateForAllNodes(Transition::TRANSITION_DEACTIVATE)) {
230  RCLCPP_ERROR(get_logger(), "Failed to pause nodes: aborting pause");
231  return false;
232  }
233  message("Managed nodes have been paused");
234  system_active_ = false;

236 }

238 bool
239 LifecycleManager::resume()
240 {
241  message("Resuming managed nodes...");
242  if (!changeStateForAllNodes(Transition::TRANSITION_ACTIVATE)) {
243  RCLCPP_ERROR(get_logger(), "Failed to resume nodes: aborting resume");
244  return false;
245  }
246  message("Managed nodes are active");
247  system_active_ = true;
248  return true;
