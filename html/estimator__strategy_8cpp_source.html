26 Eigen::MatrixXf EstimatorStrategy::triangulate(
27  RecognizedText left_button,
28  RecognizedText right_button)
29 {
30  CameraGeometry * geo = CameraGeometry::get_instance();
31  Eigen::MatrixXf result(left_button.features.size(), 3);
32 
33  for (size_t i = 0; i < left_button.features.size(); i++) {
34  Eigen::Vector2f left_pt(left_button.features[i].x, left_button.features[i].y);
35  Eigen::Vector2f right_pt(right_button.features[i].x, right_button.features[i].y);
36 
37  result.row(i) =
38  geo->get_cam_to_opt() * geo->reconstruct(
39  geo->get_cam_f(CameraGeometry::left_zed),
40  geo->get_cam_base_line(CameraGeometry::right_to_left_zed),
41  geo->get_cam_ppx(CameraGeometry::left_zed),
42  geo->get_cam_ppy(CameraGeometry::left_zed),
43  left_pt, right_pt);
44  }
45 
46  return result;
47 }
