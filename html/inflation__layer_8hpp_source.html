103  inline unsigned char computeCost(double distance) const
104  {
105  unsigned char cost = 0;
106  if (distance == 0) {
107  cost = LETHAL_OBSTACLE;
108  } else if (distance * resolution_ <= inscribed_radius_) {
109  cost = INSCRIBED_INFLATED_OBSTACLE;
110  } else {
111  // make sure cost falls off by Euclidean distance
112  double factor =
113  exp(-1.0 * cost_scaling_factor_ * (distance * resolution_ - inscribed_radius_));
114  cost = static_cast<unsigned char>((INSCRIBED_INFLATED_OBSTACLE - 1) * factor);
115  }
116  return cost;
117  }
