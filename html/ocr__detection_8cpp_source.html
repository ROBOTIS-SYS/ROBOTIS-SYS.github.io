99 void OcrDetection::pre_processing(
100  cv::Mat & src)
101 {
102  src.copyTo(host_pre_mat_);
103  input_img_.upload(host_pre_mat_);
104  cv::cuda::resize(input_img_, resized_input_img_, cv::Size(830, 480), 0, 0);
105  cv::cuda::split(resized_input_img_, input_channels_); // HWC -> CHW
106  resized_input_img_buffer_.convertTo(tensor_img_, CV_32FC3, 1.f / 255.f);
107  cv::cuda::subtract(
108  tensor_img_,
109  cv::Scalar(
110  normalize_sub_vals_[0], normalize_sub_vals_[1],
111  normalize_sub_vals_[2]), tensor_img_, cv::noArray(), -1);
112  cv::cuda::divide(
113  tensor_img_,
114  cv::Scalar(
115  normalize_div_vals_[0], normalize_div_vals_[1],
116  normalize_div_vals_[2]), tensor_img_, 1, -1);
117  auto * dataPointer = tensor_img_.ptr<void>();
118  auto error = cudaMemcpyAsync(
119  (*device_tensor_buffer_)[0].deviceBuffer.data(),
120  dataPointer,
121  tensor_img_.cols * tensor_img_.rows * tensor_img_.channels() * sizeof(float),
122  cudaMemcpyDeviceToDevice,
123  cuda_stream_);
124 }

126 void OcrDetection::post_processing(
127  cv::Mat & src,
128  std::shared_ptr<std::vector<samplesCommon::ManagedBuffer>> input_buffer,
129  std::vector<cv::Mat> & img_list,
130  std::vector<std::vector<std::vector<int>>> & boxes)
131 {

133  host_post_mat_.data,
134  (*input_buffer)[1].deviceBuffer.data(),

139  cv::Mat pred_map;
140  cv::Mat mask;

143  cv::getStructuringElement(cv::MORPH_RECT, cv::Size(2, 2));

161 void OcrDetection::boxes_from_bitmap(
162  cv::Mat & pred_mat,
163  cv::Mat & mask,
164  int padding_size,
165  std::vector<std::vector<std::vector<int>>> & boxes,
166  std::vector<double> & scores)
167 {
168  int max_candidates = 1000;
169  float min_size = 3.0;
170  float box_thresh = 0.5;
171  float unclip_ratio = 2.0;
172  std::string score_mode = "fast";
173 
174  int height = mask.rows;
175  int width = mask.cols;
176 
177  std::vector<std::vector<cv::Point>> contours;
178  std::vector<cv::Vec4i> hierarchy;
179 
180  cv::findContours(mask, contours, hierarchy, cv::RETR_LIST, cv::CHAIN_APPROX_SIMPLE);
181  int num_contours =
182  contours.size() >= max_candidates ? max_candidates : contours.size();
183 
184  for (size_t i = 0; i < num_contours; i++) {
185  if (contours[i].size() <= 4) {
186  continue;
187  }
188  cv::RotatedRect bounding_box = cv::minAreaRect(contours[i]); // center, size(w,h), angle
189  if (bounding_box.size.width * bounding_box.size.height > 100) {
190  float ssid;
191  float score;
192  std::vector<std::vector<float>> box_points;
193  get_mini_boxes(bounding_box, box_points, ssid);
194  if (ssid < min_size) {
195  continue;
196  }
197  box_score_fast(pred_mat, box_points, score);
198  if (box_thresh > score) {
199  continue;
200  }
201  cv::RotatedRect points = unclip(box_points, unclip_ratio);
202  if (points.size.height < 1.001 && points.size.width < 1.001) {
203  continue;
204  }
205  cv::RotatedRect clipbox = points;
206  std::vector<std::vector<float>> clip_box_points;
207  get_mini_boxes(clipbox, clip_box_points, ssid);
208  if (ssid < min_size + 2) {
209  continue;
210  }
211 
212  int dest_width = pred_mat.cols;
213  int dest_height = pred_mat.rows;
214  std::vector<std::vector<int>> intcliparray;
215 
216  for (int num_pt = 0; num_pt < 4; num_pt++) {
217  std::vector<int> a{
218  static_cast<int>(clampf(
219  roundf(
220  clip_box_points[num_pt][0] / static_cast<float>(width) *
221  static_cast<float>(dest_width)),
222  0,
223  static_cast<float>(dest_width))),
224  static_cast<int>(clampf(
225  roundf(
226  clip_box_points[num_pt][1] / static_cast<float>(height) *
227  static_cast<float>(dest_height)),
228  0,
229  static_cast<float>(dest_height)))};
230  intcliparray.push_back(a);
231  a.clear();
232  }
233  boxes.push_back(intcliparray);
234  intcliparray.clear();
235  clip_box_points.clear();
236  box_points.clear();
237  }
238  }
239  contours.clear();
240  hierarchy.clear();
241 }

243 void OcrDetection::reorder_points_clockwise(
244  std::vector<std::vector<int>> pts,
245  std::vector<std::vector<int>> & out_pts)
246 {
247  std::vector<std::vector<int>> box = pts;
248  std::sort(box.begin(), box.end(), compare_x_int);
249  std::vector<std::vector<int>> leftmost = {box[0], box[1]};
250  std::vector<std::vector<int>> rightmost = {box[2], box[3]};
251  if (leftmost[0][1] > leftmost[1][1]) {
252  std::swap(leftmost[0], leftmost[1]);
253  }
254  if (rightmost[0][1] > rightmost[1][1]) {
255  std::swap(rightmost[0], rightmost[1]);
256  }
257  out_pts = {leftmost[0], rightmost[0], rightmost[1], leftmost[1]};
258 
259  box.clear();
260  leftmost.clear();
261  rightmost.clear();
262 }

264 void OcrDetection::FilterTagDetRes(
265  std::vector<std::vector<std::vector<int>>> boxes,
266  std::vector<std::vector<std::vector<int>>> & resized_boxes,
267  cv::Mat src)
268 {
269  int oriimg_h = src.rows;
270  int oriimg_w = src.cols;
271  float ratio_h = 480.0 / static_cast<float>(oriimg_h);
272  float ratio_w = 830.0 / static_cast<float>(oriimg_w);
273 
274  for (int n = 0; n < boxes.size(); n++) {
275  reorder_points_clockwise(boxes[n], boxes[n]);
276  for (int m = 0; m < boxes[0].size(); m++) {
277  boxes[n][m][0] /= ratio_w;
278  boxes[n][m][1] /= ratio_h;
279  boxes[n][m][0] = static_cast<int>(
280  _min(
281  _max(boxes[n][m][0], 0),
282  oriimg_w - 1));
283  boxes[n][m][1] = static_cast<int>(
284  _min(
285  _max(boxes[n][m][1], 0),
286  oriimg_h - 1));
287  }
288  }
289  for (int n = 0; n < boxes.size(); n++) {
290  int rect_width, rect_height;
291  rect_width = static_cast<int>(sqrt(
292  pow(boxes[n][0][0] - boxes[n][1][0], 2) +
293  pow(boxes[n][0][1] - boxes[n][1][1], 2)));
294  rect_height = static_cast<int>(sqrt(
295  pow(boxes[n][0][0] - boxes[n][3][0], 2) +
296  pow(boxes[n][0][1] - boxes[n][3][1], 2)));
297  if (rect_width <= 4 || rect_height <= 4) {
298  continue;
299  }
300  resized_boxes.push_back(boxes[n]);
301  }
302 }

304 void OcrDetection::get_mini_boxes(
305  cv::RotatedRect bounding_box,
306  std::vector<std::vector<float>> & vec_points,
307  float & ssid)
308 {
309  ssid = std::max(bounding_box.size.width, bounding_box.size.height);
310 
311  cv::Mat points;
312  cv::boxPoints(bounding_box, points);
313  Mat2Vector(points, vec_points);
314  std::sort(vec_points.begin(), vec_points.end(), compare_x_float);
315 
316  std::vector<float> idx1 = vec_points[0], idx2 = vec_points[1], idx3 = vec_points[2],
317  idx4 = vec_points[3];
318  if (vec_points[3][1] <= vec_points[2][1]) {
319  idx2 = vec_points[3];
320  idx3 = vec_points[2];
321  } else {
322  idx2 = vec_points[2];
323  idx3 = vec_points[3];
324  }
325  if (vec_points[1][1] <= vec_points[0][1]) {
326  idx1 = vec_points[1];
327  idx4 = vec_points[0];
328  } else {
329  idx1 = vec_points[0];
330  idx4 = vec_points[1];
331  }
332 
333  vec_points[0] = idx1;
334  vec_points[1] = idx2;
335  vec_points[2] = idx3;
336  vec_points[3] = idx4;
337 
338  idx1.clear();
339  idx2.clear();
340  idx3.clear();
341  idx4.clear();
342 }

344 void OcrDetection::get_rotate_crop_image(
345  cv::Mat & src,
346  cv::Mat & crop_image,
347  std::vector<std::vector<int>> box)
348 {
349  cv::Mat image;
350  src.copyTo(image);
351  std::vector<std::vector<int>> points = box;
352 
353  int x_collect[4] = {box[0][0], box[1][0], box[2][0], box[3][0]};
354  int y_collect[4] = {box[0][1], box[1][1], box[2][1], box[3][1]};
355  int left = static_cast<int>(
356  *std::min_element(x_collect, x_collect + 4));
357  int right = static_cast<int>(
358  *std::max_element(x_collect, x_collect + 4));
359  int top = static_cast<int>(
360  *std::min_element(y_collect, y_collect + 4));
361  int bottom = static_cast<int>(
362  *std::max_element(y_collect, y_collect + 4));
363 
364  cv::Mat img_crop;
365  image(cv::Rect(left, top, right - left, bottom - top)).copyTo(img_crop);
366 
367  for (int i = 0; i < points.size(); i++) {
368  points[i][0] -= left;
369  points[i][1] -= top;
370  }
371 
372  int img_crop_width = int(
373  sqrt(
374  pow(points[0][0] - points[1][0], 2) +
375  pow(points[0][1] - points[1][1], 2)));
376  int img_crop_height = int(
377  sqrt(
378  pow(points[0][0] - points[3][0], 2) +
379  pow(points[0][1] - points[3][1], 2)));
380 
381  cv::Point2f pts_std[4];
382  pts_std[0] = cv::Point2f(0., 0.);
383  pts_std[1] = cv::Point2f(img_crop_width, 0.);
384  pts_std[2] = cv::Point2f(img_crop_width, img_crop_height);
385  pts_std[3] = cv::Point2f(0.f, img_crop_height);
386 
387  cv::Point2f pointsf[4];
388  pointsf[0] = cv::Point2f(points[0][0], points[0][1]);
389  pointsf[1] = cv::Point2f(points[1][0], points[1][1]);
390  pointsf[2] = cv::Point2f(points[2][0], points[2][1]);
391  pointsf[3] = cv::Point2f(points[3][0], points[3][1]);
392 
393  cv::Mat M = cv::getPerspectiveTransform(pointsf, pts_std);
394 
395  cv::Mat dst_img;
396  cv::warpPerspective(
397  img_crop,
398  dst_img,
399  M,
400  cv::Size(img_crop_width, img_crop_height),
401  cv::BORDER_REPLICATE);
402 
403  if (static_cast<float>(dst_img.rows) >=
404  static_cast<float>(dst_img.cols) * 3.0)
405  {
406  cv::Mat tmp_img;
407  find_text(dst_img, tmp_img);
408  tmp_img.copyTo(crop_image);
409  } else {
410  dst_img.copyTo(crop_image);
411  }
412  points.clear();
413 }

416 void OcrDetection::box_score_fast(
417  cv::Mat & pred_mat,
418  std::vector<std::vector<float>> & vec_points,
419  float & score)
420 {
421  int width = pred_mat.cols;
422  int height = pred_mat.rows;
423 
424  float box_x[4] = {
425  vec_points[0][0],
426  vec_points[1][0],
427  vec_points[2][0],
428  vec_points[3][0]};
429  float box_y[4] = {
430  vec_points[0][1],
431  vec_points[1][1],
432  vec_points[2][1],
433  vec_points[3][1]};
434 
435  int xmin = clamp(
436  static_cast<int>(
437  std::floor(*(std::min_element(box_x, box_x + 4)))),
438  0,
439  width - 1);
440  int xmax = clamp(
441  static_cast<int>(
442  std::ceil(*(std::max_element(box_x, box_x + 4)))),
443  0,
444  width - 1);
445  int ymin = clamp(
446  static_cast<int>(
447  std::floor(*(std::min_element(box_y, box_y + 4)))),
448  0,
449  height - 1);
450  int ymax = clamp(
451  static_cast<int>(
452  std::ceil(*(std::max_element(box_y, box_y + 4)))),
453  0,
454  height - 1);
455 
456  cv::Mat mask = cv::Mat::zeros(ymax - ymin + 1, xmax - xmin + 1, CV_8UC1);
457 
458  cv::Point root_point[4];
459  root_point[0] = cv::Point(
460  static_cast<int>(vec_points[0][0]) - xmin,
461  static_cast<int>(vec_points[0][1]) - ymin);
462  root_point[1] = cv::Point(
463  static_cast<int>(vec_points[1][0]) - xmin,
464  static_cast<int>(vec_points[1][1]) - ymin);
465  root_point[2] = cv::Point(
466  static_cast<int>(vec_points[2][0]) - xmin,
467  static_cast<int>(vec_points[2][1]) - ymin);
468  root_point[3] = cv::Point(
469  static_cast<int>(vec_points[3][0]) - xmin,
470  static_cast<int>(vec_points[3][1]) - ymin);
471 
472  const cv::Point * ppt[1] = {root_point};
473  int npt[] = {4};
474  cv::fillPoly(mask, ppt, npt, 1, cv::Scalar(1));
475 
476  cv::Mat croppedImg;
477  pred_mat(cv::Rect(xmin, ymin, xmax - xmin + 1, ymax - ymin + 1)).copyTo(croppedImg);
478  score = cv::mean(croppedImg, mask)[0];
479 }

481 void OcrDetection::find_text(cv::Mat src, cv::Mat & dst)
482 {
483  cv::Mat gray, bin, bin_inv;
484  cv::cvtColor(
485  src,
486  gray,
487  cv::COLOR_BGR2GRAY);
488  cv::threshold(gray, bin, 127, 255, cv::THRESH_BINARY);
489  cv::threshold(gray, bin_inv, 127, 255, cv::THRESH_BINARY_INV);
490  std::vector<std::vector<cv::Point>> contours, contours_inv, selected_contour;
491  get_contour(bin, contours);
492  get_contour(bin_inv, contours_inv);
493  selected_contour = contours.size() > contours_inv.size() ? contours : contours_inv;
494  std::vector<cv::Mat> ho_img;
495  if (selected_contour.size() > 0) {
496  for (auto & cnt : selected_contour) {
497  cv::Rect tmp_bbox = cv::boundingRect(cnt);
498  cv::Rect crop_bbox(
499  tmp_bbox.x,
500  tmp_bbox.y,
501  tmp_bbox.width,
502  tmp_bbox.height);
503  int resize_width = static_cast<int>(tmp_bbox.width * 48 / tmp_bbox.height);
504  cv::Mat crop_img = src(crop_bbox);
505  cv::resize(crop_img, crop_img, cv::Size(resize_width, 48), 0, 0, cv::INTER_AREA);
506  ho_img.push_back(crop_img);
507  }
508  for (size_t i = 1; i < ho_img.size(); i++) {
509  cv::hconcat(ho_img[i], ho_img[0], ho_img[0]);
510  }
511  ho_img[0].copyTo(dst);
512  }
513  contours.clear();
514  contours_inv.clear();
515  selected_contour.clear();
516  ho_img.clear();
517 }

575 cv::RotatedRect OcrDetection::unclip(
576  std::vector<std::vector<float>> box,
577  const float & unclip_ratio)
578 {
579  float distance = 1.0;
580  int pts_num = 4;
581  float area = 0.0f;
582  float dist = 0.0f;
583  for (int i = 0; i < pts_num; i++) {
584  area += box[i][0] * box[(i + 1) % pts_num][1] -
585  box[i][1] * box[(i + 1) % pts_num][0];
586  dist += sqrtf(
587  (box[i][0] - box[(i + 1) % pts_num][0]) *
588  (box[i][0] - box[(i + 1) % pts_num][0]) +
589  (box[i][1] - box[(i + 1) % pts_num][1]) *
590  (box[i][1] - box[(i + 1) % pts_num][1]));
591  }
