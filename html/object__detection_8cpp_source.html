212 void ObjectDetection::decode_outputs(
213  float * prob,
214  std::vector<Object> & objects,
215  float scale,
216  const int img_w,
217  const int img_h)
218 {
219  std::vector<Object> proposals;
220  std::vector<int> strides = {8, 16, 32};
221  std::vector<GridAndStride> grid_strides;
222 
223  generate_grids_and_strides(strides, grid_strides);
224  generate_yolox_proposals(grid_strides, prob, bbox_conf_thresh_, proposals);
225 
226  qsort_descent_inplace(proposals);
227 
228  std::vector<int> picked;
229  nms_sorted_bboxes(proposals, picked, nms_thresh_);
230 
231  int count = picked.size();
232  // std::cout << " picked size : " << count << std::endl;
233  objects.resize(count);
234 
235  for (int i = 0; i < count; ++i) {
236  objects[i] = proposals[picked[i]];
237 
238  float x0 = (objects[i].rect.x) / scale;
239  float y0 = (objects[i].rect.y) / scale;
240  float x1 = (objects[i].rect.x + objects[i].rect.width) / scale;
241  float y1 = (objects[i].rect.y + objects[i].rect.height) / scale;
242 
243  x0 = std::max(std::min(x0, static_cast<float>(img_w - 1)), 0.f);
244  y0 = std::max(std::min(y0, static_cast<float>(img_h - 1)), 0.f);
245  x1 = std::max(std::min(x1, static_cast<float>(img_w - 1)), 0.f);
246  y1 = std::max(std::min(y1, static_cast<float>(img_h - 1)), 0.f);
247 
248  objects[i].rect.x = x0;
249  objects[i].rect.y = y0;
250  objects[i].rect.width = x1 - x0;
251  objects[i].rect.height = y1 - y0;
252  }
253 }

271 void ObjectDetection::generate_yolox_proposals(
272  std::vector<GridAndStride> grid_strides,
273  float * feature_array,
274  float prob_threshold,
275  std::vector<Object> & objects)
276 {
277  const int num_anchors = grid_strides.size();
278 
279  for (int anchor_idx = 0; anchor_idx < num_anchors; anchor_idx++) {
280  const int grid0 = grid_strides[anchor_idx].grid0;
281  const int grid1 = grid_strides[anchor_idx].grid1;
282  const int stride = grid_strides[anchor_idx].stride;
283 
284  // num_bbox_conf_ : bbox[4] + conf[1] = 5
285  const int basic_pos = anchor_idx * (num_class_ + num_bbox_conf_);
286 
287  // yolox/models/yolo_head.py decode logic
288  float x_center = (feature_array[basic_pos + 0] + grid0) * stride;
289  float y_center = (feature_array[basic_pos + 1] + grid1) * stride;
290  float w = exp(feature_array[basic_pos + 2]) * stride;
291  float h = exp(feature_array[basic_pos + 3]) * stride;
292  float x0 = x_center - w * 0.5f;
293  float y0 = y_center - h * 0.5f;
294 
295  float box_objectness = feature_array[basic_pos + 4];
296 
297  for (int class_idx = 0; class_idx < num_class_; class_idx++) {
298  float box_cls_score = feature_array[basic_pos + num_bbox_conf_ + class_idx];
299  float box_prob = box_objectness * box_cls_score;
300  if (box_prob > prob_threshold) {
301  Object obj;
302  obj.rect.x = x0;
303  obj.rect.y = y0;
304  obj.rect.width = w;
305  obj.rect.height = h;
306  obj.label = class_idx;
307  obj.prob = box_prob;
308 
309  objects.push_back(obj);
310  }
311  }
312  }
313 }

315 void ObjectDetection::nms_sorted_bboxes(
316  const std::vector<Object> & faceobjects,
317  std::vector<int> & picked,
318  float nms_threshold)
319 {
320  picked.clear();
321 
322  const int n = faceobjects.size();
323 
324  std::vector<float> areas(n);
325 
326  for (int i = 0; i < n; i++) {
327  areas[i] = faceobjects[i].rect.area();
328  }
329 
330  for (int i = 0; i < n; i++) {
331  const Object & a = faceobjects[i];
332 
333  bool keep = true;
334 
335  for (int j = 0; j < static_cast<int>(picked.size()); j++) {
336  const Object & b = faceobjects[picked[j]];
337 
338  // intersection over union
339  float inter_area = this->intersection_area(a, b);
340  float union_area = areas[i] + areas[picked[j]] - inter_area;
341 
342  // float IoU = inter_area / union_area
343  if ((inter_area / union_area) > nms_threshold) {
344  keep = false;
345  }
346  }
347 
348  if (keep == true) {
349  picked.push_back(i);
350  }
351  }
352 }

354 void ObjectDetection::qsort_descent_inplace(std::vector<Object> & faceobjects, int left, int right)
355 {
356  int i = left;
357  int j = right;
358  float p = faceobjects[(left + right) / 2].prob;
359 
360  while (i <= j) {
361  while (faceobjects[i].prob > p) {
362  i++;
363  }
364 
365  while (faceobjects[j].prob < p) {
366  j--;
367  }
368 
369  if (i <= j) {
370  // swap
371  std::swap(faceobjects[i], faceobjects[j]);
372 
373  i++;
374  j--;
375  }
376  }
377 
378  #pragma omp parallel sections
379  {
380  #pragma omp section
381  {
382  if (left < j) {qsort_descent_inplace(faceobjects, left, j);}
383  }
384  #pragma omp section
385  {
386  if (i < right) {qsort_descent_inplace(faceobjects, i, right);}
387  }
388  }
389 }

413 cv::Mat ObjectDetection::padding_img(cv::Mat && img)
414 {
415  auto src = std::move(img);
416  int width = src.cols;
417  int height = src.rows;
418  int dt = (width - height) / 2;
419  std::array<int, 2> dx_dy;
420 
421  if (dt >= 0) {
422  dx_dy = {0, abs(dt)};
423  } else {
424  dx_dy = {abs(dt), 0};
425  }
426 
427  auto p_left = dx_dy[0];
428  auto p_right = dx_dy[0];
429  auto p_top = dx_dy[1];
430  auto p_bot = dx_dy[1];
431 
432  cv::Mat pad_img;
433  cv::Mat resized_pad_img = cv::Mat(patch_size_, CV_8UC3);
434 
435  cv::copyMakeBorder(
436  src,
437  pad_img,
438  p_top,
439  p_bot,
440  p_left,
441  p_right,
442  cv::BORDER_CONSTANT,
443  cv::Scalar(0, 0, 0));
444 
445  cv::resize(pad_img, resized_pad_img, resized_pad_img.size());
446 
447  return std::move(resized_pad_img);
448 }
