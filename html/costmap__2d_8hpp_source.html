342  template<typename data_type>
343  void copyMapRegion(
344  data_type * source_map, unsigned int sm_lower_left_x,
345  unsigned int sm_lower_left_y,
346  unsigned int sm_size_x, data_type * dest_map, unsigned int dm_lower_left_x,
347  unsigned int dm_lower_left_y, unsigned int dm_size_x, unsigned int region_size_x,
348  unsigned int region_size_y)
349  {
350  // we'll first need to compute the starting points for each map
351  data_type * sm_index = source_map + (sm_lower_left_y * sm_size_x + sm_lower_left_x);
352  data_type * dm_index = dest_map + (dm_lower_left_y * dm_size_x + dm_lower_left_x);
353 
354  // now, we'll copy the source map into the destination map
355  for (unsigned int i = 0; i < region_size_y; ++i) {
356  memcpy(dm_index, sm_index, region_size_x * sizeof(data_type));
357  sm_index += sm_size_x;

388  template<class ActionType>
389  inline void raytraceLine(
390  ActionType at, unsigned int x0, unsigned int y0, unsigned int x1,
391  unsigned int y1,
392  unsigned int max_length = UINT_MAX)
393  {
394  int dx = x1 - x0;
395  int dy = y1 - y0;
396 
397  unsigned int abs_dx = abs(dx);
398  unsigned int abs_dy = abs(dy);
399 
400  int offset_dx = sign(dx);
401  int offset_dy = sign(dy) * size_x_;
402 
403  unsigned int offset = y0 * size_x_ + x0;
404 
405  // we need to chose how much to scale our dominant dimension,
406  // based on the maximum length of the line
407  double dist = std::hypot(dx, dy);
408  double scale = (dist == 0.0) ? 1.0 : std::min(1.0, max_length / dist);
409 
410  // if x is dominant
411  if (abs_dx >= abs_dy) {
412  int error_y = abs_dx / 2;
413  bresenham2D(
414  at, abs_dx, abs_dy, error_y, offset_dx, offset_dy, offset,
415  (unsigned int)(scale * abs_dx));
416  return;
417  }
418 
419  // otherwise y is dominant
420  int error_x = abs_dy / 2;
421  bresenham2D(
422  at, abs_dy, abs_dx, error_x, offset_dy, offset_dx, offset,
423  (unsigned int)(scale * abs_dy));
424  }
425 
426 private:
430  template<class ActionType>
431  inline void bresenham2D(
432  ActionType at, unsigned int abs_da, unsigned int abs_db, int error_b,
433  int offset_a,
434  int offset_b, unsigned int offset,
435  unsigned int max_length)
436  {
437  unsigned int end = std::min(max_length, abs_da);
438  for (unsigned int i = 0; i < end; ++i) {
439  at(offset);
440  offset += offset_a;
441  error_b += abs_db;
442  if ((unsigned int)error_b >= abs_da) {
443  offset += offset_b;
444  error_b -= abs_da;
445  }
446  }
447  at(offset);
448  }
