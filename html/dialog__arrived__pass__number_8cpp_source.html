138 void DialogArrivedPassNumber::num_clicked(const int num)
139 {
140  if (input_pass_numbers_.size() < PASSCODE_LEN) {
141  input_pass_numbers_.push_back(num);
142  set_passcode_bullet(input_pass_numbers_.size(), true);
143  }
144  if (input_pass_numbers_.size() == PASSCODE_LEN) {
145  try_cnt++;
146  if (check_passcode()) {
147  try_cnt = 0;
148  this->close();
149  this->setResult((QDialog::Accepted));
150  result_code_ = SUCCEED;
151  return;
152  } else {
153  if (try_cnt >= MAX_ALLOWED_ATTEMPTS) {
154  try_cnt = 0;
155  result_code_ = EXCEEDED_MAX_ATTEMPS;
156  this->close();
157  this->setResult((QDialog::Rejected));
158  return;
159  }
160  set_input_pass_numbers_clear();
161  ui->lblText->setText(tr("비밀번호가 틀렸습니다."));
162  std::thread th_invalid_room(invalid_room_thread, this);
163  th_invalid_room.detach();
164  }
165  }
166 }

184 bool DialogArrivedPassNumber::check_passcode()
185 {
186  if (input_pass_numbers_.size() < PASSCODE_LEN) {
187  return false;
188  }
189  std::string input_passcode_ = "";
190  for (int code : input_pass_numbers_) {
191  input_passcode_ += std::to_string(code);
192  }
193  if (pass_code_.compare(input_passcode_) == 0) {
194  return true;
195  }
196  return false;
197 }
