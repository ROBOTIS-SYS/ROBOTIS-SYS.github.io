35 bool GcKnockMotionControl::configure()
36 {
37  RCLCPP_INFO_STREAM(node_->get_logger(), "Ready to Gc Manipulator Knock Control.");
38 
39  // Action interface name
40  std::string knock_action_name =
41  node_->get_parameter(ros_param_root_ + "knock_action_name").as_string();
42  RCLCPP_INFO_STREAM(node_->get_logger(), "Action name is " << knock_action_name << ".");
43  // Action component name
44  knock_component_name_ =
45  node_->get_parameter(ros_param_root_ + "knock_component_name").as_string();
46  if (!robot_handler_->check_component_name(knock_component_name_)) {
47  RCLCPP_ERROR_STREAM(
48  node_->get_logger(),
49  "Requested knock component name '" << knock_component_name_ <<
50  "' is not included in this robot.");
51  return false;
52  }
53  RCLCPP_INFO_STREAM(
54  node_->get_logger(),
55  "Knock component name is " << knock_component_name_ << ".");
56 
57  // manipulator_base name
58  manipulator_base_name_ =
59  node_->get_parameter(ros_param_root_ + "manipulator_base_name").as_string();
60  if (!robot_handler_->check_component_name(manipulator_base_name_)) {
61  RCLCPP_ERROR_STREAM(
62  node_->get_logger(),
63  "Requested manipulator_base_name '" << manipulator_base_name_ <<
64  "' is not included in this robot.");
65  return false;
66  }
67  RCLCPP_INFO_STREAM(
68  node_->get_logger(),
69  "manipulator_base_name is " << manipulator_base_name_ << ".");
70 
71  // Action Preparing time
72  prepare_time_ =
73  node_->get_parameter(ros_param_root_ + "prepare_time").as_double();
74  RCLCPP_INFO_STREAM(node_->get_logger(), "Prepare time is " << prepare_time_ << "s.");
75 
76  // Knock height from manipulator base

78  node_->get_parameter(ros_param_root_ + "knock_h").as_double();

80

82  node_->get_parameter(ros_param_root_ + "start_tip_angle").as_double() * M_PI / 180.0;
83  RCLCPP_INFO_STREAM(
84  node_->get_logger(), "Knock start tip angle is " << start_tip_angle_ * 180.0 / M_PI << "deg.");

87  node_->get_parameter(ros_param_root_ + "knock_velocity").as_double();
88  RCLCPP_INFO_STREAM(
89  node_->get_logger(), "Knock motion velocity is " << knock_velocity_ << "rad/s.");

91  node_->get_parameter(ros_param_root_ + "knock_acceleration").as_double();
92  RCLCPP_INFO_STREAM(
93  node_->get_logger(), "Knock motion acceleration is " << knock_acceleration_ << "rad/s.");
94 
95  // Action server
96  using namespace std::placeholders;
97  action_server_ = rclcpp_action::create_server<KnockMotionAction>(
98  node_->get_node_base_interface(),
99  node_->get_node_clock_interface(),
100  node_->get_node_logging_interface(),
101  node_->get_node_waitables_interface(),
102  std::string(node_->get_name()) + knock_action_name,
103  std::bind(&GcKnockMotionControl::goal_callback, this, _1, _2),
104  std::bind(&GcKnockMotionControl::cancel_callback, this, _1),
105  std::bind(&GcKnockMotionControl::aceepted_callback, this, _1));
106 
107  /* Publisher */
108  if (robot_handler_->is_debug_mode()) {
109  rclcpp::QoS qos(rclcpp::KeepLast(1));
110  door_pose_pub_ =
111  node_->create_publisher<geometry_msgs::msg::PoseStamped>(
112  std::string(node_->get_name()) + "/knock/door_pose", qos);
113  start_pose_pub_ =
114  node_->create_publisher<geometry_msgs::msg::PoseStamped>(
115  std::string(node_->get_name()) + "/knock/start_pose", qos);

122  }
123 
124  RCLCPP_INFO_STREAM(node_->get_logger(), "Success to configure [Gc Knock Control]");
125  return true;

149 void GcKnockMotionControl::update_cancel(rm2_control_module::ControlModule::CancelResult result)
150 {
151  KnockMotionAction::Result::SharedPtr action_result =
152  std::make_shared<KnockMotionAction::Result>();
153  manipulator_interfaces::msg::Result result_state;
154 
155  if (result == CancelResult::JOINT_LIMIT) {
156  result_state.set__val(manipulator_interfaces::msg::Result::COLLISION_CHECKING_UNAVAILABLE);
157  RCLCPP_INFO_STREAM(
158  node_->get_logger(),
159  "Knock contorl canceled : joint limit over.");
160  } else if (result == CancelResult::TOO_FAST_MOVE) {
161  result_state.set__val(manipulator_interfaces::msg::Result::CONTROL_FAILED);
162  RCLCPP_INFO_STREAM(
163  node_->get_logger(),
164  "Knock contorl canceled : too fast move failure");
165  } else if (result == CancelResult::INVALID_GOAL_VALUES) {
166  result_state.set__val(manipulator_interfaces::msg::Result::CONTROL_FAILED);
167  RCLCPP_INFO_STREAM(
168  node_->get_logger(),
169  "Knock contorl canceled : invalid goal values failure");
170  } else if (result == CancelResult::IK_FAIL) {
171  result_state.set__val(manipulator_interfaces::msg::Result::NO_IK_SOLUTION);
172  RCLCPP_INFO_STREAM(
173  node_->get_logger(),
174  "Knock contorl canceled : ik solving failure");
175  } else if (result == CancelResult::PREEMPTED) {
176  result_state.set__val(manipulator_interfaces::msg::Result::PREEMPTED);
177  RCLCPP_INFO_STREAM(
178  node_->get_logger(),
179  "Knock contorl canceled : preempted by the other controller");
180  } else if (result == CancelResult::CHECKER) {
181  result_state.set__val(manipulator_interfaces::msg::Result::GOAL_IN_COLLISION);
182  RCLCPP_INFO_STREAM(
183  node_->get_logger(),
184  "Knock contorl canceled : lidar");
185  } else {
186  result_state.set__val(manipulator_interfaces::msg::Result::FAILURE);
187  RCLCPP_INFO_STREAM(
188  node_->get_logger(),
189  "Knock contorl canceled : fail.");
190  }
191  action_result->set__result_state(result_state);
192 
193  if (actived_goal_) {
194  actived_goal_->abort(action_result);
195  }
196  actived_goal_.reset();
197 }

217 bool GcKnockMotionControl::calc_door_position(
218  std_msgs::msg::Header header,
219  geometry_msgs::msg::Polygon door)
220 {
221  // Translate
222  rclcpp::Time door_detected_time = header.stamp;
223  std::string door_base_frame = header.frame_id;
224  rm2_common::TaskValues door_base_task_values = robot_handler_->get_present_task_value(
225  door_detected_time.seconds(),
226  door_base_frame);
227  mani_base_task_values_ = robot_handler_->get_present_task_value(
228  door_detected_time.seconds(),
229  manipulator_base_name_);
230 
231  if (door.points.size() != 4) {
232  return false;
233  }
234 
235  std::vector<rm2_common::Vector> door_points_pre;
236  for (auto point : door.points) {
237  door_points_pre.push_back(
238  mani_base_task_values_.pose.inverse() *
239  door_base_task_values.pose *
240  rm2_common::Vector(
241  static_cast<double>(point.x),
242  static_cast<double>(point.y),
243  static_cast<double>(point.z)));
244  }
245 
246  // <<< +
247  // left top(0) ^*---* right top(1)
248  // ^| |
249  // left bot(2) +* * right bot(3)
250  std::deque<rm2_common::Vector> door_points;
251 
252  // First point
253  door_points.push_back(door_points_pre.at(0));
254  // set 2 top point (Z)
255  for (size_t index = 1; index < door_points_pre.size(); index++) {
256  if (door_points.at(0).z() <= static_cast<double>(door_points_pre.at(index).z())) {
257  door_points.push_front(
258  mani_base_task_values_.pose.inverse() *
259  door_base_task_values.pose *
260  rm2_common::Vector(door_points_pre.at(index)));
261  } else {
262  door_points.push_back(
263  rm2_common::Vector(door_points_pre.at(index)));
264  }
265  }
266 
267  if (door_points.at(0).y() < door_points.at(1).y()) {
268  rm2_common::Vector tmp = door_points.at(0);
269  door_points.at(0) = door_points.at(1);
270  door_points.at(1) = tmp;
271  }
272 
273  // Set bottom left and right
274  if (door_points.at(2).y() < door_points.at(3).y()) {
275  rm2_common::Vector tmp = door_points.at(2);
276  door_points.at(2) = door_points.at(3);
277  door_points.at(3) = tmp;
278  }
279 
280  // door Corners
281  // <<< +
282  // left top(0) ^*---* right top(1)
283  // ^| |
284  // left bot(2) +* * right bot(3)
285  // Set top left and right
286  RCLCPP_INFO_STREAM(node_->get_logger(), "DOOR POINTS");
287  rm2_common::Vector door_left_top = door_points.at(0);
288  rm2_common::Vector door_right_top = door_points.at(1);
289  rm2_common::Vector door_left_bottom = door_points.at(2);
290  rm2_common::Vector door_right_bottom = door_points.at(3);
291  std::cout << door_left_top.data.transpose() << " * ---- * " <<
292  door_right_top.data.transpose() << std::endl;
293  std::cout << door_left_bottom.data.transpose() << " * ---- * " <<
294  door_right_bottom.data.transpose() << std::endl;
295 
296  double door_angle = atan2(
297  door_left_bottom.y() - door_right_bottom.y(),
298  door_left_bottom.x() - door_right_bottom.x());
299  std::cout << "door angle : " << door_angle << std::endl;
300  rm2_common::Vector door_normal =
301  rm2_common::Rotation::RPY(0.0, 0.0, door_angle - M_PI_2) * rm2_common::Vector(1.0, 0.0, 0.0);
302 
303  double distance = door_normal.dot(door_left_bottom);
304  yaw_angle_to_door_ = door_angle - M_PI_2;
305  door_position_ = distance * door_normal;
306  door_position_.z(knock_h_);
307  std::cout << "Door Position from manipulator base: " << door_position_.data.transpose() <<
308  std::endl;
309 
310  double x_max = std::max(door_left_bottom.x(), door_right_bottom.x());
311  double x_min = std::min(door_left_bottom.x(), door_right_bottom.x());
312  double y_max = std::max(door_left_bottom.y(), door_right_bottom.y());
313  double y_min = std::min(door_left_bottom.y(), door_right_bottom.y());
314  if (door_position_.x() >= x_min &&
315  door_position_.x() <= x_max &&
316  door_position_.y() >= y_min &&
317  door_position_.y() <= y_max)
318  {
319  return true;
320  } else {
321  return false;
322  }
323 }

340 rclcpp_action::GoalResponse GcKnockMotionControl::goal_callback(
341  const rclcpp_action::GoalUUID & uuid,
342  std::shared_ptr<const manipulator_interfaces::action::GcKnockMotion::Goal> goal)
343 {
344  (void) uuid;
345  RCLCPP_INFO_STREAM(node_->get_logger(), "Received 'knock' action goal");
346  std::lock_guard<std::mutex> lock(mutex_);
347  // Door pose
348  if (!calc_door_position(goal->header, goal->door_position)) {
349  RCLCPP_ERROR_STREAM(node_->get_logger(), "Door is out of knocking range.");
350  return rclcpp_action::GoalResponse::REJECT;
351  }
352  // Door
353  rm2_common::TaskValues door_task_values_from_manipulator_base;
354  door_task_values_from_manipulator_base.pose.position = door_position_;
355  door_task_values_from_manipulator_base.pose.orientation =
356  rm2_common::Rotation::Yaw(yaw_angle_to_door_);
357  door_task_values_from_manipulator_base.stoped();
358 
359  // knock direction
360  rm2_common::Pose knock_direction_rate(
361  rm2_common::Vector(
362  goal->knock_direction.x * goal->knock_position_rate,
363  0.0,
364  goal->knock_direction.y * goal->knock_position_rate),
365  rm2_common::Rotation::Pitch(-start_tip_angle_));
366 
368  // Present pose (Start)
369  auto start_task_values = robot_handler_->get_previous_goal_task_value(knock_component_name_);
370 
371  double r, p, y;
372  start_task_values.pose.orientation.get_rpy(r, p, y);
373  std::cout << "Start Pose : " << start_task_values.pose.position.data.transpose() << " / " <<
374  r / M_PI * 180.0 << ", " << p / M_PI * 180.0 << ", " << y / M_PI * 180.0 << std::endl;
375  // Present joint values (Start)
376  rm2_common::JointValues start_joint_values = robot_handler_->get_previous_goal_joint_value();
377  std::cout << "Start Joint : " << start_joint_values.position.data.transpose() << std::endl;
379 
381  // Prepare Pose
382  rm2_common::TaskValues prepare_task_values;
383  prepare_task_values.pose =
384  mani_base_task_values_.pose *
385  door_task_values_from_manipulator_base.pose *
386  knock_direction_rate.inverse();
387  prepare_task_values.stoped(); // Stop
388 
389  prepare_task_values.pose.orientation.get_rpy(r, p, y);
390  std::cout << "prepare Pose : " << prepare_task_values.pose.position.data.transpose() << " / " <<
391  r / M_PI * 180.0 << ", " << p / M_PI * 180.0 << ", " << y / M_PI * 180.0 << std::endl;
392  // Prepare Pose joint vavlue
393  rm2_common::JointValues ik_present_values = start_joint_values;
394  ik_present_values.position(0) = yaw_angle_to_door_;
395  rm2_common::JointValues prepare_joint_values;
396  if (!robot_handler_->solve_inverse_kinematics(
397  knock_component_name_, prepare_task_values, &prepare_joint_values, ik_present_values))
398  {
399  RCLCPP_ERROR_STREAM(
400  node_->get_logger(), "Fail to calculate Inverse kinematics on prepare position.");
401  return rclcpp_action::GoalResponse::REJECT;
402  }
403  std::cout << "prepare Joint : " << prepare_joint_values.position.data.transpose() << std::endl;
405 
407  // Knocking pose Remapping
408  rm2_common::TaskValues knock_task_values;
409  rm2_common::Pose knock_direction(
410  rm2_common::Vector(
411  goal->knock_direction.x,
412  0.0,
413  goal->knock_direction.y),
414  rm2_common::Rotation::Pitch(goal->knock_direction.theta));
415  knock_task_values.pose = prepare_task_values.pose * knock_direction;
416  knock_task_values.stoped(); // Stop
417 
418  knock_task_values.pose.orientation.get_rpy(r, p, y);
419  std::cout << "Knock Pose : " << knock_task_values.pose.position.data.transpose() << " / " <<
420  r / M_PI * 180.0 << ", " << p / M_PI * 180.0 << ", " << y / M_PI * 180.0 << std::endl;
421  // Knocking pose Remapping
422  rm2_common::JointValues knock_joint_values;
423  if (!robot_handler_->solve_inverse_kinematics(
424  knock_component_name_, knock_task_values, &knock_joint_values, ik_present_values))
425  {
426  RCLCPP_ERROR_STREAM(
427  node_->get_logger(), "Fail to calculate Inverse kinematics on knock position.");
428  return rclcpp_action::GoalResponse::REJECT;
429  }
430  std::cout << "Knock Joint : " << knock_joint_values.position.data.transpose() << std::endl;
432 
433  if (robot_handler_->is_debug_mode()) {
435  geometry_msgs::msg::PoseStamped door_pose_msg;
436  door_pose_msg.header.stamp = node_->now();
437  door_pose_msg.header.frame_id = manipulator_base_name_;
438  door_pose_msg.pose = trans_pose_to_msg(
439  door_task_values_from_manipulator_base.pose.position,
440  door_task_values_from_manipulator_base.pose.orientation);
441  door_pose_pub_->publish(door_pose_msg);
442 
443  geometry_msgs::msg::PoseStamped start_pose_msg;
444  start_pose_msg.header.stamp = node_->now();
445  start_pose_msg.header.frame_id = robot_handler_->get_robot_model().get_base_component()->first;
446  start_pose_msg.pose = trans_pose_to_msg(
447  start_task_values.pose.position,
448  start_task_values.pose.orientation);
449  start_pose_pub_->publish(start_pose_msg);
450 
451  geometry_msgs::msg::PoseStamped prepare_pose_msg;
452  prepare_pose_msg.header.stamp = node_->now();
453  prepare_pose_msg.header.frame_id =
454  robot_handler_->get_robot_model().get_base_component()->first;
455  prepare_pose_msg.pose = trans_pose_to_msg(
456  prepare_task_values.pose.position,
457  prepare_task_values.pose.orientation);
458  prepare_pose_pub_->publish(prepare_pose_msg);
459 
460  geometry_msgs::msg::PoseStamped knock_pose_msg;
461  knock_pose_msg.header.stamp = node_->now();
462  knock_pose_msg.header.frame_id = robot_handler_->get_robot_model().get_base_component()->first;
463  knock_pose_msg.pose = trans_pose_to_msg(
464  knock_task_values.pose.position,
465  knock_task_values.pose.orientation);
466  knock_pose_pub_->publish(knock_pose_msg);
468  }
469 
470  // Trajctory Queue
471  std::queue<std::shared_ptr<rm2_trajectory::JointVelocityProfile>> empty;
472  knock_traj_queue_.swap(empty);
473  // prepare motion trajectory
474  auto prepare_traj_mini = std::make_shared<rm2_trajectory::JointMinimumJerk>();
475  prepare_traj_mini->set_move_time(prepare_time_);
476  prepare_traj_mini->calculation(
477  start_joint_values.position, prepare_joint_values.position,
478  start_joint_values.velocity, prepare_joint_values.velocity);
479  knock_traj_queue_.push(prepare_traj_mini);
480 
481  // knock motion trajectory
482  std::cout << "[calc] knock Number : " << goal->knock_num << std::endl;
483  for (int num = 0; num < goal->knock_num; num++) {
484  std::cout << "[calc] knock cnt : " << num << std::endl;
485 // auto traj_knock_trap = std::make_shared<rm2_trajectory::TaskTrapezoidal>();
486 
487  auto traj_knock_trap = std::make_shared<rm2_trajectory::JointTrapezoidal>();
488  traj_knock_trap->set_profile(knock_velocity_, knock_acceleration_);
489  traj_knock_trap->calculation(
490  prepare_joint_values.position, knock_joint_values.position,
491  prepare_joint_values.velocity, knock_joint_values.velocity);
492  knock_traj_queue_.push(traj_knock_trap);
493 
494  auto traj_back_trap = std::make_shared<rm2_trajectory::JointTrapezoidal>();
495  traj_back_trap->set_profile(knock_velocity_, knock_acceleration_);
496  traj_back_trap->calculation(
497  knock_joint_values.position, prepare_joint_values.position,
498  knock_joint_values.velocity, prepare_joint_values.velocity);
499  knock_traj_queue_.push(traj_back_trap);
500  }
501 
502  // back motion trajectory
503  auto back_traj_mini = std::make_shared<rm2_trajectory::JointMinimumJerk>();
504  back_traj_mini->set_move_time(prepare_time_);
505  back_traj_mini->calculation(
506  prepare_joint_values.position, start_joint_values.position,
507  prepare_joint_values.velocity, start_joint_values.velocity);
508  knock_traj_queue_.push(back_traj_mini);
509 
510  // set start time
511  robot_handler_->set_start_time(node_->now().seconds());
512  return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
513 }

515 rclcpp_action::CancelResponse GcKnockMotionControl::cancel_callback(
516  const std::shared_ptr<rclcpp_action::ServerGoalHandle<KnockMotionAction>> goal_handle)
517 {
518  RCLCPP_INFO_STREAM(node_->get_logger(), "Got request to cancel goal");
519  std::lock_guard<std::mutex> lock(mutex_);
520  try {
521  if (actived_goal_ == goal_handle) {
522  stop_update();
523
