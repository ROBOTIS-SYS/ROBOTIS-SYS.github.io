82 pcl::PointCloud<pcl::PointXYZ> geometry_calculation_lib::conditional_removal(
83  pcl::PointCloud<pcl::PointXYZ> pc, double x_max, double x_min, double y_max, double y_min)
84 {
85  // eq - equal ( = )
86  // ne - not equal ( <> )
87  // lt - little ( < )
88  // le - little or equal ( <= )
89  // gt - greater ( > )
90  // ge - greater or equal ( >= )
91  pcl::ConditionAnd<pcl::PointXYZ>::Ptr range_cond(new pcl::ConditionAnd<pcl::PointXYZ>());
92  range_cond->addComparison(
93  pcl::FieldComparison<pcl::PointXYZ>::Ptr(
94  new pcl::FieldComparison<pcl::PointXYZ>("x", pcl::ComparisonOps::GT, x_min)));
95  range_cond->addComparison(
96  pcl::FieldComparison<pcl::PointXYZ>::Ptr(
97  new pcl::FieldComparison<pcl::PointXYZ>("x", pcl::ComparisonOps::LT, x_max)));
98  range_cond->addComparison(
99  pcl::FieldComparison<pcl::PointXYZ>::Ptr(
100  new pcl::FieldComparison<pcl::PointXYZ>("y", pcl::ComparisonOps::GT, y_min)));
101  range_cond->addComparison(
102  pcl::FieldComparison<pcl::PointXYZ>::Ptr(
103  new pcl::FieldComparison<pcl::PointXYZ>("y", pcl::ComparisonOps::LT, y_max)));
104 
105  pcl::ConditionalRemoval<pcl::PointXYZ> cond_rem;
106  cond_rem.setInputCloud(pc.makeShared()); // 입력Ptr
107  cond_rem.setCondition(range_cond); // 조건 설정
108  cond_rem.setKeepOrganized(true); //
109  cond_rem.filter(pc); // 필터 적용
110  return pc;
111 }

169 double geometry_calculation_lib::iterative_closest_point(
170  pcl::PointCloud<pcl::PointXYZ> lidar,
171  std::shared_ptr<pcl::PointCloud<pcl::PointXYZ>> temp_ptr,
172  std::shared_ptr<geometry_msgs::msg::Pose> matching_pose_ptr,
173  int iter, double fitness_epsilon, double correspondence_distance,
174  double trans_epsilon, double trans_rot_epsilon)
175 {
176  pcl::IterativeClosestPoint<pcl::PointXYZ, pcl::PointXYZ> icp;
177  icp.setInputSource(temp_ptr->makeShared());
178  icp.setInputTarget(lidar.makeShared());
179 
183  icp.setMaximumIterations(iter);
191  icp.setEuclideanFitnessEpsilon(fitness_epsilon);
197  icp.setMaxCorrespondenceDistance(correspondence_distance);
204  // The epsilon (difference) between the previous transformation
205  // and the current estimated transformation is smaller than
206  // an user imposed value (via setTransformationEpsilon)
207  icp.setTransformationEpsilon(trans_epsilon);
214  // The epsilon (difference) between the previous transformation
215  // and the current estimated transformation is smaller than
216  // an user imposed value (via setTransformationEpsilon)
217  icp.setTransformationRotationEpsilon(trans_rot_epsilon);
218 
219 // std::cout << " Iterations : " << icp.getMaximumIterations() <<
220 // " Euclidean Fitness : " << icp.getEuclideanFitnessEpsilon() <<
221 // " Max Checking Distance : " << icp.getMaxCorrespondenceDistance() <<
222 // " Transformation Pos Epsilon : " << icp.getTransformationEpsilon() <<
223 // " Transformation Rot Epsilon : " << icp.getTransformationRotationEpsilon() << std::endl;
224 
225  pcl::PointCloud<pcl::PointXYZ> final;
226  icp.align(final);
227  *temp_ptr = final;
228 
229  // Distance
230  auto transfrom = icp.getFinalTransformation();
231  Eigen::Vector3f position = transfrom.block(0, 3, 3, 1);
232  Eigen::Matrix3f orientaion = transfrom.block(0, 0, 3, 3);
233  Eigen::Quaternionf quater(orientaion);
234  matching_pose_ptr->position.x = static_cast<double>(position(0));
235  matching_pose_ptr->position.y = static_cast<double>(position(1));
236  matching_pose_ptr->position.z = static_cast<double>(position(2));
237  matching_pose_ptr->orientation.w = static_cast<double>(quater.w());
238  matching_pose_ptr->orientation.x = static_cast<double>(quater.x());
239  matching_pose_ptr->orientation.y = static_cast<double>(quater.y());
240  matching_pose_ptr->orientation.z = static_cast<double>(quater.z());
241 
242  return icp.getFitnessScore();
243 }

245 double geometry_calculation_lib::multi_iterative_closest_point(
246  pcl::PointCloud<pcl::PointXYZ> lidar,
247  std::shared_ptr<pcl::PointCloud<pcl::PointXYZ>> temp_ptr,
248  std::shared_ptr<geometry_msgs::msg::Pose> matching_pose_ptr,
249  double search_range, int multi_search_num,
250  int iter, double fitness_epsilon,
251  double correspondence_distance,
252  double trans_epsilon, double trans_rot_epsilon,
253  double offset_x, double offset_y)
254 {
255  std::vector<double> angles;
256  std::vector<std::shared_ptr<geometry_msgs::msg::Pose>> matching_pose_ptrs;
257  std::vector<std::shared_ptr<pcl::PointCloud<pcl::PointXYZ>>> pointcloud_template_ptrs;
258  std::vector<std::future<double>> async_futures;
259 
260  double init_point_interval = search_range / static_cast<double>(multi_search_num);
261 
262  for (int y_step = 0; y_step < multi_search_num; y_step++) {
263  for (int x_step = 0; x_step < multi_search_num; x_step++) {
264  int i = y_step * multi_search_num + x_step;
265  matching_pose_ptrs.push_back(std::make_shared<geometry_msgs::msg::Pose>());
266  pointcloud_template_ptrs.push_back(std::make_shared<pcl::PointCloud<pcl::PointXYZ>>());
267  // transform matrix
268  Eigen::Affine3d transform = Eigen::Affine3d::Identity();
269  auto starting_point_x = offset_x - search_range / 2 + x_step * init_point_interval;
270  auto starting_point_y = offset_y - search_range / 2 + y_step * init_point_interval;
271  transform.translation() << starting_point_x, starting_point_y, 0.0;
272  pcl::transformPointCloud(*temp_ptr, *pointcloud_template_ptrs.at(i), transform);
273  // async
274  async_futures.push_back(
275  std::async(
276  std::launch::async,
277  std::bind(
278  &geometry_calculation_lib::iterative_closest_point,
279  lidar, pointcloud_template_ptrs.at(i), matching_pose_ptrs.at(i),
280  iter, fitness_epsilon, correspondence_distance,
281  trans_epsilon, trans_rot_epsilon)));
282  }
283  }
284 
285  // get
286  size_t best_score_index = 0;
287  double best_score = 100;
288  try {
289  for (size_t i = 0; i < async_futures.size(); i++) {
290  double score = async_futures.at(i).get();
291  if (best_score > score) {
292  best_score = score;
293  best_score_index = i;
294  }
295  }
296  } catch (const std::future_error & e) {
297  std::cout << "futrue error : " << e.what() << std::endl;
298  return 100;
299  }
300 
302  int best_x_step = best_score_index % multi_search_num;
303  int best_y_step = best_score_index / multi_search_num;
304  Eigen::Affine3d offset_transform = Eigen::Affine3d::Identity();
305  offset_transform.translation() <<
306  offset_x - search_range / 2 + best_x_step * init_point_interval,
307  offset_y - search_range / 2 + best_y_step * init_point_interval,
308  0.0;
309 
310  Eigen::Affine3d matching_transform = Eigen::Affine3d::Identity();
311  // position
312  matching_transform.translation() =
313  geometry_calculation_lib::position(matching_pose_ptrs.at(best_score_index)->position);
314  // orientation
315  Eigen::Quaterniond matching_quater =
316  geometry_calculation_lib::orientation(matching_pose_ptrs.at(best_score_index)->orientation);
317  matching_transform.rotate(matching_quater.toRotationMatrix());
318 
319  Eigen::Affine3d result_transform = matching_transform * offset_transform;
320 
321  result_transform.translation();
322 
323  // position
324  matching_pose_ptr->position.x = result_transform.translation()(0);
325  matching_pose_ptr->position.y = result_transform.translation()(1);
326  matching_pose_ptr->position.z = 0.0;
327  // orientation
328  Eigen::Quaterniond quater(result_transform.rotation());
329  matching_pose_ptr->orientation.w = quater.w();
330  matching_pose_ptr->orientation.x = quater.x();
331  matching_pose_ptr->orientation.y = quater.y();
332  matching_pose_ptr->orientation.z = quater.z();
333  pcl::transformPointCloud(
334  *pointcloud_template_ptrs.at(best_score_index),
335  *temp_ptr,
336  result_transform * matching_transform.inverse());
337 
338  return best_score;
339 }

375 double geometry_calculation_lib::yaw(geometry_msgs::msg::Quaternion orientation)
376 {
377  Eigen::Quaterniond q;
378  q.w() = orientation.w;
379  q.x() = orientation.x;
380  q.y() = orientation.y;
381  q.z() = orientation.z;
382  Eigen::Matrix3d data = q.toRotationMatrix();
383  double pitch, yaw;
384  pitch = atan2(-data(2, 0), sqrt(data(0, 0) * data(0, 0) + data(1, 0) * data(1, 0)));
385  if (fabs(pitch) > M_PI / 2) {
386  yaw = atan2(-data(0, 1), data(1, 1));
387  } else {
388  yaw = atan2(data(1, 0), data(0, 0));
389  }
390  return yaw;
391 }
