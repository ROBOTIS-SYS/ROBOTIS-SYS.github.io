121 {

133  } else {

135  double tmp_min_y = last_min_y_;
136  double tmp_max_x = last_max_x_;
137  double tmp_max_y = last_max_y_;
138  last_min_x_ = *min_x;
139  last_min_y_ = *min_y;
140  last_max_x_ = *max_x;
141  last_max_y_ = *max_y;
142  *min_x = std::min(tmp_min_x, *min_x) - inflation_radius_;
143  *min_y = std::min(tmp_min_y, *min_y) - inflation_radius_;
144  *max_x = std::max(tmp_max_x, *max_x) + inflation_radius_;
145  *max_y = std::max(tmp_max_y, *max_y) + inflation_radius_;

164 void
165 InflationLayer::updateCosts(
166  antis_nav_costmap_2d::Costmap2D & master_grid, int min_i, int min_j,
167  int max_i,
168  int max_j)
169 {
170  std::lock_guard<Costmap2D::mutex_t> guard(*getMutex());
171  if (!enabled_ || (cell_inflation_radius_ == 0)) {
172  return;
173  }
174 
175  // make sure the inflation list is empty at the beginning of the cycle (should always be true)
176  for (auto & dist : inflation_cells_) {
177  RCLCPP_FATAL_EXPRESSION(
178  rclcpp::get_logger("antis_nav_costmap_2d"),
179  !dist.empty(), "The inflation list must be empty at the beginning of inflation");
180  }
181 
182  unsigned char * master_array = master_grid.getCharMap();
183  unsigned int size_x = master_grid.getSizeInCellsX(), size_y = master_grid.getSizeInCellsY();
184 
185  if (seen_.size() != size_x * size_y) {
186  RCLCPP_WARN(
187  rclcpp::get_logger(
188  "antis_nav_costmap_2d"), "InflationLayer::updateCosts(): seen_ vector size is wrong");
189  seen_ = std::vector<bool>(size_x * size_y, false);
190  }
191 
192  std::fill(begin(seen_), end(seen_), false);
193 
194  // We need to include in the inflation cells outside the bounding
195  // box min_i...max_j, by the amount cell_inflation_radius_. Cells
196  // up to that distance outside the box can still influence the costs
197  // stored in cells inside the box.
198  min_i -= static_cast<int>(cell_inflation_radius_);
199  min_j -= static_cast<int>(cell_inflation_radius_);
200  max_i += static_cast<int>(cell_inflation_radius_);
201  max_j += static_cast<int>(cell_inflation_radius_);
202 
203  min_i = std::max(0, min_i);
204  min_j = std::max(0, min_j);
205  max_i = std::min(static_cast<int>(size_x), max_i);
206  max_j = std::min(static_cast<int>(size_y), max_j);
207 
208  // Inflation list; we append cells to visit in a list associated with
209  // its distance to the nearest obstacle
210  // We use a map<distance, list> to emulate the priority queue used before,
211  // with a notable performance boost
212 
213  // Start with lethal obstacles: by definition distance is 0.0
214  auto & obs_bin = inflation_cells_[0];
215  for (int j = min_j; j < max_j; j++) {
216  for (int i = min_i; i < max_i; i++) {
217  int index = static_cast<int>(master_grid.getIndex(i, j));
218  unsigned char cost = master_array[index];
219  if (cost == LETHAL_OBSTACLE || (inflate_around_unknown_ && cost == NO_INFORMATION)) {
220  obs_bin.emplace_back(index, i, j, i, j);
221  }
222  }
223  }
224 
225  // Process cells by increasing distance; new cells are appended to the
226  // corresponding distance bin, so they
227  // can overtake previously inserted but farther away cells
228  for (const auto & dist_bin : inflation_cells_) {
229  for (std::size_t i = 0; i < dist_bin.size(); ++i) {
230  // Do not use iterator or for-range based loops to
231  // iterate though dist_bin, since it's size might
232  // change when a new cell is enqueued, invalidating all iterators
233  unsigned int index = dist_bin[i].index_;
234 
235  // ignore if already visited
236  if (seen_[index]) {
237  continue;
238  }
239 
240  seen_[index] = true;
241 
242  unsigned int mx = dist_bin[i].x_;
243  unsigned int my = dist_bin[i].y_;
244  unsigned int sx = dist_bin[i].src_x_;
245  unsigned int sy = dist_bin[i].src_y_;
246 
247  // assign the cost associated with the distance from an obstacle to the cell
248  unsigned char cost = costLookup(mx, my, sx, sy);
249  unsigned char old_cost = master_array[index];
250  if (old_cost == NO_INFORMATION &&
251  (inflate_unknown_ ? (cost > FREE_SPACE) : (cost >= INSCRIBED_INFLATED_OBSTACLE)))
252  {
253  master_array[index] = cost;
254  } else {
255  master_array[index] = std::max(old_cost, cost);
256  }
257 
258  // attempt to put the neighbors of the current cell onto the inflation list
259  if (mx > 0) {
260  enqueue(index - 1, mx - 1, my, sx, sy);
261  }
262  if (my > 0) {
263  enqueue(index - size_x, mx, my - 1, sx, sy);
264  }
265  if (mx < size_x - 1) {
266  enqueue(index + 1, mx + 1, my, sx, sy);
267  }
268  if (my < size_y - 1) {
269  enqueue(index + size_x, mx, my + 1, sx, sy);
270  }
271  }
272  }
273 
274  for (auto & dist : inflation_cells_) {
275  dist.clear();
276  dist.reserve(200);
277  }
278 }

289 void
290 InflationLayer::enqueue(
291  unsigned int index, unsigned int mx, unsigned int my,
292  unsigned int src_x, unsigned int src_y)
293 {
294  if (!seen_[index]) {
295  // we compute our distance table one cell further than the
296  // inflation radius dictates so we can make the check below
297  double distance = distanceLookup(mx, my, src_x, src_y);
298 
299  // we only want to put the cell in the list if it is within
300  // the inflation radius of the obstacle point
301  if (distance > cell_inflation_radius_) {
302  return;
303  }
304 
305  const unsigned int r = cell_inflation_radius_ + 2;
306 
307  // push the cell data onto the inflation list and mark
308  inflation_cells_[distance_matrix_[mx - src_x + r][my - src_y + r]].emplace_back(
309  index, mx, my, src_x, src_y);
310  }
311 }

313 void
314 InflationLayer::computeCaches()
315 {
316  std::lock_guard<Costmap2D::mutex_t> guard(*getMutex());
317  if (cell_inflation_radius_ == 0) {
318  return;
319  }
320 
321  cache_length_ = cell_inflation_radius_ + 2;
322 
323  // based on the inflation radius... compute distance and cost caches
324  if (cell_inflation_radius_ != cached_cell_inflation_radius_) {
325  cached_costs_.resize(cache_length_ * cache_length_);
326  cached_distances_.resize(cache_length_ * cache_length_);
327 
328  for (unsigned int i = 0; i < cache_length_; ++i) {
329  for (unsigned int j = 0; j < cache_length_; ++j) {
330  cached_distances_[i * cache_length_ + j] = hypot(i, j);
331  }
332  }
333 
334  cached_cell_inflation_radius_ = cell_inflation_radius_;
335  }
336 
337  for (unsigned int i = 0; i < cache_length_; ++i) {
338  for (unsigned int j = 0; j < cache_length_; ++j) {
339  cached_costs_[i * cache_length_ + j] = computeCost(cached_distances_[i * cache_length_ + j]);
340  }
341  }
342 
343  int max_dist = generateIntegerDistances();
344  inflation_cells_.clear();
345  inflation_cells_.resize(max_dist + 1);
346  for (auto & dist : inflation_cells_) {
347  dist.reserve(200);
348  }
349 }

351 int
352 InflationLayer::generateIntegerDistances()
353 {
354  const int r = cell_inflation_radius_ + 2;
355  const int size = r * 2 + 1;
356 
357  std::vector<std::pair<int, int>> points;
358 
359  for (int y = -r; y <= r; y++) {
360  for (int x = -r; x <= r; x++) {
361  if (x * x + y * y <= r * r) {
362  points.emplace_back(x, y);
363  }
364  }
365  }
366 
367  std::sort(
368  points.begin(), points.end(),
369  [](const std::pair<int, int> & a, const std::pair<int, int> & b) -> bool {
370  return a.first * a.first + a.second * a.second < b.first * b.first + b.second * b.second;
371  }
372  );
373 
374  std::vector<std::vector<int>> distance_matrix(size, std::vector<int>(size, 0));
375  std::pair<int, int> last = {0, 0};
376  int level = 0;
377  for (auto const & p : points) {
378  if (p.first * p.first + p.second * p.second !=
379  last.first * last.first + last.second * last.second)
380  {
381  level++;
382  }
383  distance_matrix[p.first + r][p.second + r] = level;
384  last = p;
385  }
386 
387  distance_matrix_ = distance_matrix;
388  return level;
389 }
