70 geometry_msgs::msg::PoseArray PoseEstimator::get_pose_array()
71 {
72  geometry_msgs::msg::PoseArray result;
73  result.header.stamp = p_input_cloud_.header.stamp;
74 
75  if (p_button_poses_->size() == 0) {
76  geometry_msgs::msg::Pose empty;
77  empty.position.x = 0.0;
78  empty.position.y = 0.0;
79  empty.position.z = 0.0;
80  empty.orientation.w = 1.0;
81  empty.orientation.x = 0.0;
82  empty.orientation.y = 0.0;
83  empty.orientation.z = 0.0;
84  result.poses.assign(1, empty);
85  } else {
86  for (auto it = p_button_poses_->begin(); it != p_button_poses_->end(); it++) {
87  result.poses.push_back(it->second.pose());
88  }
89  }
90  return result;
91 }

106 perception_interfaces::msg::DetectedButtonArray PoseEstimator::get_detected_button_array()
107 {
108  perception_interfaces::msg::DetectedButtonArray result;
109  result.header.stamp = p_input_cloud_.header.stamp;
110 
111  for (size_t i = 0; i < p_left_buttons_->size(); i++) {
112  for (size_t j = 0; j < p_right_buttons_->size(); j++) {
113  std::string left_label = p_left_buttons_->at(i).class_name;
114  if (left_label.compare(p_right_buttons_->at(j).class_name) == 0) {
115  if (p_button_poses_->find(left_label) != p_button_poses_->end()) {
116  result.button_name.push_back(left_label);
117  result.button_state.push_back(p_left_buttons_->at(i).button_state);
118  result.button_pose.push_back(p_button_poses_->at(left_label).pose());
119  }
120  }
121  }
122  }
123 
124  return result;
125 }

142 perception_interfaces::msg::DetectedButtonArray PoseEstimator::get_detected_button_array(
143  std::string button_label,
144  bool btn_state)
145 {
146  perception_interfaces::msg::DetectedButtonArray result;
147  result.header.stamp = p_input_cloud_.header.stamp;
148 
149  for (auto it = p_button_poses_->begin(); it != p_button_poses_->end(); it++) {
150  result.button_name.push_back(it->first);
151  if (button_label == it->first) {
152  result.button_state.push_back(btn_state);
153  } else {
154  result.button_state.push_back(false);
155  }
156  result.button_pose.push_back(it->second.pose());
157  }
158 
159  return result;
