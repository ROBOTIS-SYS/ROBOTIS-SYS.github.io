109 void LayeredCostmap::updateMap(double robot_x, double robot_y, double robot_yaw)
110 {
111  // Lock for the remainder of this function, some plugins (e.g. VoxelLayer)
112  // implement thread unsafe updateBounds() functions.
113  std::unique_lock<Costmap2D::mutex_t> lock(*(costmap_.getMutex()));
114 
115  // if we're using a rolling buffer costmap...
116  // we need to update the origin using the robot's position
117  if (rolling_window_) {
118  double new_origin_x = robot_x - costmap_.getSizeInMetersX() / 2;
119  double new_origin_y = robot_y - costmap_.getSizeInMetersY() / 2;
120  costmap_.updateOrigin(new_origin_x, new_origin_y);
121  }
122 
123  if (isOutofBounds(robot_x, robot_y)) {
124  RCLCPP_WARN(
125  rclcpp::get_logger("antis_nav_costmap_2d"),
126  "Robot is out of bounds of the costmap!");
127  }
128 
129  if (plugins_.size() == 0) {
130  return;
131  }
132 
133  minx_ = miny_ = std::numeric_limits<double>::max();
134  maxx_ = maxy_ = std::numeric_limits<double>::lowest();
135 
136  for (vector<std::shared_ptr<Layer>>::iterator plugin = plugins_.begin();
137  plugin != plugins_.end(); ++plugin)
138  {
139  double prev_minx = minx_;
140  double prev_miny = miny_;
141  double prev_maxx = maxx_;
142  double prev_maxy = maxy_;
143  (*plugin)->updateBounds(robot_x, robot_y, robot_yaw, &minx_, &miny_, &maxx_, &maxy_);
144  if (minx_ > prev_minx || miny_ > prev_miny || maxx_ < prev_maxx || maxy_ < prev_maxy) {
145  RCLCPP_WARN(
146  rclcpp::get_logger(
147  "antis_nav_costmap_2d"), "Illegal bounds change, was [tl: (%f, %f), br: (%f, %f)], but "
148  "is now [tl: (%f, %f), br: (%f, %f)]. The offending layer is %s",
149  prev_minx, prev_miny, prev_maxx, prev_maxy,
150  minx_, miny_, maxx_, maxy_,
151  (*plugin)->getName().c_str());
152  }
153  }
154 
155  int x0, xn, y0, yn;
156  costmap_.worldToMapEnforceBounds(minx_, miny_, x0, y0);
157  costmap_.worldToMapEnforceBounds(maxx_, maxy_, xn, yn);
158 
159  x0 = std::max(0, x0);
160  xn = std::min(static_cast<int>(costmap_.getSizeInCellsX()), xn + 1);
161  y0 = std::max(0, y0);
162  yn = std::min(static_cast<int>(costmap_.getSizeInCellsY()), yn + 1);
163 
164  RCLCPP_DEBUG(
165  rclcpp::get_logger(
166  "antis_nav_costmap_2d"), "Updating area x: [%d, %d] y: [%d, %d]", x0, xn, y0, yn);
167 
168  if (xn < x0 || yn < y0) {
169  return;
170  }
171 
172  costmap_.resetMap(x0, y0, xn, yn);
173  for (vector<std::shared_ptr<Layer>>::iterator plugin = plugins_.begin();
174  plugin != plugins_.end(); ++plugin)
175  {
176  (*plugin)->updateCosts(costmap_, x0, y0, xn, yn);
177  }
178 
179  bx0_ = x0;
180  bxn_ = xn;
181  by0_ = y0;
182  byn_ = yn;
183 
184  initialized_ = true;
185 }

187 bool LayeredCostmap::isCurrent()
188 {
189  current_ = true;
190  for (vector<std::shared_ptr<Layer>>::iterator plugin = plugins_.begin();
191  plugin != plugins_.end(); ++plugin)
192  {
193  current_ = current_ && (*plugin)->isCurrent();
194  }
195  return current_;
