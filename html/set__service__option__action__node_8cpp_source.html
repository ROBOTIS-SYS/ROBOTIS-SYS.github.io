105 bool SetServiceOptionAction::parse_yaml(std::string file_path)
106 {
107  try {
108  YAML::Node file = YAML::LoadFile(file_path.c_str());
109 
110  for (YAML::iterator it = file.begin(); it != file.end(); it++) {
111  std::string opt_name = it->first.as<std::string>();
112  std::cout << "\n----------" << std::endl;
113  std::cout << "opt_name : " << opt_name << std::endl;
114  if (opt_name == "move_sector") {
115  if (it->second.IsMap() == true) {
116  parse_sector_yaml(it->second);
117  }
118  } else {
119  for (YAML::iterator itt = it->second.begin(); itt != it->second.end();
120  itt++)
121  {
122  std::string param_name = itt->first.as<std::string>();
123  std::cout << "-- " << param_name << " : ";
124  if (itt->second.IsScalar()) { // scalar
125  set_blackboard(param_name, itt->second.as<std::string>());
126 
127  } else if (itt->second.IsMap()) { // map
128  YAML::Node sub_node = itt->second;
129  auto sub_node_first_it = sub_node.begin();
130  std::string type = sub_node_first_it->first.as<std::string>();
131 
132  if (type == "ros_param") {
133  std::string ros_param_value = "";
134  for (YAML::iterator sub_it = sub_node_first_it;
135  sub_it != sub_node.end(); sub_it++)
136  {
137  make_ros_param_value(sub_it->second, ros_param_value);
138  }
139  if (ros_param_value != "") {
140  set_blackboard(param_name, ros_param_value);
141  }
142  } else if (type == "geometry_point") {
143  set_blackboard_point(
144  param_name,
145  sub_node_first_it->second["x"].as<double>(),
146  sub_node_first_it->second["y"].as<double>(),
147  sub_node_first_it->second["z"].as<double>());
148  } else if (type == "geometry_pose2d") {
149  set_blackboard_2d_pose(
150  param_name, sub_node_first_it->second["x"].as<double>(),
151  sub_node_first_it->second["y"].as<double>(),
152  sub_node_first_it->second["theta"].as<double>());
153  } else {
154  continue;
155  }
156  } else if (itt->second.IsSequence()) { // array
157  } else {
158  }
159  }
160  }
161  }
162  } catch (const std::exception & e) {
163  std::stringstream log_stream;
164  log_stream << "Fail to load process data file. : " << e.what();
165  RCLCPP_WARN_STREAM(get_logger(), log_stream.str());
166  return false;
167  }
168  return true;
169 }

171 bool SetServiceOptionAction::parse_sector_yaml(YAML::Node & sector_node)
172 {
173  if (sector_node.IsMap() == false) {
174  return false;
175  }
176 
177  // 타입별 체크
178  for (YAML::iterator it = sector_node.begin(); it != sector_node.end(); it++) {
179  // index에 지정된 파라미터를 하나씩 파싱
180  std::string sector_type = it->first.as<std::string>(); // ex. push
181  YAML::Node sector_param_node = it->second;
182 
183  for (auto params_it = sector_param_node.begin();
184  params_it != sector_param_node.end(); params_it++)
185  {
186  params_it->first.as<int>(); // index, not use
187  YAML::Node param_node = params_it->second;
188  if (param_node.IsMap() == false) { // 형식 검사
189  continue;
190  }
191  // 층 정보 설정
192  std::vector<std::string> floor_list =
193  params_it->second["floor"].as<std::vector<std::string>>();
194 
195  if (floor_list.size() == 0) { // 필수 파라미터(층) 검사
196  // 비어 있는 경우 모든 층에 설정 적용
197  floor_list.push_back("all");
198  }
199 
200  // 하위 항목을 재귀적으로 탐색하여 blackboard 키를 생성
201  // blackboard 값 종류에 따라 설정
202  // Map: 위치별 정보 탐색
203  // Scalar : 정보 저장
204  for (auto param_it = param_node.begin(); param_it != param_node.end();
205  param_it++)
206  {
207  std::string param_name = param_it->first.as<std::string>(); // ex. outside등
208  if (param_it->second.IsScalar()) { // scalar, ex.moving_time
209  for (auto floor : floor_list) {
210  std::string param_key = "sector_" + sector_type + "_" + floor + "_" + param_name;
211  std::cout << "-- " << param_key << " : ";
212  set_blackboard(param_key, param_it->second.as<std::string>());
213  }
214  } else if (param_it->second.IsMap()) { // map -> location, ex. inside/outside
215  YAML::Node param_sub_node = param_it->second;
216  for (auto param_sub_it = param_sub_node.begin();
217  param_sub_it != param_sub_node.end(); param_sub_it++)
218  {
219  std::string param_sub_name = param_sub_it->first.as<std::string>();
220  if (param_sub_it->second.IsScalar()) { // scalar, 기본 blackboard 형식
221  for (auto floor : floor_list) {
222  std::string param_key = "sector_" + sector_type + "_" + floor +
223  "_" + param_name + "_" + param_sub_name;
224  std::cout << "-- " << param_key << " : ";
225  set_blackboard(param_key, param_sub_it->second.as<std::string>());
226  }
227  } else if (param_sub_it->second.IsMap()) { // map, 특수 blackboard 형식
228  YAML::Node sub_node = param_sub_it->second;
229  auto sub_node_first_it = sub_node.begin();
230  std::string param_type =
231  sub_node_first_it->first.as<std::string>();
232 
233  if (param_type == "ros_param") {
234  std::string ros_param_value = "";
235  for (auto sub_it = sub_node_first_it; sub_it != sub_node.end(); sub_it++) {
236  make_ros_param_value(sub_it->second, ros_param_value);
237  }
238  if (ros_param_value != "") {
239  for (auto floor : floor_list) {
240  std::string param_key = "sector_" + sector_type + "_" +
241  floor + "_" + param_name + "_" + param_sub_name;
242  std::cout << "-- " << param_key << " : ";
243  set_blackboard(param_key, ros_param_value);
244  }
245  }
246  } else if (param_type == "geometry_point") {
247  for (auto floor : floor_list) {
248  std::string param_key = "sector_" + sector_type + "_" +
249  floor + "_" + param_name + "_" + param_sub_name;
250  std::string param_value = sub_node_first_it->second["x"].as<std::string>() + ";" +
251  sub_node_first_it->second["y"].as<std::string>() + ";" +
252  sub_node_first_it->second["z"].as<std::string>();
253  std::cout << "-- " << param_key << " : ";
254  set_blackboard(param_key, param_value);
255 // set_blackboard_point(
256 // param_name, sub_node_first_it->second["x"].as<double>(),
257 // sub_node_first_it->second["y"].as<double>(),
258 // sub_node_first_it->second["z"].as<double>());
259  }
260  } else if (param_type == "geometry_pose2d") {
261  for (auto floor : floor_list) {
262  std::string param_key = "sector_" + sector_type + "_" +
263  floor + "_" + param_name + "_" + param_sub_name;
264  std::string param_value = sub_node_first_it->second["x"].as<std::string>() + ";" +
265  sub_node_first_it->second["y"].as<std::string>() + ";" +
266  sub_node_first_it->second["theta"].as<std::string>();
267  std::cout << "-- " << param_key << " : ";
268  set_blackboard(param_key, param_value);
269 // set_blackboard_2d_pose(
270 // param_name, sub_node_first_it->second["x"].as<double>(),
271 // sub_node_first_it->second["y"].as<double>(),
272 // sub_node_first_it->second["theta"].as<double>());
273  }
274  } else {
275  continue;
276  }
277  }
278  }
279  }
280  }
281  }
282  }
283  return true;
