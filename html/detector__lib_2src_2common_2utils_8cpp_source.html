83 Eigen::Matrix3d PerceptionUtils::euler_to_rot(
84  double x_deg,
85  double y_deg,
86  double z_deg,
87  ROTATION_ORDER order)
88 {
89  double x_rad = x_deg / 180 * pi_;
90  double y_rad = y_deg / 180 * pi_;
91  double z_rad = z_deg / 180 * pi_;
92 
93  Eigen::AngleAxisd aad_z(z_rad, Eigen::Vector3d::UnitZ());
94  Eigen::AngleAxisd aad_y(y_rad, Eigen::Vector3d::UnitY());
95  Eigen::AngleAxisd aad_x(x_rad, Eigen::Vector3d::UnitX());
96 
97  Eigen::Quaternion<double> q;
98  switch (order) {
99  case ROTATION_ORDER::XYZ:
100  q = aad_x * aad_y * aad_z;
101  break;
102  case ROTATION_ORDER::ZYX:
103  q = aad_z * aad_y * aad_x;
104  break;
105  default: // default order : zyx
106  q = aad_z * aad_y * aad_x;
107  break;
108  }
109 
110  return q.matrix();
111 }

113 Eigen::Matrix4d PerceptionUtils::rpyxyz_to_tmat(std::vector<double> & rpyxyz, ROTATION_ORDER order)
114 {
115  if (rpyxyz.size() < 2) {
116  return Eigen::Matrix4d::Identity();
117  }
118 
119  double roll = rpyxyz.at(0);
120  double pitch = rpyxyz.at(1);
121  double yaw = rpyxyz.at(2);
122  double x = rpyxyz.at(3);
123  double y = rpyxyz.at(4);
124  double z = rpyxyz.at(5);
125 
126  Eigen::Matrix3d rotation = euler_to_rot(roll, pitch, yaw, order);
127  Eigen::Vector3d translation(x, y, z);
128  Eigen::Matrix4d transformation = Eigen::Matrix4d::Identity();
129  transformation.block<3, 3>(0, 0) = rotation;
130  transformation.block<3, 1>(0, 3) = translation;
131 
132  return transformation;
133 }

160 std::vector<std::string> PerceptionUtils::split_string(const std::string & input, char delimiter)
161 {
162  std::vector<std::string> tokens;
163  size_t start = 0, end = 0;
164 
165  while ((end = input.find(delimiter, start)) != std::string::npos) {
166  tokens.push_back(input.substr(start, end - start));
167  start = end + 1;
168  }
169 
170  tokens.push_back(input.substr(start));
171  return tokens;
172 }

213 bool PerceptionUtils::calculate_intersected_point_between_line_and_plane(
214  Eigen::Vector3f & line_p1,
215  Eigen::Vector3f & line_p2,
216  Eigen::Vector3f & plane_normal,
217  Eigen::Vector3f & plane_center,
218  float & t,
219  Eigen::Vector3f & inter_pt)
220 {
221  Eigen::Vector3f line_p21;
222  float num, den;
223  float fabsden, fabstolerance;
224 
225  // Compute line vector
226  line_p21 = line_p2 - line_p1;
227  // Compute denominator. If ~0, line and plane are parallel.
228  num = plane_normal.dot(plane_center) - line_p1.dot(plane_normal);
229  den = plane_normal.dot(line_p21);
230  // If denominator with respect to numerator is "zero", then the line and
231  // plane are considered parallel.
232  // trying to avoid an expensive call to fabs()
233  if (den < 0.0) {
234  fabsden = -den;
235  } else {
236  fabsden = den;
237  }
238  if (num < 0.0) {
239  fabstolerance = -num * 1.0e-06;
240  } else {
241  fabstolerance = num * 1.0e-06;
242  }
243  if (fabsden <= fabstolerance) {
244  t = 1.0e+299;
245  return false;
246  }
247  // valid intersection
248  t = num / den;
249  inter_pt = line_p1 + t * line_p21;
250 
251  if (t >= 0.0 && t <= 1.0) {
252  return true;
253  } else {
254  return false;
255  }
256 }

258 bool PerceptionUtils::evaluate_plane_equation(
259  Eigen::Vector3f & pt,
260  Eigen::Vector3f & plane_normal,
261  Eigen::Vector3f & plane_center)
262 {
263  // ax + by + cz + d <= 0 인지를 확인
264  float d = -plane_normal.dot(plane_center);
265  float result = pt.dot(plane_normal) + d;
266  return (result <= 1e-5) ? true : false;
267 }
