221 void PageInputDeliverySwingDoor::set_tray_target_btn(

234  bool next_btn_enabled = false;

236  bool merged;
237  bool phone_num_added;
238  for (size_t i = 0; i < btn_list.size(); i++) {

241  btn_list[i]->setIcon(QIcon(":/icons_600p/icon_dest.png"));

246  }
247  lbl_phone_num_list[i]->setText(target_phone_number.c_str());
248  }
249 
250  dest_added = (btn_list[i]->text().compare(DEFAULT_BTN_TEXT) != 0);
251  merged = (
252  (ros_node_->config_cargo_merge_12_.compare("Y") == 0 && i == 0) ||
253  (ros_node_->config_cargo_merge_34_.compare("Y") == 0 && i == 2));
254  phone_num_added = (lbl_phone_num_list[i]->text().compare("") != 0);
255 
256  if (!dest_added) {
257  lbl_x_list[i]->setVisible(false);
258  btn_list[i]->setStyleSheet(make_style_sheet_qstr(merged, false, phone_num_added));
259  } else {
260  next_btn_enabled = true;
261  if ((ros_node_->config_cargo_merge_12_.compare("Y") == 0 && i == 1) ||
262  (ros_node_->config_cargo_merge_34_.compare("Y") == 0 && i == 3))
263  {
264  lbl_x_list[i]->setVisible(false);
265  } else {
266  lbl_x_list[i]->setVisible(true);
267  }
268  btn_list[i]->setStyleSheet(make_style_sheet_qstr(merged, true, phone_num_added));
269  }

380 
381  std::string destination = "";

383 
384  bool dialog_accepted;
385 
386  // Target name 또는 Room number 입력받기
387  if (ros_node_->input_dest_page_name_.compare("SelectTarget") == 0) {
388  dialog_accepted = get_user_input_target_name(destination);
389  } else {
390  dialog_accepted = get_user_input_room_number(destination);
391  }
392 
393  // 수령인 확인 배송일 경우 휴대전화번호 입력받기
394  if (ros_node_->is_registered_delivery_mode()) {
395  if (dialog_accepted) {
396  dialog_accepted = get_user_input_phone_number(phone_number);
397  }
398  }
399 
400  if (dialog_accepted) {
401  set_tray_target_btn(tray_num, destination, phone_number);
402 
403  if (tray_num == 1 && ros_node_->config_cargo_merge_12_.compare("Y") == 0) {
404  set_tray_target_btn(2, destination, phone_number);
405  } else if (tray_num == 3 && ros_node_->config_cargo_merge_34_.compare("Y") == 0) {
406  set_tray_target_btn(4, destination, phone_number);

408  }

442 void PageInputDeliverySwingDoor::on_btnNext_clicked()
443 {
444  std::vector<QPushButton *> btn_list = {ui->btnDest1, ui->btnDest2, ui->btnDest3, ui->btnDest4};
445  std::vector<QLabel *> lbl_phone_num_list = {
446  ui->lblPhoneNum1, ui->lblPhoneNum2, ui->lblPhoneNum3, ui->lblPhoneNum4};
447 
448  RCLCPP_INFO(ros_node_->get_logger(), "[InputDelivery] NEXT clicked");
449 
450  if (ros_node_->cargo_lock_state_[0] == 0 || ros_node_->cargo_lock_state_[1] == 0 ||
451  ros_node_->cargo_lock_state_[2] == 0 || ros_node_->cargo_lock_state_[3] == 0)
452  {
453  is_tray_closing_ = true;
454  std::thread close_thread(tray_close_thread, this);
455  close_thread.detach();
456 
457  QGraphicsBlurEffect * gbe = new QGraphicsBlurEffect();
458  gbe->setBlurRadius(10);
459  this->setGraphicsEffect(gbe);
460  dlg_tray_moving_->set_closing_text();
461  dlg_tray_moving_->exec();
462 
463  while (is_tray_closing_) {
464  std::this_thread::sleep_for(std::chrono::duration<int>(1));
465  }
466  this->setGraphicsEffect(nullptr);
467 
468  RCLCPP_INFO(ros_node_->get_logger(), "cargo_closed ");
469  }
470 
471  // 서랍 닫기 성공 여부 확인
472  if (ros_node_->cargo_lock_state_[0] == 1 && ros_node_->cargo_lock_state_[1] == 1 &&
473  ros_node_->cargo_lock_state_[2] == 1 && ros_node_->cargo_lock_state_[3] == 1)
474  {
475  if (ros_node_->input_dest_page_name_.compare("SelectTarget") == 0) {
476  std::vector<aimbot_state_interfaces::msg::Destination> destination_list;
477  ros_node_->get_target_list(destination_list);
478  if (destination_list.size() == 0) {
479  return;
480  }
481  for (const auto & it : destination_list) {
482  for (size_t i = 0; i < btn_list.size(); i++) {
483  std::string input_dest = btn_list[i]->text().toStdString();
484  std::string input_phonenum = btn_list[i]->text().toStdString();
485  if (it.name.compare(input_dest) == 0) {
486  ros_node_->cargo_goal_[i] = it.key;
487  ros_node_->cargo_phonenum_[i] = input_phonenum;
488  }
489  }
490  }
491  } else {
492  for (size_t i = 0; i < btn_list.size(); i++) {
493  if (btn_list[i]->text().compare(DEFAULT_BTN_TEXT) != 0) {
494  std::string input_dest = btn_list[i]->text().toStdString();
495  std::string input_phonenum = btn_list[i]->text().toStdString();
496  ros_node_->cargo_goal_[i] = input_dest;
497  ros_node_->cargo_phonenum_[i] = remove_hyphens(input_phonenum);
498  }
499  }
500  }
501 
502  std::string opt = "";
503  std::string input_dest;
504  std::string input_phonenum;
505  for (size_t i = 0; i < btn_list.size(); i++) {
506  input_dest = btn_list[i]->text().toStdString();
507  input_phonenum = lbl_phone_num_list[i]->text().toStdString();
508  opt +=
509  ((btn_list[i]->text().compare(DEFAULT_BTN_TEXT)) ?
510  input_dest + std::string(1, DEST_PHONENUM_DELIM) + input_phonenum : "") +
511  "|";
512  }
513  emit pageChanged("DeliveryStart", opt);
514  } else {
515  RCLCPP_INFO(
516  ros_node_->get_logger(), "[InputDelivery] 열린 서랍 있음. (%d, %d, %d, %d)",
517  ros_node_->cargo_lock_state_[0],
518  ros_node_->cargo_lock_state_[1],
519  ros_node_->cargo_lock_state_[2],
520  ros_node_->cargo_lock_state_[3]);
521 
522  // emit showDialog("TrayStillOpen");
523 
524  if (close_cargo_cnt_ > 5) { // 재시도 5번 실패시 FMS 알람
525  std::list<eve_server_interfaces::msg::AdditionalField> additional_info;
526  eve_server_interfaces::msg::AdditionalField addi_field_msg;
527  addi_field_msg.name = "TITLE";
528  addi_field_msg.value = "서랍 닫기 실패";
529  additional_info.push_back(addi_field_msg);
530
