54 bool CostmapTopicCollisionChecker::isCollisionFree(
55  const geometry_msgs::msg::Pose2D & pose)
56 {
57  try {
58  if (scorePose(pose) >= LETHAL_OBSTACLE) {
59  return false;
60  }
61  return true;
62  } catch (const IllegalPoseException & e) {
63  RCLCPP_ERROR(rclcpp::get_logger(name_), "%s", e.what());
64  return false;
65  } catch (const CollisionCheckerException & e) {
66  RCLCPP_ERROR(rclcpp::get_logger(name_), "%s", e.what());
67  return false;
68  } catch (...) {
69  RCLCPP_ERROR(rclcpp::get_logger(name_), "Failed to check pose score!");
70  return false;
71  }
72 }

74 double CostmapTopicCollisionChecker::scorePose(
75  const geometry_msgs::msg::Pose2D & pose)
76 {
77  try {
78  collision_checker_.setCostmap(costmap_sub_.getCostmap());
79  } catch (const std::runtime_error & e) {
80  throw CollisionCheckerException(e.what());
81  }
82 
83  unsigned int cell_x, cell_y;
84  if (!collision_checker_.worldToMap(pose.x, pose.y, cell_x, cell_y)) {
85  RCLCPP_DEBUG(rclcpp::get_logger(name_), "Map Cell: [%d, %d]", cell_x, cell_y);
86  throw IllegalPoseException(name_, "Pose Goes Off Grid.");
87  }
88 
89  return collision_checker_.footprintCost(getFootprint(pose));
90 }

92 Footprint CostmapTopicCollisionChecker::getFootprint(const geometry_msgs::msg::Pose2D & pose)
93 {
94  Footprint footprint;
95  if (!footprint_sub_.getFootprint(footprint)) {
96  throw CollisionCheckerException("Current footprint not available.");
97  }
98 
99  Footprint footprint_spec;
100  unorientFootprint(footprint, footprint_spec);
101  transformFootprint(pose.x, pose.y, pose.theta, footprint_spec, footprint);
102 
103  return footprint;
104 }

106 void CostmapTopicCollisionChecker::unorientFootprint(
107  const std::vector<geometry_msgs::msg::Point> & oriented_footprint,
108  std::vector<geometry_msgs::msg::Point> & reset_footprint)
109 {
110  geometry_msgs::msg::PoseStamped current_pose;
111  if (!antis_nav_util::getCurrentPose(
112  current_pose, tf_, global_frame_, robot_base_frame_,
113  transform_tolerance_))
114  {
115  throw CollisionCheckerException("Robot pose unavailable.");
116  }
117 
118  double x = current_pose.pose.position.x;
119  double y = current_pose.pose.position.y;
120  double theta = tf2::getYaw(current_pose.pose.orientation);
121 
122  Footprint temp;
123  transformFootprint(-x, -y, 0, oriented_footprint, temp);
124  transformFootprint(0, 0, -theta, temp, reset_footprint);
125 }
