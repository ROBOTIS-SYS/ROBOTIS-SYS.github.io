52 BT::NodeStatus OpenTrayAction::tick()
53 {
54  std::cout << std::endl;
55  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
56  RCLCPP_INFO_STREAM(get_logger(), "<TICK> OpenTrayAction");
57  }
58 
59  int tray_num;
60  if (!getInput<int>("tray_num", tray_num)) {
61  return BT::NodeStatus::FAILURE;
62  }
63 
64  std::string open_or_close;
65  if (!getInput<std::string>("open_or_close", open_or_close)) {
66  return BT::NodeStatus::FAILURE;
67  }
68 
69  if (open_or_close == "open") {
70  // cargo_unlock_cmd init publisher
71  std::string topic_name = "/aimbot_base/cargo_unlock_cmd";
72  if (cargo_unlock_cmd_pub_ == nullptr) {
73  bool on_blackboard =
74  config().blackboard->get<decltype(cargo_unlock_cmd_pub_)>(
75  topic_name, cargo_unlock_cmd_pub_);
76 
77  if (on_blackboard == false) {
78  cargo_unlock_cmd_pub_ = node_->create_publisher<aimbot_hw_interfaces::msg::CargoUnlockCmd>(
79  topic_name, 10);
80  config().blackboard->set<decltype(cargo_unlock_cmd_pub_)>(
81  topic_name, cargo_unlock_cmd_pub_);
82  }
83  }
84 
85  open_tray(tray_num);
86  } else if (open_or_close == "close") {
87  // cargo_lock_cmd init publisher
88  std::string topic_name = "/aimbot_base/cargo_lock_cmd";
89  if (cargo_lock_cmd_pub_ == nullptr) {
90  bool on_blackboard =
91  config().blackboard->get<decltype(cargo_lock_cmd_pub_)>(
92  topic_name, cargo_lock_cmd_pub_);
93 
94  if (on_blackboard == false) {
95  cargo_lock_cmd_pub_ = node_->create_publisher<aimbot_hw_interfaces::msg::CargoUnlockCmd>(
96  topic_name, 10);
97  config().blackboard->set<decltype(cargo_lock_cmd_pub_)>(
98  topic_name, cargo_lock_cmd_pub_);
99  }
100  }
101 
102  close_tray(tray_num);
103  }
104 
105 
106  if (isHaltRequested()) {
107  return BT::NodeStatus::IDLE;
108  }
109 
110  return BT::NodeStatus::SUCCESS;
111 }
