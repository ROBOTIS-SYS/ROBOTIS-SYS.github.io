49 {
50  if (data_[EV_STATUS_1] & 0b10111110) {
51  return false;
52  }
53  if (!static_cast<bool>(data_[EV_STATUS_1] & 0b00000001)) {
54  return false;
55  }
56  if (data_[EV_STATUS_2] & 0b00111111) {

60  return false;

63  return false;
64  }
65  if (data_[EV_STATUS_4] & 0b11100000) {
66  return false;
67  }

137 EvInfo::door_state EvInfo::ev_door_state(door_side door_side_)
138 {
139  uint8_t door_state_dol;
140  uint8_t door_state_dcl;
141  door_state door_state_;
142  if (door_side_ == door_side::FRONT) {
143  door_state_dol = data_[EV_STATUS_3] & 0b00000010;
144  door_state_dcl = data_[EV_STATUS_4] & 0b00000010;
145  if (door_state_dol == 0b00000010) {
146  door_state_ = door_state::OPEN;
147  } else if (door_state_dcl == 0b00000010) {
148  door_state_ = door_state::CLOSE;
149  } else {
150  if (last_door_state_ == door_state::CLOSE || last_door_state_ == door_state::OPENING) {
151  door_state_ = door_state::OPENING;
152  } else if (last_door_state_ == door_state::OPEN || last_door_state_ == door_state::CLOSING) {
153  door_state_ = door_state::CLOSING;
154  } else {
155  door_state_ = door_state::UNKNOWN;
156  }
157  }
158  } else {
159  door_state_dol = data_[EV_STATUS_3] & 0b00000001;
160  door_state_dcl = data_[EV_STATUS_4] & 0b00000001;
161  if (door_state_dol == 0b00000001) {
162  door_state_ = door_state::OPEN;
163  } else if (door_state_dcl == 0b00000001) {
164  door_state_ = door_state::CLOSE;
165  } else {
166  if (last_door_state_ == door_state::CLOSE || last_door_state_ == door_state::OPENING) {
167  door_state_ = door_state::OPENING;
168  } else if (last_door_state_ == door_state::OPEN || last_door_state_ == door_state::CLOSING) {
169  door_state_ = door_state::CLOSING;
170  } else {
171  door_state_ = door_state::UNKNOWN;
172  }
173  }
174  }
175  last_door_state_ = door_state_;
176  return door_state_;
177 }

252 {

255  real_floor = static_cast<int>(protocol_floor) - bottom_floor_ + 1;
256  } else {
257  if (protocol_floor <= abs(bottom_floor_)) {
258  real_floor = static_cast<int>(protocol_floor) + bottom_floor_ - 1;
259  } else {
