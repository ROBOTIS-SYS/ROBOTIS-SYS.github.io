54 void TeleopKeyboard::process()
55 {
56  uint8_t status = 0;
57  double target_linear_velocity = 0.0;
58  double target_angular_velocity = 0.0;
59 
60  geometry_msgs::msg::Twist goal_base_velocity;
61 
62  try {
63  print_msgs();
64  while (1) {
65  unsigned char ch = static_cast<char>(std::getchar());
66 
67  if (ch == 'w') {
68  target_linear_velocity =
69  check_linear_velocity_limit(target_linear_velocity + LIN_VEL_STEP_SIZE);
70  status += 1;
71  print_vels(target_linear_velocity, target_angular_velocity);
72  } else if (ch == 'x') {
73  target_linear_velocity =
74  check_linear_velocity_limit(target_linear_velocity - LIN_VEL_STEP_SIZE);
75  status += 1;
76  print_vels(target_linear_velocity, target_angular_velocity);
77  } else if (ch == 'a') {
78  target_angular_velocity =
79  check_angular_velocity_limit(target_angular_velocity + ANG_VEL_STEP_SIZE);
80  status = status + 1;
81  print_vels(target_linear_velocity, target_angular_velocity);
82  } else if (ch == 'd') {
83  target_angular_velocity =
84  check_angular_velocity_limit(target_angular_velocity - ANG_VEL_STEP_SIZE);
85  status = status + 1;
86  print_vels(target_linear_velocity, target_angular_velocity);
87  } else if (ch == 's') {
88  target_linear_velocity = 0.0;
89  target_angular_velocity = 0.0;
90  print_vels(target_linear_velocity, target_angular_velocity);
91  } else {
92  if (ch == 'q') {
93  printf("Pressed Quit \n");
94  throw 0;
95  }
96  }
97 
98  if (status == 20) {
99  print_msgs();
100  status = 0;
101  }
102 
103  geometry_msgs::msg::Twist twist;
104 
105  twist.linear.x = target_linear_velocity;
106  twist.linear.y = 0.0;
107  twist.linear.z = 0.0;
108 
109  twist.angular.x = 0.0;
110  twist.angular.y = 0.0;
111  twist.angular.z = target_angular_velocity;
112 
113  cmd_vel_pub_->publish(twist);
114  }
115  } catch (int e) {
116  // finally
117  geometry_msgs::msg::Twist twist;
118 
119  twist.linear.x = 0.0;
120  twist.linear.y = 0.0;
121  twist.linear.z = 0.0;
122 
123  twist.angular.x = 0.0;
124  twist.angular.y = 0.0;
125  twist.angular.z = 0.0;
126 
127  cmd_vel_pub_->publish(twist);
128 
129  // Terminate the node
130  this->~TeleopKeyboard();
131  }
132 }

161 double TeleopKeyboard::constrain(double input_vel, double low_bound, double high_bound)
162 {
163  if (input_vel < low_bound) {
164  input_vel = low_bound;
165  } else if (input_vel > high_bound) {
166  input_vel = high_bound;
167  } else {
168  input_vel = input_vel;
169  }
170 
171  return input_vel;
172 }
