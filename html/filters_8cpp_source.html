52 geometry_msgs::msg::Pose LowPassFilter::get_filtering_data(
53  geometry_msgs::msg::Pose raw_data,
54  double sampling_time)
55 {
56  // Trans
57  Eigen::Vector3d position_raw;
58  position_raw << raw_data.position.x, raw_data.position.y, raw_data.position.z;
59  Eigen::Quaterniond orientation_raw(
60  raw_data.orientation.w,
61  raw_data.orientation.x,
62  raw_data.orientation.y,
63  raw_data.orientation.z);
64  // filter
65  if (first_input_) {
66  previous_filtering_position_ = position_raw;
67  previous_filtering_orientation_ = orientation_raw;
68  first_input_ = false;
69  } else {
70  // Position
71  previous_filtering_position_ =
72  ((tau_ * previous_filtering_position_) + (sampling_time * position_raw)) /
73  (tau_ + sampling_time);
74 
75  // Orientation
76  double low = std::max<double>(std::min<double>(hbias_ - hrange_ / 2.0, 1.0), 0.0);
77  double high = std::max<double>(std::min<double>(hbias_ + hrange_ / 2.0, 1.0), 0.0);
78  double hrangeLimited = high - low;
79 
80  // the angle (in radian) between two rotations
81  double distance = previous_filtering_orientation_.dot(orientation_raw) / 2.0;
82  double hlpf = distance / M_PI * hrangeLimited + low;
83 
84  previous_filtering_orientation_ =
85  previous_filtering_orientation_.slerp(hlpf, orientation_raw);
86  }
87  geometry_msgs::msg::Pose result;
88  result.position.x = previous_filtering_position_(0);
89  result.position.y = previous_filtering_position_(1);
90  result.position.z = previous_filtering_position_(2);
91  result.orientation.w = previous_filtering_orientation_.w();
92  result.orientation.x = previous_filtering_orientation_.x();
93  result.orientation.y = previous_filtering_orientation_.y();
94  result.orientation.z = previous_filtering_orientation_.z();
95  return result;
96 }

98 Eigen::Vector3d PoseAverage::to_rpy(Eigen::Quaterniond quat)
99 {
100  Eigen::Matrix3d data = quat.toRotationMatrix();
101  Eigen::Vector3d rpy;
102  rpy(1) =
103  atan2(-data(2, 0), sqrt(data(0, 0) * data(0, 0) + data(1, 0) * data(1, 0)));
104  if (fabs(rpy(1)) > (M_PI / 2) ) {
105  rpy(0) = 0.0;
106  rpy(2) = atan2(-data(0, 1), data(1, 1));
107  } else {
108  rpy(0) = atan2(data(2, 1), data(2, 2));
109  rpy(2) = atan2(data(1, 0), data(0, 0));
110  }
111  return rpy;
112 }

114 Eigen::Quaterniond PoseAverage::to_quat(Eigen::Vector3d rpy)
115 {
116  double ca1, cb1, cc1, sa1, sb1, sc1;
117  ca1 = cos(rpy(2)); sa1 = sin(rpy(2));
118  cb1 = cos(rpy(1));sb1 = sin(rpy(1));
119  cc1 = cos(rpy(0));sc1 = sin(rpy(0));
120  Eigen::Matrix3d mat;
121  mat <<
122  ca1 * cb1, ca1 * sb1 * sc1 - sa1 * cc1, ca1 * sb1 * cc1 + sa1 * sc1,
123  sa1 * cb1, sa1 * sb1 * sc1 + ca1 * cc1, sa1 * sb1 * cc1 - ca1 * sc1,
124  -sb1, cb1 * sc1, cb1 * cc1;
125  return Eigen::Quaterniond(mat);
126 }

131 geometry_msgs::msg::Pose PoseAverage::get_average_data(geometry_msgs::msg::Pose raw_data, int cnt)
132 {
133  Eigen::Vector3d position_raw;
134  position_raw << raw_data.position.x, raw_data.position.y, raw_data.position.z;
135  Eigen::Quaterniond orientation_raw(
136  raw_data.orientation.w,
137  raw_data.orientation.x,
138  raw_data.orientation.y,
139  raw_data.orientation.z);
140  Eigen::Vector3d orientation_rpy_raw = to_rpy(orientation_raw);
141 
142  if (first_input_) {
143  sum_position_ = position_raw;
144  sum_orientation_rpy_ = orientation_rpy_raw;
145  first_input_ = false;
146  } else {
147  sum_position_ += position_raw;
148  sum_orientation_rpy_ += orientation_rpy_raw;
149  }
150  Eigen::Vector3d position = sum_position_ / cnt;
151  Eigen::Vector3d orientation_rpy = sum_orientation_rpy_ / cnt;
152  Eigen::Quaterniond orientaion = to_quat(orientation_rpy);
153 
154  geometry_msgs::msg::Pose result;
155  result.position.x = position(0);
156  result.position.y = position(1);
157  result.position.z = position(2);
158  result.orientation.w = orientaion.w();
159  result.orientation.x = orientaion.x();
160  result.orientation.y = orientaion.y();
161  result.orientation.z = orientaion.z();
162  return result;
163 }
