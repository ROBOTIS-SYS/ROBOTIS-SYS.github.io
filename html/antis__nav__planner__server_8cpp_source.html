66 antis_nav_util::CallbackReturn
67 PlannerServer::on_configure(const rclcpp_lifecycle::State & state)
68 {
69  RCLCPP_INFO(get_logger(), "Configuring");
70 
71  costmap_ros_->on_configure(state);
72  costmap_ = costmap_ros_->getCostmap();
73 
74  RCLCPP_DEBUG(
75  get_logger(), "Costmap size: %d,%d",
76  costmap_->getSizeInCellsX(), costmap_->getSizeInCellsY());
77 
78  tf_ = costmap_ros_->getTfBuffer();
79 
80  get_parameter("planner_plugins", planner_ids_);
81  if (planner_ids_ == default_ids_) {
82  for (size_t i = 0; i < default_ids_.size(); ++i) {
83  declare_parameter(default_ids_[i] + ".plugin", default_types_[i]);
84  }
85  }
86  planner_types_.resize(planner_ids_.size());
87 
88  auto node = shared_from_this();
89 
90  for (size_t i = 0; i != planner_ids_.size(); i++) {
91  try {
92  planner_types_[i] = antis_nav_util::get_plugin_type_param(
93  node, planner_ids_[i]);
94  antis_nav_core::GlobalPlanner::Ptr planner =
95  gp_loader_.createUniqueInstance(planner_types_[i]);
96  RCLCPP_INFO(
97  get_logger(), "Created global planner %s of type %s",
98  planner_ids_[i].c_str(), planner_types_[i].c_str());
99  planner->configure(node, planner_ids_[i], tf_, costmap_ros_);
100  planners_.insert({planner_ids_[i], planner});
101  } catch (const pluginlib::PluginlibException & ex) {
102  RCLCPP_FATAL(
103  get_logger(), "Failed to create global planner. Exception: %s",
104  ex.what());
105  return antis_nav_util::CallbackReturn::FAILURE;
106  }
107  }
108 
109  for (size_t i = 0; i != planner_ids_.size(); i++) {
110  planner_ids_concat_ += planner_ids_[i] + std::string(" ");
111  }
112 
113  RCLCPP_INFO(
114  get_logger(),
115  "Planner Server has %s planners available.", planner_ids_concat_.c_str());
116 
117  double expected_planner_frequency;
118  get_parameter("expected_planner_frequency", expected_planner_frequency);
119  if (expected_planner_frequency > 0) {
120  max_planner_duration_ = 1 / expected_planner_frequency;
121  } else {
122  RCLCPP_WARN(
123  get_logger(),
124  "The expected planner frequency parameter is %.4f Hz. The value should to be greater"
125  " than 0.0 to turn on duration overrrun warning messages", expected_planner_frequency);
126  max_planner_duration_ = 0.0;
127  }
128 
129  // Initialize pubs & subs
130  plan_publisher_ = create_publisher<nav_msgs::msg::Path>("ev_path", 1);
131 
132  // Create the action server that we implement with our navigateToPose method
133  action_server_goal_ = std::make_unique<ActionServerToGoal>(
134  rclcpp_node_,
135  "compute_path_to_goal",
136  std::bind(&PlannerServer::computePath, this));
137 
138  action_server_step_by_step_ = std::make_unique<ActionServerThroughGoals>(
139  rclcpp_node_,
140  "compute_path_through_goals_step_by_step",
141  std::bind(&PlannerServer::computePathThroughGoalsStepByStep, this));
142 
143  action_server_at_once_ = std::make_unique<ActionServerThroughGoals>(
144  rclcpp_node_,
145  "compute_path_through_goals_at_once",
146  std::bind(&PlannerServer::computePathThroughGoalsAtOnce, this));
147 
148  return antis_nav_util::CallbackReturn::SUCCESS;
149 }

218 void PlannerServer::computePathThroughGoalsStepByStep()
219 {
220  RCLCPP_INFO_STREAM(get_logger(), "[planner server] Start computing path (step by step)");
221  auto start_time = steady_clock_.now();
222 
223  // Initialize the ComputePathToPose goal and result
224  auto goal = action_server_step_by_step_->get_current_goal();
225  auto result = std::make_shared<ActionThroughGoals::Result>();
226  nav_msgs::msg::Path concat_path;
227 
228  try {
229  if (isServerInactive(action_server_step_by_step_) ||
230  isCancelRequested(action_server_step_by_step_))
231  {
232  return;
233  }
234 
235  waitForCostmap();
236 
237  getPreemptedGoalIfRequested(action_server_step_by_step_, goal);
238 
239  if (goal->goals.size() == 0) {
240  RCLCPP_WARN(
241  get_logger(),
242  "[planner server] There are no viapoints more.");
243  action_server_step_by_step_->terminate_current();
244  }
245 
246  // Use start pose if provided otherwise use current robot pose
247  geometry_msgs::msg::PoseStamped start;
248  if (!getStartPose(action_server_step_by_step_, start)) {
249  return;
250  }
251 
252  // Get consecutive paths through these points
253  geometry_msgs::msg::PoseStamped curr_start, curr_goal;
254  for (unsigned int i = 0; i != goal->goals.size(); i++) {
255  // Get starting point
256  if (i == 0) {
257  curr_start = start;
258  } else {
259  curr_start = goal->goals[i - 1];
260  }
261  curr_goal = goal->goals[i];
262 
263  // Transform them into the global frame
264  if (!transformPosesToGlobalFrame(action_server_step_by_step_, curr_start, curr_goal)) {
265  return;
266  }
267 
268  // Get plan from start -> goal
269  nav_msgs::msg::Path curr_path = getPlan(curr_start, curr_goal, goal->planner_id);
270 
271  // check path for validity
272  if (!validatePath(action_server_step_by_step_, curr_goal, curr_path, goal->planner_id)) {
273  return;
274  }
275 
276  // Concatenate paths together
277  concat_path.poses.insert(
278  concat_path.poses.end(), curr_path.poses.begin(), curr_path.poses.end());
279  concat_path.header = curr_path.header;
280  }
281 
282  // Publish the plan for visualization purposes
283  result->path = concat_path;
284  publishPlan(result->path);
285 
286  auto cycle_duration = steady_clock_.now() - start_time;
287  result->planning_time = cycle_duration;
288 
289  if (max_planner_duration_ && cycle_duration.seconds() > max_planner_duration_) {
290  RCLCPP_WARN(
291  get_logger(),
292  "[planner server] Planner loop missed its desired rate of %.4f Hz.",
293  1 / max_planner_duration_);
294  RCLCPP_WARN(
295  get_logger(),
296  "[planner server] Current loop rate is %.4f Hz",
297  1 / cycle_duration.seconds());
298  }
299 
300  action_server_step_by_step_->succeeded_current(result);
301  RCLCPP_INFO(get_logger(), "Path planning finished");
302  } catch (std::exception & ex) {
303  RCLCPP_WARN(
304  get_logger(),
305  "[planner server] %s plugin failed to plan through %zu points",
306  goal->planner_id.c_str(), goal->goals.size());
307  RCLCPP_WARN(
308  get_logger(), "with final goal (%.2f, %.2f): \"%s\"",
309  goal->goals.back().pose.position.x, goal->goals.back().pose.position.y, ex.what());
310 
311  action_server_step_by_step_->terminate_current();
312  }
313 }

406 void PlannerServer::computePath()
407 {
408  RCLCPP_INFO_STREAM(get_logger(), "[planner server] Start computing path");
409 
410  auto start_time = steady_clock_.now();
411 
412  // Initialize the ComputePathToPose goal and result
413  auto goal = action_server_goal_->get_current_goal();
414  auto result = std::make_shared<ActionToGoal::Result>();
415 
416  try {
417  if (isServerInactive(action_server_goal_) || isCancelRequested(action_server_goal_)) {
418  return;
419  }
420 
421  waitForCostmap();
422 
423  getPreemptedGoalIfRequested(action_server_goal_, goal);
424 
425  // Use start pose if provided otherwise use current robot pose
426  geometry_msgs::msg::PoseStamped start;
427  if (!getStartPose(action_server_goal_, start)) {
428  return;
429  }
430 
431  // Transform them into the global frame
432  geometry_msgs::msg::PoseStamped goal_pose = goal->pose;
433  if (!transformPosesToGlobalFrame(action_server_goal_, start, goal_pose)) {
434  return;
435  }
436 
437  result->path = getPlan(start, goal_pose, goal->planner_id);
438 
439  if (!validatePath(action_server_goal_, goal_pose, result->path, goal->planner_id)) {
440  return;
441  }
442 
443  // Publish the plan for visualization purposes
444  publishPlan(result->path);
445 
446  auto cycle_duration = steady_clock_.now() - start_time;
447  result->planning_time = cycle_duration;
448 
449  if (max_planner_duration_ && cycle_duration.seconds() > max_planner_duration_) {
450  RCLCPP_WARN(
451  get_logger(),
452  "[planner server] Planner loop missed its desired rate of %.4f Hz.",
453  1 / max_planner_duration_);
454  RCLCPP_WARN(
455  get_logger(),
456  "[planner server] Current loop rate is %.4f Hz",
457  1 / cycle_duration.seconds());
458  }
459 
460  action_server_goal_->succeeded_current(result);
461  RCLCPP_INFO(
462  get_logger(), "Path planning finished");
463  } catch (std::exception & ex) {
464  RCLCPP_WARN(
465  get_logger(), "[planner server] %s plugin failed to plan calculation to (%.2f, %.2f): \"%s\"",
466  goal->planner_id.c_str(), goal->pose.pose.position.x,
467  goal->pose.pose.position.y, ex.what());
468  action_server_goal_->terminate_current();
469  }
470 }

547 template<typename T>
548 bool PlannerServer::transformPosesToGlobalFrame(
549  std::unique_ptr<antis_nav_util::SimpleActionServer<T>> & action_server,
550  geometry_msgs::msg::PoseStamped & curr_start,
551  geometry_msgs::msg::PoseStamped & curr_goal)
552 {

554  !costmap_ros_->transformPoseToGlobalFrame(curr_goal, curr_goal))

556  RCLCPP_WARN(

558  action_server->terminate_current();
559  return false;
560  }
561 
562  return true;

565 template<typename T>
566 bool PlannerServer::transformPosesToGlobalFrame(
567  std::unique_ptr<antis_nav_util::SimpleActionServer<T>> & action_server,
568  geometry_msgs::msg::PoseStamped & curr_start,
569  std::vector<geometry_msgs::msg::PoseStamped> & goals)
570 {
571  if (!costmap_ros_->transformPoseToGlobalFrame(curr_start, curr_start)) {
572  RCLCPP_WARN(
573  get_logger(), "Could not transform the start pose in the costmap frame");
574  action_server->terminate_current();
575  return false;
576  }
577 
578  for (unsigned int i = 0; i != goals.size(); i++) {
579  geometry_msgs::msg::PoseStamped curr_goal = goals[i];
580  if (!costmap_ros_->transformPoseToGlobalFrame(curr_goal, curr_goal)) {
581  RCLCPP_WARN(
582  get_logger(), "Could not transform the goal pose in the costmap frame");
583  action_server->terminate_current();
584  return false;
585  }
586  goals[i] = curr_goal;
587  }
588  return true;
589 }

591 nav_msgs::msg::Path
592 PlannerServer::getPlan(
593  const geometry_msgs::msg::PoseStamped & start,
594  const geometry_msgs::msg::PoseStamped & goal,
595  const std::string & planner_id)
596 {
597  if (planners_.find(planner_id) != planners_.end()) {
598  return planners_[planner_id]->createPlan(start, goal);
599  } else {
600  if (planners_.size() == 1 && planner_id.empty()) {
601  RCLCPP_WARN_ONCE(
602  get_logger(), "[planner server] No planners specified in action call. "
603  "Server will use only plugin %s in server."
604  " This warning will appear once.", planner_ids_concat_.c_str());
605  return planners_[planners_.begin()->first]->createPlan(start, goal);
606  } else {
607  RCLCPP_ERROR(
608  get_logger(), "[planner server] planner %s is not a valid planner. "
609  "Planner names are: %s", planner_id.c_str(),
610  planner_ids_concat_.c_str());
611  }
612  }
613 
614  return nav_msgs::msg::Path();
615 }

643 void
