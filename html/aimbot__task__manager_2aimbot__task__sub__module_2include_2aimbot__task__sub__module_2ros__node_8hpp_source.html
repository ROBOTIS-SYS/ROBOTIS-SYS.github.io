75 {

78  auto array_items = BT::splitString(key, '|');

143 template<>
144 inline std::vector<rclcpp::Parameter> convertFromString(StringView key)
145 {
146  // parameter type, value
147  auto parts = BT::splitString(key, ';');
148  std::vector<rclcpp::Parameter> output;
149  for (auto part : parts) {
150  auto param_set = BT::splitString(part, ',');
151  // name, type(BOOL, INT, DOUBLE, STRING), value
152  if (param_set.size() != 3) {
153  continue;
154  }
155 
156  if (param_set[1] == "BOOL") {
157  std::string param_name = convertFromString<std::string>(param_set[0]);
158  bool param_value = convertFromString<bool>(param_set[2]);
159  auto param = rclcpp::Parameter(param_name, param_value);
160  output.push_back(param);
161  } else if (param_set[1] == "INT") {
162  std::string param_name = convertFromString<std::string>(param_set[0]);
163  int param_value = convertFromString<int>(param_set[2]);
164  auto param = rclcpp::Parameter(param_name, param_value);
165  output.push_back(param);
166  } else if (param_set[1] == "DOUBLE") {
167  std::string param_name = convertFromString<std::string>(param_set[0]);
168  double param_value = convertFromString<double>(param_set[2]);
169  auto param = rclcpp::Parameter(param_name, param_value);
170  output.push_back(param);
171  } else if (param_set[1] == "STRING") {
172  std::string param_name = convertFromString<std::string>(param_set[0]);
173  std::string param_value = convertFromString<std::string>(param_set[2]);
174  auto param = rclcpp::Parameter(param_name, param_value);
175  output.push_back(param);
176  } else if (param_set[1] == "BYTE_ARRAY") {
177  std::string param_name = convertFromString<std::string>(param_set[0]);
178  std::vector<uint8_t> param_value = getArrayFromString<uint8_t>(param_set[2]);
179  auto param = rclcpp::Parameter(param_name, param_value);
180  output.push_back(param);
181  } else if (param_set[1] == "BOOL_ARRAY") {
182  std::string param_name = convertFromString<std::string>(param_set[0]);
183  std::vector<bool> param_value = getArrayFromString<bool>(param_set[2]);
184  auto param = rclcpp::Parameter(param_name, param_value);
185  output.push_back(param);
186  } else if (param_set[1] == "INT_ARRAY") {
187  std::string param_name = convertFromString<std::string>(param_set[0]);
188  std::vector<int64_t> param_value = getArrayFromString<int64_t>(param_set[2]);
189  auto param = rclcpp::Parameter(param_name, param_value);
190  output.push_back(param);
191  } else if (param_set[1] == "DOUBLE_ARRAY") {
192  std::string param_name = convertFromString<std::string>(param_set[0]);
193  std::vector<double> param_value = getArrayFromString<double>(param_set[2]);
194  auto param = rclcpp::Parameter(param_name, param_value);
195  output.push_back(param);
196  } else if (param_set[1] == "STRING_ARRAY") {
197  std::string param_name = convertFromString<std::string>(param_set[0]);
198  std::vector<std::string> param_value = getArrayFromString<std::string>(param_set[2]);
199  auto param = rclcpp::Parameter(param_name, param_value);
200  output.push_back(param);
201  }
202  }
203  return output;
204 }

206 template<>
207 inline task_manager_interfaces::msg::Blackboards convertFromString(StringView key)
208 {
209  // parameter type, value
210  auto blackboards = BT::splitString(key, '/');
211  task_manager_interfaces::msg::Blackboards output;
212  for (auto blackboard : blackboards) {
213  auto blackboard_set = BT::splitString(blackboard, ',');
214  // key, type(BOOL, INT, DOUBLE, STRING, ...), value
215  if (blackboard_set.size() != 3) {
216  continue;
217  }
218 
219  using BLACKBOARD_TYPE = task_manager_interfaces::msg::Blackboard;
220 
221  task_manager_interfaces::msg::Blackboard bb;
222  bb.blackboard_key = convertFromString<std::string>(blackboard_set[0]);
223  bb.blackboard_value = convertFromString<std::string>(blackboard_set[2]);
224 
225  if (blackboard_set[1] == "BOOL") {
226  bb.blackboard_type = BLACKBOARD_TYPE::BOOL;
227  } else if (blackboard_set[1] == "INT") {
228  bb.blackboard_type = BLACKBOARD_TYPE::INT;
229  } else if (blackboard_set[1] == "DOUBLE") {
230  bb.blackboard_type = BLACKBOARD_TYPE::DOUBLE;
231  } else if (blackboard_set[1] == "STRING") {
232  bb.blackboard_type = BLACKBOARD_TYPE::STRING;
233  } else if (blackboard_set[1] == "GEOMETRY_MSGS_POSE2D") {
234  bb.blackboard_type = BLACKBOARD_TYPE::GEOMETRY_MSGS_POSE2D;
235  } else if (blackboard_set[1] == "GEOMETRY_MSGS_POSE") {
236  bb.blackboard_type = BLACKBOARD_TYPE::GEOMETRY_MSGS_POSE;
237  } else if (blackboard_set[1] == "DOUBLE_VECOTR") {
238  bb.blackboard_type = BLACKBOARD_TYPE::DOUBLE_VECTOR;
239  } else if (blackboard_set[1] == "STRING_VECOTR") {
240  bb.blackboard_type = BLACKBOARD_TYPE::STRING_VECTOR;
241  }
242 
243  output.data.push_back(bb);
244  }

366 class RosTreeNode
367 {
368 public:
369  virtual bool init(
370  rclcpp::Node::SharedPtr node, BT::Blackboard::Ptr blackboard,
371  rclcpp::Node::SharedPtr log_node = nullptr)
372  {
373  // do something
374  if (node == nullptr) {
375  std::cout << "There is no ROS Node, failed to init" << std::endl;
376  return false;
377  }
378 
379  node_ = node;
380 
381  if (log_node == nullptr) {
382  log_node_ = node;
383  } else {
384  log_node_ = log_node;
385  }
386 
387  if (use_status_publisher_ == true) {
388  bool on_blackboard = blackboard->get<rclcpp::Publisher<std_msgs::msg::String>::SharedPtr>(
389  "tree_status", status_pub_);
390  if (on_blackboard == false) {
391  status_pub_ = node_->create_publisher<std_msgs::msg::String>("tree_status", 10);
392  blackboard->set<rclcpp::Publisher<std_msgs::msg::String>::SharedPtr>(
393  "tree_status",
394  status_pub_);
395  }
396  }
397 
398  bool result = on_init();
399 
400  return result;
401  }
402 
403  bool publish_tree_status(const std::string & tree_name, const std::string & tree_status)
404  {
405  if (node_ == nullptr || use_status_publisher_ == false) {
406  return false;
407  }
408 
409  std_msgs::msg::String msg;
410  std::stringstream ss;
411  ss << "[" << tree_name << "] " << tree_status;
412  msg.data = ss.str();
413 
414  status_pub_->publish(msg);
415 
416  RCLCPP_INFO(get_logger(), ss.str());
417 
418  return true;
419  }
420 
421  rclcpp::Logger get_logger() const
422  {
423  return log_node_->get_logger();
424  }
425 
426  const int ROS_TIMEOUT = 200;
427 
428 protected:
429  virtual bool on_init() {return true;}
430 
431  // The node that will be used for any ROS operations
432  rclcpp::Node::SharedPtr node_;
433  rclcpp::Node::SharedPtr log_node_;
434  rclcpp::Publisher<std_msgs::msg::String>::SharedPtr status_pub_;
435 
436 protected:
437  bool use_status_publisher_ = false;
438  bool enable_log_ = true;
