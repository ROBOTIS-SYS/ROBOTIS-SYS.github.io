43 int Segmentation::initialize()
44 {
45  CameraGeometry * geo = CameraGeometry::get_instance();
46  int x_res = geo->get_cam_img_size(CameraGeometry::left_zed).width;
47  int y_res = geo->get_cam_img_size(CameraGeometry::left_zed).height * 2;
48 
49  int load_state;
50  load_state = model_load();
51  if (load_state >= 0) {
52  host_pre_mat_ = cv::cuda::HostMem(
53  cv::Size(x_res, y_res),
54  CV_8UC3,
55  cv::cuda::HostMem::PAGE_LOCKED);
56  host_post_mat_ = cv::cuda::HostMem(
57  cv::Size(512, 512),
58  CV_32FC1,
59  cv::cuda::HostMem::PAGE_LOCKED);
60  cv::Size original_size = cv::Size(x_res, y_res);
61  cv::Size resize_size = cv::Size(512, 512);
62  device_original_mat_ = cv::cuda::GpuMat(original_size, CV_8UC3);
63  device_gray_mat_ = cv::cuda::GpuMat(original_size, CV_8UC1);
64  device_transpose_mat_ = cv::cuda::GpuMat(resize_size, CV_8UC3);
65  device_normalize_mat_ = cv::cuda::GpuMat(resize_size, CV_32FC3);
66  device_seg_result_32_ = cv::cuda::GpuMat(resize_size, CV_32SC1);
67  device_seg_result_8_ = cv::cuda::GpuMat(resize_size, CV_8UC1);
68 
69  for (size_t i = 0; i < 50; ++i) {
70  patched_images_.push_back(
71  cv::cuda::GpuMat(cv::Size(54, 54), CV_8UC1));
72  }
73 
74  for (size_t i = 0; i < 3; ++i) {
75  tensor_mat_.push_back(cv::cuda::GpuMat(resize_size, CV_32FC1));
76  }
77 
78  cudaStreamCreate(&cuda_stream_);
79  cv_cuda_stream = cv::cuda::StreamAccessor::wrapStream(cuda_stream_);
80  return load_state;
81  }
82 }

94 void Segmentation::model_run(cv::Mat & src)
95 {
96  src.copyTo(host_pre_mat_);
97  device_original_mat_.upload(host_pre_mat_, cv_cuda_stream);
98  image_processing_with_cuda_.convert_bgr2gray(
99  device_original_mat_,
100  device_gray_mat_,
101  cuda_stream_);
102  pre_processing(device_original_mat_, tensor_mat_);
103  image_processing_with_cuda_.convert_tensor2device_buffer(
104  tensor_mat_,
105  device_tensor_buffer_,
106  cuda_stream_);
107 
108  std::vector<void *> prediction_bindings;
109  for (size_t i = 0; i < device_tensor_buffer_->size(); ++i) {
110  prediction_bindings.emplace_back(
111  (*device_tensor_buffer_)[i].deviceBuffer.data());
112  }
113  trt_engine_->runInference(prediction_bindings, cuda_stream_);
114  post_processing(src, device_tensor_buffer_);
115  cudaStreamSynchronize(cuda_stream_);
116  prediction_bindings.clear();
117 }

183 void Segmentation::post_processing(
184  cv::Mat & src,
185  std::shared_ptr<std::vector<samplesCommon::ManagedBuffer>> input_buffer)
186 {
187  std::vector<std::vector<cv::Point2i>> tmp_feature_pts;
188  cv::Mat tmp_result = cv::Mat(cv::Size(512, 512), CV_32FC1);
189  // Convert device buffer to cv::cuda::GpuMat
190  if (trt_engine_->output_dim_[0].size() == 4) {
191  image_processing_with_cuda_.convert_device_buffer2host_buffer(input_buffer, cuda_stream_);
192  image_processing_with_cuda_.convert_hostBuffer_to_cv_mat(input_buffer, 1, tmp_result);
193  } else {
194  image_processing_with_cuda_.convert_device_buffer2cv(
195  input_buffer,
196  device_seg_result_32_,
197  cuda_stream_);
198  // Convert convert 32S to 8UC1
199  device_seg_result_32_.convertTo(device_seg_result_8_, CV_8UC1);
200  // Create button class image
201  image_processing_with_cuda_.look_up_table(
202  device_seg_result_8_,
203  device_seg_result_8_,
204  cuda_stream_);
205  device_seg_result_8_.download(host_post_mat_, cv_cuda_stream);
206  tmp_result = host_post_mat_.createMatHeader();
207  }
208  // Get feature points & patch images
209  if (contours_.size() > 0 ||
210  boundRect_.size() > 0 ||
211  feature_pts_.size() > 0)
212  {
213  clear();
214  }
215  tmp_result.copyTo(segmentation_result_);
216  image_processing_with_cuda_.get_contours(tmp_result, contours_);
217  image_processing_with_cuda_.get_feature_pts(
218  tmp_result,
219  src,
220  contours_,
221  boundRect_,
222  tmp_feature_pts);
223  image_processing_with_cuda_.get_patch_images(
224  device_gray_mat_, boundRect_, tmp_feature_pts, feature_pts_, patched_images_);
225  // show_patched_images();
226  // show_feature_images(src);
227  tmp_feature_pts.clear();
228 }
