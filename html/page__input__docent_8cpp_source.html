369 void PageInputDocent::fill_list()
370 {
371  QList<std::string> items;
372  for (const auto & it : docent_destination_list_) {
373  bool already_added = false;
374  for (int i = 0; i < items.size(); ++i) {
375  if (items.at(i) == get_title_from_config(it.key, ros_node_->curr_language_)) {
376  already_added = true;
377  break;
378  }
379  }
380 
381  if (already_added == false) {
382  items.append(get_title_from_config(it.key, ros_node_->curr_language_));
383  }
384  }
385 
386  if (items.size() > 0) {
387  ui->listTarget->clear();
388  }
389  for (int i = 0; i < items.size(); ++i) {
390  QListWidgetItem * newItem = new QListWidgetItem(items.at(i).c_str());
391  newItem->setTextAlignment(Qt::AlignVCenter | Qt::AlignHCenter);
392  ui->listTarget->addItem(newItem);
393  }
394 }

630 std::string PageInputDocent::get_title_from_config(std::string key, std::string lang)
631 {
632  YAML::Node dest_config = doc_["destination_config"];
633  for (YAML::iterator it = dest_config.begin(); it != dest_config.end(); ++it) {
634  try {
635  YAML::Node destination = dest_config[it->first];
636  if (destination["key"].as<std::string>() == key) {
637  return destination["title"][lang.c_str()].as<std::string>();
638  }
639  } catch (YAML::Exception & e) {
640  std::cerr << "YAML Exception: " << e.what() << std::endl;
641  return "";
642  }
643  }
