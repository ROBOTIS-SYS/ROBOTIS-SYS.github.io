61 bool RobotModel::add_component(const Component & component, const std::string & parent_name)
62 {
63  // Set parent
64  ComponentMap::iterator parent = this->components_.find(parent_name);
65  if (parent == components_.end()) {
66  return false;
67  }
68  // Set number in joint value array
69  int num_in_joint_array;
70  if (component.get_joint().get_type() != Joint::Fixed) {
71  num_in_joint_array = static_cast<int>(dof_);
72  } else {
73  num_in_joint_array = -1;
74  }
75  // Insert Component to ComponentMap
76  std::pair<ComponentMap::iterator, bool> insert_result;
77  insert_result = components_.insert(
78  std::make_pair(
79  component.get_name(),
80  ConnectedComponent(
81  component,
82  parent,
83  num_in_joint_array)));
84  if (!insert_result.second) {
85  return false;
86  }
87 
88  // Set parent child
89  parent->second.children.push_back(insert_result.first);
90  // Set number params
91  num_of_components_++;
92  if (component.get_joint().get_type() != Joint::Fixed) {
93  dof_++;
94  }
95  return true;
96 }

103 bool RobotModel::check_joint_limit(
104  JointValues values,
105  std::vector<std::string> * joint_name_over_limit)
106 {
107  joint_name_over_limit->clear();
108  auto it_components = components_.begin();
109  it_components++;
110 
111  for (auto it_components = components_.begin(); it_components != components_.end();
112  it_components++)
113  {
114  int index = it_components->second.num_in_joint_array;
115 
116  if (it_components->second.component.get_joint().get_type() != Joint::Fixed) {
117  if (!it_components->second.component.check_joint_limit(
118  values.position(index),
119  values.velocity(index),
120  values.effort(index)))
121  {
122  joint_name_over_limit->push_back(it_components->second.component.get_joint().get_name());
123  }
124  }
125  }
126  if (joint_name_over_limit->empty()) {
127  return true;
128  } else {
129  return false;
130  }
131 }

133 bool RobotModel::add_child_component_recursively(
134  ComponentMap::const_iterator parent_component,
135  const std::string & parent_name)
136 {
137  ComponentMap::const_iterator child;
138  for (unsigned int i = 0; i < parent_component->second.children.size(); i++) {
139  child = parent_component->second.children.at(i);
140  if (!this->add_component(child->second.component, parent_name)) {
141  return false;
142  }
143  if (!this->add_child_component_recursively(child, child->first)) {
144  return false;
145  }
146  }
147  return true;
148 }

151 {

157  }
158 
159  for (auto child_it : components_.at(my_component_name).children) {
160  get_joint_names(child_it->first, joint_names);
