42 template<class T0, class T1>
43 struct TimeMatch
44 {
45  TimeMatch() {}
46  TimeMatch(T0 p0, T1 p1)
47  {
48  p_0 = p0;
49  p_1 = p1;
50  }
51  T0 p_0;
52  T1 p_1;
53  bool find_idx_pair(
54  const std::deque<T0> & queue_0,
55  const std::deque<T1> & queue_1,
56  std::pair<int, int> & output)
57  {
58  auto iter_p0 = std::find(queue_0.begin(), queue_0.end(), p_0);
59  if (iter_p0 == queue_0.end()) {return false;}
60  auto iter_p1 = std::find(queue_1.begin(), queue_1.end(), p_1);
61  if (iter_p1 == queue_1.end()) {return false;}
62 
63  output.first = static_cast<int>(iter_p0 - queue_0.begin());
64  output.second = static_cast<int>(iter_p1 - queue_1.begin());
65  return true;
66  }
67 };

107 std::vector<std::pair<int, int>> OdomSyncBase::time_matching_ptr2idx(
108  std::deque<FlagOdom::SharedPtr> & odom_queue_1,
109  std::deque<FlagOdom::SharedPtr> & odom_queue_2)
110 {

112  std::vector<TimeMatch<FlagOdom::SharedPtr, FlagOdom::SharedPtr>> vec_cpy;
113  vec_cpy = time_matching_pair_ptr_.get_vector();
114  for (int i = 0; i < vec_cpy.size(); i++) {
115  std::pair<int, int> temp_pair;
116  if (vec_cpy[i].find_idx_pair(odom_queue_1, odom_queue_2, temp_pair)) {
117  rtn_pairs.emplace_back(temp_pair);
118  }
119  }
120  return rtn_pairs;
121 }

153 class RapidMonitorOdomSync : public OdomSyncBase
154 {
155 public:
156  explicit RapidMonitorOdomSync(uint32_t time_pair_size)
157  : OdomSyncBase(time_pair_size) {}
158  ~RapidMonitorOdomSync() override {}
159  bool synchronize_odom(
160  std::deque<FlagOdom::SharedPtr> & odom_queue_1,
161  std::deque<FlagOdom::SharedPtr> & odom_queue_2) override
162  {
163  bool updated = false;
164 
165  rclcpp::Duration best_time_diff = rclcpp::Duration(1, 0);
166  int best_index = 0;
167  int odom_2_ref_index = 0;
168 
169  // TODO(Sangbeom Woo): : flag를 활용해서 효율적인 pair 매칭구현
170 
171  /* odom_queue_1과 odom_queue_2의 페어 매치 미확인 인덱스 확인 */
172  std::deque<FlagOdom::SharedPtr>::iterator it_o2 = odom_queue_2.end();
173  std::deque<FlagOdom::SharedPtr>::iterator it_o1 = odom_queue_1.end();
174  // 1. odom_queue_2의 미확인구한: 미확인 ~ 최근 - 1
175  // : 최근은 매칭이 아직 이루어졌는지 속단하기 힘듬.
176  // std::cout << "tp1" << std::endl;
177  for (it_o2 = odom_queue_2.begin(); it_o2 != odom_queue_2.end(); it_o2++) {
178  if (!((*it_o2)->is_pair_checked_) && (it_o2 != (odom_queue_2.end() - 1))) {
179  // 평가지표 초기화
180  rclcpp::Duration best_time_diff = rclcpp::Duration(1, 0);
181  std::deque<FlagOdom::SharedPtr>::iterator best_it = odom_queue_1.end();
182  // 현재 odom2 기준시간 지정
183  rclcpp::Time ref_time((*it_o2)->msg.header.stamp);
184  // std::cout << "tp2" << std::endl;
185  // 2. odom_queue_1의 미확인구간: 미확인 ~ 최근
186  for (it_o1 = odom_queue_1.begin(); it_o1 != odom_queue_1.end(); it_o1++) {
187  if (!((*it_o1)->is_pair_checked_)) {
188  rclcpp::Time odom_1_curr_time((*it_o1)->msg.header.stamp);
189 
190  rclcpp::Duration time_diff(1, 0);
191  // std::cout << "tp3" << std::endl;
192  // 3. for(odom_queue_1의 미확인구간), for(odom_queue_1의 미확인구간)에 걸쳐서
193  // 검사후 매칭 확인
194  // rule: 시간차 abs비교, 최적 시간차 인덱스 검색,
195  // 만약 시간차가 threshold 이상으로 크면 drop
196  // 검색이 끝난 페어는 flag를 저장해줄 것.
197  try {
198  if (odom_1_curr_time > ref_time) {
199  time_diff = odom_1_curr_time - ref_time;
200  } else {
201  time_diff = ref_time - odom_1_curr_time;
202  }
203  } catch (const std::exception & ex) {
204  std::cout << ex.what() << ", dropping this iteration" << std::endl;
205 
206  continue;
207  }
208  // no throw guarantee 처리
209  if (time_diff < best_time_diff) {
210  best_it = it_o1;
211  best_time_diff = time_diff;
212  }
213  }
214  }
215  // 검사완료 저장
216  // std::cout << "iterator position from end: " <<
217  // std::distance(odom_queue_1.begin(), best_it) << std::endl;
218  if (best_it != odom_queue_1.end()) {(*best_it)->is_pair_checked_ = true;}
219  // 검사완료 저장
220  if (it_o2 != odom_queue_2.end()) {(*it_o2)->is_pair_checked_ = true;}
221 
222  bool time_compressed = false;
223  if (it_o2 != odom_queue_2.end() &&
224  best_it != odom_queue_1.end() &&
225  time_matching_pair_ptr_.size() != 0)
226  {
227  rclcpp::Duration odom1_dur(1, 0);
228  rclcpp::Time odom1_post((*best_it)->msg.header.stamp);
229  rclcpp::Time odom1_pre(time_matching_pair_ptr_.back().p_0->msg.header.stamp);
230  odom1_dur = odom1_post - odom1_pre;
231 
232  rclcpp::Duration odom2_dur(1, 0);
233  rclcpp::Time odom2_post((*it_o2)->msg.header.stamp);
234  rclcpp::Time odom2_pre(time_matching_pair_ptr_.back().p_0->msg.header.stamp);
235  odom2_dur = odom2_post - odom2_pre;
236 
237  time_compressed =
238  std::fabs((odom1_dur.seconds() - odom2_dur.seconds()) / odom2_dur.seconds()) >
239  time_error_ratio_th_;
240  if (time_compressed) {
241  std::cout << "Compressed time sync detected!" << std::endl;
242  }
243  }
244  if (best_time_diff < time_th_ && !time_compressed) {
245  FlagOdom::SharedPtr o1_ptr(*best_it);
246  FlagOdom::SharedPtr o2_ptr(*it_o2);
247 
248  updated = true;
249  time_matching_pair_ptr_.enqueue(
250  TimeMatch<FlagOdom::SharedPtr, FlagOdom::SharedPtr>(
251  o1_ptr,
252  o2_ptr));
253 
254  // 페어링 플래그 저장
255  if (best_it != odom_queue_1.end()) {(*best_it)->is_paired_ = true;}
256  if (it_o2 != odom_queue_2.end()) {(*it_o2)->is_paired_ = true;}
257  }
258  }
259  }
260 
261  time_matching_pair_ = time_matching_ptr2idx(odom_queue_1, odom_queue_2);
262 
263  return updated;
264  }
265 };
