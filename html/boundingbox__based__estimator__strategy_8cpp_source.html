28 int BoundingBoxBasedEstimatorStrategy::calculate(
29  std::shared_ptr<std::vector<RecognizedText>> & left_buttons,
30  std::shared_ptr<std::vector<RecognizedText>> & right_buttons,
31  std::shared_ptr<std::map<std::string, ButtonPose>> & button_poses)
32 {
33  std::vector<std::string> labels(left_buttons->size());
34  std::vector<ButtonPose> poses(left_buttons->size());
35 #pragma omp parallel num_threads(MAX_THREADS)
36  {
37 #pragma omp for
38  for (size_t i = 0; i < left_buttons->size(); i++) {
39  std::string left_label = left_buttons->at(i).class_name;
40  auto corresponding_one =
41  std::find_if(
42  right_buttons->begin(), right_buttons->end(),
43  [&left_label](const RecognizedText btn) {
44  return btn.class_name == left_label;
45  });
46  if (corresponding_one != right_buttons->end()) {
47  labels[i] = left_label;
48  poses[i] = calculate_pose(left_buttons->at(i), *corresponding_one);
49  }
50  }
51  }
52 
53  button_poses->clear();
54  for (size_t i = 0; i < labels.size(); i++) {
55  if (labels[i].empty() == false) {
56  if (is_valid(poses[i].center)) {
57  button_poses->insert(std::make_pair(labels[i], poses[i]));
58  }
59  }
60  }
61 
62  return PERCEPTION_STATE::SUCCESS_NO_ISSUE;
63 }

65 Eigen::Vector3f BoundingBoxBasedEstimatorStrategy::calculate_plane_normal(
66  std::vector<Eigen::Vector3f> plane_pts)
67 {
68  Eigen::Vector3f result(0.0, 0.0, 1.0);
69 
70  if (plane_pts.size() > 0) {
71  Eigen::Vector3f p1 = plane_pts.at(0);
72  Eigen::Vector3f p2 = plane_pts.at(1);
73  Eigen::Vector3f p3 = plane_pts.at(2);
74 
75  Eigen::Vector3f v21 = p2 - p1;
76  Eigen::Vector3f v31 = p3 - p1;
77  v21.normalize();
78  v31.normalize();
79 
80  result = v21.cross(v31);
81  }
82 
83  return result;
84 }

86 ButtonPose BoundingBoxBasedEstimatorStrategy::calculate_pose(
87  RecognizedText left_button,
88  RecognizedText right_button)
89 {
90  // Result pose
91  ButtonPose pose;
92 
93  // Set normal (Orientaion)
94  pose.normal << 0.0, 0.0, 1.0;
95 
96  Eigen::MatrixXf recon_pts = triangulate(left_button, right_button);
97 
98  std::vector<Eigen::Vector3f> corner_vec;
99  corner_vec.push_back(recon_pts.row(0));
100  corner_vec.push_back(recon_pts.row(1));
101  corner_vec.push_back(recon_pts.row(3));
102 
103  // calculate normal
104  pose.normal = calculate_plane_normal(corner_vec);
105 
106  // calculate center
107  switch (pose_return_type_) {
108  case TOP_LEFT:
109  case TOP_RIGHT:
110  case CENTER:
111  case BOTTOM_RIGHT:
112  case BOTTOM_LEFT:
113  pose.center = recon_pts.row(pose_return_type_);
114  break;
115  case TOP_CENTER:
116  pose.center = (recon_pts.row(0) + recon_pts.row(1)) * 0.5;
117  break;
118  case BOTTOM_CENTER:
119  pose.center = (recon_pts.row(3) + recon_pts.row(4)) * 0.5;
120  break;
121  default:
122  pose.center = recon_pts.row(2);
123  break;
124  }
125 
126  return pose;
127 }
