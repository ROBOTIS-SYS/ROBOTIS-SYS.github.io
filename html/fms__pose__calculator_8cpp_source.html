143 bool FMSPoseCalculator::update()
144 {
145  CameraGeometry * geo = CameraGeometry::get_instance();
146  cv::Mat cv_left_image, cv_right_image;
147  rclcpp::Time left_stamp, right_stamp;
148  std::string dummy;
149  bool ack_left_img, ack_right_img;
150 
151  if (this->is_data_available(left_img_sub_msg_name_) &&
152  this->is_data_available(right_img_sub_msg_name_))
153  {
154  ack_left_img =
155  this->get_data_image(left_img_sub_msg_name_, cv_left_image, left_stamp, dummy);
156  ack_right_img =
157  this->get_data_image(right_img_sub_msg_name_, cv_right_image, right_stamp, dummy);
158  } else {
159  return false;
160  }
161 
162  if (ack_left_img && ack_right_img) {
163  // overlay step
164  Eigen::Matrix<float, 3, 4> Pl = geo->get_cam_proj_mat(CameraGeometry::left_zed);
165  Eigen::Matrix<float, 3, 4> Prl = geo->get_cam_proj_mat(CameraGeometry::right_to_left_zed);
166 
167  Eigen::Vector2f proj_left_pt1 =
168  geo->project_to_image(Pl, virtual_tip_start_pt_);
169  Eigen::Vector2f proj_left_pt2 =
170  geo->project_to_image(Pl, virtual_tip_end_pt_);
171  Eigen::Vector2f proj_right_pt1 =
172  geo->project_to_image(Prl, virtual_tip_start_pt_);
173  Eigen::Vector2f proj_right_pt2 =
174  geo->project_to_image(Prl, virtual_tip_end_pt_);
175 
176  cv::line(
177  cv_left_image,
178  cv::Point(proj_left_pt1(0), proj_left_pt1(1)),
179  cv::Point(proj_left_pt2(0), proj_left_pt2(1)),
180  cv::Scalar(0, 0, 255),
181  15);
182  cv::line(
183  cv_right_image,
184  cv::Point(proj_right_pt1(0), proj_right_pt1(1)),
185  cv::Point(proj_right_pt2(0), proj_right_pt2(1)),
186  cv::Scalar(0, 0, 255),
187  15);
188 
189  cv::Mat overlay_image =
190  visualizer_->convert_rgba_imgs_to_rgb_img(cv_left_image, cv_right_image);
191 
192  this->run_streaming(
193  overlay_image,
194  left_stamp,
195  zed_frame_id_,
196  zed_streaming_pub_msg_name_,
197  zed_streaming_image_ratio_);
198  }
199 
200  return true;
201 }

203 void FMSPoseCalculator::fms_robot_agent_callback(
204  const std::shared_ptr<perception_interfaces::srv::PoseEstimation::Request> req,
205  std::shared_ptr<perception_interfaces::srv::PoseEstimation::Response> res)
206 {
207  CameraGeometry * geo = CameraGeometry::get_instance();
208  perception_interfaces::msg::BoundingBox left_bbox = req->left_camera;
209  perception_interfaces::msg::BoundingBox right_bbox = req->right_camera;
210 
211  Eigen::Vector2f lp1(left_bbox.left_top_x, left_bbox.left_top_y);
212  Eigen::Vector2f lp2(left_bbox.right_bottom_x, left_bbox.right_bottom_y);
213  Eigen::Vector2f rp1(right_bbox.left_top_x, right_bbox.left_top_y);
214  Eigen::Vector2f rp2(right_bbox.right_bottom_x, right_bbox.right_bottom_y);
215 
216  RCLCPP_INFO_STREAM(this->get_logger(), "[Received left bbox pixel info]");
217  RCLCPP_INFO_STREAM(
218  this->get_logger(),
219  "top left point: " << lp1(0) << ", " << lp1(1) <<
220  ", bottom right point: " << lp2(0) << ", " << lp2(1) << ".");
221 
222  RCLCPP_INFO_STREAM(this->get_logger(), "[Received right bbox pixel info]");
223  RCLCPP_INFO_STREAM(
224  this->get_logger(),
225  "top left point: " << rp1(0) << ", " << rp1(1) <<
226  ", bottom right point: " << rp2(0) << ", " << rp2(1) << ".");
227 
228  Eigen::Vector2f lcp = (lp1 + lp2) / 2;
229  Eigen::Vector2f rcp = (rp1 + rp2) / 2;
230 
231  // 비율 계산
232  cv::Size img_size = geo->get_cam_img_size(CameraGeometry::left_zed);
233  lcp(0) *= img_size.width;
234  lcp(1) *= img_size.height * 2;
235  rcp(0) *= img_size.width;
236  rcp(1) *= img_size.height * 2;
237  rcp(1) -= img_size.height;
238 
239  Eigen::Vector3f cp =
240  geo->get_cam_to_opt() * geo->reconstruct(
241  geo->get_cam_f(CameraGeometry::left_zed),
242  geo->get_cam_base_line(CameraGeometry::right_to_left_zed),
243  geo->get_cam_ppx(CameraGeometry::left_zed),
244  geo->get_cam_ppy(CameraGeometry::left_zed),
245  lcp, rcp);
246  if (cp.hasNaN() || cp.norm() == 0) {
247  res->result = perception_interfaces::srv::PoseEstimation::Response::RESULT_INVALID_VALUE;
248  } else {
249  res->result = perception_interfaces::srv::PoseEstimation::Response::RESULT_SUCCESS;
250  }
251 
252  res->pose.header.stamp = this->get_clock()->now();
253  res->pose.header.frame_id = zed_frame_id_;
254  ButtonPose pose;
255  pose.center = cp;
256  pose.normal = geo->get_cam_to_opt() * Eigen::Vector3f(0, 0, 1);
257  res->pose.pose = pose.pose();
258 
259  RCLCPP_INFO_STREAM(
260  this->get_logger(),
261  "FMS pose estimation result: " << cp(0) << ", " << cp(1) << ", " << cp(2) << ".");
262 }
