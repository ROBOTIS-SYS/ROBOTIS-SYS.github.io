108 void EVStateManager::update()
109 {
110  // 서비스 상태 체크
111  if (robot_service_->get_current_sequence() == EVRobotService::NO_SERVICE) {
112  return;
113  }
114 
115  // 엘리베이터 상태 업데이트는 callback을 통해 EVState에서 자동으로 처리
116  // EVState를 이용해 서비스 상태 변경 확인
117  bool is_changed = robot_service_->handle_ev_state(ev_status_list_);
118 
119  // 서비스 상태 업데이트
120  if (robot_service_->is_sequence_updated() || is_changed) {
121  publish_ev_sequence();
122  }
123 
124  // 서비스 신호 업데이트
125  if (robot_service_->is_signal_updated()) {
126  EVRobotService::SignalState signal = robot_service_->get_signal();
127  publish_ev_signal_sequence(signal);
128  }
129 }

178 void EVStateManager::call_robot_service_callback(
179  const std::shared_ptr<elevator_interfaces::srv::CallRobotService::Request>

187 
188  publish_log(
189  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
190  aimbot_log_interfaces::msg::Log::TYPE_WARN, "EV Manager",
191  " 로봇 서비스 호출 콜백");

197  ev_state_it.second->clear_ev_state();
198  }
199 
200  int8_t ev_num =
201  (req->ev_num == 0) ? DEFAULT_EV_NUM : req->ev_num;
202 
203  bool result = robot_service_->set_robot_service(
204  ev_num, req->call_floor,
205  req->dest_floor);
206 
207 // // 서비스 시작 시 EV 상태 초기화
208 // if (result == true) {
209 // for (auto ev_state_it : ev_status_list_) {
210 // ev_state_it.second->clear_ev_state();
211 // }
212 // }
213 
214  res->result = result;
215 }

217 void EVStateManager::call_robot_service_in_ev_callback(
218  const std::shared_ptr<
219  elevator_interfaces::srv::CallRobotServiceInEV::Request>
220  req,
221  std::shared_ptr<elevator_interfaces::srv::CallRobotServiceInEV::Response>
222  res)
223 {
224  RCLCPP_INFO(
225  this->get_logger(),
226  "INFO call robot service in ev : ev_num(%d), dest_floor(%s)",
227  req->ev_num, req->dest_floor.c_str());
228 
229  publish_log(
230  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
231  aimbot_log_interfaces::msg::Log::TYPE_WARN, "EV Manager",
232  " 로봇 서비스 (in EV) 호출 콜백");
233 
234  if (req->ev_num == 0) {
235  res->result = false;
236  publish_log(
237  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
238  aimbot_log_interfaces::msg::Log::TYPE_WARN, "EV Manager",
239  " 로봇 서비스 (in EV) 호출 실패 !! (ev_num == 0) ");
240  return;
241  }
242 
243  // 기존 서비스 초기화
244  robot_service_->clear_robot_service();
245 
246  bool result = robot_service_->set_robot_service(
247  req->ev_num, req->dest_floor,
248  req->dest_floor);
249  // 서비스 시작 시 EV 상태 초기화
250  if (result == true) {
251  for (auto ev_state_it : ev_status_list_) {
252  ev_state_it.second->clear_ev_state();
253  }
254  }
255  publish_ev_sequence();
256 
257  res->result = result;
258 }

282 void EVStateManager::set_robot_service_callback(
283  const std::shared_ptr<elevator_interfaces::srv::SetRobotService::Request>
284  req,
285  std::shared_ptr<elevator_interfaces::srv::SetRobotService::Response> res)
286 {
287  RCLCPP_INFO(
288  this->get_logger(), "INFO set robot service (%s), last seq (%s)",
289  req->robot_status.c_str(),
290  robot_service_->get_current_sequence_name().c_str());
291 
292  // Getting Off, Tasking On만 처리, 그 외는 종료
293  if (req->robot_status == "Taking On") {
294  bool result = robot_service_->set_robot_action(EVRobotService::TAKING_ON);
295  res->result = true;
296  res->ev_service_status = robot_service_->get_current_sequence_name();
297  return;
298  } else if (req->robot_status == "Getting Off") {
299  bool result = robot_service_->set_robot_action(EVRobotService::GETTING_OFF);
300  res->result = true;
301  res->ev_service_status = robot_service_->get_current_sequence_name();
302  return;
303  } else {
304  RCLCPP_WARN(
305  this->get_logger(),
306  "WARN set_robot_service : robot_status(%s) is not valied!",
307  req->robot_status.c_str());
308  }
309 
310  res->ev_service_status = robot_service_->get_current_sequence_name();
311  res->result = false;
312 }

315 void EVStateManager::ev_status_callback(
316  const perception_interfaces::msg::EVStatus::SharedPtr msg)
317 {
318  // service 동작 중이 아닌 경우 업데이트 안함
319  if (robot_service_->get_current_sequence() == EVRobotService::NO_SERVICE) {
320  return;
321  }
322 
323  // EV를 대기중, 이동중인 경우 상태 체크
324  // if (robot_service_->get_current_sequence() == EVRobotService::MOVING) {
325  // ev_state_->update_ev_state(*msg);
326  // }
327 
328  // EV 상태 업데이트
329  bool alternately = robot_in_ev_ ? alternately_ev_in_ : alternately_ev_out_;
330  std::string ev_num = std::to_string(robot_service_->get_current_allocated_ev_num());
331  if (ev_status_list_.empty()) {
332  // 정보 생성 및 업데이트 후 리스트에 추가
333  auto ev_status = std::make_shared<EVState>();
334  ev_status->print_log =
335  std::bind(&EVStateManager::print_log, this, std::placeholders::_1);
336  ev_status->update_ev_state(*msg, alternately);
337  ev_status_list_[ev_num] = ev_status;
338  } else {
339  ev_status_list_.begin()->second->update_ev_state(*msg, alternately);
340  }
341 }

343 void EVStateManager::ev_status_array_callback(
344  const perception_interfaces::msg::EVStatusArray::SharedPtr msg)
345 {
346  // service 동작 중이 아닌 경우 업데이트 안함
347  if (robot_service_->get_current_sequence() == EVRobotService::NO_SERVICE) {
348  return;
349  }
350 
351  // msg 체크
352  if (msg->ev_num.size() == 0 || msg->ev_num.size() != msg->ev_status.size()) {
353  return;
354  }
355 
356  // EV 상태 업데이트
357  bool alternately = robot_in_ev_ ? alternately_ev_in_ : alternately_ev_out_;
358  std::string called_ev_num = std::to_string(robot_service_->get_current_allocated_ev_num());
359  for (size_t ix = 0; ix < msg->ev_num.size(); ix++) {
360  std::string ev_num = (msg->ev_num.at(ix) == NONE_STRING) ? called_ev_num : msg->ev_num.at(ix);
361  auto status_it = ev_status_list_.find(ev_num);
362  if (status_it != ev_status_list_.end()) {
363  // 정보 업데이트
364  status_it->second->update_ev_state(msg->ev_status.at(ix), alternately);
365  } else {
366  // 정보 생성 및 업데이트 후 리스트에 추가
367  auto ev_status = std::make_shared<EVState>();
368  ev_status->print_log =
369  std::bind(&EVStateManager::print_log, this, std::placeholders::_1);
370  ev_status->update_ev_state(msg->ev_status.at(ix), alternately);
371  ev_status_list_[ev_num] = ev_status;
372  }
373  }
374 }
