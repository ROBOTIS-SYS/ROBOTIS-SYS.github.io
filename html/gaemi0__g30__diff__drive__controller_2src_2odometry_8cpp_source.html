48 bool Odometry::update(double left_pos, double right_pos, const rclcpp::Time & time)
49 {
50  // We cannot estimate the speed with very small time intervals:
51  const double dt = time.seconds() - timestamp_.seconds();
52  if (dt < 0.0001) {
53  return false; // Interval too small to integrate with
54  }
55 
56  // Get current wheel joint positions:
57  const double left_wheel_cur_pos = left_pos * left_wheel_radius_;
58  const double right_wheel_cur_pos = right_pos * right_wheel_radius_;
59 
60  // Estimate velocity of wheels using old and current position:
61  const double left_wheel_est_vel = left_wheel_cur_pos - left_wheel_old_pos_;
62  const double right_wheel_est_vel = right_wheel_cur_pos - right_wheel_old_pos_;
63 
64  // Update old position with current:
65  left_wheel_old_pos_ = left_wheel_cur_pos;
66  right_wheel_old_pos_ = right_wheel_cur_pos;
67 
68  // Compute linear and angular diff:
69  const double linear = (right_wheel_est_vel + left_wheel_est_vel) * 0.5;
70  // Now there is a bug about scout angular velocity
71  const double angular = (right_wheel_est_vel - left_wheel_est_vel) / wheel_separation_;
72 
73  // Integrate odometry:
74  integrateExact(linear, angular);
75 
76  timestamp_ = time;
77 
78  // Estimate speeds using a rolling mean to filter them out:
79  linear_accumulator_.accumulate(linear / dt);
80  angular_accumulator_.accumulate(angular / dt);
81 
82  linear_ = linear_accumulator_.getRollingMean();
83  angular_ = angular_accumulator_.getRollingMean();
84 
85  return true;
86 }

122 void Odometry::integrateRungeKutta2(double linear, double angular)
123 {
124  const double direction = heading_ + angular * 0.5;
125 
127  x_ += linear * cos(direction);
128  y_ += linear * sin(direction);
129  heading_ += angular;
130 }

132 void Odometry::integrateExact(double linear, double angular)
133 {
134  if (fabs(angular) < 1e-6) {
135  integrateRungeKutta2(linear, angular);
136  } else {
138  const double heading_old = heading_;
139  const double r = linear / angular;
140  heading_ += angular;
141  x_ += r * (sin(heading_) - sin(heading_old));
142  y_ += -r * (cos(heading_) - cos(heading_old));
143  }
144 }
