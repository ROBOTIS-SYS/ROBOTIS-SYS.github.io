42 double FloorEstimationByAtmPressure::calculate_atmospheric_pressure_derivative(
43  double time,
44  double present_atm_pressure)
45 {
46  static double previous_time = time;
47  static double previous_atm_pressure = present_atm_pressure;
48 
49  double atm_pressure_derivative;
50  if (fabs(time - previous_time) < 0.001) {
51  atm_pressure_derivative = 0;
52  } else {
53  atm_pressure_derivative = (present_atm_pressure - previous_atm_pressure) /
54  (time - previous_time);
55  }
56 
57  // RCLCPP_INFO_STREAM(logger_, "Current atm_pressure_derivative: " << atm_pressure_derivative);
58 
59  previous_time = time;
60  previous_atm_pressure = present_atm_pressure;
61 
62  return fabs(atm_pressure_derivative);
63 }

65 bool FloorEstimationByAtmPressure::set_altitudes_relative_to_initial_floor()
66 {
67  double cumulative_height = 0;
68  double altitudes_on_initial_floor = -1.0;
69  relative_floor_altitudes_ = floor_altitudes_;
70  for (size_t i = 0; i < floor_altitudes_.size(); ++i) {
71  relative_floor_altitudes_.at(i).second = cumulative_height;
72  cumulative_height += floor_altitudes_.at(i).second;
73  if (relative_floor_altitudes_.at(i).first == initial_floor_name_) {
74  altitudes_on_initial_floor = relative_floor_altitudes_.at(i).second;
75  }
76  }
77  if (altitudes_on_initial_floor < 0) {
78  RCLCPP_ERROR_STREAM(
79  logger_,
80  "Initial floor name not found in floor altitudes.");
81  return false;
82  }
83  for (size_t i = 0; i < relative_floor_altitudes_.size(); ++i) {
84  relative_floor_altitudes_.at(i).second -= altitudes_on_initial_floor;
85  }
86  return true;
87 }

94 std::string FloorEstimationByAtmPressure::estimate_floor_based_on_current_altitude(
95  double present_altitude)
96 {
97  std::string current_floor;
98  double min_difference = std::numeric_limits<double>::max();
99  for (const auto & pair : relative_floor_altitudes_) {
100  double difference = std::abs(pair.second - present_altitude);
101  if (difference < min_difference) {
102  min_difference = difference;
103  current_floor = pair.first;
104  }
105  }
106  // RCLCPP_INFO_STREAM(logger_, "Current calculated altitude: " << present_altitude);
107  // RCLCPP_INFO_STREAM(logger_, "Current floor calculation: " << current_floor);
108 
109  return current_floor;
110 }

116 bool FloorEstimationByAtmPressure::initialize(
117  double altitude_to_pressure_ratio,
118  std::string initial_floor,
119  std::string floor_height_path,
120  double pressure_derivative_threshold,
121  int moving_average_window_size)
122 {
123  altitude_to_pressure_ratio_ = altitude_to_pressure_ratio;
124  initial_floor_name_ = initial_floor;
125  current_floor_name_ = initial_floor;
126  pressure_filter_ptr_ = std::make_shared<MovingAverageFilter>(moving_average_window_size);
127  pressure_derivative_threshold_ = pressure_derivative_threshold;
128  RCLCPP_INFO_STREAM(logger_, "altitude_to_pressure_ratio_: " << altitude_to_pressure_ratio_);
129  RCLCPP_INFO_STREAM(logger_, "initial_floor_name_: " << initial_floor_name_);
130  RCLCPP_INFO_STREAM(logger_, "current_floor_name_: " << current_floor_name_);
131  RCLCPP_INFO_STREAM(logger_, "pressure_derivative_threshold_: " << pressure_derivative_threshold_);
132  RCLCPP_INFO_STREAM(logger_, "moving_average_window_size: " << moving_average_window_size);
133  try {
134  YAML::Node node = YAML::LoadFile(floor_height_path.c_str());
135  YAML::Node floor_height_node = node["floor_height_m"];
136 
137  for (YAML::iterator yaml_it = floor_height_node.begin(); yaml_it != floor_height_node.end();
138  ++yaml_it)
139  {
140  double altitude = yaml_it->second.as<double>();
141  if (altitude <= 0) {
142  RCLCPP_ERROR_STREAM(
143  logger_,
144  "Invalid altitude for floor '" << yaml_it->first.as<std::string>() << "': " <<
145  altitude);
146  return false;
147  }
148 
149  floor_altitudes_.push_back(
150  std::make_pair(yaml_it->first.as<std::string>(), altitude));
151  RCLCPP_INFO_STREAM(
152  logger_,
153  "Altitude of Floor '" << yaml_it->first.as<std::string>() << "' is " <<
154  yaml_it->second.as<double>() << "'.");
155  }
156  } catch (const std::exception & e) {
157  RCLCPP_ERROR_STREAM(
158  logger_,
159  "Fail to load Floor Height data. : " << e.what());
160  return false;
161  }
162  RCLCPP_INFO_STREAM(
163  logger_,
164  "Loaded floor number is " << floor_altitudes_.size());
165 
166  return set_altitudes_relative_to_initial_floor();
167 }

169 std::string FloorEstimationByAtmPressure::get_current_floor(
170  const sensor_msgs::msg::FluidPressure::SharedPtr msg)
171 {
172  rclcpp::Time time = msg->header.stamp;
173  double filtered_pressure = pressure_filter_ptr_->apply(msg->fluid_pressure);
174 
175  // RCLCPP_INFO_STREAM(logger_, "Current filtered pressure: " << filtered_pressure);
176 
177  if (calculate_atmospheric_pressure_derivative(time.seconds(), filtered_pressure) <=
178  pressure_derivative_threshold_)
179  {
180  // RCLCPP_INFO_STREAM(logger_, "DIFF!!!!!!!!!!!!!! ");
181  set_initial_pressure(filtered_pressure);
182  }
183 
184  return estimate_current_floor(filtered_pressure);
185 }
