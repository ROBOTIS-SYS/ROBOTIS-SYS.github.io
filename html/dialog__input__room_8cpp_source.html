86 void DialogInputRoom::on_btnAccept_clicked()
87 {
88  if (ui->lblText->text().compare(tr("사용할 수 없는 객실 번호에요.")) == 0) {
89  return;
90  }
91 
92  // debug 모드에서는 체크 안함
93  if (ros_node->debug_ == true) {
94  this->close();
95  this->setResult((QDialog::Accepted));
96  return;
97  }
98 
99  if (destination_list.size() == 0) {
100  ros_node->get_target_list(destination_list);
101  }
102 
103  if (destination_list.size() > 0) {
104  for (const auto & it : destination_list) {
105  if (ui->lblRoom->text().toStdString().compare(it.key) == 0) {
106  this->close();
107  this->setResult((QDialog::Accepted));
108  return;
109  }
110  }
111  }
112 
113  ui->lblUnderBar->setStyleSheet(
114  "QLabel { border-image: url(:/images/underbar_red.png) 0 0 0 0 stretch stretch; }");
115  ui->lblText->setText(tr("사용할 수 없는 객실 번호에요."));
116  std::thread th_invalid_room(invalid_room_thread, this);
117  th_invalid_room.detach();
118 }
