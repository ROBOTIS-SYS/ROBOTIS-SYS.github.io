Usage

Install

# Install Ubuntu 20.04 Focal Fossa
(https://releases.ubuntu.com/20.04/)
# Install ROS 2 Foxy
$ wget https://raw.githubusercontent.com/ROBOTIS-GIT/robotis_tools/master/install_ros2_foxy.sh \
&& chmod 755 ./install_ros2_foxy.sh \
&& bash ./install_ros2_foxy.sh
# Install Aimbot package
$ sudo apt install git
$ cd ~/colcon_ws/src
$ git clone https://github.com/ROBOTIS-SYS/aimbot_base.git -b develop
$ git clone https://github.com/ROBOTIS-SYS/aimbot_interfaces.git -b feature-simple-path-planner
$ git clone https://github.com/ROBOTIS-SYS/aimbot_utils.git
# Install GurumDDS
$ cd ~/colcon_ws/src/aimbot_base
$ sudo mkdir /etc/flame
$ sudo cp dds.lic /etc/flame/dds.lic
$ sudo apt install ros-foxy-rmw-gurumdds-cpp
$ sh -c "echo \"export RMW_IMPLEMENTATION=rmw_gurumdds_cpp\" >> ~/.bashrc"
$ sh -c "echo \"export ROS_DOMAIN_ID=15\" >> ~/.bashrc" # Domain Number 15
$ source ~/.bashrc
# Install camera related packages
https://github.com/IntelRealSense/librealsense/blob/master/doc/distribution_linux.md#installing-the-packages
# Install Lidar-related packages
(YD Lidar)
$ sudo apt install cmake pkg-config python swig python3-pip
$ cd ~/colcon_ws/src/aimbot_utils/YDLidar-SDK/
$ mkdir build && cd build
$ cmake ..
$ make
$ sudo make install
(Hokuyo Lidar)
# urg node package will be installed by rosdep later on
# Install ROS package dependencies
$ sudo apt install python3-rosdep2
$ cd ~/colcon_ws
$ rosdep update && rosdep install --from-paths src --ignore-src -r --rosdistro foxy -y
# Compile all installed packages
$ cd ~/colcon_ws
$ colcon build --symlink-install
$ source ~/.bashrc
# Configure USB port setting
$ sudo usermod -a -G dialout $USER
$ sudo reboot

Run

# Run Bringup (HW, Scan, Camera)
(prototype1, prototype3)
$ ros2 launch aimbot_bringup bringup.launch.py
$ ros2 launch aimbot_bringup load_start_hw_interface.launch.py
(prototype2)
$ ros2 launch aimbot_bringup bringup_hokuyo.launch.py
$ ros2 launch aimbot_bringup load_start_hw_interface.launch.py
# Run Hardware Interface
$ ros2 launch aimbot_bringup hw.launch.py
# Run Lidar Scan
(prototype1, prototype3)
$ ros2 launch aimbot_bringup lidar_yd.launch.py
(prototype2)
$ ros2 launch aimbot_bringup lidar_hokuyo.launch.py
# Run Camera
$ ros2 launch aimbot_bringup camera.launch.py
# Run Gazebo
$ ros2 launch aimbot_gazebo empty_world.launch.py # in the emtpy world
$ ros2 launch aimbot_gazebo turtlebot3_world.launch.py # in the turtlebot3 world
# Run Rviz
$ ros2 launch aimbot_description rviz.launch.py
$ ros2 launch aimbot_description rviz_fake.launch.py # without a physical or simulation robot
# Run Teleop
(Keyboard)
$ ros2 run aimbot_teleop teleop_keyboard
(Joystick)
$ ros2 run joy joy_node
$ ros2 run aimbot_teleop teleop_joystick
# Run SLAM
$ ros2 launch aimbot_slam cartographer.launch.py # use_sim_time:=true if using sim
$ ros2 run nav2_map_server map_saver_cli -f ~/map # save the newly made map
# Run Navigation
$ ros2 launch aimbot_navigation navigation2.launch.py # use_sim_time:=true if using sim
# Run Xacro File Converter
$ ros2 launch aimbot_description xacro_converter.launch.py # convert urdf.xacro to urdf
