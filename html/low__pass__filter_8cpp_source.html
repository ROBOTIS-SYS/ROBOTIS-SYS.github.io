82 rm2_common::Rotation RotationLowPassFilter::get_filtering_data(
83  rm2_common::Rotation raw_data,
84  double sampling_time)
85 {
86  if (first_input_) {
87  previous_filtering_data_ = raw_data;
88  first_input_ = false;
89  } else {
90  double low = std::max<double>(std::min<double>(hbias_ - hrange_ / 2.0, 1.0), 0.0);
91  double high = std::max<double>(std::min<double>(hbias_ + hrange_ / 2.0, 1.0), 0.0);
92  double hrangeLimited = high - low;
93 
94  // the angle (in radian) between two rotations
95  double distance = previous_filtering_data_.angle_distance(raw_data);
96  double hlpf = distance / M_PI * hrangeLimited + low;
97 
98  previous_filtering_data_ = previous_filtering_data_.slerp(hlpf, raw_data);
99  }
100 
101  return previous_filtering_data_;
102 }
