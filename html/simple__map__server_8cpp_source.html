35 SimpleMapServer::SimpleMapServer(
36  rclcpp_lifecycle::LifecycleNode::SharedPtr node_ptr, std::string map_name,
37  std::string yaml_filename, std::string topic_name,
38  std::string frame_id)
39 {
40  node_ptr_ = node_ptr;
41  map_name_ = map_name;
42  frame_id_ = frame_id;
43 
44  if (!load_map_from_yaml(yaml_filename)) {
45  throw std::runtime_error("Failed to load map yaml file: " + yaml_filename);
46  }
47 
48  // Make name prefix for services
49  const std::string service_prefix = map_name_ + "_server" + std::string("/");
50 
51  using namespace std::chrono_literals;
52  using namespace std::placeholders;
53  // Create a service that provides the occupancy grid
54  occ_service_ = node_ptr_->create_service<nav_msgs::srv::GetMap>(
55  service_prefix + std::string(service_name_),
56  std::bind(&SimpleMapServer::get_map_callback, this, _1, _2, _3));
57 
58  // Create a publisher using the QoS settings to emulate a ROS1 latched topic
59  occ_pub_ = node_ptr_->create_publisher<nav_msgs::msg::OccupancyGrid>(
60  topic_name,
61  rclcpp::QoS(rclcpp::KeepLast(1)).transient_local().reliable());
62 }

78 bool SimpleMapServer::load_map_from_yaml(const std::string & yaml_file)
79 {
80  switch (nav2_map_server::loadMapFromYaml(yaml_file, msg_)) {
81  case nav2_map_server::MAP_DOES_NOT_EXIST:
82  // response->result = nav2_msgs::srv::LoadMap::Response::RESULT_MAP_DOES_NOT_EXIST;
83  return false;
84  case nav2_map_server::INVALID_MAP_METADATA:
85  // response->result = nav2_msgs::srv::LoadMap::Response::RESULT_INVALID_MAP_METADATA;
86  return false;
87  case nav2_map_server::INVALID_MAP_DATA:
88  // response->result = nav2_msgs::srv::LoadMap::Response::RESULT_INVALID_MAP_DATA;
89  return false;
90  case nav2_map_server::LOAD_MAP_SUCCESS:
91  // Correcting msg_ header when it belongs to spiecific node
92  update_msg_header();
93  }
94 
95  return true;
96 }
