26 bool GcGeometryIkSolver::initialize()
27 {
28 // base = robot_model_.get_component("base_frame")->second.component;
29 
30  for (auto it = robot_model_.get_component("manipulator_base");
31  it != robot_model_.get_base_component();
32  it = it->second.parent)
33  {
34  if (it->second.component.get_joint().get_type() != rm2_robot_model::Joint::RotAxis ||
35  it->second.component.get_joint().get_type() != rm2_robot_model::Joint::TransAxis)
36  {
37  base_pose_ = it->second.component.get_pose(0.0) * base_pose_;
38  }
39  }
40  std::cout << "Manipulator base pose : " << base_pose_.position.data.transpose() << std::endl;
41  joint1_ = robot_model_.get_component("link1")->second.component;
42  joint2_ = robot_model_.get_component("link2")->second.component;
43  joint3_ = robot_model_.get_component("link3")->second.component;
44  joint4_ = robot_model_.get_component("link4")->second.component;
45  joint5_ = robot_model_.get_component("link5")->second.component;
46 
47  auto end_effector_its = robot_model_.get_component("link5")->second.children;
48  for (uint i = 0; i < end_effector_its.size(); i++) {
49  end_effectors_.insert(
50  std::make_pair(
51  end_effector_its.at(i)->first,
52  end_effector_its.at(i)->second.component));
53  }
54 
55 // tip = robot_model_.get_component("tip")->second.component;
56 // knock_tip = robot_model_.get_component("knock_tip")->second.component;
57 
58  link2_length_ = joint3_.get_joint().get_origin_from_parent().position.norm();
59  link3_length_ = joint4_.get_joint().get_origin_from_parent().position.norm();
60  delta_joint3_ = atan2(
61  joint4_.get_joint().get_origin_from_parent().position.z(),
62  sqrt(
63  joint4_.get_joint().get_origin_from_parent().position.x() *
64  joint4_.get_joint().get_origin_from_parent().position.x() +
65  joint4_.get_joint().get_origin_from_parent().position.y() *
66  joint4_.get_joint().get_origin_from_parent().position.y()));
67  return true;
68 }

73 bool GcGeometryIkSolver::solve(
74  const std::string target_component_name,
75  const rm2_kinematics_solver_modules::IKSolverModule::TaskValues target_task_values,
76  rm2_kinematics_solver_modules::IKSolverModule::JointValues * joint_values,
77  JointValues current_joint_values)
78 {
79  /* To Do : add twist solving */
80 // std::cout << "---------IK---------" << std::endl;
81  // val for result
82  JointValues target_joint_value(robot_model_.get_dof());
83 
84  // target from manipulator_base
85  rm2_common::Pose goal_pose_from_mani_base = base_pose_.inverse() * target_task_values.pose;
86 
87  // Link5 Pose
88  Pose link5_pose;
89  if (end_effectors_.find(target_component_name) == end_effectors_.end()) {
90  std::cout <<
91  "[IK Solver Failure] please set tip name 'tip' or children of link5." << std::endl;
92  return false;
93  } else if (target_component_name == "tip") {
94  double goal_roll, goal_pitch, goal_yaw;
95  rm2_common::Rotation(
96  goal_pose_from_mani_base.orientation.w(),
97  goal_pose_from_mani_base.orientation.x(),
98  goal_pose_from_mani_base.orientation.y(),
99  goal_pose_from_mani_base.orientation.z()).get_rpy(
100  goal_roll, goal_pitch, goal_yaw);
101  goal_pose_from_mani_base.orientation = rm2_common::Rotation::RPY(0.0, 0.0, goal_yaw);
102  link5_pose = goal_pose_from_mani_base * end_effectors_.at("tip").get_pose(0.0).inverse();
103  } else {
104  link5_pose = goal_pose_from_mani_base *
105  end_effectors_.at(target_component_name).get_pose(0.0).inverse();
106  }
107 // std::cout << "link5_pose\n" << link5_pose.position.data.transpose() << "\n" <<
108 // link5_pose.orientation.data << std::endl;
109 
110  // Joint1 angle
111  target_joint_value.position(0) = atan2(link5_pose.position.y(), link5_pose.position.x());
112 
113  // Joint2 Pose
114  Pose joint2_pose =
115  joint1_.get_pose(target_joint_value.position(0)) * joint2_.get_joint().get_origin_from_parent();
116 // std::cout << "joint2_pose\n" << joint2_pose.position.data.transpose() << "\n" <<
117 // joint2_pose.orientation.data << std::endl;
118 
119  // Pose from joint2 to link5
120  Pose pose_form_joint2_to_link5 = joint2_pose.inverse() * link5_pose;
121 
122  // Gc Manipulator tip RPY
123  // pitch + yaw
124  double tip_roll, tip_pitch, tip_yaw;
125  if (!gc_manipulator_orientation(
126  pose_form_joint2_to_link5.orientation,
127  tip_roll, tip_pitch, tip_yaw))
128  {
129  std::cout << "[IK Solver Failure] Wrong goal orientation. (" <<
130  tip_roll << ",\t " << tip_pitch << ",\t " << tip_yaw << ")\n" <<
131  pose_form_joint2_to_link5.orientation.data << std::endl;
132  return false;
133  }
134 
135  // Joint5 angle
136  target_joint_value.position(4) = tip_yaw;
137 
138  // link4 Pose
139  Pose link4_pose = link5_pose * joint5_.get_pose(target_joint_value.position(4)).inverse();
140 // std::cout << "link4_pose\n" << link4_pose.position.data.transpose() << "\n" <<
141 // link4_pose.orientation.data << std::endl;
142 
143  // Pose from joint2 to joint4
144  Pose pose_from_joint2_to_link4 = joint2_pose.inverse() * link4_pose;
145 // std::cout << "pose_from_joint2_to_joint4\n" <<
146 // pose_from_joint2_to_link4.position.data.transpose() << "\n" <<
147 // pose_from_joint2_to_link4.orientation.data << std::endl;
148 
149  double alpha = atan2(
150  pose_from_joint2_to_link4.position.z(),
151  sqrt(
152  pose_from_joint2_to_link4.position.x() * pose_from_joint2_to_link4.position.x() +
153  pose_from_joint2_to_link4.position.y() * pose_from_joint2_to_link4.position.y()));
154 // std::cout << "alpha: " << alpha * 180 / M_PI << std::endl;
155 
156  double distance = pose_from_joint2_to_link4.position.norm();
157 // std::cout << "distance: " << distance << std::endl;
158 
159  double beta =
160  acos(
161  -(link3_length_ * link3_length_ - distance * distance - link2_length_ * link2_length_) /
162  (2 * distance * link2_length_));
163 // std::cout << "beta: " << beta * 180 / M_PI << std::endl;
164  if (std::isnan(beta) || std::isinf(beta)) {
165  std::cout << "[IK Solver Failure] Too far or close target." << std::endl;
166  return false;
167  }
168 
169  target_joint_value.position(1) = M_PI_2 - (alpha + beta);
170 
171  double phi = acos(
172  (distance * distance - link2_length_ * link2_length_ - link3_length_ * link3_length_) /
173  (-2 * link2_length_ * link3_length_));
174 // std::cout << "phi: " << phi * 180 / M_PI << std::endl;
175 
176  target_joint_value.position(2) = M_PI_2 - phi + delta_joint3_;
177 
178  // Joint 4 angle
179  target_joint_value.position(3) =
180  tip_pitch - target_joint_value.position(1) - target_joint_value.position(2);
181 
182  if (std::isnan(target_joint_value.position[0]) ||
183  std::isnan(target_joint_value.position[1]) ||
184  std::isnan(target_joint_value.position[2]) ||
185  std::isnan(target_joint_value.position[3]) ||
186  std::isnan(target_joint_value.position[4]) )
187  {
188  printf("Target angle value is NAN!!\n");
189  return false;
190  }
191 
192  if (std::isinf(target_joint_value.position[0]) ||
193  std::isinf(target_joint_value.position[1]) ||
194  std::isinf(target_joint_value.position[2]) ||
195  std::isinf(target_joint_value.position[3]) ||
196  std::isinf(target_joint_value.position[4]) )
197  {
198  printf("Target angle value is INF!!\n");
199  return false;
200  }
201 
202  *joint_values = target_joint_value;
203  return true;
204 }

206 bool GcGeometryIkSolver::gc_manipulator_orientation(
207  rm2_common::Rotation orientation,
208  double & roll,
209  double & pitch,
210  double & yaw)
211 {
212  // Pitch * Yaw <Wrong solver>
213  // (cosP 0 sinP) (cosY -sinY 0) (cosPcosY -cosPsinY sinP)
214  // (0 1 0 ) x (sinY cosY 0) = (sinY cosY 0 )
215  // (-sinP 0 cosP) (0 0 1) (-sinPcosY sinPsinY cosP)
216  // Yaw * Pitch * Roll(E)
217  // (cosY -sinY 0) (cosP 0 sinP) (cosYcosP -sinY cosYsinP)
218  // (sinY cosY 0) x (0 1 0 ) = (sinYcosP cosY sinYsinP)
219  // (0 0 1) (-sinP 0 cosP) (-sinP 0 cosP)
220  if (orientation(2, 1) < 1e-3) {
221  roll = 0.0;
222  pitch = atan2(-orientation(2, 0), orientation(2, 2));
223  yaw = atan2(-orientation(0, 1), orientation(1, 1));
224  return true;
225  } else {
226  orientation.get_rpy(roll, pitch, yaw);
227 // if (roll < 1e-3) {
228 // return true;
229 // }
230  return false;
231  }
232 }
