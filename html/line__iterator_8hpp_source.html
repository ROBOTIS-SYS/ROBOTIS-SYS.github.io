42 class LineIterator
43 {
44 public:
45  LineIterator(int x0, int y0, int x1, int y1)
46  : x0_(x0),
47  y0_(y0),
48  x1_(x1),
49  y1_(y1),
50  x_(x0), // X and Y start of at first endpoint.
51  y_(y0),
52  deltax_(abs(x1 - x0)),
53  deltay_(abs(y1 - y0)),
54  curpixel_(0)
55  {
56  if (x1_ >= x0_) { // The x-values are increasing
57  xinc1_ = 1;
58  xinc2_ = 1;
59  } else { // The x-values are decreasing
60  xinc1_ = -1;
61  xinc2_ = -1;
62  }
63 
64  if (y1_ >= y0_) { // The y-values are increasing
65  yinc1_ = 1;
66  yinc2_ = 1;
67  } else { // The y-values are decreasing
68  yinc1_ = -1;
69  yinc2_ = -1;
70  }
71 
72  if (deltax_ >= deltay_) { // There is at least one x-value for every y-value
73  xinc1_ = 0; // Don't change the x when numerator >= denominator
74  yinc2_ = 0; // Don't change the y for every iteration
75  den_ = deltax_;
76  num_ = deltax_ / 2;
77  numadd_ = deltay_;
78  numpixels_ = deltax_; // There are more x-values than y-values
79  } else { // There is at least one y-value for every x-value
80  xinc2_ = 0; // Don't change the x for every iteration
81  yinc1_ = 0; // Don't change the y when numerator >= denominator
82  den_ = deltay_;
83  num_ = deltay_ / 2;
84  numadd_ = deltax_;
85  numpixels_ = deltay_; // There are more y-values than x-values
86  }
87  }
88 
89  bool isValid() const
90  {
91  return curpixel_ <= numpixels_;
92  }
93 
94  void advance()
95  {
96  num_ += numadd_; // Increase the numerator by the top of the fraction
97  if (num_ >= den_) { // Check if numerator >= denominator
98  num_ -= den_; // Calculate the new numerator value
99  x_ += xinc1_; // Change the x as appropriate
100  y_ += yinc1_; // Change the y as appropriate
101  }
102  x_ += xinc2_; // Change the x as appropriate
103  y_ += yinc2_; // Change the y as appropriate
104 
105  curpixel_++;
106  }
107 
108  int getX() const
109  {
110  return x_;
111  }
112  int getY() const
113  {
114  return y_;
115  }
116

120  }

123  return y0_;
124  }

126  int getX1() const
127  {
128  return x1_;
129  }
130  int getY1() const
131  {
132  return y1_;
133  }
134

140

150  int den_, num_, numadd_, numpixels_;
151 };
