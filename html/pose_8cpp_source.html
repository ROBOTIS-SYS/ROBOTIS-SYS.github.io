36 double Vector::norm(double eps) const
37 {
38  double tmp1;
39  double tmp2;
40  tmp1 = fabs(data(0));
41  tmp2 = fabs(data(1));
42  if (tmp1 >= tmp2) {
43  tmp2 = fabs(data(2));
44  if (tmp1 >= tmp2) {
45  if (tmp1 < eps) {
46  // only to everything exactly zero case, all other are handled correctly
47  return 0;
48  }
49  return tmp1 * sqrt(1 + pow(data(1) / data(0), 2) + pow(data(2) / data(0), 2));
50  } else {
51  return tmp2 * sqrt(1 + pow(data(0) / data(2), 2) + pow(data(1) / data(2), 2));
52  }
53  } else {
54  tmp1 = fabs(data(2));
55  if (tmp2 > tmp1) {
56  return tmp2 * sqrt(1 + pow(data(0) / data(1), 2) + pow(data(2) / data(1), 2));
57  } else {
58  return tmp1 * sqrt(1 + pow(data(0) / data(2), 2) + pow(data(1) / data(2), 2));
59  }
60  }
61 }

86 double Rotation::get_axis_angle(Vector & axis, double eps) const
87 {
88  double angle, x, y, z; // variables for result
89  double epsilon = eps; // margin to allow for rounding errors
90  double epsilon2 = eps * 10; // margin to distinguish between 0 and 180 degrees
91 
92  if ((std::abs(data(0, 1) - data(1, 0)) < epsilon) &&
93  (std::abs(data(0, 2) - data(2, 0)) < epsilon) &&
94  (std::abs(data(1, 2) - data(2, 1)) < epsilon))
95  {
96  // singularity found
97  // first check for identity matrix which must have +1 for all terms
98  // in leading diagonal and zero in other terms
99  if ((std::abs(data(0, 1) + data(1, 0)) < epsilon2) &&
100  (std::abs(data(0, 2) + data(2, 0)) < epsilon2) &&
101  (std::abs(data(1, 2) + data(2, 1)) < epsilon2) &&
102  (std::abs(data(0, 0) + data(1, 1) + data(2, 2) - 3) < epsilon2))
103  {
104  // this singularity is identity matrix so angle = 0, axis is arbitrary
105  // Choose 0, 0, 1 to pass orocos tests
106  axis = Vector(0, 0, 1);
107  angle = 0.0;
108  return angle;
109  }
110 
111  // otherwise this singularity is angle = 180
112  angle = M_PI;
113  double xx = (data(0, 0) + 1) / 2;
114  double yy = (data(1, 1) + 1) / 2;
115  double zz = (data(2, 2) + 1) / 2;
116  double xy = (data(0, 1) + data(1, 0)) / 4;
117  double xz = (data(0, 2) + data(2, 0)) / 4;
118  double yz = (data(1, 2) + data(2, 1)) / 4;
119 
120  if ((xx > yy) && (xx > zz)) {
121  // data(0,0) is the largest diagonal term
122  x = sqrt(xx);
123  y = xy / x;
124  z = xz / x;
125  } else if (yy > zz) {
126  // data(1,1) is the largest diagonal term
127  y = sqrt(yy);
128  x = xy / y;
129  z = yz / y;
130  } else {
131  // data(2,2) is the largest diagonal term so base result on this
132  z = sqrt(zz);
133  x = xz / z;
134  y = yz / z;
135  }
136  axis = Vector(x, y, z);
137  return angle; // return 180 deg rotation
138  }
139 
140  double f = (data(0, 0) + data(1, 1) + data(2, 2) - 1) / 2;
141 
142  x = (data(2, 1) - data(1, 2));
143  y = (data(0, 2) - data(2, 0));
144  z = (data(1, 0) - data(0, 1));
145  axis = Vector(x, y, z);
146  angle = atan2(axis.norm() / 2, f);
147  axis.normalize();
148  return angle;
149 }

151 Rotation Rotation::EulerZYZ(double Alfa, double Beta, double Gamma)
152 {
153  double sa, ca, sb, cb, sg, cg;
154  sa = sin(Alfa); ca = cos(Alfa);
155  sb = sin(Beta); cb = cos(Beta);
156  sg = sin(Gamma); cg = cos(Gamma);
157  return Rotation(
158  ca * cb * cg - sa * sg, -ca * cb * sg - sa * cg, ca * sb,
159  sa * cb * cg + ca * sg, -sa * cb * sg + ca * cg, sa * sb,
160  -sb * cg, sb * sg, cb
161  );
162 }

164 void Rotation::get_euler_zyz(double & alpha, double & beta, double & gamma) const
165 {
166  double epsilon = 1E-12;
167  if (fabs(data(2, 2)) > 1 - epsilon) {
168  gamma = 0.0;
169  if (data(2, 2) > 0) {
170  beta = 0.0;
171  alpha = atan2(data(1, 0), data(0, 0));
172  } else {
173  beta = M_PI;
174  alpha = atan2(-data(1, 0), -data(0, 0));
175  }
176  } else {
177  alpha = atan2(data(1, 2), data(0, 2));
178  beta = atan2(sqrt(data(2, 0) * data(2, 0) + data(2, 1) * data(2, 1)), data(2, 2));
179  gamma = atan2(data(2, 1), -data(2, 0));
180  }
181 }

183 Rotation Rotation::RPY(double roll, double pitch, double yaw)
184 {
185  double ca1, cb1, cc1, sa1, sb1, sc1;
186  ca1 = cos(yaw); sa1 = sin(yaw);
187  cb1 = cos(pitch);sb1 = sin(pitch);
188  cc1 = cos(roll);sc1 = sin(roll);
189  return Rotation(
190  ca1 * cb1, ca1 * sb1 * sc1 - sa1 * cc1, ca1 * sb1 * cc1 + sa1 * sc1,
191  sa1 * cb1, sa1 * sb1 * sc1 + ca1 * cc1, sa1 * sb1 * cc1 - ca1 * sc1,
192  -sb1, cb1 * sc1, cb1 * cc1);
193 }

195 void Rotation::get_rpy(double & roll, double & pitch, double & yaw) const
196 {
197  double epsilon = 1E-12;
198  pitch = atan2(-data(2, 0), sqrt(data(0, 0) * data(0, 0) + data(1, 0) * data(1, 0)));
199  if (fabs(pitch) > (M_PI / 2 - epsilon) ) {
200  yaw = atan2(-data(0, 1), data(1, 1));
201  roll = 0.0;
202  } else {
203  roll = atan2(data(2, 1), data(2, 2));
204  yaw = atan2(data(1, 0), data(0, 0));
205  }

275 Eigen::Matrix3d rodriguesRotationMatrix(Vector axis, double angle)
276 {
277  Eigen::Matrix3d skew_symmetric_matrix = Eigen::Matrix3d::Zero();
278  Eigen::Matrix3d rotation_matrix = Eigen::Matrix3d::Zero();
279  Eigen::Matrix3d Identity_matrix = Eigen::Matrix3d::Identity();
280 
281  skew_symmetric_matrix = skewSymmetricMatrix(axis);
282  rotation_matrix = Identity_matrix +
283  skew_symmetric_matrix * sin(angle) +
284  skew_symmetric_matrix * skew_symmetric_matrix * (1 - cos(angle));
285  return rotation_matrix;
286 }

288 Vector matrixLogarithm(Eigen::Matrix3d rotation_matrix)
289 {
290  Eigen::Matrix3d R = rotation_matrix;
291  Eigen::Vector3d l = Eigen::Vector3d::Zero();
292  Eigen::Vector3d rotation_vector = Eigen::Vector3d::Zero();
293 
294  double theta = 0.0;
295  // double diag = 0.0;
296  bool diagonal_matrix = R.isDiagonal();
297 
298  l << R(2, 1) - R(1, 2),
299  R(0, 2) - R(2, 0),
300  R(1, 0) - R(0, 1);
301  theta = atan2(l.norm(), R(0, 0) + R(1, 1) + R(2, 2) - 1);
302 
303  if (R.isIdentity()) {
304  rotation_vector.setZero(3);
305  return Vector(rotation_vector);
306  }
307 
308  if (diagonal_matrix == true) {
309  rotation_vector << R(0, 0) + 1, R(1, 1) + 1, R(2, 2) + 1;
310  rotation_vector = rotation_vector * M_PI_2;
311  } else {
312  rotation_vector = theta * (l / l.norm());
313  }
314 
315  return Vector(rotation_vector);
316 }
