506 bool
507 AmclNode::getOdomPose(
508  geometry_msgs::msg::PoseStamped & odom_pose,
509  double & x, double & y, double & yaw,
510  const rclcpp::Time & sensor_timestamp, const std::string & frame_id)
511 {
512  // Get the robot's pose
513  geometry_msgs::msg::PoseStamped ident;
514  ident.header.frame_id = antis_nav_util::strip_leading_slash(frame_id);
515  ident.header.stamp = sensor_timestamp;
516  tf2::toMsg(tf2::Transform::getIdentity(), ident.pose);
517 
518  try {
519  tf_buffer_->transform(ident, odom_pose, odom_frame_id_);
520  } catch (tf2::TransformException & e) {
521  ++scan_error_count_;
522  if (scan_error_count_ % 20 == 0) {
523  RCLCPP_ERROR(
524  get_logger(), "(%d) consecutive laser scan transforms failed: (%s)", scan_error_count_,
525  e.what());
526  }
527  return false;
528  }
529 
530  scan_error_count_ = 0; // reset since we got a good transform
531  x = odom_pose.pose.position.x;
532  y = odom_pose.pose.position.y;
533  yaw = tf2::getYaw(odom_pose.pose.orientation);
534 
535  return true;
536 }

538 pf_vector_t
539 AmclNode::uniformPoseGenerator(void * arg)
540 {
541  map_t * map = reinterpret_cast<map_t *>(arg);
542 
543 #if NEW_UNIFORM_SAMPLING
544  unsigned int rand_index = drand48() * free_space_indices.size();
545  std::pair<int, int> free_point = free_space_indices[rand_index];
546  pf_vector_t p;
547  p.v[0] = MAP_WXGX(map, free_point.first);
548  p.v[1] = MAP_WYGY(map, free_point.second);
549  p.v[2] = drand48() * 2 * M_PI - M_PI;
550 #else
551  double min_x, max_x, min_y, max_y;
552 
553  min_x = (map->size_x * map->scale) / 2.0 - map->origin_x;
554  max_x = (map->size_x * map->scale) / 2.0 + map->origin_x;
555  min_y = (map->size_y * map->scale) / 2.0 - map->origin_y;
556  max_y = (map->size_y * map->scale) / 2.0 + map->origin_y;
557 
558  pf_vector_t p;
559 
560  RCLCPP_DEBUG(get_logger(), "Generating new uniform sample");
561  for (;; ) {
562  p.v[0] = min_x + drand48() * (max_x - min_x);
563  p.v[1] = min_y + drand48() * (max_y - min_y);
564  p.v[2] = drand48() * 2 * M_PI - M_PI;
565  // Check that it's a free cell
566  int i, j;
567  i = MAP_GXWX(map, p.v[0]);
568  j = MAP_GYWY(map, p.v[1]);
569  if (MAP_VALID(map, i, j) && (map->cells[MAP_INDEX(map, i, j)].occ_state == -1)) {
570  break;
571  }
572  }
573 #endif
574  return p;
575 }

577 double gaussianRandom(double standard_deviation)
578 {
579  static std::random_device rd;
580  static std::mt19937 gen(rd());
581 
582  std::normal_distribution<double> dist(0, standard_deviation);

587 pf_vector_t
588 AmclNode::gaussianPoseGenerator(void * arg)
589 {
590  map_t * map = reinterpret_cast<map_t *>(arg);
591 
592  pf_vector_t p;
593 
594  for (;; ) {
595  p.v[0] = current_pose.v[0] + gaussianRandom(stdev);
596  p.v[1] = current_pose.v[1] + gaussianRandom(stdev);
597  p.v[2] = current_pose.v[2] + gaussianRandom(rot_stdev) * 2 * M_PI;
598 
599  // SBWOO: check if there is no occupied cell between particle and robot
600  double diff_x;
601  double diff_y;
602  diff_x = p.v[0] - current_pose.v[0];
603  diff_y = p.v[1] - current_pose.v[1];
604  double diff_abs = std::sqrt(std::pow(diff_x, 2) + std::pow(diff_y, 2));
605  double diff_angle = std::atan2(diff_y, diff_x);
606 
607  double occ_range = map_calc_range_exclude_unknown(
608  map, current_pose.v[0], current_pose.v[1], diff_angle, diff_abs + 0.3);
609 
610  if (occ_range > diff_abs) {break;}
611 
612  // check if it is free cell
613  // int x = MAP_GXWX(map, p.v[0]);
614  // int y = MAP_GYWY(map, p.v[1]);
615 
616  // if (MAP_VALID(map, x, y) && (map->cells[MAP_INDEX(map, x, y)].occ_state == -1)) {break;}
617  }
618  return p;
619 }

689 void
690 AmclNode::handleInitialPose(geometry_msgs::msg::PoseWithCovarianceStamped & msg)
691 {
692  // In case the client sent us a pose estimate in the past, integrate the
693  // intervening odometric change.
694  geometry_msgs::msg::TransformStamped tx_odom;
695  try {
696  rclcpp::Time rclcpp_time = now();
697  tf2::TimePoint tf2_time(std::chrono::nanoseconds(rclcpp_time.nanoseconds()));
698 
699  // Check if the transform is available
700  tx_odom = tf_buffer_->lookupTransform(
701  base_frame_id_, tf2_ros::fromMsg(msg.header.stamp),
702  base_frame_id_, tf2_time, odom_frame_id_);
703  } catch (tf2::TransformException & e) {
704  // If we've never sent a transform, then this is normal, because the
705  // global_frame_id_ frame doesn't exist. We only care about in-time
706  // transformation for on-the-move pose-setting, so ignoring this
707  // startup condition doesn't really cost us anything.
708  if (sent_first_transform_) {
709  RCLCPP_WARN(get_logger(), "Failed to transform initial pose in time (%s)", e.what());
710  }
711  tf2::impl::Converter<false, true>::convert(tf2::Transform::getIdentity(), tx_odom.transform);
712  }
713 
714  tf2::Transform tx_odom_tf2;
715  tf2::impl::Converter<true, false>::convert(tx_odom.transform, tx_odom_tf2);
716 
717  tf2::Transform pose_old;
718  tf2::impl::Converter<true, false>::convert(msg.pose.pose, pose_old);
719 
720  tf2::Transform pose_new = pose_old * tx_odom_tf2;
721 
722  // Transform into the global frame
723 
724  RCLCPP_INFO(
725  get_logger(), "Setting pose (%.6f): %.3f %.3f %.3f",
726  now().nanoseconds() * 1e-9,
727  pose_new.getOrigin().x(),
728  pose_new.getOrigin().y(),
729  tf2::getYaw(pose_new.getRotation()));
730 
731  // Re-initialize the filter
732  pf_vector_t pf_init_pose_mean = pf_vector_zero();
733  pf_init_pose_mean.v[0] = pose_new.getOrigin().x();
734  pf_init_pose_mean.v[1] = pose_new.getOrigin().y();
735  pf_init_pose_mean.v[2] = tf2::getYaw(pose_new.getRotation());
736 
737  pf_matrix_t pf_init_pose_cov = pf_matrix_zero();
738  // Copy in the covariance, converting from 6-D to 3-D
739  for (int i = 0; i < 2; i++) {
740  for (int j = 0; j < 2; j++) {
741  pf_init_pose_cov.m[i][j] = msg.pose.covariance[6 * i + j];
742  }
743  }
744 
745  pf_init_pose_cov.m[2][2] = msg.pose.covariance[6 * 5 + 5];
746 
747  pf_init(pf_, pf_init_pose_mean, pf_init_pose_cov);
748  pf_init_ = false;
749  init_pose_received_on_inactive = false;
750  initial_pose_is_known_ = true;
751 }

946  } else {
947  if (force_update_buffer >= 1) {
948  force_update_buffer--;
949  }
950  }
951

953

957 bool AmclNode::updateFilter(
958  const int & laser_index,
959  const sensor_msgs::msg::LaserScan::ConstSharedPtr & laser_scan,
960  const pf_vector_t & pose)
961 {
962  antis_nav_amcl::LaserData ldata;
963  ldata.laser = lasers_[laser_index];
964  ldata.range_count = laser_scan->ranges.size();
965  // To account for lasers that are mounted upside-down, we determine the
966  // min, max, and increment angles of the laser in the base frame.
967  //
968  // Construct min and max angles of laser, in the base_link frame.
969  // Here we set the roll pich yaw of the lasers. We assume roll and pich are zero.
970  geometry_msgs::msg::QuaternionStamped min_q, inc_q;
971  min_q.header.stamp = laser_scan->header.stamp;
972  min_q.header.frame_id = antis_nav_util::strip_leading_slash(laser_scan->header.frame_id);
973  min_q.quaternion = orientationAroundZAxis(laser_scan->angle_min);
974 
975  inc_q.header = min_q.header;
976  inc_q.quaternion = orientationAroundZAxis(laser_scan->angle_min + laser_scan->angle_increment);
977  try {
978  tf_buffer_->transform(min_q, min_q, base_frame_id_);
979  tf_buffer_->transform(inc_q, inc_q, base_frame_id_);
980  } catch (tf2::TransformException & e) {
981  RCLCPP_WARN(
982  get_logger(), "Unable to transform min/max laser angles into base frame: %s",
983  e.what());
984  return false;
985  }
986  double angle_min = tf2::getYaw(min_q.quaternion);
987  double angle_increment = tf2::getYaw(inc_q.quaternion) - angle_min;
988 
989  // wrapping angle to [-pi .. pi]
990  angle_increment = fmod(angle_increment + 5 * M_PI, 2 * M_PI) - M_PI;
991 
992  RCLCPP_DEBUG(
993  get_logger(), "Laser %d angles in base frame: min: %.3f inc: %.3f", laser_index, angle_min,
994  angle_increment);
995 
996  // Apply range min/max thresholds, if the user supplied them
997  if (laser_max_range_ > 0.0) {
998  ldata.range_max = std::min(laser_scan->range_max, static_cast<float>(laser_max_range_));
999  } else {
1000  ldata.range_max = laser_scan->range_max;
1001  }
1002  double range_min;
1003  if (laser_min_range_ > 0.0) {
1004  range_min = std::max(laser_scan->range_min, static_cast<float>(laser_min_range_));
1005  } else {
1006  range_min = laser_scan->range_min;
1007  }
1008 
1009  // The LaserData destructor will free this memory
1010  ldata.ranges = new double[ldata.range_count][2];
1011  for (int i = 0; i < ldata.range_count; i++) {
1012  // amcl doesn't (yet) have a concept of min range. So we'll map short
1013  // readings to max range.
1014  if (laser_scan->ranges[i] <= range_min) {
1015  ldata.ranges[i][0] = ldata.range_max;
1016  } else {
1017  ldata.ranges[i][0] = laser_scan->ranges[i];
1018  }
1019  // Compute bearing
1020  ldata.ranges[i][1] = angle_min +
1021  (i * angle_increment);
1022  }
1023  lasers_[laser_index]->sensorUpdate(pf_, reinterpret_cast<antis_nav_amcl::LaserData *>(&ldata));
1024 
1025  /************* SBWOO custom code bit *************/
1026  RCLCPP_DEBUG(
1027  get_logger(), "sensor have best match score with value %f",
1028  lasers_[laser_index]->get_max_match_score());
1029  /************* SBWOO custom code bit *************/
1030 
1031  lasers_update_[laser_index] = false;
1032  pf_odom_pose_ = pose;
1033  return true;
1034 }

1104 void
1105 AmclNode::publishAmclPose(
1106  const sensor_msgs::msg::LaserScan::ConstSharedPtr & laser_scan,
1107  const std::vector<amcl_hyp_t> & hyps, const int & max_weight_hyp)
1108 {
1109  // If initial pose is not known, AMCL does not know the current pose
1110  if (!initial_pose_is_known_) {
1111  if (checkElapsedTime(2s, last_time_printed_msg_)) {
1112  RCLCPP_WARN(
1113  get_logger(), "ACML cannot publish a pose or update the transform. "
1114  "Please set the initial pose...");
1115  last_time_printed_msg_ = now();
1116  }
1117  return;
1118  }
1119 
1120  auto p = std::make_unique<geometry_msgs::msg::PoseWithCovarianceStamped>();
1121  // Fill in the header
1122  p->header.frame_id = global_frame_id_;
1123  p->header.stamp = laser_scan->header.stamp;
1124  // Copy in the pose
1125  p->pose.pose.position.x = hyps[max_weight_hyp].pf_pose_mean.v[0];
1126  p->pose.pose.position.y = hyps[max_weight_hyp].pf_pose_mean.v[1];
1127  p->pose.pose.orientation = orientationAroundZAxis(hyps[max_weight_hyp].pf_pose_mean.v[2]);
1128  // Copy in the covariance, converting from 3-D to 6-D
1129  pf_sample_set_t * set = pf_->sets + pf_->current_set;
1130  for (int i = 0; i < 2; i++) {
1131  for (int j = 0; j < 2; j++) {
1132  // Report the overall filter covariance, rather than the
1133  // covariance for the highest-weight cluster
1134  // p->covariance[6*i+j] = hyps[max_weight_hyp].pf_pose_cov.m[i][j];
1135  p->pose.covariance[6 * i + j] = set->cov.m[i][j];
1136  }
1137  }
1138  p->pose.covariance[6 * 5 + 5] = set->cov.m[2][2];
1139  float temp = 0.0;
1140  for (auto covariance_value : p->pose.covariance) {
1141  temp += covariance_value;
1142  }
1143  temp += p->pose.pose.position.x + p->pose.pose.position.y;
1144  if (!std::isnan(temp)) {
1145  RCLCPP_DEBUG(get_logger(), "Publishing pose");
1146  last_published_pose_ = *p;
1147  first_pose_sent_ = true;
1148  pose_pub_->publish(std::move(p));
1149  } else {
1150  RCLCPP_WARN(
1151  get_logger(), "AMCL covariance or pose is NaN, likely due to an invalid "
1152  "configuration or faulty sensor measurements! Pose is not available!");
1153  }
1154 
1155  RCLCPP_DEBUG(
1156  get_logger(), "New pose: %6.3f %6.3f %6.3f",
1157  hyps[max_weight_hyp].pf_pose_mean.v[0],
1158  hyps[max_weight_hyp].pf_pose_mean.v[1],
1159  hyps[max_weight_hyp].pf_pose_mean.v[2]);
