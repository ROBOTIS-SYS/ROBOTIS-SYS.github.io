104 void OcrRecognition::model_run(
105  std::vector<cv::Mat> & src,
106  std::vector<std::string> & rec_result_txt,
107  std::vector<std::vector<std::vector<int>>> & input_boxes,
108  std::vector<std::vector<std::vector<int>>> & output_boxes,
109  std::vector<RecognizedText> & result_recognized_txt)
110 {
111  for (size_t i = 0; i < src.size(); i++) {
112  if (src[i].rows > 20 && src[i].cols > 40) {
113  std::string tmp_result_txt;
114  pre_processing_test(src[i]);
115  trt_engine_->runInference(prediction_bindings_, cuda_stream_);
116  cudaStreamSynchronize(cuda_stream_);
117  if (post_processing(src[i], device_tensor_buffer_, tmp_result_txt)) {
118  RecognizedText tmp_recognized_txt;
119  tmp_recognized_txt.class_name = tmp_result_txt;
120  int center_x =
121  (input_boxes[i][0][0] + input_boxes[i][1][0] + input_boxes[i][2][0] +
122  input_boxes[i][3][0]) * 0.25;
123  int center_y =
124  (input_boxes[i][0][1] + input_boxes[i][1][1] + input_boxes[i][2][1] +
125  input_boxes[i][3][1]) * 0.25;
126  tmp_recognized_txt.features.push_back(
127  cv::Point2i(
128  input_boxes[i][0][0],
129  input_boxes[i][0][1]));
130  tmp_recognized_txt.features.push_back(
131  cv::Point2i(
132  input_boxes[i][1][0],
133  input_boxes[i][1][1]));
134  tmp_recognized_txt.features.push_back(cv::Point2i(center_x, center_y));
135  tmp_recognized_txt.features.push_back(
136  cv::Point2i(
137  input_boxes[i][2][0],
138  input_boxes[i][2][1]));
139  tmp_recognized_txt.features.push_back(
140  cv::Point2i(
141  input_boxes[i][3][0],
142  input_boxes[i][3][1]));
143  result_recognized_txt.push_back(tmp_recognized_txt);
144  tmp_recognized_txt.features.clear();
145  }
146  }
147  }

156 void OcrRecognition::pre_processing_test(
157  cv::Mat & src)
158 {
159  std::vector<int> rec_image_shape_ = {3, 48, 320};
160  int imgH = rec_image_shape_[1];
161  int imgW = rec_image_shape_[2];
162  float max_wh_ratio = imgW * 1.0 / imgH;
163  int h = src.rows;
164  int w = src.cols;
165  float wh_ratio = w * 1.0 / h;
166  std::vector<float> mean_ = {0.5f, 0.5f, 0.5f};
167  std::vector<float> scale_ = {1 / 0.5f, 1 / 0.5f, 1 / 0.5f};
168  max_wh_ratio = std::max(max_wh_ratio, wh_ratio);
169  cv::Mat resize_img;
170  resize(src, resize_img, max_wh_ratio, rec_image_shape_);
171  normalize(&resize_img, mean_, scale_, true);
172  convert_mat2vec(resize_img, h_input_pin);
173  auto error = cudaMemcpyAsync(
174  (*device_tensor_buffer_)[0].deviceBuffer.data(),
175  h_input_pin,
176  1 * 3 * imgH * imgW * sizeof(float),
177  cudaMemcpyHostToDevice,
178  cuda_stream_);
179  rec_image_shape_.clear();
180  mean_.clear();
181  scale_.clear();
182 }

184 void OcrRecognition::transpose(cv::Mat & src, std::vector<int> rec_image_shape)
185 {
186  int imgC, imgH, imgW;
187  int resize_w;
188  imgC = rec_image_shape[0];
189  imgH = rec_image_shape[1];
190  imgW = rec_image_shape[2];
191  float ratio = static_cast<float>(src.cols) / static_cast<float>(src.rows);
192  if (ceilf(imgH * ratio) > imgW) {
193  resize_w = imgW;
194  } else {
195  resize_w = static_cast<int>(ceilf(imgH * ratio));
196  }
197  if (resize_w > imgW) {
198  resize_w = imgW;
199  }
200  src.copyTo(host_pre_mat_);
201  input_img_.upload(host_pre_mat_);
202  cv::cuda::resize(
203  input_img_,
204  input_img_,
205  cv::Size(resize_w, imgH),
206  0.f,
207  0.f,
208  cv::INTER_LINEAR);
209  cv::cuda::copyMakeBorder(
210  input_img_,
211  resized_input_img_,
212  0,
213  0,
214  0,
215  static_cast<int>(imgW - input_img_.cols),
216  cv::BORDER_CONSTANT,
217  {127, 127, 127});
218  // image_processing_with_cuda_.normalize(
219  // resized_input_img_,
220  // norm_input_img_);
221  cv::cuda::split(resized_input_img_, input_channels_); // HWC -> CHW
222  resized_input_img_buffer_.convertTo(tensor_img_, CV_32FC3, 1.f / 255.f);
223  cv::cuda::subtract(
224  tensor_img_,
225  cv::Scalar(
226  normalize_sub_vals_[0], normalize_sub_vals_[1],
227  normalize_sub_vals_[2]), tensor_img_, cv::noArray(), -1);
228  cv::cuda::divide(
229  tensor_img_,
230  cv::Scalar(
231  normalize_div_vals_[0], normalize_div_vals_[1],
232  normalize_div_vals_[2]), tensor_img_, 1, -1);
233  auto * dataPointer = tensor_img_.ptr<void>();
234  auto error = cudaMemcpyAsync(
235  (*device_tensor_buffer_)[0].deviceBuffer.data(),
236  dataPointer,
237  tensor_img_.cols * tensor_img_.rows * tensor_img_.channels() * sizeof(float),
238  cudaMemcpyDeviceToDevice,
239  cuda_stream_);
240 }

283 void OcrRecognition::normalize(
284  cv::Mat * im,
285  const std::vector<float> & mean,
286  const std::vector<float> & scale,
287  const bool is_scale)
288 {
289  double e = 1.0;
290  if (is_scale) {
291  e /= 255.0;
292  }
293  (*im).convertTo(*im, CV_32FC3, e);
294  std::vector<cv::Mat> bgr_channels(3);
295  cv::split(*im, bgr_channels);
296  for (auto i = 0; i < bgr_channels.size(); i++) {
297  bgr_channels[i].convertTo(
298  bgr_channels[i],
299  CV_32FC1,
300  1.0 * scale[i],
301  (0.0 - mean[i]) * scale[i]);
302  }
303  cv::merge(bgr_channels, *im);
304  bgr_channels.clear();
305 }

322 void OcrRecognition::filter_text(
323  std::vector<RecognizedText> & recognized_text,
324  const std::vector<std::string> & target_list,
325  double ratio_thr)
326 {
327  for (auto target_name : target_list) {
328  for (size_t i = 0; i < recognized_text.size(); ++i) {
329  // std::cout << "Target Name : " << target_name << std::endl;
330  std::string tmp_text = recognized_text[i].class_name;
331  // std::cout << "Before Text : " << recognized_text[i].class_name << std::endl;
332 
333  double ratio = 0;
334  auto matcher = difflib::MakeSequenceMatcher<>(target_name, tmp_text);
335  auto match_list = matcher.get_matching_blocks();
336  ratio = matcher.ratio();
337 
338  // std::cout << ratio << std::endl;
339  if (ratio > ratio_thr) {
340  recognized_text[i].class_name = target_name;
341  }
342  std::cout << "After Text : " << recognized_text[i].class_name << std::endl;
343  }

347 bool OcrRecognition::post_processing(
348  cv::Mat & src,
349  std::shared_ptr<std::vector<samplesCommon::ManagedBuffer>> input_buffer,
350  std::string & rec_result_txt)
351 {
352  int string_size_count = 0;
353  const int limit_string_size = 1;
354  std::string pre_txt = "";
355  cudaMemcpyAsync(
356  (*input_buffer)[1].hostBuffer.data(),
357  (*input_buffer)[1].deviceBuffer.data(),
358  40 * 3690 * sizeof(float),
359  cudaMemcpyDeviceToHost,
360  cuda_stream_);
361  std::vector<std::vector<float>> rec_result;
362  for (int i = 0; i < 40; i++) {
363  std::vector<float> tmp_vector;
364  tmp_vector.resize(3690);
365  memcpy(
366  tmp_vector.data(),
367  reinterpret_cast<const char *>(
368  (*input_buffer)[1].hostBuffer.data() + i * 3690 * sizeof(float)),
369  3690 * sizeof(float));
370  rec_result.push_back(tmp_vector);
371  int nArgMax = std::max_element(
372  tmp_vector.begin(),
373  tmp_vector.end()) - tmp_vector.begin();
374  if (tmp_vector[nArgMax] > 0.8) {
375  if (nArgMax != 0) {
376  if (pre_txt == class_characters_[nArgMax]) {
377  continue;
378  } else {
379  rec_result_txt += class_characters_[nArgMax];
380  pre_txt = class_characters_[nArgMax];
381  string_size_count++;
382  }
383  }
384  }
385  tmp_vector.clear();
386  }
387  rec_result.clear();
388  if (string_size_count > limit_string_size) {
389  return true;
390  } else {
391  return false;
392  }
393 }
