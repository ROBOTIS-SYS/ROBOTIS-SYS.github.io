99 void GcInitEndMotionControl::update()
100 {
101  if (move_flag_ == false && joint_motion_queue_.size() == 0) {
102  // stop update
103  send_action_success();
104  stop_update();
105  } else if (move_flag_ == false && joint_motion_queue_.size() != 0) {
106  // make trajectory
107  make_trajectory();
108  } else {
109  bool finish = false;
110  if (unlimited_falg_) {
111  auto goal = joint_trajectory_ptr_->get(robot_handler_->get_tick_time(), finish);
112  if (finish == true) {
113  // end of trajectory
114  trajectory_time_ = 0.0;
115  move_flag_ = false;
116  } else {
117  robot_handler_->set_goal_joint_value_unlimited(goal);
118  }
119  } else {
120  auto goal = joint_trajectory_ptr_->get(robot_handler_->get_tick_time(), finish);
121 // auto goal = task_trajectory_ptr_->get(robot_handler_->get_tick_time(), finish);
122  if (finish == true) {
123  // end of trajectory
124  trajectory_time_ = 0.0;
125  move_flag_ = false;
126  } else {
127  robot_handler_->set_goal_joint_value(goal);
128 // robot_handler_->set_goal_task_value(tip_name_, goal);
129  }
130  }
131  }
132  send_feedback();
133 }

135 void GcInitEndMotionControl::update_cancel(CancelResult result)
136 {
137 // Init-end action 도중에 다시 Init-end action이 들어올경우,
138 // Goal callback에서 motion_queue를 체우고, start_update를 실행할때,
139 // update_cancel이 호출되어, queue를 지워버리는 현상이 일어난다.
140 // 그를 방지하기위해 cancel에서는 queue를 지우지 않는다.
141 // for (size_t i = 0; i < joint_motion_queue_.size(); i++) {
142 // joint_motion_queue_.pop();
143 // }
144  move_flag_ = false;
145 
146  GcInitEndMotionAction::Result::SharedPtr action_result =
147  std::make_shared<GcInitEndMotionAction::Result>();
148  manipulator_interfaces::msg::Result result_state;
149 
150  if (result == CancelResult::JOINT_LIMIT) {
151  result_state.set__val(manipulator_interfaces::msg::Result::COLLISION_CHECKING_UNAVAILABLE);
152  RCLCPP_INFO_STREAM(
153  node_->get_logger(),
154  "Init contorl canceled : joint limit over.");
155  } else if (result == CancelResult::TOO_FAST_MOVE) {
156  result_state.set__val(manipulator_interfaces::msg::Result::CONTROL_FAILED);
157  RCLCPP_INFO_STREAM(
158  node_->get_logger(),
159  "Init contorl canceled : too fast move failure");
160  } else if (result == CancelResult::INVALID_GOAL_VALUES) {
161  result_state.set__val(manipulator_interfaces::msg::Result::CONTROL_FAILED);
162  RCLCPP_INFO_STREAM(
163  node_->get_logger(),
164  "Init contorl canceled : invalid goal values failure");
165  } else if (result == CancelResult::IK_FAIL) {
166  result_state.set__val(manipulator_interfaces::msg::Result::NO_IK_SOLUTION);
167  RCLCPP_INFO_STREAM(
168  node_->get_logger(),
169  "Init contorl canceled : ik solving failure");
170  } else if (result == CancelResult::PREEMPTED) {
171  result_state.set__val(manipulator_interfaces::msg::Result::PREEMPTED);
172  RCLCPP_INFO_STREAM(
173  node_->get_logger(),
174  "Init contorl canceled : preempted by the other controller");
175  } else if (result == CancelResult::CHECKER) {
176  result_state.set__val(manipulator_interfaces::msg::Result::GOAL_IN_COLLISION);
177  RCLCPP_INFO_STREAM(
178  node_->get_logger(),
179  "Init contorl canceled : lidar");
180  } else {
181  result_state.set__val(manipulator_interfaces::msg::Result::FAILURE);
182  RCLCPP_INFO_STREAM(
183  node_->get_logger(),
184  "Init contorl canceled : fail.");
185  }
186 
187  action_result->set__result_state(result_state);
188 
189  if (actived_goal_) {
190  actived_goal_->abort(action_result);
191  }
192  actived_goal_.reset();
193 }

204 rclcpp_action::GoalResponse GcInitEndMotionControl::goal_callback(
205  const rclcpp_action::GoalUUID & uuid, std::shared_ptr<const GcInitEndMotionAction::Goal> goal)
206 {
207  (void)uuid;
208  RCLCPP_INFO_STREAM(node_->get_logger(), "Received new init or end action goal");
209  std::lock_guard<std::mutex> lock(mutex_);
210 
211  for (size_t i = 0; i < joint_motion_queue_.size(); i++) {
212  joint_motion_queue_.pop();
213  }
214  move_flag_ = false;
215  JointMotion m;
216  m.joint_values = robot_handler_->get_previous_goal_joint_value();
217  m.joint_values.stoped();
218  // Check stored state
219  bool stored_state = false;
220  if (!robot_handler_->get_robot_model().check_joint_limit(m.joint_values)) {
221  stored_state = true;
222  }
223 
224  // true -> task / false -> joint
225  bool task_ok_state = rm2_common::equal(
226  m.joint_values.position(1) + m.joint_values.position(2) + m.joint_values.position(3),
227  0.0,
228  0.01);
229 
230  if (goal->motion_name == goal->INIT_MOTION) {
231  // set init motion
232  RCLCPP_INFO_STREAM(node_->get_logger(), "SET INIT POSE");
233  was_init_motion_ = true;
234  // start time
235  m.start_time = 0.0;
236  // motion time
237  m.move_time = goal->time_from_start;
238  // joint angle
239  m.joint_values.position = init_position_;
240  m.joint_values.position = m.joint_values.position * DEG2RAD;
241  if (stored_state || (!task_ok_state)) {
242  m.unlimited = true; // joint
243  } else {
244  m.unlimited = false; // task
245  }
246  joint_motion_queue_.push(m);
247 
248  } else if (goal->motion_name == goal->END_MOTION) {
249  // set end motion
250  RCLCPP_INFO_STREAM(node_->get_logger(), "SET END POSE");
251  was_init_motion_ = false;
252  if (!stored_state) {
253  if (!rm2_common::equal(m.joint_values.position(0), 0.0)) {
254  // Time rate
255  // 전체 이동시간 중 init에서 end로의 이동이 차지하는 비율
256  double time_rate = (init_position_.at(1) - end_position_.at(1)) /
257  (robot_handler_->get_previous_goal_joint_value().position(1) - end_position_.at(1));
259  RCLCPP_INFO_STREAM(node_->get_logger(), "Pass through init position");
260  // start time
261  m.start_time = 0.0;
262  // motion time
263  m.move_time = goal->time_from_start * (1 - time_rate);
264  RCLCPP_INFO_STREAM(node_->get_logger(), "Init time : " << m.move_time);
265  // joint angle
266  m.joint_values.position = init_position_;
267  m.joint_values.position = m.joint_values.position * DEG2RAD;
268  if (!task_ok_state) {
269  m.unlimited = true; // joint
270  } else {
271  m.unlimited = false; // task
272  }
273  joint_motion_queue_.push(m);
275  // start time
276  m.start_time = 0.0;
277  // motion time
278  m.move_time = goal->time_from_start * time_rate;
279  RCLCPP_INFO_STREAM(node_->get_logger(), "End time : " << m.move_time);
280  // joint angle
281  m.joint_values.position = end_position_;
282  m.joint_values.position = m.joint_values.position * DEG2RAD;
283  m.unlimited = true;
284  joint_motion_queue_.push(m);
285  } else {
287  RCLCPP_INFO_STREAM(node_->get_logger(), "Doesn't pass through init position");
288  // start time
289  m.start_time = 0.0;
290  // motion time
291  m.move_time = goal->time_from_start;
292  RCLCPP_INFO_STREAM(node_->get_logger(), "End time : " << m.move_time);
293  // joint angle
294  m.joint_values.position = end_position_;
295  m.joint_values.position = m.joint_values.position * DEG2RAD;
296  m.unlimited = true;
297  joint_motion_queue_.push(m);
298  }
299  } else {
300  RCLCPP_INFO_STREAM(node_->get_logger(), "Already Stored. skip the End motion.");
301  // start time
302  m.start_time = 0.0;
303  // motion time
304  m.move_time = 1.0;
305  // joint angle
306  m.joint_values.position = end_position_;
307  m.joint_values.position = m.joint_values.position * DEG2RAD;
308  m.unlimited = true;
309  joint_motion_queue_.push(m);
310  }
311  } else {
312  // reject : motion name is not exist
313  RCLCPP_ERROR_STREAM(node_->get_logger(), "Motion name is not exist");
314  return rclcpp_action::GoalResponse::REJECT;
315  }
316 
317  return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
318 }

320 rclcpp_action::CancelResponse GcInitEndMotionControl::cancel_callback(
321  const std::shared_ptr<rclcpp_action::ServerGoalHandle<GcInitEndMotionAction>> goal_handle)
322 {
323  RCLCPP_INFO_STREAM(node_->get_logger(), "Got request to cancel goal");
324  std::lock_guard<std::mutex> lock(mutex_);
325  try {
326  if (actived_goal_ == goal_handle) {
327  stop_update();
328 
329  for (size_t i = 0; i < joint_motion_queue_.size(); i++) {
330  joint_motion_queue_.pop();
331  }
332  move_flag_ = false;
333  RCLCPP_INFO_STREAM(
334  node_->get_logger(),
335  "Canceling active action goal because cancel callback received.");
336 
337  // Mark the current goal as canceled
338  auto action_res = std::make_shared<GcInitEndMotionAction::Result>();
339  manipulator_interfaces::msg::Result result_state;
340  result_state.set__val(manipulator_interfaces::msg::Result::CANCELED);
341  action_res->set__result_state(result_state);
342  if (actived_goal_) {
343  actived_goal_->canceled(action_res);
344  }
345  } else {
346  // Reset current goal
347  actived_goal_.reset();
348  RCLCPP_ERROR_STREAM(
349  node_->get_logger(), "Fail to Canceling active action goal due to goal handle mismatch.");
350  return rclcpp_action::CancelResponse::REJECT;
351  }
352  } catch (std::runtime_error e) {
353  // Reset current goal
354  actived_goal_.reset();
355  RCLCPP_ERROR_STREAM(node_->get_logger(), "Runtime error : " << e.what());
356  return rclcpp_action::CancelResponse::REJECT;
357  }
358  // Reset current goal
359  actived_goal_.reset();
360  return rclcpp_action::CancelResponse::ACCEPT;
361 }

403 bool GcInitEndMotionControl::make_trajectory()
404 {
405  // make joint trajectory
406  if (joint_motion_queue_.size() <= 0) {
407  RCLCPP_ERROR_STREAM(node_->get_logger(), "Joint Motion Queue is empty");
408  return false;
409  }
410  RCLCPP_INFO_STREAM(node_->get_logger(), "Make Trajectoty.");
411 
412  auto j = joint_motion_queue_.front();
413  joint_motion_queue_.pop();
414 
415  // set start time
416  if (j.start_time == 0.0) {
417  j.start_time = node_->now().seconds();
418  }
419  robot_handler_->set_start_time(j.start_time);
420  unlimited_falg_ = j.unlimited;
421 
422  rm2_common::JointValues start_waypoint = robot_handler_->get_previous_goal_joint_value();
423  joint_trajectory_ptr_->set_move_time(j.move_time);
424  if (!joint_trajectory_ptr_->calculation(
425  start_waypoint.position, j.joint_values.position,
426  start_waypoint.velocity, j.joint_values.velocity))
427  {return false;}
428 
429  // set move flag
430  move_flag_ = true;
431  trajectory_time_ = j.move_time;
432 
433  return true;
434 }
