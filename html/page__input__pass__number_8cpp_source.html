117

126  next_page_name_ = list[1].toStdString();
127  }
128

130  option_ = "";
131  for (int i = 2; i < list.size(); i++) {

134  } else {

241 void PageInputPassNumber::num_clicked(const int num)
242 {
243  if (timer_->remainingTime() < 3 * 1000) {
244  timer_->start(3 * 1000);
245  }
246 
247  if (vt_pass.size() >= PASS_MAX_NUM) {
248  std::string user_pass = "";
249  for (auto d : vt_pass) {
250  user_pass += std::to_string(d);
251  }
252  return;
253  }
254  pass_labels1.at(vt_pass.size())->setStyleSheet(
255  "QLabel {"
256  " background: qlineargradient(x1: 0, y1: 0, x2: 0, y2: 1, stop: 0 #FFFFFF, stop: 1 #434343);"
257  " border: 0px solid #858585;"
258  " border-radius: 20px;"
259  "}");
260  pass_labels2.at(vt_pass.size())->setStyleSheet(
261  "QLabel {"
262  " background: qlineargradient(x1: 0, y1: 0, x2: 0, y2: 1, stop: 0 #0E0D0D, stop: 1 #434343);"
263  " border: 0px solid #858585;"
264  " border-radius: 18px;"
265  "}");
266  vt_pass.push_back(num);
267 
268  if (vt_pass.size() >= PASS_MAX_NUM) {
269  std::string user_pass = "";
270  for (auto d : vt_pass) {
271  user_pass += std::to_string(d);
272  }
273 
274  if (user_pass != ros_node_->config_password_) {
275  ui->lblText->setText(tr("잘못된 비밀번호에요."));
276  std::thread th_wrong_pass(wrong_pass_thread, this);
277  th_wrong_pass.detach();
278  set_pass_clear();
279  } else {
280  emit pageChanged(next_page_name_, back_page_name_);
281  }
282  }
283 }
