47 enum class TaskSequence
48 {
49  Init,
50  // 만약 로봇이 접혀있는 상태라면 Init Pose를 거치는 경로 생성
51  Prepare,
52  // expected_button_pose와 present_pose가 로봇의 경계면을 통과한다면 경계면에 중간 경유점을 생성
53  Tracking,
54  // button의 pose에 따라 tracking하며 이동하는 단계로
55  // 이때 마지막 목적지에서의 속도는 profile.velocity이다.
56  Push,
57  // push_velocity로 등속운동을 하는 구간이다. sensorxel값이 threshold를 넘으면 back으로 넘어간다.
58  BackAndSuccess,
59  // push를 멈추고 뒤로 빼는단계이다. 목적지는 push_start_pose_*back_rate_이다.
60  BackAndFail,
61  // push를 멈추고 뒤로 빼는단계이다. 목적지는 push_start_pose_*back_rate_이다.
62  End,
63  // 모든 작업이 종료된 단계
64  Canceled
65 };
66 // 로봇의 경계면은 xy평면내에서 base_link를 중심으로 robot_radius를 반지름으로하고,
67 // 높이는 robot_height의 원통의 표면을 의미한다.
68 // 이때 원통의 위에 manipulator_tip이 존재하면 경유점을 생성한다.
69 // Approach, Push 단계에서는 success, fail, cancel 모든 결과에서 여기로 넘어와서 꼭 후진을 한다.

73  SUCCESS,
