32 void CameraGeometry::set_cam_info(
33  const std::string & cam_name,
34  sensor_msgs::msg::CameraInfo & left_cam_info,
35  sensor_msgs::msg::CameraInfo & right_cam_info,
36  const std::string & frame_id)
37 {
38  std::string left_cam_name = "left_" + cam_name;
39  std::string right_cam_name = "right_" + cam_name;
40  std::string right_to_left_cam_name = "right_to_left_" + cam_name;
41 
42  image_size_[left_cam_name] = cv::Size(
43  left_cam_info.width,
44  left_cam_info.height);
45 
46  intrinsic_[left_cam_name] = Eigen::Matrix3f::Identity();
47  intrinsic_[left_cam_name](0, 0) = left_cam_info.p[0];
48  intrinsic_[left_cam_name](1, 1) = left_cam_info.p[5];
49  intrinsic_[left_cam_name](0, 2) = left_cam_info.p[2];
50  intrinsic_[left_cam_name](1, 2) = left_cam_info.p[6];
51 
52  intrinsic_[right_cam_name] = Eigen::Matrix3f::Identity();
53  intrinsic_[right_cam_name](0, 0) = right_cam_info.p[0];
54  intrinsic_[right_cam_name](1, 1) = right_cam_info.p[5];
55  intrinsic_[right_cam_name](0, 2) = right_cam_info.p[2];
56  intrinsic_[right_cam_name](1, 2) = right_cam_info.p[6];
57 
58  // Set Extrinsic Params
59  extrinsic_[right_to_left_cam_name] = Eigen::Matrix4f::Identity();
60  extrinsic_[right_to_left_cam_name](0, 3) =
61  right_cam_info.p[3] / left_cam_info.p[0] * -1000;
62 
63  // Build Projection Matrix
64  proj_mat_[left_cam_name] =
65  intrinsic_[left_cam_name] *
66  Eigen::Matrix<float, 3, 4>::Identity();
67  proj_mat_[right_cam_name] =
68  intrinsic_[right_cam_name] *
69  Eigen::Matrix<float, 3, 4>::Identity() *
70  extrinsic_[right_to_left_cam_name];
71  proj_mat_[right_to_left_cam_name] =
72  intrinsic_[right_cam_name] *
73  Eigen::Matrix<float, 3, 4>::Identity() *
74  extrinsic_[right_to_left_cam_name].inverse();
75 
76  frame_id_[left_cam_name] = frame_id;
77 }

138 Eigen::MatrixXf CameraGeometry::project_to_image(
139  Eigen::Matrix<float, 3, 4> & proj,
140  Eigen::MatrixXf & pt)
141 {
142  int n = pt.rows();
143  Eigen::MatrixXf pt_homo = Eigen::MatrixXf::Ones(n, 4);
144  pt_homo.block(0, 0, n, 3) = pt;
145 
146  Eigen::MatrixXf reproj_pts = proj * pt_homo.transpose();
147  Eigen::MatrixXf result = reproj_pts.topRows(2).array().rowwise() / reproj_pts.row(2).array();
148  return result;
149 }

160 Eigen::Vector3f CameraGeometry::reconstruct(
161  Eigen::Matrix<float, 3, 4> & lproj,
162  Eigen::Matrix<float, 3, 4> & rproj,
163  Eigen::Vector2f & lpt,
164  Eigen::Vector2f & rpt)
165 {
166  float lu = lpt(0);
167  float lv = lpt(1);
168  float ru = rpt(0);
169  float rv = rpt(1);
170 
171  Eigen::Vector3f result(0.0, 0.0, 0.0);
172  Eigen::Matrix<float, 4, 3> a_mat;
173  Eigen::Vector4f b_mat;
174 
175  a_mat <<
176  lu * lproj(2, 0) - lproj(0, 0), lu * lproj(2, 1) - lproj(0, 1), lu * lproj(2, 2) - lproj(0, 2),
177  lv * lproj(2, 0) - lproj(1, 0), lv * lproj(2, 1) - lproj(1, 1), lv * lproj(2, 2) - lproj(1, 2),
178  ru * rproj(2, 0) - rproj(0, 0), ru * rproj(2, 1) - rproj(0, 1), ru * rproj(2, 2) - rproj(0, 2),
179  rv * rproj(2, 0) - rproj(1, 0), rv * rproj(2, 1) - rproj(1, 1), rv * rproj(2, 2) - rproj(1, 2);
180 
181  b_mat <<
182  -lproj(0, 3) + lu * lproj(2, 3),
183  -lproj(1, 3) + lv * lproj(2, 3),
184  -rproj(0, 3) + ru * rproj(2, 3),
185  -rproj(1, 3) + rv * rproj(2, 3);
186 
187  return (a_mat.transpose() * a_mat).ldlt().solve(a_mat.transpose() * b_mat);
188 }

190 Eigen::Vector3f CameraGeometry::reconstruct(
191  float f, float base_line, float ppx, float ppy,
192  Eigen::Vector2f & lpt,
193  Eigen::Vector2f & rpt)
194 {
195  float lu = lpt(0);
196  float lv = lpt(1);
197  float ru = rpt(0);
198  float rv = rpt(1);
199 
200  float disparity = lu - ru;
201  if (abs(disparity) < 1) {
202  disparity = 1.0;
203  }
204 
205  float pz = (f * base_line) / disparity;
206  float py = (lv - ppy) * pz / f;
207  float px = (lu - ppx) * pz / f;
208 
209  return Eigen::Vector3f(px, py, pz);
210 }

212 void CameraGeometry::create_frustum(const std::string & cam_name)
213 {
214  Frustum frustum;
215 
216  Eigen::Matrix<float, 3, 4> proj_mat = get_cam_proj_mat(cam_name);
217  Eigen::Matrix3f rot = get_cam_ext_mat(cam_name).block<3, 3>(0, 0);
218  Eigen::Vector3f trans = get_cam_ext_mat(cam_name).block<3, 1>(0, 3);
219 
220  // 카메라 원점
221  Eigen::Vector3f cam_orin(0, 0, 0);
222  frustum.origin = cam_orin;
223 
224  // 최대 거리에 있는 가상면 생성
225  Eigen::Vector3f vp_nor(0, 0, 1);
226  Eigen::Vector3f vp_cen(0, 0, 5);
227  frustum.far_plane = std::make_pair(vp_cen, vp_nor);
228 
229  // 절두체 생성
230  Eigen::Matrix3f intrinsic = proj_mat.block<3, 3>(0, 0);
231  cv::Size img_size = get_cam_img_size(cam_name);
232  std::vector<Eigen::Vector3f> img_corners {
233  {0, 0, 1},
234  {img_size.width - 1, 0, 1},
235  {img_size.width - 1, img_size.height - 1, 1},
236  {0, img_size.height - 1, 1}};
237 
238  // 절두체 뒷면의 끝점 4점 계산
239  std::vector<Eigen::Vector3f> rear_corners;
240  for (auto corner : img_corners) {
241  Eigen::Vector3f frustum_line = intrinsic.inverse() * corner;
242  Eigen::Vector3f intersected_pt = vp_cen(2) * frustum_line;
243  rear_corners.push_back(intersected_pt);
244  frustum.edges.push_back(frustum_line);
245  }
246 
247  // 절두체 4면 정의 (위, 아래, 양 옆)
248  for (int curr_id = 0; curr_id < rear_corners.size(); curr_id++) {
249  int next_id = (curr_id + 1) % rear_corners.size();
250  // 각 면의 중심점 계산
251  Eigen::Vector3f p1 = rear_corners[curr_id];
252  Eigen::Vector3f p2 = rear_corners[next_id];
253  Eigen::Vector3f cen = (p1 + p2 + cam_orin) / 3.0;
254 
255  // 각 면의 법선 계산
256  Eigen::Vector3f vec1 = p1 - cam_orin;
257  Eigen::Vector3f vec2 = p2 - cam_orin;
258  Eigen::Vector3f nor = vec2.cross(vec1).normalized();
259 
260  frustum.other_planes.push_back(std::make_pair(cen, nor));
261  }
262 
263  frustum_[cam_name] = frustum;
264 }
