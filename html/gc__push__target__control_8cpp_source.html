92 void GcPushTargetControl::update()
93 {
94  get_sensor_data_wih_coeff();
95 
96  if (init_flag_ == true) {
97  if (buf_value_.size() < buf_cnt_) {
98  RCLCPP_ERROR_STREAM(node_->get_logger(), "sensorxel_force_ : " << sensorxel_force_);
99  if (sensorxel_force_ >= 0.0) {
100  RCLCPP_ERROR_STREAM(node_->get_logger(), "sensorxel_force_ added!!!");
101  buf_value_.push_back(sensorxel_force_);
102  }
103  } else {
104  sort(buf_value_.begin(), buf_value_.end());
105  init_thres_ = buf_value_.at(buf_cnt_ / 2);
106  RCLCPP_ERROR_STREAM(node_->get_logger(), "init threshold value : " << init_thres_);
107  init_flag_ = false;
108  robot_handler_->set_start_time(node_->now().seconds());
109  }
110  } else {
111  if (trajectory_state_ == APRO_TRAJ) {
112  bool goal_finish = false;
113  rm2_common::TaskWaypoint goal = task_trajectory_ptr_->get(
114  robot_handler_->get_tick_time(), goal_finish);
115 
116  auto limit_over_joints = robot_handler_->check_joint_limit(target_component_name_, goal);
117  if (!limit_over_joints.empty()) {
118  RCLCPP_WARN_STREAM(node_->get_logger(), "Limit over during approach motion.");
119  trajectory_state_ = BACK_TRAJ;
120  success_flag_ = false;
121  if (make_back_task_trajectory() == false) {
122  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to make BACK trajectory.");
123  send_action_fail();
124  stop_update();
125  }
126  } else {
127  robot_handler_->set_goal_task_value_unlimited(target_component_name_, goal);
128  }
129 
130  if (goal_finish) {
131  trajectory_state_ = PUSH_TRAJ;
132  publish_log(
133  "[Aproach Trajectory END] sensor value: " + std::to_string(sensorxel_force_) +
134  " init: " + std::to_string(init_thres_) +
135  " thres + init: " + std::to_string(push_force_threshold_ + init_thres_));
136  robot_handler_->set_start_time();
137  }
138 
139  // check push target condition
140  if (sensorxel_force_ > push_force_threshold_ + init_thres_) {
141  trajectory_state_ = BACK_TRAJ;
142  publish_log(
143  "[Over Thresshold during Aproach Motion] sensor value: " +
144  std::to_string(sensorxel_force_) +
145  " init: " + std::to_string(init_thres_) +
146  " thres + init: " + std::to_string(push_force_threshold_ + init_thres_));
147  pushed_waypoint_ = robot_handler_->get_present_task_value(target_component_name_);
148  if (make_back_task_trajectory() == false) {
149  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to make PUSH trajectory.");
150  send_action_fail();
151  stop_update();
152  }
153  }
154  } else if (trajectory_state_ == PUSH_TRAJ) {
155  auto tick_goal = robot_handler_->get_previous_goal_task_value(target_component_name_);
156  tick_goal.pose.position += push_direction_ * push_velocity_x_coeff_ * control_period_;
157  tick_goal.velocity.linear = push_direction_ * push_velocity_x_coeff_;
158 
159  auto limit_over_joints = robot_handler_->check_joint_limit(target_component_name_, tick_goal);
160  if (!limit_over_joints.empty()) {
161  RCLCPP_WARN_STREAM(node_->get_logger(), "Limit over during push motion.");
162  trajectory_state_ = BACK_TRAJ;
163  success_flag_ = false;
164  if (make_back_task_trajectory() == false) {
165  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to make BACK trajectory.");
166  send_action_fail();
167  stop_update();
168  }
169  } else {
170  robot_handler_->set_goal_task_value_unlimited(target_component_name_, tick_goal);
171  }
172 
173  if (robot_handler_->get_tick_time() > push_time_) {
174  trajectory_state_ = BACK_TRAJ;
175  success_flag_ = false;
176  publish_log(
177  "[Push Trajectory END] sensor value: " + std::to_string(sensorxel_force_) +
178  " init: " + std::to_string(init_thres_) +
179  " thres + init: " + std::to_string(push_force_threshold_ + init_thres_));
180  if (make_back_task_trajectory() == false) {
181  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to make BACK trajectory.");
182  send_action_fail();
183  stop_update();
184  }
185  }
186 
187  // check push target condition
188  if (sensorxel_force_ > push_force_threshold_ + init_thres_) {
189  trajectory_state_ = BACK_TRAJ;
190  publish_log(
191  "[Over Thresshold during Push Motion] sensor value: " + std::to_string(sensorxel_force_) +
192  " init: " + std::to_string(init_thres_) +
193  " thres + init: " + std::to_string(push_force_threshold_ + init_thres_));
194  pushed_waypoint_ = robot_handler_->get_present_task_value(target_component_name_);
195  RCLCPP_INFO_STREAM(
196  this->node_->get_logger(),
197  "Success to call service, Value : " << this->sensorxel_force_);
198  if (make_back_task_trajectory() == false) {
199  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to make trajectory.");
200  send_action_fail();
201  stop_update();
202  }
203  }
204  } else if (trajectory_state_ == BACK_TRAJ) {
205  bool trajectory_finish = false;
206  auto goal = task_trajectory_ptr_->get(robot_handler_->get_tick_time(), trajectory_finish);
207  if (trajectory_finish) {
208  if (success_flag_) {
209  send_action_success();
210  stop_update();
211  } else {
212  send_action_fail();
213  stop_update();
214  }
215  }
216  robot_handler_->set_goal_task_value(target_component_name_, goal);
217  }
218  }
219  send_feedback();
220 }

222 void GcPushTargetControl::update_cancel(CancelResult result)
223 {
224  GcPushTargetAction::Result::SharedPtr action_result =
225  std::make_shared<GcPushTargetAction::Result>();
226  manipulator_interfaces::msg::Result result_state;
227 
228  if (result == CancelResult::JOINT_LIMIT) {
229  result_state.set__val(manipulator_interfaces::msg::Result::COLLISION_CHECKING_UNAVAILABLE);
230  RCLCPP_INFO_STREAM(
231  node_->get_logger(),
232  "Push contorl canceled : joint limit over.");
233  } else if (result == CancelResult::TOO_FAST_MOVE) {
234  result_state.set__val(manipulator_interfaces::msg::Result::CONTROL_FAILED);
235  RCLCPP_INFO_STREAM(
236  node_->get_logger(),
237  "Push contorl canceled : too fast move failure");
238  } else if (result == CancelResult::INVALID_GOAL_VALUES) {
239  result_state.set__val(manipulator_interfaces::msg::Result::CONTROL_FAILED);
240  RCLCPP_INFO_STREAM(
241  node_->get_logger(),
242  "Push contorl canceled : invalid goal values failure");
243  } else if (result == CancelResult::IK_FAIL) {
244  result_state.set__val(manipulator_interfaces::msg::Result::NO_IK_SOLUTION);
245  RCLCPP_INFO_STREAM(
246  node_->get_logger(),
247  "Push contorl canceled : ik solving failure");
248  } else if (result == CancelResult::PREEMPTED) {
249  result_state.set__val(manipulator_interfaces::msg::Result::PREEMPTED);
250  RCLCPP_INFO_STREAM(
251  node_->get_logger(),
252  "Push contorl canceled : preempted by the other controller");
253  } else if (result == CancelResult::CHECKER) {
254  result_state.set__val(manipulator_interfaces::msg::Result::GOAL_IN_COLLISION);
255  RCLCPP_INFO_STREAM(
256  node_->get_logger(),
257  "Push contorl canceled : lidar");
258  } else {
259  result_state.set__val(manipulator_interfaces::msg::Result::FAILURE);
260  RCLCPP_INFO_STREAM(
261  node_->get_logger(),
262  "Push contorl canceled : fail.");
263  }
264  action_result->set__result_state(result_state);
265 
266  if (actived_goal_) {
267  actived_goal_->abort(action_result);
268  }
269  actived_goal_.reset();
270 }

285 rm2_common::TaskValues GcPushTargetControl::set_target_pose()
286 {
287  rm2_common::TaskValues present_value;
288  present_value = robot_handler_->get_previous_goal_task_value(target_component_name_);
289 
290  rm2_common::Vector push_vector(push_velocity_x_coeff_, 0.0, 0.0);
291  push_vector = present_value.pose.orientation * push_vector;

299 bool GcPushTargetControl::make_back_task_trajectory()
300 {
301  // set start waypoint
302  rm2_common::TaskValues start =
303  robot_handler_->get_previous_goal_task_value(target_component_name_);
304 
305  // set goal waypoint
306  rm2_common::TaskValues goal;
307  rm2_common::Vector back_vector(-push_goal_distance_ * back_rate_, 0.0, back_z_traj_);
308  rm2_common::TaskValues s = robot_handler_->get_present_task_value(target_component_name_);
309  back_vector = s.pose.orientation * back_vector;
310  goal.pose.position = s.pose.position + back_vector;
311  goal.pose.orientation = s.pose.orientation;
312 
313  // set start & move time
314  robot_handler_->set_start_time(node_->now().seconds());
315  task_trajectory_ptr_->set_move_time(back_time_);
316 
317  // make trajectory
318  if (!task_trajectory_ptr_->calculation(start.pose, goal.pose, start.velocity, goal.velocity)) {
319  RCLCPP_ERROR_STREAM(this->node_->get_logger(), "Fail to calculate back trajectory.");
320  return false;
321  }
322  return true;
323 }

325 rclcpp_action::GoalResponse GcPushTargetControl::goal_callback(
326  const rclcpp_action::GoalUUID & uuid,
327  std::shared_ptr<const GcPushTargetAction::Goal> goal)
328 {
329  (void) uuid;
330  trajectory_state_ = APRO_TRAJ;
331  sensorxel_force_ = 0.0;
332  init_thres_ = 0.0;
333 
334  RCLCPP_INFO_STREAM(node_->get_logger(), "Received new action goal");
335  std::lock_guard<std::mutex> lock(mutex_);
336 
337  if (!robot_handler_->check_component_name(goal->component_name)) {
338  RCLCPP_ERROR_STREAM(node_->get_logger(), "NO componet name.");
339  return rclcpp_action::GoalResponse::REJECT;
340  }
341 
342  init_flag_ = true;
343  buf_value_.clear();
344  target_component_name_ = goal->component_name;
345  push_force_threshold_ = goal->force_threshold; // + init_thres_;
346  push_velocity_x_coeff_ = goal->velocity_x_coeff;
347  push_goal_distance_ = goal->goal_distance;
348 
349  lpf_sensorxel_.set_tau(0.1);
350  lpf_sensorxel_.set_first_input_flag(true);
351 
352  auto start_waypoint = robot_handler_->get_previous_goal_task_value(target_component_name_);
353 
354  push_direction_ = start_waypoint.pose.orientation * rm2_common::Vector(1.0, 0.0, 0.0);
355  rm2_common::Vector push_vector = push_goal_distance_ * push_rate_ * push_direction_;
356 
357  rm2_common::TaskValues goal_value = start_waypoint;
358  goal_value.pose.position += push_vector;
359 
360  rm2_common::Twist target_vel = rm2_common::Twist(
361  push_velocity_x_coeff_ * push_direction_,
362  rm2_common::Vector::Zero());
363  task_trajectory_ptr_->set_move_time(apro_time_);
364  task_trajectory_ptr_->calculation(
365  start_waypoint.pose, goal_value.pose,
366  rm2_common::Twist::Zero(), target_vel);
367 
368  success_flag_ = true;
369 
370  // set start
371  robot_handler_->set_start_time(node_->now().seconds());
372  return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
373 }

375 rclcpp_action::CancelResponse GcPushTargetControl::cancel_callback(
376  const std::shared_ptr<rclcpp_action::ServerGoalHandle<GcPushTargetAction>> goal_handle)
377 {
378  RCLCPP_INFO_STREAM(node_->get_logger(), "Got request to cancel goal");
379  std::lock_guard<std::mutex> lock(mutex_);
380 
381  try {
382  if (actived_goal_ == goal_handle) {
383  stop_update();
384 
385  RCLCPP_INFO_STREAM(
386  node_->get_logger(),
387  "Canceling active action goal because cancel callback received.");
388 
389  // Mark the current goal as canceled
390  auto action_res = std::make_shared<GcPushTargetAction::Result>();
391  manipulator_interfaces::msg::Result result_state;
392  result_state.set__val(manipulator_interfaces::msg::Result::CANCELED);
393  action_res->set__result_state(result_state);
394  if (actived_goal_) {
395  actived_goal_->canceled(action_res);
396  }
397  } else {
398  // Reset current goal
399  actived_goal_.reset();
400  RCLCPP_ERROR_STREAM(
401  node_->get_logger(), "Fail to Canceling active action goal due to goal handle mismatch.");
402  return rclcpp_action::CancelResponse::REJECT;
403  }
404  // Reset current goal
405  actived_goal_.reset();
406  } catch (std::runtime_error e) {
407  RCLCPP_ERROR_STREAM(node_->get_logger(), "Runtime error : " << e.what());
408  return rclcpp_action::CancelResponse::REJECT;
409  }
410  // Reset current goal
411  actived_goal_.reset();
412  return rclcpp_action::CancelResponse::ACCEPT;
413 }

435 void GcPushTargetControl::send_action_success()
436 {
437  // clear
438  sensorxel_force_ = 0.0;
439  trajectory_state_ = PUSH_TRAJ;
440 
441  auto result = std::make_shared<GcPushTargetAction::Result>();
442 
443  manipulator_interfaces::msg::Result result_state;
444  result_state.set__val(manipulator_interfaces::msg::Result::SUCCESS);
445  result->set__result_state(result_state);
446 
447  auto final_waypoint = robot_handler_->get_present_task_value(target_component_name_);
448 
449  rm2_common::Pose rm2_delta = pushed_waypoint_.pose.inverse() * final_waypoint.pose;
450 
451  geometry_msgs::msg::Pose delta;
452  delta.position.x = rm2_delta.position.x();
453  delta.position.y = rm2_delta.position.y();
454  delta.position.z = rm2_delta.position.z();
455  delta.orientation.x = rm2_delta.orientation.x();
456  delta.orientation.y = rm2_delta.orientation.y();
457  delta.orientation.z = rm2_delta.orientation.z();
458  delta.orientation.w = rm2_delta.orientation.w();
459  result->set__delta_pose(delta);
460 
461  RCLCPP_INFO_STREAM(
462  node_->get_logger(),
463  "Delta :\n(" << rm2_delta.position.data.transpose() << "),\n(" <<
464  rm2_delta.orientation.data << ")");
465 
466  if (actived_goal_) {
467  actived_goal_->succeed(result);
468  }
469  actived_goal_.reset();
470 }

492 void GcPushTargetControl::get_sensor_data_wih_coeff()
493 {
494  double value = robot_handler_->get_sensor_value(SENSOR_NAME);
495  // RCLCPP_INFO_STREAM(node_->get_logger(), SENSOR_NAME << " : " << value);
496 
497  const double Vin = 3.3; // 3.3v input
498  const double RM = 10000.0; // 10K ohm
499 
500  // RFSR = (Vin / Vout - 1) * RM
501  // Force = 1.0 / RFSR * 1000 --> g
502  double Vout = value / 4096.0 * Vin;
503  double RFSR = (Vin / Vout - 1) * RM;
504  double raw_data = 1.0 / RFSR * 1000.0;
505 
506  // this->sensorxel_force_ = lpf_sensorxel_.get_filtering_data(raw_data);
507  sensorxel_force_ = raw_data;
508 
509  RCLCPP_INFO_STREAM(
510  this->node_->get_logger(),
511  "Success to call service, Value : " << this->sensorxel_force_ << " init : " <<
512  this->init_thres_);
513 }
