73 void MinimumJerkControl::update_cancel(rm2_control_module::ControlModule::CancelResult result)
74 {
75  JointGoalAction::Result::SharedPtr action_result = std::make_shared<JointGoalAction::Result>();
76  manipulator_interfaces::msg::Result result_state;
77 
78  if (result == CancelResult::JOINT_LIMIT) {
79  result_state.set__val(manipulator_interfaces::msg::Result::COLLISION_CHECKING_UNAVAILABLE);
80  RCLCPP_INFO_STREAM(
81  node_->get_logger(),
82  "Joint minimum jerk contorl canceled : joint limit over.");
83  } else if (result == CancelResult::IK_FAIL) {
84  result_state.set__val(manipulator_interfaces::msg::Result::NO_IK_SOLUTION);
85  RCLCPP_INFO_STREAM(
86  node_->get_logger(),
87  "Joint minimum jerk contorl canceled : ik solving failure");
88  } else if (result == CancelResult::PREEMPTED) {
89  result_state.set__val(manipulator_interfaces::msg::Result::PREEMPTED);
90  RCLCPP_INFO_STREAM(
91  node_->get_logger(),
92  "Joint minimum jerk contorl canceled : preempted by the other controller");
93  } else if (result == CancelResult::CHECKER) {
94  result_state.set__val(manipulator_interfaces::msg::Result::GOAL_IN_COLLISION);
95  RCLCPP_INFO_STREAM(
96  node_->get_logger(),
97  "Joint minimum jerk contorl canceled : lidar");
98  } else {
99  result_state.set__val(manipulator_interfaces::msg::Result::FAILURE);
100  RCLCPP_INFO_STREAM(
101  node_->get_logger(),
102  "Joint minimum jerk contorl canceled : fail.");
103  }
104 
105  action_result->set__result_state(result_state);
106 
107  if (actived_goal_) {
108  actived_goal_->abort(action_result);
109  }
110  actived_goal_.reset();
111 }

120 rclcpp_action::GoalResponse MinimumJerkControl::goal_callback(
121  const rclcpp_action::GoalUUID & uuid, std::shared_ptr<const JointGoalAction::Goal> goal)
122 {
123  RCLCPP_INFO_STREAM(node_->get_logger(), "Received new action goal");
124  std::lock_guard<std::mutex> lock(mutex_);
126  /* Validate Goal Time */
128  rclcpp::Time start_time = static_cast<rclcpp::Time>(goal->header.stamp);
129  // If the starting time it set to 0.0, it means the controller should start it now.
130  // Otherwise we check if the trajectory ends before the current time,
131  // in which case it can be ignored.
132  if (start_time.seconds() != 0.0) {
133  rclcpp::Time end_time = start_time +
134  rclcpp::Duration::from_seconds(goal->goal_waypoint.time_from_start);
135  if (end_time < node_->now()) {
136  RCLCPP_ERROR_STREAM(
137  node_->get_logger(),
138  "Received trajectory with non zero time start time (" << start_time.seconds() <<
139  ") that ends on the past (" << end_time.seconds() << ")");
140  return rclcpp_action::GoalResponse::REJECT;
141  }
142  } else {
143  start_time = node_->now();
144  }
145 
147  /* Validate joint size and names */
149  if (goal->joint_names.empty()) {
150  RCLCPP_ERROR_STREAM(node_->get_logger(), "Empty joint names on incoming trajectory.");
151  return rclcpp_action::GoalResponse::REJECT;
152  }
153 
155  /* Validate joint waypoint size with names */
157  // Positions
158  if (goal->goal_waypoint.positions.size() != goal->joint_names.size()) {
159  RCLCPP_ERROR_STREAM(
160  node_->get_logger(),
161  "Mismatch between joint_names (" << goal->joint_names.size() <<
162  ") and positions (" << goal->goal_waypoint.positions.size() << ") at point.");
163  return rclcpp_action::GoalResponse::REJECT;
164  }
165  if (goal->goal_waypoint.velocities.size() != goal->joint_names.size()) {
166  RCLCPP_ERROR_STREAM(
167  node_->get_logger(),
168  "Mismatch between joint_names (" << goal->joint_names.size() <<
169  ") and velocities (" << goal->goal_waypoint.velocities.size() << ") at point.");
170  return rclcpp_action::GoalResponse::REJECT;
171  }
172  if (goal->goal_waypoint.accelerations.size() != goal->joint_names.size()) {
173  RCLCPP_ERROR_STREAM(
174  node_->get_logger(),
175  "Mismatch between joint_names (" << goal->joint_names.size() <<
176  ") and accelerations (" << goal->goal_waypoint.accelerations.size() << ") at point.");
177  return rclcpp_action::GoalResponse::REJECT;
178  }
179  if (goal->goal_waypoint.effort.size() != goal->joint_names.size()) {
180  RCLCPP_ERROR_STREAM(
181  node_->get_logger(),
182  "Mismatch between joint_names (" << goal->joint_names.size() <<
183  ") and effort (" << goal->goal_waypoint.effort.size() << ") at point.");
184  return rclcpp_action::GoalResponse::REJECT;
185  }
186 
187  // start waypoint
188  rm2_common::JointValues start_waypoint = robot_handler_->get_previous_goal_joint_value();
189  rm2_common::JointValues goal_waypoint = start_waypoint;
190 
191  if (robot_handler_->get_joint_names() == goal->joint_names) { // check
192  goal_waypoint.position = goal->goal_waypoint.positions;
193  goal_waypoint.velocity = goal->goal_waypoint.velocities;
194  goal_waypoint.effort = goal->goal_waypoint.effort;
195  } else {
196  std::vector<std::string> local_joint_names = robot_handler_->get_joint_names();
197  std::vector<std::string> msg_joint_names = goal->joint_names;
198  for (auto i = 0ul; i < msg_joint_names.size(); ++i) {
199  if (std::find(
200  local_joint_names.begin(), local_joint_names.end(),
201  msg_joint_names[i]) == local_joint_names.end() )
202  {
203  RCLCPP_ERROR_STREAM(
204  node_->get_logger(),
205  "joint '" << msg_joint_names[i].c_str() << "' doesn't match the controller's joints.");
206  return rclcpp_action::GoalResponse::REJECT;
207  }
208  for (auto j = 0ul; j < local_joint_names.size(); ++j) {
209  if (msg_joint_names[i] == local_joint_names[j]) {
210  goal_waypoint.position[j] = goal->goal_waypoint.positions[i];
211  break;
212  }
213  }
214  }
215  }
216 
217  // set start time
218  robot_handler_->set_start_time(start_time.seconds());
219  if (!joint_trajectory_ptr_->set_move_time(goal->goal_waypoint.time_from_start)) {
220  RCLCPP_ERROR_STREAM(
221  node_->get_logger(), "Move time error : " << goal->goal_waypoint.time_from_start);
222  return rclcpp_action::GoalResponse::REJECT;
223  }
224 
225  if (!joint_trajectory_ptr_->calculation(
226  start_waypoint.position, goal_waypoint.position,
227  start_waypoint.velocity, goal_waypoint.velocity))
228  {
229  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to calculate joint trajectory");
230  return rclcpp_action::GoalResponse::REJECT;
231  }
232 
233  goal_for_feedback_ = trans_handler_to_msg(goal_waypoint, goal->goal_waypoint.time_from_start);
234 // start_update();
235 // RCLCPP_INFO_STREAM(node_->get_logger(), "Start Move");
236  return rclcpp_action::GoalResponse::ACCEPT_AND_EXECUTE;
237 }

239 rclcpp_action::CancelResponse MinimumJerkControl::cancel_callback(
240  const std::shared_ptr<rclcpp_action::ServerGoalHandle<MinimumJerkControl::JointGoalAction>>
241  goal_handle)
242 {
243  RCLCPP_INFO_STREAM(node_->get_logger(), "Got request to cancel goal");
244  std::lock_guard<std::mutex> lock(mutex_);
245  try {
246  if (actived_goal_ == goal_handle) {
247  // Controller uptime
248  // Enter hold current position mode
249  stop_update();
250 
251  RCLCPP_DEBUG_STREAM(
252  node_->get_logger(), "Canceling active action goal because cancel callback received");
253 
254  // Mark the current goal as canceled
255  auto action_res = std::make_shared<JointGoalAction::Result>();
256  manipulator_interfaces::msg::Result result_state;
257  result_state.set__val(manipulator_interfaces::msg::Result::CANCELED);
258  action_res->set__result_state(result_state);
259  if (actived_goal_) {
260  actived_goal_->canceled(action_res);
261  }
262  } else {
263  // Reset current goal
264  actived_goal_.reset();
265  RCLCPP_ERROR_STREAM(
266  node_->get_logger(), "Fail to Canceling active action goal due to goal handle mismatch.");
267  return rclcpp_action::CancelResponse::REJECT;
268  }
269  } catch (std::runtime_error e) {
270  // Reset current goal
271  actived_goal_.reset();
272  RCLCPP_ERROR_STREAM(node_->get_logger(), "Runtime error : " << e.what());
273  return rclcpp_action::CancelResponse::REJECT;
274  }
275  // Reset current goal
276  actived_goal_.reset();
277  return rclcpp_action::CancelResponse::ACCEPT;
278 }

292  feedback->header.stamp = node_->now();
293 
294  auto desired_value = robot_handler_->get_previous_goal_joint_value();
295  feedback->desired_waypoint =
296  trans_handler_to_msg(desired_value, robot_handler_->get_tick_time());
297 
298  auto present_value = robot_handler_->get_present_joint_value();
299  feedback->present_waypoint =
300  trans_handler_to_msg(present_value, robot_handler_->get_tick_time());
301 
302  if (actived_goal_) {
303  actived_goal_->publish_feedback(feedback);
304  }

320 manipulator_interfaces::msg::JointWaypoint MinimumJerkControl::trans_handler_to_msg(
321  rm2_common::JointWaypoint waypoint, double time_from_start)
322 {
