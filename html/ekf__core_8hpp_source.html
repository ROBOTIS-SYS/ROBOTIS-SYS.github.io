25 class EKFEstimator
26 {
27 public:
28  EKFEstimator();
29 /* state
30 * x = [p v q] = [x y z vx vy vz qx qy qz qw]
31 * dx = [dp dv dth] = [dx dy dz dvx dvy dvz dthx dthy dthz]
32 *
33 * pos_k = pos_{k-1} + vel_k * dt + (1/2) * (Rot(q_{k-1}) acc_{k-1}^{imu} - g) *dt^2
34 * vel_k = vel_{k-1} + (Rot(quat_{k-1})) acc_{k-1}^{imu} - g) *dt
35 * quat_k = Rot(w_{k-1}^{imu}*dt)*quat_{k-1}
36 *
37 * covariance
38 * P_{k} = F_k P_{k-1} F_k^T + L Q_k L^T
39 */
40  void predictionUpdate(
41  const double current_time_imu,
42  const Eigen::Vector3d & gyro,
43  const Eigen::Vector3d & linear_acceleration);
44 
45 /*
46 * y = pobs = [xobs yobs zobs]
47 *
48 * K = P_k H^T (H P_k H^T + R)^{-1}
49 *
50 * dx = K (y_k - p_k )
51 *
52 * p_x = p_{k-1} + dp_k
53 * v_k = v_{k-1} + dv_k
54 * q_k = Rot(dth) q_{k-1}
55 *
56 * P_k = (I - KH)*P_{k-1}
57 */
58 
59  void OdomPredictionUpdate(
60  const Eigen::Vector3d & x,
61  const Eigen::Quaterniond & quat_wdt,

72
