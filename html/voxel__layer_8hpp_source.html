103  inline bool worldToMap3DFloat(
104  double wx, double wy, double wz, double & mx, double & my,
105  double & mz)
106  {
107  if (wx < origin_x_ || wy < origin_y_ || wz < origin_z_) {
108  return false;
109  }
110  mx = ((wx - origin_x_) / resolution_);
111  my = ((wy - origin_y_) / resolution_);
112  mz = ((wz - origin_z_) / z_resolution_);
113  if (mx < size_x_ && my < size_y_ && mz < size_z_) {
114  return true;
115  }
116 
117  return false;
118  }
119 
120  inline bool worldToMap3D(
121  double wx, double wy, double wz, unsigned int & mx, unsigned int & my,
122  unsigned int & mz)
123  {
124  if (wx < origin_x_ || wy < origin_y_ || wz < origin_z_) {
125  return false;
126  }
127 
128  mx = static_cast<unsigned int>((wx - origin_x_) / resolution_);
129  my = static_cast<unsigned int>((wy - origin_y_) / resolution_);
130  mz = static_cast<unsigned int>((wz - origin_z_) / z_resolution_);
131 
132  if (mx < size_x_ && my < size_y_ && mz < (unsigned int)size_z_) {
133  return true;
134  }
135 
136  return false;
137  }
138 
139  inline void mapToWorld3D(
140  unsigned int mx, unsigned int my, unsigned int mz, double & wx,
141  double & wy,
142  double & wz)
143  {
144  // returns the center point of the cell
145  wx = origin_x_ + (mx + 0.5) * resolution_;
146  wy = origin_y_ + (my + 0.5) * resolution_;
147  wz = origin_z_ + (mz + 0.5) * z_resolution_;
148  }
149 
150  inline double dist(double x0, double y0, double z0, double x1, double y1, double z1)
151  {
152  return sqrt((x1 - x0) * (x1 - x0) + (y1 - y0) * (y1 - y0) + (z1 - z0) * (z1 - z0));
153  }
154 
155  double getSizeInMetersZ() const
156  {
157  return (size_z_ - 1 + 0.5) * z_resolution_;
158  }
