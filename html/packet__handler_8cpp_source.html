193 int PacketHandler::tx_packet(PortHandler * port, uint8_t * txpacket)
194 {
195  uint16_t total_packet_length = 0;
196  int written_packet_length = 0;
197 
198  if (port->is_using_) {
199  return COMM_PORT_BUSY;
200  }
201  port->is_using_ = true;
202 
203  // byte stuffing for header
204  // addStuffing(txpacket);
205 
206  // check max packet length
207  total_packet_length = DXL_MAKEDWORD(
208  DXL_MAKEWORD(txpacket[PKT_LENGTH_4], txpacket[PKT_LENGTH_3]),
209  DXL_MAKEWORD(txpacket[PKT_LENGTH_2], txpacket[PKT_LENGTH_1])) + 9;
210  // 5: STX TYPE CRC_H CRC_L ETX + 4
211  if (total_packet_length > TXPACKET_MAX_LEN) {
212  port->is_using_ = false;
213  return COMM_TX_ERROR;
214  }
215  // make packet header
216  /*
217  txpacket[PKT_HEADER0] = 0xFF;
218  txpacket[PKT_HEADER1] = 0xFF;
219  txpacket[PKT_HEADER2] = 0xFD;
220  txpacket[PKT_RESERVED] = 0x00;
221  */
222 
223  // add CRC16
224  uint16_t crc = update_crc(0xFFFF, txpacket + 1, total_packet_length - 4);
225  // 4: STX + CRC16_L + CRC16_H + ETX
226  txpacket[total_packet_length - 3] = DXL_HIBYTE(crc);
227  txpacket[total_packet_length - 2] = DXL_LOBYTE(crc);
228 
229 // for (int i = 0; i < 13; i++) {
230 // printf("%#X\n", txpacket[i]);
231 // }
232 
233  // tx packet
234 
235  port->clear_port();
236  written_packet_length = port->write_port(txpacket, total_packet_length);
237  if (total_packet_length != written_packet_length) {
238  std::cout << YELLOW << "total_packet_length : " << total_packet_length << C_END << std::endl;
239  std::cout << GREEN << "written_packet_length : " <<
240  std::to_string(written_packet_length) << C_END << std::endl;
241  port->is_using_ = false;
242  return COMM_TX_FAIL;
243  }
244  return COMM_SUCCESS;
245 }

247 int PacketHandler::rx_packet(PortHandler * port, uint8_t * rxpacket)
248 {
249  int result = COMM_TX_FAIL;
250 
251  int16_t rx_length = 0;
252  uint16_t wait_length = 11;
253  // minimum length (// STX, TYPE, LENGTH_1 ~ 4, RESERIVED_1 ~ 2, CRC16_H, CRC16_L, ETX)
254 
255  while (true) {
256  int result_recv = port->read_port(&rxpacket[rx_length], wait_length - rx_length);
257  rx_length += (result_recv < 0) ? 0 : result_recv;
258  // std::cout << "rxlen : " << rx_length << std::endl;
259 
260  if (rx_length >= wait_length) {
261  uint16_t idx = 0;
262 
263  // find packet header
264  for (idx = 0; idx < (rx_length - 3); idx++) {
265  if ((rxpacket[idx] == 0xF0) &&
266  (rxpacket[idx + 1] == 0x45 || rxpacket[idx + 1] == 0x65 || rxpacket[idx + 1] == 0x61) )
267  {
268  // std::cout << "idx : " << idx << std::endl;
269  break;
270  }
271  }
272 
273  if (idx == 0) { // found at the beginning of the packet
274 /*
275  if (rxpacket[PKT_RESERVED] != 0x00 ||
276  rxpacket[PKT_ID] > 0xFC ||
277  DXL_MAKEWORD(rxpacket[PKT_LENGTH_L], rxpacket[PKT_LENGTH_H]) > RXPACKET_MAX_LEN ||
278  rxpacket[PKT_INSTRUCTION] != 0x55)
279  {
280  // remove the first byte in the packet
281  for (uint16_t s = 0; s < rx_length - 1; s++) {
282  rxpacket[s] = rxpacket[1 + s];
283  }
284  //memcpy(&rxpacket[0], &rxpacket[idx], rx_length - idx);
285  rx_length -= 1;
286  continue;
287  }
288 */
289 
290  // re-calculate the exact length of the rx packet
291  if (wait_length != DXL_MAKEDWORD(
292  DXL_MAKEWORD(rxpacket[PKT_LENGTH_4], rxpacket[PKT_LENGTH_3]),
293  DXL_MAKEWORD(rxpacket[PKT_LENGTH_2], rxpacket[PKT_LENGTH_1])) + 9)
294  {
295  wait_length = DXL_MAKEDWORD(
296  DXL_MAKEWORD(rxpacket[PKT_LENGTH_4], rxpacket[PKT_LENGTH_3]),
297  DXL_MAKEWORD(rxpacket[PKT_LENGTH_2], rxpacket[PKT_LENGTH_1])) + 9;
298  continue;
299  }
300 
301  if (rx_length < wait_length) {
302  // check timeout
303  if (port->is_packet_timeout() == true) {
304  if (rx_length == 0) {
305  result = COMM_RX_TIMEOUT;
306  } else {
307  result = COMM_RX_CORRUPT;
308  }
309  break;
310  } else {
311  continue;
312  }
313  }
314 
315  // verify CRC16
316  uint16_t crc = DXL_MAKEWORD(rxpacket[wait_length - 2], rxpacket[wait_length - 3]);
317  if (update_crc(0xFFFF, rxpacket + 1, wait_length - 4) == crc) {
318  result = COMM_SUCCESS;
319  } else {
320  result = COMM_RX_CORRUPT;
321  }
322  break;
323  } else {
324  // remove unnecessary packets
325  std::cout << "remove unnecessary packets.." << std::endl;
326  for (uint16_t s = 0; s < rx_length - idx; s++) {
327  rxpacket[s] = rxpacket[idx + s];
328  }
329  // memcpy(&rxpacket[0], &rxpacket[idx], rx_length - idx);
330  rx_length -= idx;
331  }
332  } else {
333  // check timeout
334  if (port->is_packet_timeout() == true) {
335  if (rx_length == 0) {
336  result = COMM_RX_TIMEOUT;
337  } else {
338  result = COMM_RX_CORRUPT;
339  }
340  break;
341  }
342  }
343  usleep(0);
344  }
345 
346  port->is_using_ = false;
347 /*
348  if (result == COMM_SUCCESS) {
349  removeStuffing(rxpacket);
350  }
351 */
352 
353  return result;
354 }
