38 inline double euclidean_distance(
39  const geometry_msgs::msg::Point & pos1,
40  const geometry_msgs::msg::Point & pos2)
41 {
42  double dx = pos1.x - pos2.x;
43  double dy = pos1.y - pos2.y;
44  double dz = pos1.z - pos2.z;
45  return std::sqrt(dx * dx + dy * dy + dz * dz);
