50 // Determine the probability for the given pose
51 double
52 LikelihoodFieldModelProb::sensorFunction(LaserData * data, pf_sample_set_t * set)
53 {
54  LikelihoodFieldModelProb * self;
55  int i, j, step;
56  double z, pz;
57  double log_p;
58  double obs_range, obs_bearing;
59  double total_weight;
60  pf_sample_t * sample;
61  pf_vector_t pose;
62  pf_vector_t hit;
63 
64  self = reinterpret_cast<LikelihoodFieldModelProb *>(data->laser);
65 
66  total_weight = 0.0;
67 
68 
69  step = ceil((data->range_count) / static_cast<double>(self->max_beams_));
70 
71  // Step size must be at least 1
72  if (step < 1) {
73  step = 1;
74  }
75 
76  // Pre-compute a couple of things
77  double z_hit_denom = 2 * self->sigma_hit_ * self->sigma_hit_;
78  double z_rand_mult = 1.0 / data->range_max;
79 
80  double max_dist_prob = exp(-(self->map_->max_occ_dist * self->map_->max_occ_dist) / z_hit_denom);
81 
82  // Beam skipping - ignores beams for which a majoirty of particles do not agree with the map
83  // prevents correct particles from getting down weighted because of unexpected obstacles
84  // such as humans
85 
86  bool do_beamskip = self->do_beamskip_;
87  double beam_skip_distance = self->beam_skip_distance_;
88  double beam_skip_threshold = self->beam_skip_threshold_;
89 
90  // we only do beam skipping if the filter has converged
91  if (do_beamskip && !set->converged) {
92  do_beamskip = false;
93  }
94 
95  // we need a count the no of particles for which the beam agreed with the map
96  int * obs_count = new int[self->max_beams_]();
97 
98  // we also need a mask of which observations to integrate (to decide which beams to integrate to
99  // all particles)
100  bool * obs_mask = new bool[self->max_beams_]();
101 
102  int beam_ind = 0;
103 
104  // realloc indicates if we need to reallocate the temp data structure needed to do beamskipping
105  bool realloc = false;
106 
107  // reset the matching scores for logging
108  self->match_scores_.clear();
109 
110  if (do_beamskip) {
111  if (self->max_obs_ < self->max_beams_) {
112  realloc = true;
113  }
114 
115  if (self->max_samples_ < set->sample_count) {
116  realloc = true;
117  }
118 
119  if (realloc) {
120  self->reallocTempData(set->sample_count, self->max_beams_);
121  fprintf(stderr, "Reallocing temp weights %d - %d\n", self->max_samples_, self->max_obs_);
122  }
123  }
124 
125  // Compute the sample weights
126  for (j = 0; j < set->sample_count; j++) {
127  sample = set->samples + j;
128  pose = sample->pose;
129 
130  // Take account of the laser pose relative to the robot
131  pose = pf_vector_coord_add(self->laser_pose_, pose);
132 
133  log_p = 0;
134 
135  beam_ind = 0;
136 
137  for (i = 0; i < data->range_count; i += step, beam_ind++) {
138  obs_range = data->ranges[i][0];
139  obs_bearing = data->ranges[i][1];
140 
141  // This model ignores max range readings
142  if (obs_range >= data->range_max) {
143  continue;
144  }
145 
146  // Check for NaN
147  if (obs_range != obs_range) {
148  continue;
149  }
150 
151  pz = 0.0;
152 
153  // Compute the endpoint of the beam
154  hit.v[0] = pose.v[0] + obs_range * cos(pose.v[2] + obs_bearing);
155  hit.v[1] = pose.v[1] + obs_range * sin(pose.v[2] + obs_bearing);
156 
157  // Convert to map grid coords.
158  int mi, mj;
159  mi = MAP_GXWX(self->map_, hit.v[0]);
160  mj = MAP_GYWY(self->map_, hit.v[1]);
161 
162  // Part 1: Get distance from the hit to closest obstacle.
163  // Off-map penalized as max distance
164 
165  if (!MAP_VALID(self->map_, mi, mj)) {
166  pz += self->z_hit_ * max_dist_prob;
167  } else {
168  z = self->map_->cells[MAP_INDEX(self->map_, mi, mj)].occ_dist;
169  if (z < beam_skip_distance) {
170  obs_count[beam_ind] += 1;
171  }
172  pz += self->z_hit_ * exp(-(z * z) / z_hit_denom);
173  }
174 
175  // Gaussian model
176  // NOTE: this should have a normalization of 1/(sqrt(2pi)*sigma)
177 
178  // Part 2: random measurements
179  pz += self->z_rand_ * z_rand_mult;
180 
181  assert(pz <= 1.0);
182  assert(pz >= 0.0);
183 
184  // TODO(?): outlier rejection for short readings
185 
186  if (!do_beamskip) {
187  log_p += log(pz);
188  } else {
189  self->temp_obs_[j][beam_ind] = pz;
190  }
191  }
192 
193  self->match_scores_.push_back(log_p);
194 
195  if (!do_beamskip) {
196  sample->weight *= exp(log_p);
197  total_weight += sample->weight;
198  }
199  }
200 
201  if (do_beamskip) {
202  int skipped_beam_count = 0;
203  for (beam_ind = 0; beam_ind < self->max_beams_; beam_ind++) {
204  if ((obs_count[beam_ind] / static_cast<double>(set->sample_count)) > beam_skip_threshold) {
205  obs_mask[beam_ind] = true;
206  } else {
207  obs_mask[beam_ind] = false;
208  skipped_beam_count++;
209  }
210  }
211 
212  // we check if there is at least a critical number of beams that agreed with the map
213  // otherwise it probably indicates that the filter converged to a wrong solution
214  // if that's the case we integrate all the beams and hope the filter might converge to
215  // the right solution
216  bool error = false;
217 
218  if (skipped_beam_count >= (beam_ind * self->beam_skip_error_threshold_)) {
219  fprintf(
220  stderr,
221  "Over %f%% of the observations were not in the map - pf may have converged to wrong pose -"
222  " integrating all observations\n",
223  (100 * self->beam_skip_error_threshold_));
224  error = true;
225  }
226 
227  for (j = 0; j < set->sample_count; j++) {
228  sample = set->samples + j;
229  pose = sample->pose;
230 
231  log_p = 0;
232 
233  for (beam_ind = 0; beam_ind < self->max_beams_; beam_ind++) {
234  if (error || obs_mask[beam_ind]) {
235  log_p += log(self->temp_obs_[j][beam_ind]);
236  }
237  }
238 
239  sample->weight *= exp(log_p);
240 
241  total_weight += sample->weight;
242  }
243  }
244 
245  delete[] obs_count;
246  delete[] obs_mask;
247  return total_weight;
248 }
