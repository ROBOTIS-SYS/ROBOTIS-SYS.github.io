38 static std::vector<std::string>
39 get_param_names_for_node(rclcpp::Node::SharedPtr node, std::string node_name)
40 {
41  auto client = node->create_client<rcl_interfaces::srv::ListParameters>(
42  node_name + "/list_parameters");
43 
44  while (!client->wait_for_service(std::chrono::seconds(1))) {
45  if (!rclcpp::ok()) {
46  throw std::runtime_error("client interrupted while waiting for service to appear.");
47  }
48 
49  throw std::runtime_error(
50  std::string("ListParameters service for ") +
51  node_name + " not available");
52  }
53 
54  auto request = std::make_shared<rcl_interfaces::srv::ListParameters::Request>();
55  auto result_future = client->async_send_request(request);
56 
57  if (rclcpp::spin_until_future_complete(node, result_future, 1s) !=
58  rclcpp::FutureReturnCode::SUCCESS)
59  {
60  throw std::runtime_error(std::string("service call to \"") + node_name + "\" failed");
61  }
62 
63  return result_future.get()->result.names;
64 }

130 // A local version to avoid trailing zeros
131 static std::string to_string(const rcl_interfaces::msg::ParameterValue & param_value)
132 {
133  switch (param_value.type) {
134  case rcl_interfaces::msg::ParameterType::PARAMETER_NOT_SET:
135  return "NOT_SET";
136 
137  case rcl_interfaces::msg::ParameterType::PARAMETER_DOUBLE:
138  {
139  // remove trailing zeroes
140  std::ostringstream out;
141  out << param_value.double_value;
142  return out.str();
143  }
144 
145  case rcl_interfaces::msg::ParameterType::PARAMETER_STRING:
146  return std::string("\"") + param_value.string_value + std::string("\"");
147 
148  case rcl_interfaces::msg::ParameterType::PARAMETER_STRING_ARRAY:
149  {
150  std::stringstream stream;
151  stream << "[";
152  auto num_items = param_value.string_array_value.size();
153  for (unsigned i = 0; i < num_items; i++) {
154  stream << "\"" << param_value.string_array_value[i] << "\"";
155  if (i != num_items - 1) {
156  stream << ", ";
157  }
158  }
159  stream << "]";
160  return stream.str();
161  };
162 
163  case rcl_interfaces::msg::ParameterType::PARAMETER_BOOL:
164  {
165  return param_value.bool_value ? "True" : "False";
166  };
167 
168  case rcl_interfaces::msg::ParameterType::PARAMETER_BOOL_ARRAY:
169  {
170  std::stringstream stream;
171  stream << "[";
172  auto num_items = param_value.bool_array_value.size();
173  for (unsigned i = 0; i < num_items; i++) {
174  stream << (param_value.bool_array_value[i] ? "True" : "False");
175  if (i != num_items - 1) {
176  stream << ", ";
177  }
178  }
179  stream << "]";
180  return stream.str();
181  };
182 
183  default:
184  {
185  return rclcpp::Parameter{"", param_value}.value_to_string();
186  }
187  }
188 }

190 static void
191 print_yaml(
192  const std::string node_name, std::vector<std::string> & param_names,
193  const std::vector<rcl_interfaces::msg::ParameterValue> & param_values,
194  const std::vector<rcl_interfaces::msg::ParameterDescriptor> & param_descriptors, bool verbose)
195 {
196  std::cout << node_name << ":" << std::endl;
197  std::cout << " ros__parameters:" << std::endl;
198 
199  for (unsigned i = 0; i < param_names.size(); i++) {
200  auto param_str = to_string(param_values[i]);
201 
202  // Use a field width wide enough for all of the headers
203  auto fw = 30;
204 
205  if (verbose) {
206  std::cout << " " << std::left << std::setw(fw + 2) <<
207  param_names[i] + ":" << param_str << "\n";
208  } else {
209  std::cout << " " << param_names[i] << ": " << param_str << "\n";
210  }
211 
212  if (verbose) {
213  std::cout << " # " << std::left << std::setw(fw) << "Range: ";
214  if (param_descriptors[i].floating_point_range.size()) {
215  auto range = param_descriptors[i].floating_point_range[0];
216  std::cout << range.from_value << ";" <<
217  range.to_value << ";" <<
218  range.step << "\n";
219  } else if (param_descriptors[i].integer_range.size()) {
220  auto range = param_descriptors[i].integer_range[0];
221  std::cout << range.from_value << ";" <<
222  range.to_value << ";" <<
223  range.step << "\n";
224  } else {
225  std::cout << "N/A\n";
226  }
227 
228  std::cout << " # " << std::left << std::setw(fw) << "Description: " <<
229  param_descriptors[i].description << "\n";
230  std::cout << " # " << std::left << std::setw(fw) << "Additional constraints: " <<
231  param_descriptors[i].additional_constraints << "\n";
232  std::cout << " # " << std::left << std::setw(fw) << "Read-only: " <<
233  (param_descriptors[i].read_only ? "True" : "False") << "\n";
234 
235  std::cout << std::endl;
236  }
237  }
238 
239  std::cout << std::endl;
240 }

277  range.step << "|";

302 template<typename T>
303 void
304 validate(
305  boost::any & v, const std::vector<std::string> & values,
306  option_sequence<T> * /*target_type*/, int)
307 {
308  std::vector<T> result;
309  typedef std::vector<std::string> strings;
310  for (strings::const_iterator iter = values.begin(); iter != values.end(); ++iter) {
311  strings tks;
312  alg::split(tks, *iter, alg::is_any_of(","));
313  for (strings::const_iterator tk = tks.begin(); tk != tks.end(); ++tk) {
314  result.push_back(boost::lexical_cast<T>(*tk));
315  }
316  }
317  v = option_sequence<T>();
318  boost::any_cast<option_sequence<T> &>(v).values.swap(result);
319 }

329 int main(int argc, char * argv[])
330 {
331  rclcpp::init(argc, argv);
332  auto dump_params_node = rclcpp::Node::make_shared("dump_params");
333 
334  try {
335  po::options_description desc("Options");
336 
337  /* *INDENT-OFF* */
338  desc.add_options()("help,h", "Print help message")
339  ("node_names,n", po::value<option_sequence<std::string>>(),
340  "A list of comma-separated node names")
341  ("format,f", po::value<std::string>(), "The format to dump ('yaml' or 'markdown')")
342  ("verbose,v", "Verbose option")
343  ;
344  /* *INDENT-ON* */
345 
346  po::variables_map vm;
347  po::store(po::parse_command_line(argc, argv, desc), vm);
348  po::notify(vm);
349 
350  if (vm.count("help")) {
351  std::cout << "Usage: " << basename(argv[0]) << "\n";
352  std::cout << desc << "\n";
353  return 1;
354  }
355 
356  std::vector<std::string> node_names;
357 
358  if (vm.count("node_names")) {
359  node_names = vm["node_names"].as<option_sequence<std::string>>().values;
360  } else {
361  node_names = dump_params_node->get_node_names();
362  }
363 
364  bool verbose = vm.count("verbose");
365 
366  for (std::string target_node_name : node_names) {
367  // Skip hidden nodes
368  if (target_node_name[1] == '_') {
369  continue;
370  }
371 
372  try {
373  auto param_names = get_param_names_for_node(dump_params_node, target_node_name);
374  auto param_values =
375  get_param_values_for_node(dump_params_node, target_node_name, param_names);
376  auto param_descriptors = get_param_descriptors_for_node(
377  dump_params_node, target_node_name,
378  param_names);
379 
380  if (!vm.count("format")) {
381  // Default to YAML if the format hasn't been specified
382  print_yaml(target_node_name, param_names, param_values, param_descriptors, verbose);
383  } else {
384  auto format = vm["format"].as<std::string>();
385  if (format == "md" || format == "markdown") {
386  print_markdown(target_node_name, param_names, param_values, param_descriptors, verbose);
387  } else {
388  if (format != "yaml") {
389  std::cerr << "Unknown output format specified, defaulting to 'yaml'" << std::endl;
390  }
391  print_yaml(target_node_name, param_names, param_values, param_descriptors, verbose);
392  }
393  }
394  } catch (std::exception & e) {
395  std::cerr << "Error: " << e.what() << "\n" << std::endl;
396  }
397  }
398  } catch (po::error & e) {
399  std::cerr << "Error: " << e.what() << std::endl;
400  }
401 
402  rclcpp::shutdown();
403  return 0;
404 }
