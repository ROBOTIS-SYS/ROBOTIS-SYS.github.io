35 bool JacobianSolverBasic::solve(
36  std::string component_name,
37  rm2_kinematics_solver_modules::JacobianSolverModule::DataArray present_joint_position,
38  rm2_kinematics_solver_modules::JacobianSolverModule::Jacobian & jacobian)
39 {
40  // First we check all the sizes:
41  if (present_joint_position.size() != robot_model_.get_dof()) {
42  fprintf(
43  stderr,
44  "[Jacobian Solver] Mismatch between present joint position "
45  "(%u) and the dof of robot model (%u) at point.\n",
46  present_joint_position.size(), robot_model_.get_dof());
47  return false;
48  }
49 
50  // Lets search the tree-element
51  using ComponentMapIt = rm2_robot_model::ComponentMap::const_iterator;
52  ComponentMapIt it = robot_model_.get_component(component_name);
53 
54  // If segmentname is not inside the tree, back out:
55  if (it == robot_model_.get_components().end()) {
56  return false;
57  }
58 
59  // Let's make the jacobian zero:
60  rm2_common::set_to_zero(jacobian);
61  jacobian.resize(robot_model_.get_dof());
62 
63  using Pose = rm2_common::Pose;
64  ComponentMapIt base = robot_model_.get_base_component();
65  Pose pose_total = Pose::Identity();
66 
67  // Lets recursively iterate until we are in the root segment
68  while (it != base) {
69  // get the corresponding joint number for this Robot model Component:
70  int joint_num = it->second.num_in_joint_array;
71 
72  // get the pose of the segment:
73  Pose pose_local;
74  if (joint_num >= 0) {
75  pose_local = it->second.component.get_pose(present_joint_position(joint_num));
76  } else {
77  pose_local = it->second.component.get_pose(0.0);
78  }
79 
80  // calculate new T_end:
81  pose_total = pose_local * pose_total;
82 
83 // std::cout << "---" << it->second.component.get_name() <<
84 // " <" << joint_num << ">" << "---" << std::endl;
85 
86 // std::cout << "P total:\n" << pose_total.position.data << std::endl;
87 // std::cout << "R total:\n" << pose_total.orientation.data << std::endl;
88 
89 // std::cout << "P local:\n" << pose_local.position.data << std::endl;
90 // std::cout << "R local:\n" << pose_local.orientation.data << std::endl;
91 
92  // get the twist of the segment:
93  using Joint = rm2_robot_model::Joint;
94  using Twist = rm2_common::Twist;
95  if (it->second.component.get_joint().get_type() != Joint::Fixed) {
96  // Axis at it component
97  Twist twist_local = it->second.component.get_twist(present_joint_position(joint_num), 1.0);
98 
99 // std::cout << "axis:\n" << twist_local.angular.data << std::endl;
100 // std::cout << "v local:\n" << twist_local.linear.data << std::endl;
101 // std::cout << "w local:\n" << twist_local.angular.data << std::endl;
102 
103  twist_local = twist_local.convert_coordinates(
104  pose_total.position - pose_local.position,
105  pose_total.orientation.inverse());
106 
107  // store the twist in the jacobian:
108  jacobian.set_column(static_cast<unsigned int>(joint_num), twist_local);
109  } // endif
110 
111  // goto the parent
112  it = it->second.parent;
113  } // end while
114 
115  // Change the base of the complete jacobian from the endpoint to the base
116  jacobian.change_base(pose_total.orientation);
117 
118 // std::cout << "jacobian:\n" << jacobian.data << std::endl;
119 
120  return true;
121 }
