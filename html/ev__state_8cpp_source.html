126 bool EVState::is_arrived_req(
127  const std::string & target_floor, uint8_t direction, bool in_ev,
128  bool use_hall_lanrern, bool & arrived, bool & opened)
129 {
130  std::lock_guard<std::mutex> lock(mutex_);
131 
132  // check update time
133  auto dur = rclcpp::Time() - updated_time_;
134  if (dur > timeout_) {
135  std::cout << "Timeout : EV status" << std::endl;
136  return false;
137  }
138 
139  if (in_ev == false) {
140  // EV 외부
141  opened =
142  (ev_door_status_ == EVStatusMsg::DOOR_OPENED ||
143  ev_door_status_ == EVStatusMsg::DOOR_OPENING);
144  // 홀랜턴 사용 시 도착은 홀랜턴(방향)만 체크
145  if (use_hall_lanrern) {
146  arrived = (direction == ev_direction_);
147  } else { // 현재 엘리베이터 층수 및 방향 체크
148 // arrived = ((ev_floor_ == target_floor) || opened) && (direction == ev_direction_);
149  arrived = (opened) && (direction == ev_direction_);
150  }
151  } else {
152  // EV 내부
153  // robot_floor가 존재하는 경우는 ev_floor 대신 robot_floor 사용
154  std::string ev_floor = ev_floor_;
155  if (robot_floor_ != "") {
156  ev_floor = robot_floor_;
157  if (ev_floor_ != robot_floor_) {
158  if (print_log != nullptr) {
159  std::stringstream log_ss;
160  log_ss << "[DIFF FLOOR] EV floor : " << ev_floor_ << ", Robot(ATM) floor : " <<
161  robot_floor_;
162  print_log(log_ss.str());
163  }
164  }
165  }
166 
167  arrived = (ev_floor == target_floor);
168  opened =
169  (ev_door_status_ == EVStatusMsg::DOOR_OPENED ||
170  ev_door_status_ == EVStatusMsg::DOOR_OPENING);
171  }
172 
173  if (arrived && opened) {
174  if (print_log != nullptr) {
175  std::stringstream log_ss;
176  log_ss << "[Arrived] Elevator is arrived : " << target_floor << ", door : " <<
177  (opened ? "opened" : "closed");
178  print_log(log_ss.str());
179  }
180  // std::cout << "Elevator is arrived : " << target_floor << ", door : " <<
181  // (opened ? "opened" : "closed") << std::endl;
182  }
183  return true;
184 }
