94 bool SimpleGoalChecker::isGoalReached(
95  const geometry_msgs::msg::Pose & query_pose, const geometry_msgs::msg::Pose & goal_pose,
96  const geometry_msgs::msg::Twist & velocity)
97 {
98  RCLCPP_INFO_STREAM(logger_, "[simple goal checker] ---- Check Goal Reached ----");
99  if (check_xy_) {
100  double dx = query_pose.position.x - goal_pose.position.x,
101  dy = query_pose.position.y - goal_pose.position.y;
102  RCLCPP_INFO_STREAM(logger_, "[simple goal checker] --> Translational error check");
103  RCLCPP_INFO_STREAM(
104  logger_,
105  "[simple goal checker] goal tolerance square: " << xy_goal_tolerance_sq_);
106  RCLCPP_INFO_STREAM(logger_, "[simple goal checker] distance to goal: " << dx * dx + dy * dy);
107  if (dx * dx + dy * dy > xy_goal_tolerance_sq_) {
108  return false;
109  }
110  // We are within the window
111  // If we are stateful, change the state.
112  if (stateful_) {
113  RCLCPP_INFO_STREAM(
114  logger_,
115  "[simple goal checker] Convert from translation error check to rotational error check");
116  check_xy_ = false;
117  }
118  }
119  double dyaw = angles::shortest_angular_distance(
120  tf2::getYaw(query_pose.orientation),
121  tf2::getYaw(goal_pose.orientation));
122  RCLCPP_INFO_STREAM(logger_, "[simple goal checker] --> Rotational error check");
123  RCLCPP_INFO_STREAM(logger_, "[simple goal checker] yaw tolerance: " << yaw_goal_tolerance_);
124  RCLCPP_INFO_STREAM(logger_, "[simple goal checker] yaw difference to goal: " << dyaw);
125 
126  return fabs(dyaw) < yaw_goal_tolerance_;
127 }
