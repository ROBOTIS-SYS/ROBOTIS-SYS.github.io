92 bool FindDoorAction::find_door(const sensor_msgs::msg::LaserScan & scan)
93 {
94  // get param
95  double min_angle = 0.0, max_angle = 0.0;
96  if (!getInput<double>("min_angle", min_angle) || !getInput<double>("max_angle", max_angle)) {
97  return false;
98  } else {
99  min_angle = min_angle * M_PI / 180.0;
100  max_angle = max_angle * M_PI / 180.0;
101  }
102  double max_range;
103  if (!getInput<double>("max_range", max_range)) {
104  max_range = 1.0;
105  }
106 
107  bool sim_3d = false;
108  int dimension = 2;
109  if (!getInput<bool>("sim_3d", sim_3d)) {
110  sim_3d = false;
111  }
112  if (sim_3d == true) {
113  dimension = 3;
114  }
115 
116  std::vector<Eigen::VectorXd> points;
117  Eigen::VectorXd variable_temp(dimension), variable(dimension);
118 
119  // range filter : angle
120  bool include_zero = false;
121  for (size_t ix = 0; ix < scan.ranges.size(); ix++) {
122  double angle = scan.angle_min + scan.angle_increment * static_cast<double>(ix);
123  if (angle < min_angle || angle > max_angle) {
124  continue;
125  }
126 
127  if (isnan(scan.ranges.at(ix)) || scan.ranges.at(ix) >= max_range) {
128  continue;
129  }
130 
131  // change to point
132  Eigen::VectorXd point(dimension);
133  point.coeffRef(0) = scan.ranges.at(ix) * cos(angle);
134  point.coeffRef(1) = scan.ranges.at(ix) * sin(angle);
135  if (sim_3d == true) {
136  point.coeffRef(2) = 0.0;
137  }
138 
139  points.push_back(point);
140  }
141 
142  std::cout << "filtered points : " << points.size() << ", all : " << scan.ranges.size() <<
143  std::endl;
144 
145  // return false if not enough points left
146  if (points.size() < 10) {
147  geometry_msgs::msg::Pose door_pose;
148  geometry_msgs::msg::Vector3 scale;
149  publish_door_marker(door_pose, scale, true);
150  return false;
151  }
152 
153  std::cout << "first point : " << points.begin()->coeff(0) << ", " << points.begin()->coeff(1) <<
154  std::endl;
155  std::cout << "last point : " << points.rbegin()->coeff(0) << ", " << points.rbegin()->coeff(1) <<
156  std::endl;
157 
158  // ransac : find door
159  int c_max = 0;
160  int iteration = static_cast<int>(0.2 * points.size());
161  int sample_num = 3;
162  double threshold = 0.01;
163  if (!getInput<double>("threshold", threshold)) {
164  threshold = 0.01;
165  }
166  std::random_device rd;
167  std::mt19937 gen(rd());
168  std::uniform_int_distribution<int> dist(0, points.size() - 1);
169  std::uniform_real_distribution<double> dist_2(0, 1.0);
170  if (!getInput<int>("sample_num", sample_num)) {
171  sample_num = 3;
172  }
173 
174  int refine_num = 3;
175  for (int refine_ix = 0; refine_ix < refine_num; refine_ix++) {
176  for (int it = 0; it < iteration; it++) {
177  int select_index[3];
178 
179  for (int sample_ix = 0; sample_ix < sample_num; sample_ix++) {
180  // select_index[sample_ix] = dist(gen);
181  select_index[sample_ix] = static_cast<int>(dist_2(gen) * points.size());
182  }
183 
184  // calc
185  std::uniform_real_distribution<double> dist_height(0.0, 2.1);
186  Eigen::MatrixXd sample_mat(dimension, sample_num);
187  for (int v_ix = 0; v_ix < sample_num; v_ix++) {
188  Eigen::VectorXd v(dimension);
189  v = points.at(select_index[v_ix]);
190  if (sim_3d == true) {
191  v.coeffRef(2) = dist_height(gen);
192  }
193  sample_mat.col(v_ix) = v;
194  }
195 
196  Eigen::VectorXd id(sample_num);
197  id.setOnes();
198 
199  // Eigen::BDCSVD<Eigen::MatrixXd> SVD(sample_mat, Eigen::ComputeThinU |
200  // Eigen::ComputeThinV); Eigen::MatrixXd sample_iv = SVD.matrixV() *
201  // SVD.singularValues().array().inverse().matrix().asDiagonal() *
202  // SVD.matrixU().transpose();
203  // variable_temp = id.transpose() * sample_iv;
204 
205  Eigen::MatrixXd square_mat = sample_mat * sample_mat.eval().transpose();
206  variable_temp = id.transpose() * sample_mat.eval().transpose() * square_mat.inverse();
207 
208  // evaluation
209  int evaluation_count = 0;
210  for (auto v : points) {
211  double residual = fabs(variable_temp.transpose() * v - 1.0) /
212  sqrt(variable_temp.transpose() * variable_temp);
213  if (residual < threshold * (refine_num - refine_ix)) {
214  evaluation_count++;
215  }
216  }
217 
218  if (evaluation_count < c_max || evaluation_count < points.size() * 0.3) {
219  continue;
220  }
221 
222  c_max = evaluation_count;
223  variable = variable_temp;
224  }
225 
226  if (c_max == 0) {
227  std::cout << "Failed to find the foor." << std::endl;
228 
229  geometry_msgs::msg::Pose door_pose;
230  geometry_msgs::msg::Vector3 scale;
231  publish_door_marker(door_pose, scale, true);
232  return false;
233  }
234 
235  // refine
236  std::vector<Eigen::VectorXd> door_points;
237  for (int ix = 0; ix < points.size(); ix++) {
238  double residual = fabs(variable.transpose() * points.at(ix) - 1.0) /
239  sqrt(variable.transpose() * variable);
240  if (residual < threshold * (refine_num - refine_ix)) {
241  door_points.push_back(points.at(ix));
242  }
243  }
244 
245  std::cout << "result - c_max : " << c_max << ", refined points : " << door_points.size() <<
246  std::endl;
247 
248  std::cout << "variable : " << std::endl << variable << std::endl;
249 
250  points = door_points;
251  c_max = 0;
252 
253  if (points.size() < 10) {
254  geometry_msgs::msg::Pose door_pose;
255  geometry_msgs::msg::Vector3 scale;
256  publish_door_marker(door_pose, scale, true);
257  return false;
258  }
259  }
260 
261  // find center
262  double begin_y = points.begin()->coeff(1);
263  double begin_x = (1 - variable.coeff(1) * begin_y) / variable.coeff(0);
264  double end_y = points.rbegin()->coeff(1);
265  double end_x = (1 - variable.coeff(1) * end_y) / variable.coeff(0);
266 
267  double center_x = (begin_x + end_x) * 0.5;
268  double center_y = (begin_y + end_y) * 0.5;
269 
270  // std::cout << "(" << begin_x << ", " << begin_y << ") - (" << end_x << ", " << end_y << ")"
271  // <<
272  // std::endl;
273 
274  geometry_msgs::msg::Point output_point;
275  output_point.x = begin_x;
276  output_point.y = begin_y;
277  output_point.z = 0.0;
278  setOutput<geometry_msgs::msg::Point>("point_1", output_point);
279  output_point.x = end_x;
280  output_point.y = end_y;
281  output_point.z = 0.0;
282  setOutput<geometry_msgs::msg::Point>("point_2", output_point);
283  output_point.x = center_x;
284  output_point.y = center_y;
285  output_point.z = 1.05;
286  setOutput<geometry_msgs::msg::Point>("point_3", output_point);
287 
288  double diff_x = end_x - begin_x;
289  double diff_y = end_y - begin_y;
290  double width = sqrt(diff_x * diff_x + diff_y * diff_y);
291  double distance = 1 / variable.norm();
292 
293  // std::cout << "Estimated distance to door : " << distance << "m" << std::endl;
294 
295  // publish door marker
296  geometry_msgs::msg::Pose door_pose;
297  geometry_msgs::msg::Vector3 scale;
298  // scale
299  scale.x = 0.05;
300  scale.y = width;
301  scale.z = 2.1;
302  // positoin
303  door_pose.position.x = center_x;
304  door_pose.position.y = center_y;
305  door_pose.position.z = 1.05 - 0.28;
306  // orientation
307  Eigen::Quaterniond orientation;
308  Eigen::Vector3d from(0, 1, 0);
309  Eigen::Vector3d to = *points.rbegin() - *points.begin();
310  orientation.setFromTwoVectors(from, to);
311  door_pose.orientation.x = orientation.x();
312  door_pose.orientation.y = orientation.y();
313  door_pose.orientation.z = orientation.z();
314  door_pose.orientation.w = orientation.w();
315 
316  publish_door_marker(door_pose, scale);
317 
318  return true;
319 }

325  if (!getInput<double>("min_angle", min_angle) || !getInput<double>("max_angle", max_angle)) {
326  return false;
327  } else {
328  min_angle = min_angle * M_PI / 180.0;
329  max_angle = max_angle * M_PI / 180.0;
330  }
331  double max_range;
332  if (!getInput<double>("max_range", max_range)) {
333  max_range = 1.0;
334  }
335  bool sim_3d = false;
336  int dimension = 2;
337  if (!getInput<bool>("sim_3d", sim_3d)) {
338  sim_3d = false;
339  }
340  if (sim_3d == true) {
341  dimension = 3;
342  }
343 
344  // divide zones
345  double zone_angle;
346  if (!getInput<double>("zone_angle", zone_angle)) {
347  zone_angle = 2.5 * M_PI / 180.0;
348  } else {
349  zone_angle = zone_angle * M_PI / 180.0;
350  }
351  std::vector<std::vector<Eigen::VectorXd>> zone_points;
352  int zone_num = static_cast<int>((max_angle - min_angle) / zone_angle);
353  int center_zone_index = 0;
354  for (int zone_index = 0; zone_index < zone_num; zone_index++) {
355  double start_angle = min_angle + zone_angle * zone_index;
356  double end_angle = min_angle + zone_angle * (zone_index + 1);
357  int start_index = static_cast<int>((start_angle - scan.angle_min) / scan.angle_increment);
358  int end_index = static_cast<int>((end_angle - scan.angle_min) / scan.angle_increment);
359  if (start_angle <= 0 && end_angle >= 0) {
360  center_zone_index = zone_index;
361  }
362  std::vector<Eigen::VectorXd> points;
363 
364  for (int ix = start_index; ix < end_index; ix++) {
365  double angle = scan.angle_min + scan.angle_increment * static_cast<double>(ix);
366  if (angle < min_angle || angle > max_angle) {
367  continue;
368  }
369 
370  if (isnan(scan.ranges.at(ix)) || scan.ranges.at(ix) >= max_range) {
371  continue;
372  }
373 
374  // change to point
375  Eigen::VectorXd point(dimension);
376  point.coeffRef(0) = scan.ranges.at(ix) * cos(angle);
377  point.coeffRef(1) = scan.ranges.at(ix) * sin(angle);
378  if (sim_3d == true) {
379  point.coeffRef(2) = 0.0;
380  }
381 
382  points.push_back(point);
383  }
384  zone_points.push_back(points);
385 
386  // if (zone_points.at(zone_index).size() != 0) {
387  // std::cout << "Zone [" << zone_index << "] " <<
388  // zone_points.at(zone_index).at(0).coeff(0) <<
389  // " , " << zone_points.at(zone_index).at(0).coeff(1) << std::endl;
390  // } else {
391  // std::cout << "Zone [" << zone_index << "]" << std::endl;
392  // }
393  }
394 
395  // std::cout << "zone number : " << zone_points.size() << ", points : " <<
396  // zone_points.at(0).size() <<
397  // ", all : " << scan.ranges.size() << std::endl;
398 
399  // ransac
400  int sample_num = 3;
401  if (!getInput<int>("sample_num", sample_num)) {
402  sample_num = 3;
403  }
404  double threshold = 0.01;
405  if (!getInput<double>("threshold", threshold)) {
406  threshold = 0.01;
407  }
408 
409  std::vector<Eigen::VectorXd> variables;
410  for (int zone_index = 0; zone_index < zone_num; zone_index++) {
411  std::vector<Eigen::VectorXd> points = zone_points.at(zone_index);
412  Eigen::VectorXd variable(dimension);
413  if (points.size() < sample_num) {
414  variables.push_back(variable);
415  continue;
416  }
417  bool result = ransac(points, dimension, sample_num, threshold, false, variable);
418 
419  if (result == false) {
420  variable.setZero(3);
421  }
422  variables.push_back(variable);
423  }
424 
425  // grouping
426  std::vector<Eigen::VectorXd> center_points;
427  bool is_center = false;

429  center_points = zone_points.at(0);
430  for (int zone_index = 1; zone_index < zone_num; zone_index++) {
431  if (zone_index == center_zone_index) {
432  is_center = true;
433  }
434  double dot = variables.at(zone_index - 1).dot(variables.at(zone_index));
435  if (dot > cos_value) {
436  // same gropt
437  center_points.insert(
438  center_points.end(), zone_points.at(zone_index).begin(),
439  zone_points.at(zone_index).end());
440  } else {
441  // other group
442  if (is_center == false) {

445  } else {
446  break;
447  }
448  }
449  }

463  if (result == false) {

527 }

532 {

537  std::mt19937 gen(rd());
538  std::uniform_int_distribution<int> dist(0, points.size() - 1);
539  // std::uniform_real_distribution<double> dist_2(0, 1.0);
540 
541  for (int it = 0; it < iteration; it++) {
542  int select_index[3];
543 
544  for (int sample_ix = 0; sample_ix < sample_num; sample_ix++) {
545  select_index[sample_ix] = dist(gen);
546  // select_index[sample_ix] = static_cast<int>(dist_2(gen) * points.size());
547  }
548 
549  // calc
550  std::uniform_real_distribution<double> dist_height(0.0, 2.1);
551  Eigen::MatrixXd sample_mat(dimension, sample_num);
552  for (int v_ix = 0; v_ix < sample_num; v_ix++) {
553  Eigen::VectorXd v(dimension);
554  v = points.at(select_index[v_ix]);
555  if (dimension == 3) {
556  v.coeffRef(2) = dist_height(gen);
557  }
558  sample_mat.col(v_ix) = v;
559  }
560 
561  Eigen::VectorXd id(sample_num);
562  id.setOnes();
563 
564  // Eigen::BDCSVD<Eigen::MatrixXd> SVD(sample_mat, Eigen::ComputeThinU |
565  // Eigen::ComputeThinV); Eigen::MatrixXd sample_iv = SVD.matrixV() *
566  // SVD.singularValues().array().inverse().matrix().asDiagonal() *
567  // SVD.matrixU().transpose();
568  // variable_temp = id.transpose() * sample_iv;
569 
570  Eigen::MatrixXd square_mat = sample_mat * sample_mat.eval().transpose();
571  variable_temp = id.transpose() * sample_mat.eval().transpose() * square_mat.inverse();
572 
573  // variable_temp = id.transpose() * sample_mat.inverse();

577  for (auto v : points) {
578  double residual = fabs(variable_temp.transpose() * v - 1.0) /

580  if (residual < threshold) {
581  evaluation_count++;
582  }
583  }
584 
585  if (evaluation_count < c_max || evaluation_count < points.size() * 0.3) {
586  continue;
587  }
588 
589  c_max = evaluation_count;

597  // geometry_msgs::msg::Vector3 scale;
598  // publish_door_marker(door_pose, scale, true);
599  return false;
600  }
601 
602  // refine
603  if (refind == false) {
604  return true;
605  }
606 
607  // std::vector<Eigen::Vector3d> door_points;
608  std::vector<Eigen::VectorXd> door_points;
609  // for (auto v : points) {
610  for (int ix = 0; ix < points.size(); ix++) {
