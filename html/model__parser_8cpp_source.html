50 static inline rm2_robot_model::Joint to_model(urdf::JointSharedPtr jnt)
51 {
52  using Vector = rm2_common::Vector;
53  using Limit = rm2_common::Limit;
54  using Frame = rm2_common::Frame;
55  using Joint = rm2_robot_model::Joint;
56 
57  Frame origin_from_parent = to_model(jnt->parent_to_joint_origin_transform);
58 
59  fprintf(stderr, " -joint name: %s\n", jnt->name.c_str());
60  double rpy[3];
61  origin_from_parent.orientation.get_rpy(rpy[0], rpy[1], rpy[2]);
62  fprintf(
63  stderr,
64  " -joint frame :\n \t\t%lf | %lf\n \t\t%lf | %lf\n \t\t%lf | %lf\n",
65  origin_from_parent.position(0), rpy[0],
66  origin_from_parent.position(1), rpy[1],
67  origin_from_parent.position(2), rpy[2]);
68  switch (jnt->type) {
69  case urdf::Joint::FIXED: {
70  fprintf(stderr, " -joint type: FIXED\n");
71  return Joint(jnt->name, origin_from_parent);
72  }
73  case urdf::Joint::REVOLUTE: {
74  Vector axis = to_model(jnt->axis);
75  Limit limit = to_model(jnt->limits);
76  fprintf(stderr, " -joint type: REVOLUTE\n");
77  fprintf(
78  stderr, " -limit : %lf, %lf, %lf, %lf\n",
79  limit.maximum, limit.minimum,
80  limit.velocity, limit.effort);
81 
82  return Joint(jnt->name, Joint::RotAxis, axis, origin_from_parent, limit);
83  }
84  case urdf::Joint::CONTINUOUS: {
85  fprintf(stderr, " -joint type: CONTINUOUS\n");
86 // Vector axis = to_model(jnt->axis)/;
87 // Limit limit;
88 // return Joint(jnt->name, Joint::RotAxis, axis, origin_from_parent, limit);
89  return Joint(jnt->name, origin_from_parent);
90  }
91  case urdf::Joint::PRISMATIC: {
92  fprintf(stderr, " -joint type: PRISMATIC\n");
93  Vector axis = to_model(jnt->axis);
94  Limit limit = to_model(jnt->limits);
95  fprintf(
96  stderr, " -limit : %lf, %lf, %lf, %lf\n",
97  limit.maximum, limit.minimum,
98  limit.velocity, limit.effort);
99  return Joint(jnt->name, Joint::TransAxis, axis, origin_from_parent, limit);
100  }
101  default: {
102  fprintf(
103  stderr, "Converting unknown joint type of joint '%s' into a fixed joint\n",
104  jnt->name.c_str());
105  return Joint(jnt->name, origin_from_parent);
106  }
107  }

110 static inline rm2_robot_model::Inertia to_model(urdf::InertialSharedPtr iner)
111 {
112  using Vector = rm2_common::Vector;
113  using Frame = rm2_common::Frame;
114  using InertiaTensor = rm2_common::InertiaTensor;
115  using Inertia = rm2_common::Inertia;
116 
117  Frame origin = to_model(iner->origin);
118 
119  double mass = iner->mass;
120 
121  Vector com = origin.position;
122 
123  // manipulator_lib specifies the inertia matrix in the reference frame of the "link",
124  // while the urdf specifies the inertia matrix in the "inertia reference frame"
125  InertiaTensor inertia_tensor(iner->ixx, iner->iyy, iner->izz, iner->ixy, iner->ixz, iner->iyz);
126 
127  // Rotation operators are not defined for rotational inertia(inertia tensor),
128  // so we use the Inertia operators (with com = 0) as a workaround
129  Inertia rm2_inertia_wrt_com_workaround =
130  origin.orientation * Inertia(0, Vector::Zero(), inertia_tensor);
131 
132  // Note that the Inertia constructor takes the 3d inertia wrt the com
133  // while the getRotationalInertia method returns the 3d inertia wrt the frame origin
134  // (but having com = Vector::Zero() in rm2_inertia_wrt_com_workaround they match)
135  InertiaTensor rm2_inertia_wrt_com =
136  rm2_inertia_wrt_com_workaround.get_inertia_tensor();
137 
138  return Inertia(mass, com, rm2_inertia_wrt_com);
139 }

141 bool add_child_compoent(urdf::LinkConstSharedPtr root, RobotModel & model)
142 {
143  using Inertia = rm2_common::Inertia;
144  using Joint = rm2_robot_model::Joint;
145  using Component = rm2_robot_model::Component;
146 
147  std::vector<urdf::LinkSharedPtr> children = root->child_links;
148  fprintf(stderr, "Link '%s' had %zu children\n", root->name.c_str(), children.size());
149  // constructs the optional inertia
150  Inertia inertia;
151  if (root->inertial) {
152  inertia = to_model(root->inertial);
153  }
154 
155  // constructs the kdl joint
156  Joint joint = to_model(root->parent_joint);
157 
158  // construct the kdl segment
159  Component component(root->name, joint, inertia);
160 
161  // add segment to tree
162  model.add_component(component, root->parent_joint->parent_link_name);
163  fprintf(
164  stderr, " -added Link '%s' joint num : %d\n",
165  model.get_component(component.get_name())->first.c_str(),
166  model.get_component(component.get_name())->second.num_in_joint_array);
167 
168  // recurslively add all children
169  for (size_t i = 0; i < children.size(); i++) {
170  if (!add_child_compoent(children[i], model)) {
171  return false;
172  }
173  }
174  return true;
175 }

197 bool model_from_urdf(const urdf::ModelInterface & robot_model, RobotModel & model)
198 {
199  if (!robot_model.getRoot()) {
200  return false;
201  }
202 
203  model = rm2_robot_model::RobotModel(robot_model.getRoot()->name);
204 
205  // warn if root link has inertia. KDL does not support this
206  if (robot_model.getRoot()->inertial) {
207  fprintf(
208  stderr, "The root link %s has an inertia specified in the URDF,"
209  " but Manipulator library does not "
210  "support a root link with an inertia. As a workaround, you can add an extra "
211  "dummy link to your URDF.\n",
212  robot_model.getRoot()->name.c_str());
213  }
214 
215  // add all children
216  for (size_t i = 0; i < robot_model.getRoot()->child_links.size(); i++) {
217  if (!add_child_compoent(robot_model.getRoot()->child_links[i], model)) {
218  return false;
219  }
220  }
221 
222  return true;
223 }
