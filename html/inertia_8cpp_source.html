50 Inertia Inertia::reference_point(const Vector & position)
51 {
52  // mb=ma
53  // hb=(h-m*r)
54  // Ib = (Ia+r x h x + (h-m*r) x r x)
55  Vector hmr = this->h_ - this->mass_ * position;
56  Eigen::Matrix3d rcrosshcross = h_.data * position.data.transpose() -
57  position.data.dot(h_.data) * Eigen::Matrix3d::Identity();
58  Eigen::Matrix3d hmrcrossrcross = position.data * hmr.data.transpose() -
59  hmr.data.dot(position.data) * Eigen::Matrix3d::Identity();
60  InertiaTensor inertia_tensor =
61  this->inertia_tensor_ + InertiaTensor(rcrosshcross + hmrcrossrcross);
62 
63  return Inertia(this->mass_, hmr, inertia_tensor, mhi);
64 }

106 Inertia operator*(const Pose & pose, const Inertia & inertia)
107 {
108  // mb=ma
109  // hb=R*(h-m*r)
110  // Ib = R(Ia+r x h x + (h-m*r) x r x)R'
111  Pose pose_inverse = pose.inverse();
112  // Vector
113  Vector hmr = inertia.h_ - inertia.mass_ * pose_inverse.position;
114  Eigen::Matrix3d rcrosshcross = inertia.h_.data * pose_inverse.position.data.transpose() -
115  pose_inverse.position.data.dot(inertia.h_.data) * Eigen::Matrix3d::Identity();
116  Eigen::Matrix3d hmrcrossrcross = pose_inverse.position.data * hmr.data.transpose() -
117  hmr.data.dot(pose_inverse.position.data) * Eigen::Matrix3d::Identity();
118  Eigen::Matrix3d new_inwetia_tensor = pose_inverse.orientation.data *
119  (inertia.inertia_tensor_.data + rcrosshcross + hmrcrossrcross) *
120  pose_inverse.orientation.data.transpose();
121  return Inertia(inertia.mass_, pose.orientation * hmr, InertiaTensor(new_inwetia_tensor), mhi);
122 }

124 Inertia operator*(const Rotation & rotation, const Inertia & inertia)
125 {
126  // mb=ma
127  // hb=R*h
128  // Ib = R(Ia)R' with r=0
129  Eigen::Matrix3d new_inertia_tensor = rotation.data.transpose() *
130  inertia.inertia_tensor_.data * rotation.data;
131  return Inertia(inertia.mass_, rotation * inertia.h_, InertiaTensor(new_inertia_tensor), mhi);
