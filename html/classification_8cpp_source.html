123 void Classification::model_run(
124  std::vector<cv::cuda::GpuMat> & d_src,
125  std::vector<std::vector<cv::Point2i>> feature_pts,
126  int patch_size)
127 {
128  left_detected_buttons.clear();
129  right_detected_buttons.clear();
130  std::vector<std::vector<void *>> tmp_prediction_bindings;
131  for (size_t i = 0; i < patch_size; ++i) {
132  tmp_prediction_bindings.push_back(
133  {
134  (*device_tensor_buffer_array_[i])[0].deviceBuffer.data(),
135  (*device_tensor_buffer_array_[i])[1].deviceBuffer.data(),
136  (*device_tensor_buffer_array_[i])[2].deviceBuffer.data()});
137  }
138  for (size_t i = 0; i < patch_size; ++i) {
139  pre_processing(d_src[i], i);
140  cudaMemcpyAsync(
141  (*device_tensor_buffer_array_[i])[0].deviceBuffer.data(),
142  device_normal_patch_mat_continuous_[i].data,
143  patch_size_ * patch_size_ * sizeof(float),
144  cudaMemcpyDeviceToDevice,
145  cuda_stream_array_[i]);
146  trt_engine_->runInference(tmp_prediction_bindings[i], cuda_stream_array_[i]);
147  cudaStreamSynchronize(cuda_stream_array_[i]);
148  }
149  for (size_t i = 0; i < patch_size; ++i) {
150  for (size_t j = 0; j < trt_engine_->output_dim_.size(); ++j) {
151  cudaMemcpyAsync(
152  (*device_tensor_buffer_array_[i])[j + 1].hostBuffer.data(),
153  (*device_tensor_buffer_array_[i])[j + 1].deviceBuffer.data(),
154  (*device_tensor_buffer_array_[i])[j + 1].deviceBuffer.nbBytes(),
155  cudaMemcpyDeviceToHost,
156  cuda_stream_array_[i]);
157  }
158  }
159  for (size_t i = 0; i < patch_size; ++i) {
160  std::vector<int> class_result = post_processing(i);
161  std::string s_result = "";
162  if (class_result[1] == 30) {
163  s_result = button_class[class_result[0]];
164  } else {
165  s_result =
166  button_class[class_result[0]] +
167  button_class[class_result[1]];
168  }
169 
170  bool left_right_check = check_left_right_pts(feature_pts[i]);
171  RecognizedText tmp_detected_button;
172  tmp_detected_button.class_name = s_result;
173  tmp_detected_button.button_state = false;
174  tmp_detected_button.d_patch_img = d_src[i];
175  if (left_right_check) {
176  for (size_t j = 0; j < 5; ++j) {
177  tmp_detected_button.features.push_back(feature_pts[i][j] - offset_);
178  }
179  right_detected_buttons.push_back(tmp_detected_button);
180  } else {
181  for (size_t j = 0; j < 5; ++j) {
182  tmp_detected_button.features.push_back(feature_pts[i][j]);
183  }
184  left_detected_buttons.push_back(tmp_detected_button);
185  }
186  }
187 }

304 void Classification::pre_processing_3ch(cv::cuda::GpuMat & d_src, int index)
305 {
306  cv::Mat src;
307  d_src.download(src);
308 
309  std::vector<int> rec_image_shape_ = {3, 224, 224};
310  int imgH = rec_image_shape_[1];
311  int imgW = rec_image_shape_[2];
312  float max_wh_ratio = imgW * 1.0 / imgH;
313  int h = src.rows;
314  int w = src.cols;
315  float wh_ratio = w * 1.0 / h;
316  std::vector<float> mean_ = {0.5f, 0.5f, 0.5f};
317  std::vector<float> scale_ = {1 / 0.5f, 1 / 0.5f, 1 / 0.5f};
318  max_wh_ratio = std::max(max_wh_ratio, wh_ratio);
319 
320  cv::Mat resize_img;
321  resize(src, resize_img, max_wh_ratio, rec_image_shape_);
322  normalize(&resize_img, mean_, scale_, true);
323  convert_mat2vec(resize_img, h_input_pin);
324  auto error = cudaMemcpyAsync(
325  (*device_tensor_buffer_array_[index])[0].deviceBuffer.data(),
326  h_input_pin,
327  1 * 3 * imgH * imgW * sizeof(float),
328  cudaMemcpyHostToDevice,
329  cuda_stream_array_[index]);
330  rec_image_shape_.clear();
331  mean_.clear();
332  scale_.clear();

366 void Classification::normalize(
367  cv::Mat * im,
368  const std::vector<float> & mean,
369  const std::vector<float> & scale,
370  const bool is_scale)
371 {
372  double e = 1.0;
373  if (is_scale) {
374  e /= 255.0;
375  }
376  (*im).convertTo(*im, CV_32FC3, e);
377  std::vector<cv::Mat> bgr_channels(3);
378  cv::split(*im, bgr_channels);
379  for (auto i = 0; i < bgr_channels.size(); i++) {
380  bgr_channels[i].convertTo(
381  bgr_channels[i], CV_32FC1, 1.0 * scale[i],
382  (0.0 - mean[i]) * scale[i]);
383  }
384  cv::merge(bgr_channels, *im);
385  bgr_channels.clear();
386 }

400 void Classification::resize(
401  const cv::Mat & img,
402  cv::Mat & resize_img,
403  float wh_ratio,
404  const std::vector<int> & rec_image_shape)
405 {
406  int imgC, imgH, imgW;
407  imgC = rec_image_shape[0];
408  imgH = rec_image_shape[1];
409  imgW = rec_image_shape[2];
410 
411  float ratio = static_cast<float>(img.cols) / static_cast<float>(img.rows);
412  int resize_w, resize_h;
413 
414  if (ceilf(imgH * ratio) > imgW) {
415  resize_w = imgW;
416  } else {
417  resize_w = static_cast<int>(ceilf(imgH * ratio));
418  }
419  if (resize_w > imgW) {
420  resize_w = imgW;
421  }
422  cv::resize(
423  img, resize_img, cv::Size(resize_w, imgH), 0.f, 0.f,
424  cv::INTER_LINEAR);
425  cv::copyMakeBorder(
426  resize_img,
427  resize_img,
428  0,
429  0,
430  0,
431  static_cast<int>(imgW - resize_img.cols),
432  cv::BORDER_CONSTANT,
433  {127, 127, 127});
434 }
435 std::vector<float> Classification::softmax(const std::vector<float> & logits)
436 {
437  std::vector<float> probabilities(logits.size());
438  float max_logit = *std::max_element(logits.begin(), logits.end());
439  float sum_exp = 0.0;
440 
441  for (int i = 0; i < logits.size(); ++i) {
442  probabilities[i] = std::exp(logits[i] - max_logit);
443  sum_exp += probabilities[i];
444  }
445 
446  for (int i = 0; i < probabilities.size(); ++i) {
447  probabilities[i] /= sum_exp;
448  }
449 
450  return probabilities;
451 }
