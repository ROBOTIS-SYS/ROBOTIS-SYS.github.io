95 bool GcLidarLimiter::checker()
96 {
97  if (is_checking_ && scan_msg_first_input_) {
98  // check lidar time
99  if (node_->now() - rclcpp::Time(scan_msg_.header.stamp) > rclcpp::Duration(1, 0)) {
100  RCLCPP_ERROR_STREAM(node_->get_logger(), "Lidar data time over.");
101  scan_msg_first_input_ = false;
102  return true;
103  }
104  // Lidar Pose
105  rm2_common::Pose lidar_pose;
106 
107  if (this->robot_handler_->check_component_name(scan_msg_.header.frame_id)) {
108  lidar_pose = this->robot_handler_->get_present_task_value(scan_msg_.header.frame_id).pose;
109  }
110 
111 // std::cout << "------" << rclcpp::Time(
112 // scan_msg_.header.stamp).seconds() << "------" << std::endl;
113 
114  bool detected = false;
115  double min_delta = detection_range_;
116 
117  std::string detected_component_name;
118  double detected_componenet_x = 0.0;
119  double detected_componenet_y = 0.0;
120  double detected_componenet_z = 0.0;
121  rm2_common::Rotation detected_componenet_orientation;
122  double target_x = 0.0;
123  double target_y = 0.0;
124 
125  for (std::string component_name : check_components_) {
126  // Manipulator position
127  rm2_common::TaskValues component_task_value;
128 
129  if (use_prev_goal_) {
130  component_task_value = this->robot_handler_->get_previous_goal_task_value(component_name);
131  } else {
132  component_task_value = this->robot_handler_->get_present_task_value(component_name);
133  }
134 
135  // Relative Pose from lidar to tip
136  rm2_common::Pose pose_from_lidar = lidar_pose.inverse() * component_task_value.pose;
137 
138  // Component Angle and Distance
139  double angle = atan2(pose_from_lidar.position.y(), pose_from_lidar.position.x());
140 // std::cout << "angle : " << angle << std::endl;
141  double distance =
142  sqrt(
143  pose_from_lidar.position.x() * pose_from_lidar.position.x() +
144  pose_from_lidar.position.y() * pose_from_lidar.position.y());
145 // std::cout << "distance : " << distance << std::endl;
146  double detection_angle_range = detection_range_ / distance;
147  double componenet_x = distance * cos(angle);
148  double componenet_y = distance * sin(angle);
149 
150  // Scan Data
151  float min_angle = static_cast<float>(angle - detection_angle_range);
152  uint min_angle_index;
153  if (min_angle < scan_msg_.angle_min) {
154  min_angle_index = 0;
155  } else if (min_angle > scan_msg_.angle_max) {
156  min_angle_index = static_cast<uint>((scan_msg_.angle_max) / scan_msg_.angle_increment);
157  } else {
158  min_angle_index =
159  static_cast<uint>((min_angle - scan_msg_.angle_min) / scan_msg_.angle_increment);
160  }
161  float max_angle = static_cast<float>(angle + detection_angle_range);
162  uint max_angle_index;
163  if (max_angle < scan_msg_.angle_min) {
164  max_angle_index = 0;
165  } else if (max_angle > scan_msg_.angle_max) {
166  max_angle_index = static_cast<uint>((scan_msg_.angle_max) / scan_msg_.angle_increment);
167  } else {
168  max_angle_index =
169  static_cast<uint>((max_angle - scan_msg_.angle_min) / scan_msg_.angle_increment);
170  }
171 
172  for (uint index = min_angle_index; index <= max_angle_index; index++) {
173  double scan_angle =
174  static_cast<double>((index * scan_msg_.angle_increment) + scan_msg_.angle_min);
175  double scan_distance = static_cast<double>(scan_msg_.ranges[index]);
176 
177  double scan_x = scan_distance * cos(scan_angle);
178  double scan_y = scan_distance * sin(scan_angle);
179  double delta = sqrt(
180  (scan_x - componenet_x) * (scan_x - componenet_x) +
181  (scan_y - componenet_y) * (scan_y - componenet_y));
182 
183  if (delta <= min_delta && scan_distance >= distance) {
184  detected = true;
185  min_delta = delta;
186 
187  detected_component_name = component_name;
188 
189  detected_componenet_x = componenet_x;
190  detected_componenet_y = componenet_y;
191  detected_componenet_z = pose_from_lidar.position.z();
192  detected_componenet_orientation = pose_from_lidar.orientation;
193 
194  target_x = componenet_x - (detection_range_ - min_delta) / min_delta *
195  (scan_x - componenet_x);
196  target_y = componenet_y - (detection_range_ - min_delta) / min_delta *
197  (scan_y - componenet_y);
198  }
199  }
200  }
201 
202  if (detected) {
203  std::cout << "target(x,y) =(" << target_x << ", " << target_y << ")" << std::endl;
204  std::cout << "delta :" << sqrt(
205  (target_x - detected_componenet_x) * (target_x - detected_componenet_x) +
206  (target_y - detected_componenet_y) * (target_y - detected_componenet_y)) << std::endl;
207  std::cout << "move angle :" <<
208  atan2(
209  (target_y - detected_componenet_y),
210  (target_x - detected_componenet_x)) * 180 / M_PI << std::endl;
211 
212  rm2_common::Pose target_pose_from_lidar(
213  rm2_common::Vector(
214  target_x,
215  target_y,
216  detected_componenet_z),
217  detected_componenet_orientation);
218  RCLCPP_ERROR_STREAM(
219  node_->get_logger(),
220  "over lidar limit. [" << detected_component_name << "]");
221 
222  // Set Target data
223  rm2_common::TaskValues target_task_value = rm2_common::TaskValues(
224  lidar_pose * target_pose_from_lidar);
225 
226  if (robot_handler_->is_debug_mode()) {
228  std::function<geometry_msgs::msg::Pose(
229  rm2_common::Vector position,
230  rm2_common::Rotation orientation)> trans_pose_to_msg =
231  [](rm2_common::Vector position,
232  rm2_common::Rotation orientation) -> geometry_msgs::msg::Pose {
233  geometry_msgs::msg::Pose result;
234  result.position.set__x(position.x());
235  result.position.set__y(position.y());
236  result.position.set__z(position.z());
237  result.orientation.set__w(orientation.w());
238  result.orientation.set__x(orientation.x());
239  result.orientation.set__y(orientation.y());
240  result.orientation.set__z(orientation.z());
241  return result;
242  };
243 
244  geometry_msgs::msg::PoseStamped back_pose;
245  back_pose.header.stamp = node_->now();
246  back_pose.header.frame_id = robot_handler_->get_robot_model().get_base_component()->first;
247  back_pose.pose = trans_pose_to_msg(
248  target_task_value.pose.position,
249  target_task_value.pose.orientation);
250  back_pose_pub_->publish(back_pose);
252  }
253 
254  if (!robot_handler_->set_goal_task_value(detected_component_name, target_task_value)) {
255  RCLCPP_ERROR_STREAM(node_->get_logger(), "Fail to set Task values!!!!!!");
256  is_checking_ = false;
257  }
258  return false;
259  }
260  }
261  return true;
262 }
