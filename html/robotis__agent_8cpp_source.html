43  }

45 
46  init_ros_publisher();
47  init_ros_subscriber();

49  is_maintenance_mode_ = read_maintenance_mode_file();

56 void RobotisAgent::timer_callback() // 1s
57 {
58  // monitoring data (1s)
59  if (monitoring_data_.docked_status) {
60  if (docked_5s_cnt_ % 5 == 0) {
61  pub_monitoring_data();
62  }
63  docked_5s_cnt_++;
64  } else {
65  docked_5s_cnt_ = 0;
66  pub_monitoring_data();
67  }
68 
69  bool maintenance_mode = read_maintenance_mode_file();
70  if (is_maintenance_mode_ != maintenance_mode) {
71  is_maintenance_mode_ = maintenance_mode;
72  if (is_maintenance_mode_) {
73  publish_log("점검 시작", aimbot_log_interfaces::msg::Log::TYPE_INFO);
74  } else {
75  publish_log("점검 종료", aimbot_log_interfaces::msg::Log::TYPE_INFO);
76  }
77  }
78 }

91 void RobotisAgent::init_ros_subscriber()
92 {
94  // Monitoring data
95  // EMO
96  estop_state_sub_ = create_subscription<std_msgs::msg::Bool>(
97  "/aimbot_base/estop_status", rclcpp::SystemDefaultsQoS(),
98  [this](const std_msgs::msg::Bool::SharedPtr msg) -> void
99  {
100  this->monitoring_data_.emergency_status = msg->data;
101  });
102 
103  // Battery
104  battery_state_sub_ = create_subscription<sensor_msgs::msg::BatteryState>(
105  "/aimbot_base/battery_state", rclcpp::SystemDefaultsQoS(),
106  [this](
107  const sensor_msgs::msg::BatteryState::SharedPtr msg) -> void
108  {
109  this->monitoring_data_.battery_percentage = msg->percentage;
110  switch (msg->power_supply_status) {
111  case sensor_msgs::msg::BatteryState::POWER_SUPPLY_STATUS_CHARGING:
112  this->monitoring_data_.docked_status = true;
113  break;
114  case sensor_msgs::msg::BatteryState::POWER_SUPPLY_STATUS_UNKNOWN: // docking
115  this->monitoring_data_.docked_status = true;
116  break;
117  case sensor_msgs::msg::BatteryState::POWER_SUPPLY_STATUS_NOT_CHARGING:
118  this->monitoring_data_.docked_status = false;
119  break;
120  default:
121  this->monitoring_data_.docked_status = false;
122  break;
123  }
124  });
125 
126  // Motion State
127  motion_state_sub_ = create_subscription<aimbot_state_interfaces::msg::MotionState>(
128  "/motion_state", rclcpp::SystemDefaultsQoS(),
129  [this](const aimbot_state_interfaces::msg::MotionState::SharedPtr msg) -> void
130  {
131  this->monitoring_data_.position_type = "SLAM";
132  this->monitoring_data_.area_id = msg->floor;
133  this->monitoring_data_.pose_from_map_current = msg->pose_from_map_current;
134  });
135 
136  // Odom
137  odom_sub_ = create_subscription<nav_msgs::msg::Odometry>(
138  "/odom", rclcpp::SystemDefaultsQoS(),
139  [this](const nav_msgs::msg::Odometry::SharedPtr msg) -> void
140  {
141  this->monitoring_data_.linear_velocity = msg->twist.twist.linear.x;
142  this->monitoring_data_.angular_velocity = msg->twist.twist.angular.z;
143  });
144 
145  // Ultrasonic
146  for (size_t i = 0; i < ultrasonic_list_.size(); i++) {
147  monitoring_data_.ultrasonic_angle.at(i) = ultrasonic_angle_.at(i) * M_PIf32 / 180.0f;
148  ultrasonic_sub_.at(i) = create_subscription<sensor_msgs::msg::Range>(
149  ultrasonic_topic_name_.at(i), rclcpp::SystemDefaultsQoS(),
150  [this, i](const sensor_msgs::msg::Range::SharedPtr msg) -> void
151  {this->monitoring_data_.ultrasonic_range.at(i) = msg->range;});
152  }
153 
154  // Lidar
155  lidar_sub_ = create_subscription<sensor_msgs::msg::LaserScan>(
156  "/scan", rclcpp::SystemDefaultsQoS(),
157  [this](const sensor_msgs::msg::LaserScan::SharedPtr msg) -> void
158  {
159  monitoring_data_.lidar_angle.clear();
160  monitoring_data_.lidar_range.clear();
161  // left to Right CW
162  for (float angle = -M_PIf32; angle < M_PIf32; angle += 45.0f * M_PIf32 / 180.0f) {
163  float index = (angle - msg->angle_min) / msg->angle_increment;
164  if (index > 0.0f && index < msg->ranges.size()) {
165  monitoring_data_.lidar_angle.push_back(angle);
166  monitoring_data_.lidar_range.push_back(msg->ranges.at(static_cast<size_t>(index)));
167  }
168  }
169  });
170 
171  // Cargo
172  cargo_door_state_sub_ = create_subscription<aimbot_hw_interfaces::msg::CargoDoorStatus>(
173  "/aimbot_base/cargo_door_status", rclcpp::SystemDefaultsQoS(),
174  [this](const aimbot_hw_interfaces::msg::CargoDoorStatus::SharedPtr msg) -> void
175  {
176  this->monitoring_data_.tray_name.clear();
177  this->monitoring_data_.tray_locked.clear();
178  for (size_t index = 0; index < msg->state.size(); index++) {
179  this->monitoring_data_.tray_name.push_back("TRAY" + std::to_string(index + 1));
180  this->monitoring_data_.tray_locked.push_back(
181  msg->state.at(index) == aimbot_hw_interfaces::msg::CargoDoorStatus::CLOSE);
182  }
183  });
184 
185  // task list
186  task_list_sub_ = create_subscription<task_manager_interfaces::msg::TaskListItems>(
187  "/service_manager/task_list", rclcpp::SystemDefaultsQoS(),
188  [this](const task_manager_interfaces::msg::TaskListItems::SharedPtr msg) -> void
189  {
190  this->task_list_ = *msg;
191  });
192 
193  // Monitoring data
195  log_pub_ = this->create_publisher<aimbot_log_interfaces::msg::Log>("/aimbot_log", 30);
196 
197  maintenance_mode_sub_ = create_subscription<std_msgs::msg::Bool>(
198  "/maintenance_mode", rclcpp::SystemDefaultsQoS(),
199  [this](const std_msgs::msg::Bool::SharedPtr msg) -> void
200  {
201  write_maintenance_mode(msg->data);
202  });
203 }

206 void RobotisAgent::pub_monitoring_data()
207 {
208  monitoring_data_.header.stamp = now();
209 
210  if (monitoring_data_.emergency_status == true) {
211  // ERROR : 장애
212  monitoring_data_.robot_status =
213  robotis_server_interfaces::msg::MonitoringData::ROBOT_STATUS_ERROR;
214  } else if (task_list_.tasks.size() != 0) {
215  if (task_list_.tasks.at(0).task.task_name == "BACK_TO_BASE") {
216  // RETURNING : 복귀 중
217  monitoring_data_.robot_status =
218  robotis_server_interfaces::msg::MonitoringData::ROBOT_STATUS_RETURNING;
219  } else {
220  // PERFORMING_TASK : 태스크 수행 중
221  monitoring_data_.robot_status =
222  robotis_server_interfaces::msg::MonitoringData::ROBOT_STATUS_PERFORMING_TASK;
223  }
224  } else {
225  // READY : 대기 중
226  monitoring_data_.robot_status =
227  robotis_server_interfaces::msg::MonitoringData::ROBOT_STATUS_READY;
228  }
229 
230  monitoring_data_.maintenance_mode = read_maintenance_mode_file();
231 
232  // additional fields 필요시 추가
233  monitoring_data_pub_->publish(monitoring_data_);
234 }

236 bool RobotisAgent::read_maintenance_mode_file()
237 {
238  try {
239  std::string mode_file_path = ament_index_cpp::get_package_share_directory(
240  "robotis_agent") +
241  "/states/mode.yaml";
242 
243  // Node
244  YAML::Node mode_node = YAML::LoadFile(mode_file_path.c_str());
245 
246  return mode_node["maintenance_mode"].as<bool>();
247  } catch (const std::exception & e) {
248  RCLCPP_ERROR_STREAM(
249  this->get_logger(),
250  "Fail to load aimbot state config data. : " << e.what());
251  return false;
252  }

255 bool RobotisAgent::write_maintenance_mode(bool maintenance_mode)
256 {
257  try {
258  std::string mode_file_path = ament_index_cpp::get_package_share_directory(
259  "robotis_agent") +
260  "/states/mode.yaml";
261 
262  YAML::Emitter yaml_out;
263  yaml_out << YAML::BeginMap;
264  yaml_out <<
265  YAML::Key << "maintenance_mode" <<
266  YAML::Value << maintenance_mode <<
267  YAML::EndMap;
268 
269  std::ofstream fout(mode_file_path.c_str());
270  fout << yaml_out.c_str();
271  } catch (const std::exception & e) {
272  std::stringstream log_stream;
273  log_stream << "Fail to wirte mode data. : " << e.what();
274  RCLCPP_ERROR_STREAM(this->get_logger(), log_stream.str());
275  return false;
276  }
277  return true;
