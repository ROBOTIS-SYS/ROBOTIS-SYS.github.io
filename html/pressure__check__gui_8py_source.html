41 class ButterworthFilter:
42  # 로우 패스 버터워스 필터를 설계합니다.
43  def __init__(self, cutoff_frequency, sampling_frequency, order=5):
44  nyq = 0.5 * sampling_frequency
45  normal_cutoff = cutoff_frequency / nyq
46  self.b, self.a = butter(order, normal_cutoff, btype='low', analog=False)
47 
48  # 데이터에 로우 패스 필터를 적용합니다.
49  def apply(self, input_value):
50  return filtfilt(self.b, self.a, input_value)
