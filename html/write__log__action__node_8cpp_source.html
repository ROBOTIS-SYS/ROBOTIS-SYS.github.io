69 BT::NodeStatus WriteLogAction::tick()
70 {
71  std::cout << std::endl;
72  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
73  RCLCPP_INFO_STREAM(get_logger(), "<TICK> WriteLogAction");
74  }
75 
76  has_battery_state_ = false;
77 
78  std::string log_file_name;
79  if (!getInput<std::string>("log_file_name", log_file_name)) {
80  RCLCPP_ERROR(get_logger(), "There is no file name");
81  return BT::NodeStatus::FAILURE;
82  }
83 
84  double check_time = 0.5;
85 // if (!getInput<double>("check_time", check_time)) {
86 // check_time = 1.0;
87 // }
88 
89  // s -> ms
90  check_time *= 1000.0;
91  double loop_rate = 100.0; // ms
92  int loop_count = 0;
93 
94  rclcpp::WallRate loopRate(std::chrono::milliseconds(100));
95 
96  while (rclcpp::ok() && has_battery_state_ == false) {
97  // sleep
98  loopRate.sleep();
99 
100  if ((loop_rate * ++loop_count) >= check_time) {
101  break;
102  }
103  }
104 
105  std::string target_name, target_floor, current_floor, message;
106  getInput<std::string>("target_name", target_name);
107  getInput<std::string>("target_floor", target_floor);
108  getInput<std::string>("current_floor", current_floor);
109  getInput<std::string>("message", message);
110 
111  bool result = log_file(log_file_name, target_name, target_floor, current_floor, message);
112  result = publish_log(target_name, target_floor, current_floor, message);
113 
114  return result ? BT::NodeStatus::SUCCESS : BT::NodeStatus::FAILURE;
115 }

117 void WriteLogAction::battery_state_callback(
118  const sensor_msgs::msg::BatteryState::SharedPtr msg)
119 {
120  if (status() != BT::NodeStatus::RUNNING) {
121  return;
122  }
123 
124  if (has_battery_state_ == true) {
125  return;
126  }
127 
128  battery_voltage_ = static_cast<double>(msg->voltage) * 1e-2;
129  battery_status_ =
130  (msg->power_supply_status ==
131  sensor_msgs::msg::BatteryState::POWER_SUPPLY_STATUS_CHARGING) ? "Charging" : "Not Charging";
132 
133  has_battery_state_ = true;
134 }
