55 bool Engine::build(std::string onnxModelPath)
56 {
57  engine_name_ = serializeEngineOptions(options_);
58  if (doesFileExist(engine_name_)) {
59  return true;
60  }
61  auto builder =
62  std::unique_ptr<nvinfer1::IBuilder>(nvinfer1::createInferBuilder(logger_));
63  if (!builder) {
64  return false;
65  }
66  builder->setMaxBatchSize(options_.maxBatchSize);
67  auto explicitBatch =
68  1U << static_cast<uint32_t>(nvinfer1::NetworkDefinitionCreationFlag::kEXPLICIT_BATCH);
69  auto network =
70  std::unique_ptr<nvinfer1::INetworkDefinition>(builder->createNetworkV2(explicitBatch));
71  if (!network) {
72  return false;
73  }
74  auto parser =
75  std::unique_ptr<nvonnxparser::IParser>(nvonnxparser::createParser(*network, logger_));
76  if (!parser) {
77  return false;
78  }
79  std::ifstream file(onnxModelPath, std::ios::binary | std::ios::ate);
80  std::streamsize size = file.tellg();
81  file.seekg(0, std::ios::beg);
82  std::vector<char> buffer(size);
83  if (!file.read(buffer.data(), size)) {
84  throw std::runtime_error("Unable to read engine file");
85  }
86  auto parsed = parser->parse(buffer.data(), buffer.size());
87  if (!parsed) {
88  return false;
89  }
90  const auto input = network->getInput(0);
91  const auto output = network->getOutput(0);
92  const auto inputName = input->getName();
93  const auto inputDims = input->getDimensions();
94  int32_t inputC = inputDims.d[1];
95  int32_t inputH = inputDims.d[2];
96  int32_t inputW = inputDims.d[3];
97 
98  auto config = std::unique_ptr<nvinfer1::IBuilderConfig>(builder->createBuilderConfig());
99  if (!config) {
100  return false;
101  }
102  nvinfer1::IOptimizationProfile * defaultProfile = builder->createOptimizationProfile();
103  defaultProfile->setDimensions(
104  inputName,
105  nvinfer1::OptProfileSelector::kMIN,
106  nvinfer1::Dims4(1, inputC, inputH, inputW));
107  defaultProfile->setDimensions(
108  inputName,
109  nvinfer1::OptProfileSelector::kOPT,
110  nvinfer1::Dims4(1, inputC, inputH, inputW));
111  defaultProfile->setDimensions(
112  inputName,
113  nvinfer1::OptProfileSelector::kMAX,
114  nvinfer1::Dims4(options_.maxBatchSize, inputC, inputH, inputW));
115  config->addOptimizationProfile(defaultProfile);
116 
117  for (const auto & optBatchSize : options_.optBatchSizes) {
118  if (optBatchSize == 1) {
119  continue;
120  }
121  if (optBatchSize > options_.maxBatchSize) {
122  throw std::runtime_error("optBatchSize cannot be greater than maxBatchSize!");
123  }
124  nvinfer1::IOptimizationProfile * profile = builder->createOptimizationProfile();
125  profile->setDimensions(
126  inputName,
127  nvinfer1::OptProfileSelector::kMIN,
128  nvinfer1::Dims4(1, inputC, inputH, inputW));
129  profile->setDimensions(
130  inputName,
131  nvinfer1::OptProfileSelector::kOPT,
132  nvinfer1::Dims4(optBatchSize, inputC, inputH, inputW));
133  profile->setDimensions(
134  inputName,
135  nvinfer1::OptProfileSelector::kMAX,
136  nvinfer1::Dims4(options_.maxBatchSize, inputC, inputH, inputW));
137  config->addOptimizationProfile(profile);
138  }
139  config->setMaxWorkspaceSize(options_.maxWorkspaceSize);
140  if (options_.FP16) {
141  config->setFlag(nvinfer1::BuilderFlag::kFP16);
142  }
143 
144  auto profileStream = samplesCommon::makeCudaStream();
145  if (!profileStream) {
146  return false;
147  }
148  config->setProfileStream(*profileStream);
149 
150  std::unique_ptr<nvinfer1::IHostMemory> plan{builder->buildSerializedNetwork(*network, *config)};
151  if (!plan) {
152  return false;
153  }
154  std::ofstream outfile(engine_name_, std::ofstream::binary);
155  outfile.write(reinterpret_cast<const char *>(plan->data()), plan->size());
156  return true;
157 }

159 int Engine::loadNetwork(
160  std::string model_name,
161  std::shared_ptr<std::vector<samplesCommon::ManagedBuffer>> & output_tensor_buffer)
162 {
163  std::ifstream file(model_name, std::ios::binary | std::ios::ate);
164  std::streamsize size = file.tellg();
165  file.seekg(0, std::ios::beg);
166  std::vector<char> buffer(size);
167  if (!file.read(buffer.data(), size)) {
168  return PERCEPTION_STATE::ERROR_MODEL_READ;
169  }
170  initLibNvInferPlugins(&logger_, "");
171  std::unique_ptr<nvinfer1::IRuntime> runtime{nvinfer1::createInferRuntime(logger_)};
172  if (!runtime) {
173  return PERCEPTION_STATE::ERROR_MODEL_RUNTIME;
174  }
175  auto ret = cudaSetDevice(options_.deviceIndex);
176  if (ret != 0) {
177  int numGPUs;
178  cudaGetDeviceCount(&numGPUs);
179  auto errMsg = "Unable to set GPU device index to: " + std::to_string(options_.deviceIndex) +
180  ". Note, your device has " + std::to_string(numGPUs) + " CUDA-capable GPU(s).";
181  return PERCEPTION_STATE::ERROR_CUDA_NO_DEVICE;
182  }
183  engine_ = std::unique_ptr<nvinfer1::ICudaEngine>(
184  runtime->deserializeCudaEngine(buffer.data(), buffer.size()));
185  if (!engine_) {
186  return PERCEPTION_STATE::ERROR_ENGINE_INVALID_DESERIALIZE;
187  }
188  context_ = std::unique_ptr<nvinfer1::IExecutionContext>(
189  engine_->createExecutionContext());
190  if (!context_) {
191  return PERCEPTION_STATE::ERROR_ENGINE_INVALID_CONTEXT;
192  }
193  for (int i = 0; i < engine_->getNbBindings(); i++) {
194  samplesCommon::ManagedBuffer tmp_output_buff;
195  auto dims = engine_->getBindingDimensions(i);
196  if (engine_->bindingIsInput(i)) {
197  nvinfer1::Dims4 inputDims = {dims.d[0], dims.d[1], dims.d[2], dims.d[3]};
198  context_->setBindingDimensions(0, inputDims);
199  for (size_t j = 0; j < dims.nbDims; ++j) {
200  input_dim_.push_back(dims.d[j]);
201  }
202  } else {
203  std::vector<int> tmp_dims;
204  tmp_dims.reserve(dims.nbDims);
205  for (size_t j = 0; j < dims.nbDims; ++j) {
206  tmp_dims.push_back(dims.d[j]);
207  }
208  output_dim_.push_back(tmp_dims);
209  }
210  tmp_output_buff.hostBuffer.resize(dims);
211  tmp_output_buff.deviceBuffer.resize(dims);
212  int test = 0;
213  output_tensor_buffer->emplace_back(std::move(tmp_output_buff));
214  }
215  buffer.clear();
216  return PERCEPTION_STATE::SUCCESS_NO_ISSUE;
217 }

304 std::string Engine::serializeEngineOptions(const Options & options)
305 {
306  std::string engineName = "trt.engine";
307 
308  std::vector<std::string> gpuUUIDs;
309  getGPUUUIDs(gpuUUIDs);
310  if (static_cast<size_t>(options.deviceIndex) >= gpuUUIDs.size()) {
311  throw std::runtime_error("Error, provided device index is out of range!");
312  }
313  engineName += "." + gpuUUIDs[options.deviceIndex];
314  // Serialize the specified options into the filename
315  if (options.FP16) {
316  engineName += ".fp16";
317  } else {
318  engineName += ".fp32";
319  }
320  engineName += "." + std::to_string(options.maxBatchSize) + ".";
321  for (size_t i = 0; i < options_.optBatchSizes.size(); ++i) {
322  engineName += std::to_string(options_.optBatchSizes[i]);
323  if (i != options_.optBatchSizes.size() - 1) {
324  engineName += "_";
325  }
326  }
327  engineName += "." + std::to_string(options.maxWorkspaceSize);
328  return engineName;
329 }
