129 SafetyFilter::FilterResult SafetyFilter::filtering_cmd_vel(
130  geometry_msgs::msg::TwistStamped raw_data)
131 {
132  FilterResult filtered_result;
133  geometry_msgs::msg::TwistStamped filtered_cmd_vel;
134 
135  // Init
136  filtered_result.cmd_vel = raw_data;
137  filtered_result.decelration = false;
138  filtered_result.clash_time = params_.check_time;
139  filtered_result.deceleration_rate = 1.0;
140 
141  if (safety_enable_ &&
142  (filtered_result.cmd_vel.twist.linear.x != 0.0 ||
143  filtered_result.cmd_vel.twist.angular.z != 0.0))
144  {
145  // Intersection of circles
146  double v = filtered_result.cmd_vel.twist.linear.x;
147  double w = filtered_result.cmd_vel.twist.angular.z;
148 
149  if (v >= 0.0) { // Front Side
150  // front right
151  UltrasonicData::ClashResult result_fr =
152  ultrasonic_front_right_.get_min_clash_time(v, w, params_.check_time);
153  if (params_.check_time > result_fr.clash_time) {
154  filtered_result.detected_sensor_data.insert(
155  std::make_pair(
156  "FR-" + result_fr.clash_circle_name,
157  ultrasonic_front_right_.range_));
158  std::cout <<
159  "FR-" << result_fr.clash_circle_name << ":" << result_fr.clash_time <<
160  "(" << ultrasonic_front_right_.range_ << ")";
161  }
162  // front left
163  UltrasonicData::ClashResult result_fl =
164  ultrasonic_front_left_.get_min_clash_time(v, w, params_.check_time);
165  if (params_.check_time > result_fl.clash_time) {
166  filtered_result.detected_sensor_data.insert(
167  std::make_pair(
168  "FL-" + result_fl.clash_circle_name,
169  ultrasonic_front_left_.range_));
170  std::cout <<
171  " FL-" << result_fl.clash_circle_name << ":" << result_fl.clash_time <<
172  "(" << ultrasonic_front_left_.range_ << ")";
173  }
174  // right
175  UltrasonicData::ClashResult result_r =
176  ultrasonic_right_.get_min_clash_time(v, w, params_.check_time);
177  if (params_.check_time > result_r.clash_time) {
178  filtered_result.detected_sensor_data.insert(
179  std::make_pair(
180  "R-" + result_r.clash_circle_name,
181  ultrasonic_right_.range_));
182  std::cout <<
183  " R-" << result_r.clash_circle_name << ":" << result_r.clash_time <<
184  "(" << ultrasonic_right_.range_ << ")";
185  }
186  // left
187  UltrasonicData::ClashResult result_l =
188  ultrasonic_left_.get_min_clash_time(v, w, params_.check_time);
189  if (params_.check_time > result_l.clash_time) {
190  filtered_result.detected_sensor_data.insert(
191  std::make_pair(
192  "L-" + result_l.clash_circle_name,
193  ultrasonic_left_.range_));
194  std::cout <<
195  " L-" << result_l.clash_circle_name << ":" << result_l.clash_time <<
196  "(" << ultrasonic_left_.range_ << ")";
197  }
198 
199 
200  filtered_result.clash_time =
201  std::min(
202  std::min(result_r.clash_time, result_l.clash_time),
203  std::min(result_fr.clash_time, result_fl.clash_time));
204  } else if (v < 0.0) { // Back Side
205  // Back
206  UltrasonicData::ClashResult result_b =
207  ultrasonic_back_.get_min_clash_time(v, w, params_.check_time);
208  if (params_.check_time > result_b.clash_time) {
209  filtered_result.detected_sensor_data.insert(
210  std::make_pair(
211  "B-" + result_b.clash_circle_name,
212  ultrasonic_back_.range_));
213  std::cout <<
214  " B-" << result_b.clash_circle_name << ":" << result_b.clash_time <<
215  "(" << ultrasonic_back_.range_ << ")";
216  }
217  // back right
218  UltrasonicData::ClashResult result_br =
219  ultrasonic_back_right_.get_min_clash_time(v, w, params_.check_time);
220  if (params_.check_time > result_br.clash_time) {
221  filtered_result.detected_sensor_data.insert(
222  std::make_pair(
223  "BR-" + result_br.clash_circle_name,
224  ultrasonic_back_right_.range_));
225  std::cout <<
226  " BR-" << result_br.clash_circle_name << ":" << result_br.clash_time <<
227  "(" << ultrasonic_back_right_.range_ << ")";
228  }
229  // back left
230  UltrasonicData::ClashResult result_bl =
231  ultrasonic_back_left_.get_min_clash_time(v, w, params_.check_time);
232  if (params_.check_time > result_bl.clash_time) {
233  filtered_result.detected_sensor_data.insert(
234  std::make_pair(
235  "BL-" + result_bl.clash_circle_name,
236  ultrasonic_back_left_.range_));
237  std::cout <<
238  " BL-" << result_bl.clash_circle_name << ":" << result_bl.clash_time <<
239  "(" << ultrasonic_back_left_.range_ << ")";
240  }
241  filtered_result.clash_time =
242  std::min(result_b.clash_time, std::min(result_br.clash_time, result_bl.clash_time));
243  }
244 
245  // Deccelation
246  if (params_.stop_time >= filtered_result.clash_time) {
247  std::cout << " <" << v << "," << w << ">->";
248  v = 0.0;
249  w = 0.0;
250  filtered_result.deceleration_rate = 0.0;
251  filtered_result.decelration = true;
252  std::cout << " <" << v << "," << w << ">" << std::endl;
253  } else if (params_.check_time > filtered_result.clash_time) {
254  std::cout << " <" << v << "," << w << ">->";
255  v = (filtered_result.clash_time / params_.check_time) * v;
256  w = (filtered_result.clash_time / params_.check_time) * w;
257  filtered_result.deceleration_rate = filtered_result.clash_time / params_.check_time;
258  filtered_result.decelration = true;
259  std::cout << " <" << v << "," << w << ">" << std::endl;
260  }
261  filtered_result.cmd_vel.twist.linear.x = v;
262  filtered_result.cmd_vel.twist.angular.z = w;
263  }
264  return filtered_result;
265 }
