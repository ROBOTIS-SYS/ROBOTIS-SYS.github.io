67 FootprintSubscriber::getFootprint(
68  std::vector<geometry_msgs::msg::Point> & footprint,
69  rclcpp::Time & stamp,
70  rclcpp::Duration valid_footprint_timeout)
71 {
72  if (!footprint_received_) {
73  return false;
74  }
75 
76  footprint = toPointVector(
77  std::make_shared<geometry_msgs::msg::Polygon>(footprint_->polygon));
78  auto & footprint_stamp = footprint_->header.stamp;
79 
80  if (stamp - footprint_stamp > valid_footprint_timeout) {
81  return false;
82  }
83 
84  return true;
85 }
