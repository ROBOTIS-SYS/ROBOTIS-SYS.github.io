66 void CostmapSubscriber::toCostmap2D()
67 {
68  if (costmap_ == nullptr) {
69  costmap_ = std::make_shared<Costmap2D>(
70  costmap_msg_->metadata.size_x, costmap_msg_->metadata.size_y,
71  costmap_msg_->metadata.resolution, costmap_msg_->metadata.origin.position.x,
72  costmap_msg_->metadata.origin.position.y);
73  } else if (costmap_->getSizeInCellsX() != costmap_msg_->metadata.size_x || // NOLINT
74  costmap_->getSizeInCellsY() != costmap_msg_->metadata.size_y ||
75  costmap_->getResolution() != costmap_msg_->metadata.resolution ||
76  costmap_->getOriginX() != costmap_msg_->metadata.origin.position.x ||
77  costmap_->getOriginY() != costmap_msg_->metadata.origin.position.y)
78  {
79  // Update the size of the costmap
80  costmap_->resizeMap(
81  costmap_msg_->metadata.size_x, costmap_msg_->metadata.size_y,
82  costmap_msg_->metadata.resolution,
83  costmap_msg_->metadata.origin.position.x,
84  costmap_msg_->metadata.origin.position.y);
85  }
86 
87  unsigned char * master_array = costmap_->getCharMap();
88  unsigned int index = 0;
89  for (unsigned int i = 0; i < costmap_msg_->metadata.size_x; ++i) {
90  for (unsigned int j = 0; j < costmap_msg_->metadata.size_y; ++j) {
91  master_array[index] = costmap_msg_->data[index];
92  ++index;
93  }
94  }
95 }
