155  inline bool bitsBelowThreshold(unsigned int n, unsigned int bit_threshold)
156  {
157  unsigned int bit_count;
158  for (bit_count = 0; n; ) {
159  ++bit_count;
160  if (bit_count > bit_threshold) {
161  return false;
162  }
163  n &= n - 1; // clear the least significant bit set
164  }
165  return true;
166  }

168  static inline unsigned int numBits(unsigned int n)
169  {
170  unsigned int bit_count;
171  for (bit_count = 0; n; ++bit_count) {
172  n &= n - 1; // clear the least significant bit set
173  }
174  return bit_count;
175  }

177  static VoxelStatus getVoxel(
178  unsigned int x, unsigned int y, unsigned int z,
179  unsigned int size_x, unsigned int size_y, unsigned int size_z, const uint32_t * data)
180  {
181  if (x >= size_x || y >= size_y || z >= size_z) {
182  return UNKNOWN;
183  }
184  uint32_t full_mask = ((uint32_t)1 << z << 16) | (1 << z);
185  uint32_t result = data[y * size_x + x] & full_mask;
186  unsigned int bits = numBits(result);
187 
188  // known marked: 11 = 2 bits, unknown: 01 = 1 bit, known free: 00 = 0 bits
189  if (bits < 2) {
190  if (bits < 1) {
191  return FREE;
192  }
193  return UNKNOWN;
194  }
195  return MARKED;
196  }

223  template<class ActionType>
224  inline void raytraceLine(
225  ActionType at, double x0, double y0, double z0,
226  double x1, double y1, double z1, unsigned int max_length = UINT_MAX)
227  {
228  int dx = int(x1) - int(x0); // NOLINT
229  int dy = int(y1) - int(y0); // NOLINT
230  int dz = int(z1) - int(z0); // NOLINT
231 
232  unsigned int abs_dx = abs(dx);
233  unsigned int abs_dy = abs(dy);
234  unsigned int abs_dz = abs(dz);
235 
236  int offset_dx = sign(dx);
237  int offset_dy = sign(dy) * size_x_;
238  int offset_dz = sign(dz);
239 
240  unsigned int z_mask = ((1 << 16) | 1) << (unsigned int)z0;
241  unsigned int offset = (unsigned int)y0 * size_x_ + (unsigned int)x0;
242 
243  GridOffset grid_off(offset);
244  ZOffset z_off(z_mask);
245 
246  // we need to chose how much to scale our dominant dimension, based on the
247  // maximum length of the line
248  double dist = sqrt((x0 - x1) * (x0 - x1) + (y0 - y1) * (y0 - y1) + (z0 - z1) * (z0 - z1));
249  double scale = std::min(1.0, max_length / dist);
250 
251  // is x dominant
252  if (abs_dx >= max(abs_dy, abs_dz)) {
253  int error_y = abs_dx / 2;
254  int error_z = abs_dx / 2;
255 
256  bresenham3D(
257  at, grid_off, grid_off, z_off, abs_dx, abs_dy, abs_dz, error_y, error_z,
258  offset_dx, offset_dy, offset_dz, offset, z_mask, (unsigned int)(scale * abs_dx));
259  return;
260  }
261 
262  // y is dominant
263  if (abs_dy >= abs_dz) {
264  int error_x = abs_dy / 2;
265  int error_z = abs_dy / 2;
266 
267  bresenham3D(
268  at, grid_off, grid_off, z_off, abs_dy, abs_dx, abs_dz, error_x, error_z,
269  offset_dy, offset_dx, offset_dz, offset, z_mask, (unsigned int)(scale * abs_dy));
270  return;
271  }
272 
273  // otherwise, z is dominant
274  int error_x = abs_dz / 2;
275  int error_y = abs_dz / 2;
276 
277  bresenham3D(
278  at, z_off, grid_off, grid_off, abs_dz, abs_dx, abs_dy, error_x, error_y, offset_dz,
279  offset_dx, offset_dy, offset, z_mask, (unsigned int)(scale * abs_dz));
280  }

282 private:
283  // the real work is done here... 3D bresenham implementation
284  template<class ActionType, class OffA, class OffB, class OffC>
285  inline void bresenham3D(
286  ActionType at, OffA off_a, OffB off_b, OffC off_c,
287  unsigned int abs_da, unsigned int abs_db, unsigned int abs_dc,
288  int error_b, int error_c, int offset_a, int offset_b, int offset_c, unsigned int & offset,
289  unsigned int & z_mask, unsigned int max_length = UINT_MAX)
290  {
291  unsigned int end = std::min(max_length, abs_da);
292  for (unsigned int i = 0; i < end; ++i) {
293  at(offset, z_mask);
294  off_a(offset_a);
295  error_b += abs_db;
296  error_c += abs_dc;
297  if ((unsigned int)error_b >= abs_da) {
298  off_b(offset_b);
299  error_b -= abs_da;
300  }
301  if ((unsigned int)error_c >= abs_da) {
302  off_c(offset_c);
303  error_c -= abs_da;
304  }
305  }
306  at(offset, z_mask);
307  }
