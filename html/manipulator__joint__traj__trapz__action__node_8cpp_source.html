35 BT::NodeStatus ManipulatorJointTrajTrapzAction::on_tick()
36 {
37  std::vector<std::string> joint_name;
38  if (!getInput<std::vector<std::string>>("joint_name_arr", joint_name)) {
39  RCLCPP_ERROR_STREAM(node_->get_logger(), "Joint name arr is empty!!");
40  return BT::NodeStatus::FAILURE;
41  }
42  std::vector<double> joint_goal;
43  if (!getInput<std::vector<double>>("joint_goal_rad_arr", joint_goal)) {
44  RCLCPP_ERROR_STREAM(node_->get_logger(), "Joint goal arr is empty!!");
45  return BT::NodeStatus::FAILURE;
46  }
47  double velocity;
48  if (!getInput<double>("velocity", velocity)) {
49  RCLCPP_ERROR_STREAM(node_->get_logger(), "Trajectory velocity is empty!!");
50  return BT::NodeStatus::FAILURE;
51  }
52  double acceleration;
53  if (!getInput<double>("acceleration", acceleration)) {
54  RCLCPP_ERROR_STREAM(node_->get_logger(), "Trajectory acceleration is empty!!");
55  return BT::NodeStatus::FAILURE;
56  }
57  if (joint_name.size() != joint_goal.size()) {
58  RCLCPP_ERROR_STREAM(
59  node_->get_logger(),
60  "Number of Joint goal angles is not equal with Number of Joint names!!");
61  return BT::NodeStatus::FAILURE;
62  }
63 
64  // Set Ros action configure
65  std::string manipulator_namespace = "gc_manipulator";
66  std::string action_name = manipulator_namespace +
67  "/manipulator_controller/joint_trajectory_trapezoidal";
68 
69  // Create Action [.action]
70  // set Action Timeout = 10s
71  std::chrono::milliseconds action_timeout = std::chrono::seconds(10);
72  if (!create_action_client(action_name, action_timeout)) {
73  return BT::NodeStatus::FAILURE;
74  }
75 
76  // Log [.msg]
77  try {
78  bool on_blackboard = config().blackboard->get<decltype(log_pub_)>("/aimbot_log", log_pub_);
79  if (on_blackboard == false) {
80  log_pub_ = node_->create_publisher<aimbot_log_interfaces::msg::Log>("/aimbot_log", 10);
81  config().blackboard->set<decltype(log_pub_)>("/aimbot_log", log_pub_);
82  }
83  } catch (std::exception & ex) {
84  RCLCPP_ERROR_STREAM(
85  node_->get_logger(),
86  "Failed to create \"/aimbot_log\" action : " << ex.what());
87  return BT::NodeStatus::FAILURE;
88  }
89 
90  set_ros_spin_loop_rate(10);
91  goal_.header.stamp = rclcpp::Clock().now();
92  goal_.joint_names = joint_name;
93  goal_.goal_waypoint.positions = joint_goal;
94  std::vector<double> empty(joint_name.size());
95  goal_.goal_waypoint.velocities = empty;
96  goal_.goal_waypoint.accelerations = empty;
97  goal_.goal_waypoint.effort = empty;
98  goal_.profile.velocity = velocity;
99  goal_.profile.acceleration = acceleration;
100 
101  return BT::NodeStatus::RUNNING;
102 }

104 BT::NodeStatus ManipulatorJointTrajTrapzAction::on_success()
105 {
106  auto value = result_.result->result_state.val;
107  if (value == manipulator_interfaces::msg::Result::SUCCESS) {
108  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
109  RCLCPP_INFO_STREAM(node_->get_logger(), "[Joint Trapezoidal Trajectory] SUCCESS!!!");
110  }
111  publish_log("SUCCESS!!!");
112  return BT::NodeStatus::SUCCESS;
113  } else if (value == manipulator_interfaces::msg::Result::CANCELED) {
114  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
115  RCLCPP_INFO_STREAM(
116  node_->get_logger(),
117  "[Joint Trapezoidal Trajectory] canceled manipulator action.");
118  }
119  publish_log("CANCELED!");
120  return BT::NodeStatus::SUCCESS;
121  } else if (value == manipulator_interfaces::msg::Result::PREEMPTED) {
122  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
123  RCLCPP_INFO_STREAM(
124  node_->get_logger(), "[Joint Trapezoidal Trajectory] Accepted another manipulator action.");
125  }
126  publish_log(
127  "PREEMPTED : " + std::to_string(value),
128  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
129  aimbot_log_interfaces::msg::Log::TYPE_ERROR);
130  return BT::NodeStatus::SUCCESS;
131  } else {
132  RCLCPP_ERROR_STREAM(
133  node_->get_logger(), "[Joint Trapezoidal Trajectory] FAIL :( ERROR : " << value);
134  publish_log(
135  "ERROR : " + std::to_string(value),
136  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
137  aimbot_log_interfaces::msg::Log::TYPE_ERROR);
138  return BT::NodeStatus::FAILURE;
139  }
140 }
