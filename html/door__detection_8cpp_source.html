59 CallbackReturn DoorDetection::on_configure(const rclcpp_lifecycle::State & state)
60 {
61  (void)state;
62  RCLCPP_INFO_STREAM(get_logger(), "Configuring");
63 
64  // get parameter
65  check_time_ = get_parameter("check_time").as_double();
66  min_angle_ = get_parameter("min_angle").as_double() * M_PI / 180.0;
67  max_angle_ = get_parameter("max_angle").as_double() * M_PI / 180.0;
68  max_range_ = get_parameter("max_range").as_double(); // m
69  threshold_ = get_parameter("threshold").as_double(); // m
70  zone_angle_ = get_parameter("zone_angle").as_double() * M_PI / 180.0;
71  relation_angle_ = get_parameter("relation_angle").as_double() * M_PI / 180.0;
72  sim_3d_ = get_parameter("sim_3d").as_bool();
73  sample_num_ = static_cast<int>(get_parameter("sample_num").as_int());
74  lidar_height_ = get_parameter("lidar_height").as_double();
75  min_width_ = get_parameter("min_width").as_double();
76  opened_angle_ = get_parameter("opened_angle").as_double() * M_PI / 180.0;
77  robot_frame_id_ = get_parameter("transform.robot_frame_id").as_string();
78  lidar_frame_id_ = get_parameter("transform.lidar_frame_id").as_string();
79  max_door_distance_ = get_parameter("max_door_distance").as_double();
80  use_center_ = get_parameter("use_center").as_bool();
81 
82  double x, y, z, a, b, c;
83  x = get_parameter("transform.x").as_double();
84  y = get_parameter("transform.y").as_double();
85  z = get_parameter("transform.z").as_double();
86  a = get_parameter("transform.roll").as_double();
87  b = get_parameter("transform.pitch").as_double();
88  c = get_parameter("transform.yaw").as_double();
89 
90  Eigen::AngleAxisd roll_angle(a, Eigen::Vector3d::UnitX());
91  Eigen::AngleAxisd pitch_angle(b, Eigen::Vector3d::UnitY());
92  Eigen::AngleAxisd yaw_angle(c, Eigen::Vector3d::UnitZ());
93  Eigen::Quaternion<double> q = roll_angle * pitch_angle * yaw_angle;
94  Eigen::Matrix3d rot_mat = q.matrix();
95  Eigen::Vector3d trans_mat(x, y, z);
96  transformation_mat_.setIdentity(); // Set to Identity to make bottom row of Matrix 0,0,0,1
97  transformation_mat_.block<3, 3>(0, 0) = rot_mat;
98  transformation_mat_.block<3, 1>(0, 3) = trans_mat;
99 
100  laser_sub_ = create_subscription<sensor_msgs::msg::LaserScan>(
101  "/scan", rclcpp::SensorDataQoS(),
102  std::bind(&DoorDetection::laserscan_callback, this, std::placeholders::_1));
103 
104  door_normal_vector_sub_ = create_subscription<geometry_msgs::msg::Vector3>(
105  "/door_normal", rclcpp::SystemDefaultsQoS(),
106  std::bind(&DoorDetection::door_normal_vector_callback, this, std::placeholders::_1));
107  door_center_sub_ = create_subscription<geometry_msgs::msg::Pose2D>(
108  "/door_center", rclcpp::SystemDefaultsQoS(),
109  std::bind(&DoorDetection::door_center_pose_callback, this, std::placeholders::_1));
110 
111  marker_pub_ = create_publisher<visualization_msgs::msg::Marker>("door", 10);
112  door_polygon_pub_ = create_publisher<geometry_msgs::msg::PolygonStamped>("door_polygon", 10);
113  door_pub_ = create_publisher<perception_interfaces::msg::DetectedDoor>("detected_door", 10);
114 
115  tf_buffer_ = std::make_shared<tf2_ros::Buffer>(get_clock());
116  tf_listener_ = std::make_shared<tf2_ros::TransformListener>(*tf_buffer_);
117 
118  // get lidar position from tf
119  geometry_msgs::msg::TransformStamped lidar_link_transform;
120  auto trans_mat_backup = transformation_mat_;
121  try {
122  lidar_link_transform = tf_buffer_->lookupTransform(
123  robot_frame_id_, lidar_frame_id_, rclcpp::Time(0), rclcpp::Duration::from_seconds(1.0));
124 
125  geometry_msgs::msg::Pose lidar_pose;
126  lidar_pose.position.x = lidar_link_transform.transform.translation.x;
127  lidar_pose.position.y = lidar_link_transform.transform.translation.y;
128  lidar_pose.position.z = lidar_link_transform.transform.translation.z;
129  lidar_pose.orientation = lidar_link_transform.transform.rotation;
130 
131  Eigen::Quaternion<double> q;
132  Eigen::Vector3d trans_mat;
133  tf2::fromMsg(lidar_link_transform.transform.translation, trans_mat);
134  tf2::fromMsg(lidar_link_transform.transform.rotation, q);
135  Eigen::Matrix3d rot_mat = q.matrix();
136  transformation_mat_.setIdentity(); // Set to Identity to make bottom row of Matrix 0,0,0,1
137  transformation_mat_.block<3, 3>(0, 0) = rot_mat;
138  transformation_mat_.block<3, 1>(0, 3) = trans_mat;
139  RCLCPP_INFO_STREAM(get_logger(), "Transformation Mat : " << transformation_mat_);
140  } catch (tf2::TransformException & ex) {
141  RCLCPP_ERROR_STREAM(get_logger(), "Failed to get the current pose");
142  transformation_mat_ = trans_mat_backup;
143  }
144 
145  return CallbackReturn::SUCCESS;
146 }

219 bool DoorDetection::find_door(
220  const sensor_msgs::msg::LaserScan & scan,
221  perception_interfaces::msg::DetectedDoor & detected_door)
222 {
223  // get param
224  int dimension = 2;
225  if (sim_3d_ == true) {
226  dimension = 3;
227  }
228 
229  // divide zones : lidar 좌표계에서 구역 나눈 후 robot 좌표계(base_footprint)로 변환
230  std::vector<std::vector<Eigen::VectorXd>> zone_points;
231  int zone_num = static_cast<int>((max_angle_ - min_angle_) / zone_angle_);
232  int center_zone_index = 0;
233  for (int zone_index = 0; zone_index < zone_num; zone_index++) {
234  double start_angle = min_angle_ + door_direction_ + zone_angle_ * zone_index;
235  double end_angle = min_angle_ + door_direction_ + zone_angle_ * (zone_index + 1);
236  int start_index =
237  static_cast<int>((start_angle - static_cast<double>(scan.angle_min)) /
238  static_cast<double>(scan.angle_increment));
239  int end_index =
240  static_cast<int>((end_angle - static_cast<double>(scan.angle_min)) /
241  static_cast<double>(scan.angle_increment));
242  if (start_angle <= door_direction_ && end_angle >= door_direction_) {
243  center_zone_index = zone_index;
244  }
245  std::vector<Eigen::VectorXd> points;
246 
247  for (int ix = start_index; ix < end_index; ix++) {
248  double angle = static_cast<double>(scan.angle_min) +
249  static_cast<double>(scan.angle_increment) * static_cast<double>(ix);
250  if (angle < (min_angle_ + door_direction_) || angle > (max_angle_ + door_direction_)) {
251  continue;
252  }
253 
254  if (std::isnan(scan.ranges.at(static_cast<ulong>(ix))) ||
255  static_cast<double>(scan.ranges.at(static_cast<ulong>(ix))) >= max_range_)
256  {
257  continue;
258  }
259 
260  // change to point
261  Eigen::Vector4d lidar_coordinate, robot_coordinate;
262  lidar_coordinate.coeffRef(0) =
263  static_cast<double>(scan.ranges.at(static_cast<ulong>(ix))) * cos(angle);
264  lidar_coordinate.coeffRef(1) =
265  static_cast<double>(scan.ranges.at(static_cast<ulong>(ix))) * sin(angle);
266  lidar_coordinate.coeffRef(2) = 0.0;
267  lidar_coordinate.coeffRef(3) = 1.0;
268  robot_coordinate = transformation_mat_ * lidar_coordinate;
269 
270  Eigen::VectorXd point(dimension);
271  point.block<2, 1>(0, 0) = robot_coordinate.block<2, 1>(0, 0);
272  if (sim_3d_ == true) {
273  point.coeffRef(2) = 0.0;
274  }
275 
276  points.push_back(point);
277  }
278  zone_points.push_back(points);
279  }
280 
281  // ransac : 각 zone의 점들로 평면을 추출하여 평면 normal vector를 계산
282  std::vector<Eigen::VectorXd> variables;
283  for (int zone_index = 0; zone_index < zone_num; zone_index++) {
284  std::vector<Eigen::VectorXd> points = zone_points.at(static_cast<ulong>(zone_index));
285  Eigen::VectorXd variable(dimension);
286  if (static_cast<int>(points.size()) < sample_num_) {
287  variables.push_back(variable);
288  continue;
289  }
290  bool result = ransac(points, dimension, sample_num_, threshold_, false, variable);
291 
292  if (result == false) {
293  variable.setZero(3);
294  }
295  variables.push_back(variable);
296 
297  if (result) {
298  std::cout << "Normal of Zone [" << zone_index << "] " << variable.coeff(0) << " , " <<
299  variable.coeff(1) << " , " << variable.coeff(2) << std::endl;
300  } else {
301  std::cout << "Normal of Zone [" << zone_index << "]" << std::endl;
302  }
303  }
304 
305  // grouping
306  std::vector<Eigen::VectorXd> grouped_points;
307 
308  // 로봇 전면부 기준 구역 그룹화
309  if (use_center_) {
310  bool is_center = false;
311  int index_offset = 1;
312  // if the center zone is empty, select the nearest zone to the center.
313  while (zone_points.at(static_cast<ulong>(center_zone_index)).size() == 0 &&
314  index_offset < static_cast<int>(zone_num * 0.5))
315  {
316  if (
317  (center_zone_index + index_offset) < zone_num &&
318  zone_points.at(static_cast<ulong>(center_zone_index + index_offset)).size() != 0.0)
319  {
320  center_zone_index += index_offset;
321  break;
322  }
323 
324  if (
325  (center_zone_index - index_offset) > 0 &&
326  zone_points.at(static_cast<ulong>(center_zone_index - index_offset)).size() != 0.0)
327  {
328  center_zone_index -= index_offset;
329  break;
330  }
331 
332  index_offset += 1;
333  }
334 
335  // there is no center zone.
336  if (zone_points.at(static_cast<ulong>(center_zone_index)).size() == 0) {
337  detected_door.door_status = perception_interfaces::msg::DetectedDoor::UNKNOWN;
338  return false;
339  }
340 
341  grouped_points = zone_points.at(0);
342  for (int zone_index = 1; zone_index < zone_num; zone_index++) {
343  bool result = check_correlation(
344  variables.at(zone_index - 1), variables.at(
345  zone_index), relation_angle_, 1.25, false);
346 
347  if (result) {
348  // same group
349  grouped_points.insert(
350  grouped_points.end(), zone_points.at(static_cast<ulong>(zone_index)).begin(),
351  zone_points.at(static_cast<ulong>(zone_index)).end());
352 
353  std::cout << "Same zone : " << zone_index << std::endl;
354  } else {
355  // other group
356  if (is_center == false) {
357  grouped_points.clear();
358  grouped_points = zone_points.at(static_cast<ulong>(zone_index));
359  std::cout << "******************" << std::endl << "Different zone : " << zone_index <<
360  std::endl;
361  } else {
362  break;
363  }
364  }
365 
366  if (zone_index == center_zone_index) {
367  is_center = true;
368  }
369  }
370 
371  std::cout << "Center Index : " << center_zone_index << ", Size : " << grouped_points.size() <<
372  std::endl;
373 
374  if (grouped_points.size() < static_cast<ulong>(sample_num_)) {
375  geometry_msgs::msg::Pose door_pose;
376  geometry_msgs::msg::Vector3 scale;
377  publish_door_marker(door_pose, scale, true);
378 
379  detected_door.door_status = perception_interfaces::msg::DetectedDoor::UNKNOWN;
380  return false;
381  }
382  } else {
383  // 구역별 normal vector를 비교하여 그룹화, 제일 크기가 큰 그룹을 선택
384  std::vector<size_t> group_index;
385  std::vector<int> group_num;
386  // 모든 구역의 그룹을 0으로 설정(초기값), 그룹 멤버 수는 1로 설정
387  group_index.resize(zone_points.size(), 0);
388  group_num.resize(zone_points.size(), 1);
389  for (size_t zone_index = 1; zone_index < zone_points.size(); zone_index++) {
390  group_index.at(zone_index) = zone_index; // 현재 구역 인덱스로 그룹 인덱스 초기화
391  for (size_t comparison_index = 0; comparison_index < zone_index; comparison_index++) {
392  // 다른 그룹에 포함되어 있는 경우 비교하지 않음
393  if (group_index.at(comparison_index) != comparison_index) {
394  continue;
395  }
396 
397  // 다른 그룹과 비교
398  bool result = check_correlation(
399  variables.at(comparison_index), variables.at(
400  zone_index), relation_angle_, 1.25, false);
401 
402  if (result) {
403  // same group : 비교한 구역과 같은 그룹으로 설정, 그룹 카운트 증감
404  group_index.at(zone_index) = group_index.at(comparison_index);
405  group_num.at(comparison_index) = group_num.at(comparison_index) + 1;
406  group_num.at(zone_index) = 0;
407 
408  std::cout << "Index : " << zone_index << " -> group : " <<
409  group_index.at(comparison_index) << std::endl;
410  break;
411  }
412  }
413  if (group_index.at(zone_index) == zone_index) {
414  std::cout << "Index : " << zone_index << " -> group : " << group_index.at(zone_index) <<
415  std::endl;
416  }
417  }
418 
419  // 그룹 선택
420  size_t max_group_index = 0;
421  int max_count = 1;
422  for (size_t zone_index = 0; zone_index < zone_points.size(); zone_index++) {
423  if (group_num.at(zone_index) > max_count) {
424  max_group_index = zone_index;
425  max_count = group_num.at(zone_index);
426  }
427  }
428 
429  // 선택된 그룹의 점들을 grouped_points로 합침
430  for (size_t zone_index = 0; zone_index < zone_points.size(); zone_index++) {
431  if (group_index.at(zone_index) == max_group_index &&
432  zone_points.at(static_cast<ulong>(zone_index)).size() != 0)
433  {
434  grouped_points.insert(
435  grouped_points.end(), zone_points.at(static_cast<ulong>(zone_index)).begin(),
436  zone_points.at(static_cast<ulong>(zone_index)).end());
437  }
438  }
439 
440  std::cout << " ===================================== " << std::endl;
441  std::cout << "Group Index : " << max_group_index << "[" << max_count << "|" << zone_num <<
442  "], Size : " << grouped_points.size() << std::endl;
443 
444  if (grouped_points.size() < static_cast<ulong>(sample_num_)) {
445  geometry_msgs::msg::Pose door_pose;
446  geometry_msgs::msg::Vector3 scale;
447  publish_door_marker(door_pose, scale, true);
448 
449  detected_door.door_status = perception_interfaces::msg::DetectedDoor::UNKNOWN;
450  return false;
451  }
452  }
453 
454  // find door from grouping points
455  Eigen::VectorXd variable(dimension);
456 
457  bool result = ransac(grouped_points, dimension, sample_num_, threshold_, true, variable);
458 
459  if (result == false) {
460  geometry_msgs::msg::Pose door_pose;
461  geometry_msgs::msg::Vector3 scale;
462  publish_door_marker(door_pose, scale, true);
463 
464  detected_door.door_status = perception_interfaces::msg::DetectedDoor::UNKNOWN;
465  return false;
466  }
467 
468  // find door center
469  double begin_y = grouped_points.begin()->coeff(1);
470  double begin_x = (1 - variable.coeff(1) * begin_y) / variable.coeff(0);
471  double end_y = grouped_points.rbegin()->coeff(1);
472  double end_x = (1 - variable.coeff(1) * end_y) / variable.coeff(0);
473 
474  double center_x = (begin_x + end_x) * 0.5;
475  double center_y = (begin_y + end_y) * 0.5;
476 
477  double diff_x = end_x - begin_x;
478  double diff_y = end_y - begin_y;
479  double width = sqrt(diff_x * diff_x + diff_y * diff_y);
480  double distance = 1 / variable.norm(); // |ax + by + cx + 1| / sqrt(a*a + b*b + c*c)
481 
482  // publish door marker
483  geometry_msgs::msg::Pose door_pose;
484  geometry_msgs::msg::Vector3 scale;
485  // scale
486  scale.x = 0.05;
487  scale.y = width;
488  scale.z = DOOR_HEIGHT;
489  // positoin
490  door_pose.position.x = center_x;
491  door_pose.position.y = center_y;
492  door_pose.position.z = DOOR_HEIGHT * 0.5; // -lidar_height_;
493  // orientation
494  Eigen::Quaterniond orientation;
495  Eigen::Vector3d from(0, 1, 0);
496  Eigen::Vector3d to = *grouped_points.rbegin() - *grouped_points.begin();
497  orientation.setFromTwoVectors(from, to);
498  door_pose.orientation.x = orientation.x();
499  door_pose.orientation.y = orientation.y();
500  door_pose.orientation.z = orientation.z();
501  door_pose.orientation.w = orientation.w();
502 
503  publish_door_marker(door_pose, scale);
504 
505  // check min width
506  if (width < min_width_) {
507  detected_door.door_status = perception_interfaces::msg::DetectedDoor::LESS_THAN_MIN_WIDTH;
508 
509  if (prev_door_state_ != detected_door.door_status) {
510  RCLCPP_WARN_STREAM(
511  this->get_logger(), "Esitmated door width is below than min value : " << width);
512  }
513  return true;
514  }
515 
516  // check distance
517  if (distance > max_door_distance_) {
518  detected_door.door_status = perception_interfaces::msg::DetectedDoor::OUT_OF_RANGE;
519 
520  if (prev_door_state_ != detected_door.door_status) {
521  RCLCPP_WARN_STREAM(
522  this->get_logger(), "Esitmated door is out of range : " << distance);
523  }
524  return true;
525  }
526 
527  // check opened
528  Eigen::Vector3d nomailzed_variable;
529  if (dimension == 3) {
530  nomailzed_variable = variable.normalized();
531  } else {

615 bool DoorDetection::check_correlation(
616  const Eigen::VectorXd & from, const Eigen::VectorXd & to, double angle, double ratio,
617  bool check_direction)
618 {
619  if (from.norm() == 0.0 || to.norm() == 0.0) {
620  return false;
621  }
622  double correlation_cos = cos(angle); // 0.7; // 45 degree
623  double cos_value = from.dot(to) / (from.norm() * to.norm());
624  double distance_ratio =
625  to.norm() / from.norm(); // |ax + by + cx + 1| / sqrt(a*a + b*b + c*c)
626 
627  if (check_direction == false) {
628  cos_value = fabs(cos_value);
629  }
630 
631  if (cos_value > correlation_cos) {
632  if (ratio == 0.0 || (distance_ratio >= (1 - ratio) && distance_ratio < (1 + ratio))) {
633  return true;
634  }
635  }
636 
637  return false;
638 }

641  std::vector<Eigen::VectorXd> & points, int dimension, int sample_num, double threshold,
642  bool refind, Eigen::VectorXd & variable)
643 {
644  if (points.size() == 0) {
645  return false;
646  }
647 
648  Eigen::VectorXd variable_temp(dimension);
649  int c_max = 0;
650  int iteration = std::max<int>(10, static_cast<int>(0.2 * points.size()));
651  std::random_device rd;
652  std::mt19937 gen(rd());
653  std::uniform_int_distribution<int> dist(0, static_cast<int>(points.size() - 1));
654  // std::uniform_real_distribution<double> dist_2(0, 1.0);
655 
656  for (int it = 0; it < iteration; it++) {
657  std::vector<int> select_index;
658 
659  for (int sample_ix = 0; sample_ix < sample_num; sample_ix++) {
660  select_index.push_back(dist(gen));
661  // select_index[sample_ix] = static_cast<int>(dist_2(gen) * points.size());
662  }
663 
664  // calc
665  std::uniform_real_distribution<double> dist_height(0.0, 2.1);
666  Eigen::MatrixXd sample_mat(dimension, sample_num);
667  for (int v_ix = 0; static_cast<ulong>(v_ix) < select_index.size(); v_ix++) {
668  Eigen::VectorXd v(dimension);
669  v = points.at(static_cast<ulong>(select_index.at(static_cast<ulong>(v_ix))));
670  if (dimension == 3) {
671  v.coeffRef(2) = dist_height(gen);
672  }
673  sample_mat.col(v_ix) = v;
674  }
675 
676  Eigen::VectorXd id(sample_num);
677  id.setOnes();
678 
679  // Eigen::BDCSVD<Eigen::MatrixXd> SVD(sample_mat, Eigen::ComputeThinU |
680  // Eigen::ComputeThinV); Eigen::MatrixXd sample_iv = SVD.matrixV() *
681  // SVD.singularValues().array().inverse().matrix().asDiagonal() *
682  // SVD.matrixU().transpose();
683  // variable_temp = id.transpose() * sample_iv;
684 
685  Eigen::MatrixXd square_mat = sample_mat * sample_mat.eval().transpose();
686  variable_temp = id.transpose() * sample_mat.eval().transpose() * square_mat.inverse();
687 
688  // variable_temp = id.transpose() * sample_mat.inverse();
689 
690  // evaluation
691  int evaluation_count = 0;
692  for (auto v : points) {
693  double residual = fabs(variable_temp.transpose() * v - 1.0) /
694  sqrt(variable_temp.transpose() * variable_temp);
695  if (residual < threshold) {
696  evaluation_count++;
697  }
698  }
699 
700  if (evaluation_count < c_max || evaluation_count < points.size() * 0.3) {
701  continue;
702  }
703 
704  c_max = evaluation_count;
705  variable = variable_temp;
706  }
707 
708  if (c_max == 0) {
709  // std::cout << "Failed to find the foor." << std::endl;
710 
711  // geometry_msgs::msg::Pose door_pose;
712  // geometry_msgs::msg::Vector3 scale;
713  // publish_door_marker(door_pose, scale, true);
714  return false;
715  }
716 
717  // refine
718  if (refind == false) {
719  return true;
720  }
721 
722  // std::vector<Eigen::Vector3d> door_points;
723  std::vector<Eigen::VectorXd> door_points;
724  // for (auto v : points) {
725  for (int ix = 0; static_cast<ulong>(ix) < points.size(); ix++) {
726  double residual =
727  fabs(variable.transpose() * points.at(static_cast<ulong>(ix)) - 1.0) /
728  sqrt(variable.transpose() * variable);
729  if (residual < threshold) {
730  door_points.push_back(points.at(static_cast<ulong>(ix)));
731  // std::cout << "inbound : " << points.at(ix) << std::endl;
732  }
733  }
734 
735  std::cout << "result - c_max : " << c_max << ", refined points : " << door_points.size() <<
736  std::endl;
737 
738  std::cout << "variable : " << std::endl << variable << std::endl;
