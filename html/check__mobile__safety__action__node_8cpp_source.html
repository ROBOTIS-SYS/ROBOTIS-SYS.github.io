34 bool CheckMobileSafetyAction::on_init()
35 {
36  if (config().blackboard->get("enable_log", enable_log_) == false ||
37  enable_log_ == true)
38  {
39  RCLCPP_INFO_STREAM(get_logger(), "Init : CheckMobileSafetyActionNode");
40  }
41 
42  if (getInput<int>("check_type", check_type_)) {
43  switch (check_type_) {
44  case ALL:
45  motor_state_sub_ = node_->create_subscription<MotorStateMsg>(
46  "/aimbot_base/motor_state", rclcpp::SystemDefaultsQoS().keep_last(1),
47  std::bind(
48  &CheckMobileSafetyAction::motor_state_callback, this,
49  std::placeholders::_1));
50  safety_filter_sub_ = node_->create_subscription<SafetyFilterStateMsg>(
51  "tof_safety_filter_state", rclcpp::SystemDefaultsQoS().keep_last(1),
52  std::bind(
53  &CheckMobileSafetyAction::safety_filter_callback, this,
54  std::placeholders::_1));
55  break;
56 
57  case BUMPER:
58  motor_state_sub_ = node_->create_subscription<MotorStateMsg>(
59  "/aimbot_base/motor_state", rclcpp::SystemDefaultsQoS().keep_last(1),
60  std::bind(
61  &CheckMobileSafetyAction::motor_state_callback, this,
62  std::placeholders::_1));
63  break;
64 
65  case TOF:
66  safety_filter_sub_ = node_->create_subscription<SafetyFilterStateMsg>(
67  "tof_safety_filter_state", rclcpp::SystemDefaultsQoS().keep_last(1),
68  std::bind(
69  &CheckMobileSafetyAction::safety_filter_callback, this,
70  std::placeholders::_1));
71  break;
72 
73  default:
74  return false;
75  }
76  } else {
77  return false;
78  }
79 
80  return true;

84 BT::NodeStatus CheckMobileSafetyAction::tick()
85 {
86  std::cout << std::endl;
87  if (config().blackboard->get("enable_log", enable_log_) == false ||
88  enable_log_ == true)
89  {
90  RCLCPP_INFO_STREAM(get_logger(), "<TICK> CheckMobileSafetyAction");
91  }
92 
93  bumper_error_occurred_ = -1;
94  tof_error_occurred_ = false;
95  msg_received_ = false;
96 
97  if (!getInput<double>("check_latency", check_latency_)) {
98  check_latency_ = 0.1;
99  }
100 
101  while (status() == BT::NodeStatus::RUNNING) {
102  rclcpp::sleep_for(std::chrono::milliseconds(WAIT_PER));
103 
104  if (isHaltRequested()) {
105  return BT::NodeStatus::IDLE;
106  }
107 
108  if (msg_received_.load()) {
109  break;
110  }
111  }
112 
113  if (bumper_error_occurred_.load() == -1 && tof_error_occurred_.load() == false) {
114  RCLCPP_INFO_STREAM(get_logger(), "[" << name() << "] No mobile error");
115  return BT::NodeStatus::SUCCESS;
116  }
117 
118  rclcpp::sleep_for(std::chrono::milliseconds(static_cast<int>(check_latency_ * 1000)));
119 
120  if (isHaltRequested()) {
121  return BT::NodeStatus::IDLE;
122  }
123 
124  if (bumper_error_occurred_.load() == -1 && tof_error_occurred_.load() == true) {
125  setOutput<int>("error_source", 2);
126  setOutput<std::string>("error_state", "");
127  RCLCPP_WARN_STREAM(get_logger(), "[" << name() << "] Mobile error : TOF");
128  } else if (bumper_error_occurred_.load() != -1 && tof_error_occurred_.load() == false) {
129  setOutput<int>("error_source", 1);
130  if (bumper_error_occurred_.load() == 0) {
131  setOutput<std::string>("error_state", "NO_BRAKE");
132  }
133  if (bumper_error_occurred_.load() == 1) {
134  setOutput<std::string>("error_state", "FRONT_BUMPER");
135  }
136  if (bumper_error_occurred_.load() == 2) {
137  setOutput<std::string>("error_state", "REAR_BUMPER");
138  }
139  RCLCPP_WARN_STREAM(get_logger(), "[" << name() << "] Mobile error : Bumper");
140  } else if (bumper_error_occurred_.load() != -1 && tof_error_occurred_.load() == true) {
141  setOutput<int>("error_source", 0);
142  if (bumper_error_occurred_.load() == 0) {
143  setOutput<std::string>("error_state", "NO_BRAKE");
144  }
145  if (bumper_error_occurred_.load() == 1) {
146  setOutput<std::string>("error_state", "FRONT_BUMPER");
147  }
148  if (bumper_error_occurred_.load() == 2) {
149  setOutput<std::string>("error_state", "REAR_BUMPER");
150  }
151  RCLCPP_WARN_STREAM(get_logger(), "[" << name() << "] Mobile error : Bumper & TOF");
152  }
153 
154  return BT::NodeStatus::FAILURE;
155 }
