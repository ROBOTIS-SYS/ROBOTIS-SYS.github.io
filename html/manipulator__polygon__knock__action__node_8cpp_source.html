38 BT::NodeStatus ManipulatorPolygonKnockAction::on_tick()
39 {
40  std::string manipulator_namespace = "gc_manipulator";
41  std::string action_name = manipulator_namespace + "/manipulator_controller/knock";
42  std::chrono::milliseconds action_timeout = std::chrono::milliseconds(10000); // 10s
43 
44  if (!create_action_client(action_name, action_timeout)) {
45  return BT::NodeStatus::FAILURE;
46  }
47 
48  // Log [msg]
49  try {
50  bool on_blackboard = config().blackboard->get<decltype(log_pub_)>("/aimbot_log", log_pub_);
51  if (on_blackboard == false) {
52  log_pub_ = node_->create_publisher<aimbot_log_interfaces::msg::Log>("/aimbot_log", 10);
53  config().blackboard->set<decltype(log_pub_)>("/aimbot_log", log_pub_);
54  }
55  } catch (std::exception & ex) {
56  RCLCPP_ERROR_STREAM(
57  node_->get_logger(),
58  "Failed to create \"/aimbot_log\" action : " << ex.what());
59  return BT::NodeStatus::FAILURE;
60  }
61 
62  // test door open sub
63  door_opened_ = false;
64  door_open_sub_ = node_->create_subscription<std_msgs::msg::Empty>(
65  "/door_opend", 0,
66  [this](std_msgs::msg::Empty::SharedPtr msg_) {
67  (void) msg_;
68  this->door_opened_ = true;
69  }
70  );
71 
72  set_ros_spin_loop_rate(10);
73 
74  int knock_cnt;
75  if (!getInput<int>("knock_cnt", knock_cnt)) {
76  knock_cnt = 3;
77  }
78 
79  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
80  RCLCPP_INFO_STREAM(
81  node_->get_logger(), "ManipulatorKnockAction: Set knock cnt : " << knock_cnt);
82  }
83 
84  std::string frame_id;
85  if (!getInput<std::string>("frame_id", frame_id)) {
86  frame_id = "base_footprint";
87  }
88  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
89  RCLCPP_INFO_STREAM(
90  node_->get_logger(), "ManipulatorPolygonKnockAction: Set frame_id : " << frame_id);
91  }
92 
93  geometry_msgs::msg::Point32 point1;
94  if (!getInput<geometry_msgs::msg::Point32>("point1", point1)) {
95  point1.x = 0.5;
96  point1.y = 0.5;
97  point1.z = 0.0;
98  }
99  geometry_msgs::msg::Point32 point2;
100  if (!getInput<geometry_msgs::msg::Point32>("point2", point2)) {
101  point2.x = 0.5;
102  point2.y = 0.5;
103  point2.z = 2.0;
104  }
105  geometry_msgs::msg::Point32 point3;
106  if (!getInput<geometry_msgs::msg::Point32>("point3", point3)) {
107  point3.x = 0.5;
108  point3.y = -0.5;
109  point3.z = 2.0;
110  }
111  geometry_msgs::msg::Point32 point4;
112  if (!getInput<geometry_msgs::msg::Point32>("point4", point4)) {
113  point4.x = 0.5;
114  point4.y = -0.5;
115  point4.z = 0.0;
116  }
117 
118  double knock_position_rate;
119  if (!getInput<double>("knock_position_rate", knock_position_rate)) {
120  knock_position_rate = 0.8;
121  }
122  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
123  RCLCPP_INFO_STREAM(
124  node_->get_logger(),
125  "ManipulatorKnockAction: Set knock position rate : " << knock_position_rate);
126  }
127 
128  geometry_msgs::msg::Pose2D knock_direction;
129  if (!getInput<geometry_msgs::msg::Pose2D>("knock_direction", knock_direction)) {
130  knock_direction.x = 0.03;
131  knock_direction.y = -0.01;
132  knock_direction.theta = 10.0;
133  }
134 
135  // set action data
136  goal_.header.stamp = node_->now();
137  goal_.header.frame_id = frame_id;
138  goal_.door_position.points.push_back(point1);
139  goal_.door_position.points.push_back(point2);
140  goal_.door_position.points.push_back(point3);
141  goal_.door_position.points.push_back(point4);
142  goal_.knock_position_rate = knock_position_rate;
143  goal_.knock_direction = knock_direction;
144  goal_.knock_num = knock_cnt;
145 
146  return BT::NodeStatus::RUNNING;
147 }

163 BT::NodeStatus ManipulatorPolygonKnockAction::on_success()
164 {
165  auto value = result_.result->result_state.val;
166  if (value == manipulator_interfaces::msg::Result::SUCCESS) {
167  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
168  RCLCPP_INFO_STREAM(node_->get_logger(), "[Knock] SUCCESS!!!");
169  }
170  publish_log("SUCCESS!!!");
171  return BT::NodeStatus::SUCCESS;
172  } else if (value == manipulator_interfaces::msg::Result::CANCELED) {
173  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
174  RCLCPP_INFO_STREAM(node_->get_logger(), "[Knock] canceled manipulator action.");
175  }
176  publish_log("CANCELED!");
177  return BT::NodeStatus::SUCCESS;
178  } else if (value == manipulator_interfaces::msg::Result::PREEMPTED) {
179  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
180  RCLCPP_INFO_STREAM(
181  node_->get_logger(), "[Knock] Accepted another manipulator action.");
182  }
183  publish_log(
184  "PREEMPTED : " + std::to_string(value),
185  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
186  aimbot_log_interfaces::msg::Log::TYPE_ERROR);
187  return BT::NodeStatus::SUCCESS;
188  } else {
189  RCLCPP_ERROR_STREAM(
190  node_->get_logger(), "[Knock] FAIL :( ERROR : " << value);
191  publish_log(
192  "ERROR : " + std::to_string(value),
193  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
194  aimbot_log_interfaces::msg::Log::TYPE_ERROR);
195  return BT::NodeStatus::FAILURE;
196  }
197 }
