175  bool get_pid_by_name(const std::string & process_name)
176  {
177  int pid = -1;
178 
179  // get pid
180  // Open the /proc directory
181  DIR * dp = opendir("/proc");
182  if (dp != NULL) {
183  // Enumerate all entries in directory until process found
184  struct dirent * dirp;
185  while (pid < 0 && (dirp = readdir(dp))) {
186  if (isHaltRequested() == true) {
187  return false;
188  }
189  // Skip non-numeric entries
190  int id = atoi(dirp->d_name);
191  if (id > 0) {
192  // Read contents of virtual /proc/{pid}/cmdline file
193  std::string cmd_path = std::string("/proc/") + dirp->d_name + "/cmdline";
194  std::ifstream cmd_file(cmd_path.c_str());
195  std::string cmd_line;
196  getline(cmd_file, cmd_line);
197  if (!cmd_line.empty()) {
198  // Keep first cmdline item which contains the program path
199  size_t pos = cmd_line.find('\0');
200  if (pos != std::string::npos) {
201  cmd_line = cmd_line.substr(0, pos);
202  }
203  // Keep program name only, removing the path
204  pos = cmd_line.rfind('/');
205  if (pos != std::string::npos) {
206  cmd_line = cmd_line.substr(pos + 1);
207  }
208  // Compare against requested process name
209  if (process_name == cmd_line) {
210  pid = id;
211  }
212  }
213  }
214  }
215  }
216 
217  closedir(dp);
218 
219  return pid != -1;
220  }

229  RCLCPP_INFO_STREAM(get_logger(), "Process is accessable : " << process_path.str());
230  }
231 
232  return result == 0;
233  }
