127 void PageWelcome::setOption(std::string option)
128 {
129  RCLCPP_INFO(ros_node_->get_logger(), "PageWelcome : option(%s) ", option.c_str());
130 
131  if (option.compare("SoundPlay") == 0) {
132  emit playSound("Welcome_Robotis.mp3");
133  } else {
134  std::istringstream iss(option);
135  std::string token;
136  int cnt = 0;
137  while (std::getline(iss, token, '|')) {
138  cnt++;
139  if (cnt == 1 && token.compare("SoundPlay") != 0) {
140  return;
141  }
142  if (cnt == 2) {
143  try {
144  standing_still_time_ = std::stoi(token);
145  } catch (const std::exception & e) {
146  RCLCPP_ERROR(
147  ros_node_->get_logger(), "Cannot Convert string (%s) to Integer", token.c_str());
148  return;
149  }
150  }
151  emit playSound("Welcome_Robotis.mp3");
152  played_time_ = 0;
153  sound_play_timer_->start(sound_play_cycle_ * 1000);
154  }
155  }
156 }

175 std::string PageWelcome::get_service_type_in_bracket(std::string service_type)
176 {
177  std::string extracted_string;
178  size_t start_pos = service_type.find('(');
179  size_t end_pos = service_type.find(')', start_pos);
180 
181  if (start_pos != std::string::npos && end_pos != std::string::npos) {
182  // Extract the substring between the brackets
183  extracted_string = service_type.substr(start_pos + 1, end_pos - start_pos - 1);
184  } else {
185  extracted_string = "";
186  }
187  return extracted_string;
188 }
