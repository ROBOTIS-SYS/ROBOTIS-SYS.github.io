35 double EveAgentUtils::get_current_ram_usage()
36 {
37  std::ifstream file("/proc/meminfo");
38  if (!file.is_open()) {
39  return -1.0;
40  }
41  std::string line;
42  std::getline(file, line);
43  std::istringstream ss(line);
44  std::string name;
45  double memTotal = -1.0, memFree = -1.0, buffers = -1.0, cached = -1.0;
46  ss >> name >> memTotal;
47  std::getline(file, line);
48  ss.clear();
49  ss.str(line);
50  ss >> name >> memFree;
51  std::getline(file, line);
52  ss.clear();
53  ss.str(line);
54  ss >> name >> buffers;
55  std::getline(file, line);
56  ss.clear();
57  ss.str(line);
58  ss >> name >> cached;
59  file.close();
60  if (memTotal == -1.0 || memFree == -1.0 || buffers == -1.0 || cached == -1.0) {
61  return -1.0;
62  }
63  double usedMemory = memTotal - memFree - buffers - cached;
64  double ramUsage = usedMemory / memTotal * 100.0;
65  return static_cast<double>(ramUsage);
66 }

69 double EveAgentUtils::get_current_cpu_usage()
70 {
71  double cpuUsage = 0.0;
72  std::ifstream file("/proc/stat");
73  // printf("getting...");
74  if (file.is_open()) {
75  std::string line;
76  std::getline(file, line);
77 
78  std::istringstream ss(line);
79  std::string cpuLabel;
80  ss >> cpuLabel;
81 
82  if (cpuLabel == "cpu") {
83  int user, nice, system, idle, iowait, irq, softirq, steal, guest, guest_nice;
84  ss >> user >> nice >> system >> idle >> iowait >> irq >> softirq >> steal >> guest >>
85  guest_nice;
86 
87  // Wait for 100ms
88  usleep(100000);
89 
90  file.seekg(0);
91  std::getline(file, line);
92 
93  ss.str(line);
94  ss.clear();
95  ss >> cpuLabel;
96 
97  if (cpuLabel == "cpu") {
98  int prevUser, prevNice, prevSystem, prevIdle, prevIoWait, prevIrq, prevSoftIrq, prevSteal,
99  prevGuest, prevGuestNice;
100  ss >> prevUser >> prevNice >> prevSystem >> prevIdle >> prevIoWait >> prevIrq >>
101  prevSoftIrq >> prevSteal >> prevGuest >> prevGuestNice;
102 
103  int prevIdleTotal = prevIdle + prevIoWait;
104  int idleTotal = idle + iowait;
105  int prevNonIdle = prevUser + prevNice + prevSystem + prevIrq + prevSoftIrq + prevSteal;
106  int nonIdle = user + nice + system + irq + softirq + steal;
107  int prevTotal = prevIdleTotal + prevNonIdle;
108  int total = idleTotal + nonIdle;
109 
110  int totald = total - prevTotal;
111  int idled = idleTotal - prevIdleTotal;
112 
113  cpuUsage = (totald - idled) / static_cast<double>(totald) * 100.0;
114  }
115  }
116 
117  file.close();
118  }
119 
120  return static_cast<double>(cpuUsage);
121 }

123 std::tuple<bool, std::string> EveAgentUtils::sys_cmd(std::string cmd)
124 {
125  std::string output;
126 
127  cmd += " 2>&1"; // redirect stderr to stdout
128  FILE * fp = popen(cmd.c_str(), "r");
129  if (!fp) {
130  return std::make_pair(false, "");
131  }
132 
133  int fd = fileno(fp);
134  fcntl(fd, F_SETFL, O_NONBLOCK); // set file descriptor to non-blocking mode
135 
136  while (true) {
137  fd_set set;
138  FD_ZERO(&set);
139  FD_SET(fd, &set);
140 
141  struct timeval timeout;
142  timeout.tv_sec = 0;
143  timeout.tv_usec = 100; // timeout in microseconds
144 
145  int res = select(fd + 1, &set, nullptr, nullptr, &timeout);
146  if (res == -1) {
147  // select() failed
148  pclose(fp);
149  return std::make_pair(false, "Failed to read command output");
150  } else if (res == 0) {
151  // timeout occurred
152  continue;
153  } else {
154  char buffer[1024];
155  int bytes_read = fread(buffer, 1, sizeof(buffer), fp);
156  if (bytes_read == 0) {
157  break; // end of file
158  } else {
159  output.append(buffer, bytes_read);
160  }
161  }
162  }
163 
164  pclose(fp);
165  return std::make_pair(true, output);
166 }

265 EveAgentUtils::NetInfo EveAgentUtils::get_net_info(std::string eth_port)
266 {
267  EveAgentUtils::NetInfo net_info;
268  ifreq ifrq;
269  char ipstr[40];
270  char arr_mac_addr[18] = {0x00, };
271  int s = socket(AF_INET, SOCK_DGRAM, 0);
272  try {
273  strncpy(ifrq.ifr_name, eth_port.c_str(), IFNAMSIZ);
274  if (ioctl(s, SIOCGIFADDR, &ifrq) < 0) {
275  // RCLCPP_INFO_STREAM(get_logger(), "Failed to get ip_address from interface...");
276  net_info.ip_addr_ = "Unknown";
277  close(s);
278  return net_info;
279  } else {
280  inet_ntop(AF_INET, ifrq.ifr_addr.sa_data + 2, ipstr, sizeof(struct sockaddr));
281  net_info.ip_addr_ = ipstr;
282  }
283  if (ioctl(s, SIOCGIFHWADDR, &ifrq) < 0) {
284  net_info.mac_addr_ = "Unknown";
285  close(s);
286  return net_info;
287  // RCLCPP_INFO_STREAM(get_logger(), "Failed to get mac_address from interface...");
288  } else {
289  snprintf(
290  arr_mac_addr, sizeof(arr_mac_addr), "%02x:%02x:%02x:%02x:%02x:%02x",
291  (unsigned char)ifrq.ifr_hwaddr.sa_data[0],
292  (unsigned char)ifrq.ifr_hwaddr.sa_data[1],
293  (unsigned char)ifrq.ifr_hwaddr.sa_data[2],
294  (unsigned char)ifrq.ifr_hwaddr.sa_data[3],
295  (unsigned char)ifrq.ifr_hwaddr.sa_data[4],
296  (unsigned char)ifrq.ifr_hwaddr.sa_data[5]);
297  net_info.mac_addr_ = arr_mac_addr;
298  }
299  } catch (std::exception & e) {
300  std::cout << e.what() << std::endl;
301  }
302  close(s);
303  return net_info;
304 }
