40 static inline void mapToWorld3D(
41  const unsigned int mx,
42  const unsigned int my, const unsigned int mz,
43  const double origin_x, const double origin_y, const double origin_z,
44  const double x_resolution, const double y_resolution,
45  const double z_resolution,
46  double & wx, double & wy, double & wz)
47 {
48  // returns the center point of the cell
49  wx = origin_x + (mx + 0.5) * x_resolution;
50  wy = origin_y + (my + 0.5) * y_resolution;
51  wz = origin_z + (mz + 0.5) * z_resolution;
52 }

76 void voxelCallback(const antis_nav_msgs::msg::VoxelGrid::ConstSharedPtr grid)
77 {
78  if (grid->data.empty()) {
79  RCLCPP_ERROR(g_node->get_logger(), "Received empty voxel grid");
80  return;
81  }
82 
83  antis_nav_util::ExecutionTimer timer;
84  timer.start();
85 
86  RCLCPP_DEBUG(g_node->get_logger(), "Received voxel grid");
87  const std::string frame_id = grid->header.frame_id;
88  const rclcpp::Time stamp = grid->header.stamp;
89  const uint32_t * data = &grid->data.front();
90  const double x_origin = grid->origin.x;
91  const double y_origin = grid->origin.y;
92  const double z_origin = grid->origin.z;
93  const double x_res = grid->resolutions.x;
94  const double y_res = grid->resolutions.y;
95  const double z_res = grid->resolutions.z;
96  const uint32_t x_size = grid->size_x;
97  const uint32_t y_size = grid->size_y;
98  const uint32_t z_size = grid->size_z;
99 
100  g_marked.clear();
101  g_unknown.clear();
102  uint32_t num_marked = 0;
103  uint32_t num_unknown = 0;
104  for (uint32_t y_grid = 0; y_grid < y_size; ++y_grid) {
105  for (uint32_t x_grid = 0; x_grid < x_size; ++x_grid) {
106  for (uint32_t z_grid = 0; z_grid < z_size; ++z_grid) {
107  antis_nav_voxel_grid::VoxelStatus status =
108  antis_nav_voxel_grid::VoxelGrid::getVoxel(
109  x_grid, y_grid,
110  z_grid, x_size, y_size, z_size, data);
111  if (status == antis_nav_voxel_grid::UNKNOWN) {
112  Cell c;
113  c.status = status;
114  mapToWorld3D(
115  x_grid, y_grid, z_grid, x_origin, y_origin,
116  z_origin, x_res, y_res, z_res, c.x, c.y, c.z);
117 
118  g_unknown.push_back(c);
119 
120  ++num_unknown;
121  } else if (status == antis_nav_voxel_grid::MARKED) {
122  Cell c;
123  c.status = status;
124  mapToWorld3D(
125  x_grid, y_grid, z_grid, x_origin, y_origin,
126  z_origin, x_res, y_res, z_res, c.x, c.y, c.z);
127 
128  g_marked.push_back(c);
129 
130  ++num_marked;
131  }
132  }
133  }
134  }
135 
136  {
137  auto cloud = std::make_unique<sensor_msgs::msg::PointCloud>();
138  cloud->points.resize(num_marked);
139  cloud->channels.resize(1);
140  cloud->channels[0].values.resize(num_marked);
141  cloud->channels[0].name = "rgb";
142  cloud->header.frame_id = frame_id;
143  cloud->header.stamp = stamp;
144 
145  sensor_msgs::msg::ChannelFloat32 & chan = cloud->channels[0];
146  for (uint32_t i = 0; i < num_marked; ++i) {
147  geometry_msgs::msg::Point32 & p = cloud->points[i];
148  float & cval = chan.values[i];
149  Cell & c = g_marked[i];
150 
151  p.x = c.x;
152  p.y = c.y;
153  p.z = c.z;
154 
155  uint32_t r = g_colors_r[c.status] * 255.0;
156  uint32_t g = g_colors_g[c.status] * 255.0;
157  uint32_t b = g_colors_b[c.status] * 255.0;
158  // uint32_t a = g_colors_a[c.status] * 255.0;
159 
160  uint32_t col = (r << 16) | (g << 8) | b;
161  memcpy(&cval, &col, sizeof col);
162  }
163 
164  pub_marked->publish(std::move(cloud));
165  }
166 
167  {
168  auto cloud = std::make_unique<sensor_msgs::msg::PointCloud>();
169  cloud->points.resize(num_unknown);
170  cloud->channels.resize(1);
171  cloud->channels[0].values.resize(num_unknown);
172  cloud->channels[0].name = "rgb";
173  cloud->header.frame_id = frame_id;
174  cloud->header.stamp = stamp;
175 
176  sensor_msgs::msg::ChannelFloat32 & chan = cloud->channels[0];
177  for (uint32_t i = 0; i < num_unknown; ++i) {
178  geometry_msgs::msg::Point32 & p = cloud->points[i];
179  float & cval = chan.values[i];
180  Cell & c = g_unknown[i];
181 
182  p.x = c.x;
183  p.y = c.y;
184  p.z = c.z;
185 
186  uint32_t r = g_colors_r[c.status] * 255.0;
187  uint32_t g = g_colors_g[c.status] * 255.0;
188  uint32_t b = g_colors_b[c.status] * 255.0;
189  // uint32_t a = g_colors_a[c.status] * 255.0;
190 
191  uint32_t col = (r << 16) | (g << 8) | b;
192  memcpy(&cval, &col, sizeof col);
193  }
194 
195  pub_unknown->publish(std::move(cloud));
196  }
197 
198  timer.end();
199  RCLCPP_DEBUG(
200  g_node->get_logger(), "Published %d points in %f seconds",
201  num_marked + num_unknown, timer.elapsed_time_in_seconds());
202 }
