31 EvButtonDetector::EvButtonDetector(
32  const std::string & node_name, rclcpp::NodeOptions & option)
33 : perception_base::DetectorBase(node_name, option)
34 {
35  RCLCPP_INFO_STREAM(this->get_logger(), "***** EvButtonDetector *****");
36 
37  inside_detector_name_ = "ev_inside_button_detector";
38 
39  get_params();
40 
41  load_network();
42 
43  auto left_callback_group = create_callback_group(rclcpp::CallbackGroupType::MutuallyExclusive);
44  auto right_callback_group = create_callback_group(rclcpp::CallbackGroupType::MutuallyExclusive);
45  this->create_subscriber_image(left_img_sub_msg_name_, left_callback_group);
46  this->create_subscriber_image(right_img_sub_msg_name_, right_callback_group);
47  this->create_publisher_image(zed_streaming_pub_msg_name_);
48 
49  pose_estimator_ = std::make_shared<detector_lib::PoseEstimator>();
50  if (this->get_name() == inside_detector_name_) {
51  template_strategy_ = std::make_shared<detector_lib::TemplateBasedEstimatorStrategy>();
52  template_strategy_->set_template(template_pts_, template_nor_);
53  pose_estimator_->set_calculator_strategy(template_strategy_);
54  RCLCPP_INFO(
55  get_logger(), "Set as inside button detector");
56  } else {
57  boundingbox_strategy_ = std::make_shared<detector_lib::BoundingBoxBasedEstimatorStrategy>();
58  boundingbox_strategy_->set_pose_calc_location(
59  detector_lib::BoundingBoxBasedEstimatorStrategy::CENTER);
60  pose_estimator_->set_calculator_strategy(boundingbox_strategy_);
61  RCLCPP_INFO(
62  get_logger(), "Set as outside button detector");
63  }
64 
65  visualizer_ = std::make_shared<detector_lib::Visualizer>();
66 
67  ev_target_pose_sub_ = this->create_subscription<geometry_msgs::msg::PoseStamped>(
68  ev_target_pose_msg_name_,
69  perception_tools::default_qos,
70  std::bind(&EvButtonDetector::ev_target_pose_callback, this, std::placeholders::_1));
71  ev_detected_btn_arr_pub_ =
72  create_publisher<perception_interfaces::msg::DetectedButtonArray>(
73  ev_detected_button_arr_, 1);
74  RCLCPP_INFO(
75  get_logger(), "EV button detected pose array * '%s'",
76  ev_detected_button_arr_.c_str());
77  ev_detected_btn_goal_pose_pub_ =
78  create_publisher<geometry_msgs::msg::PoseStamped>(
79  ev_detected_button_goal_pose_, 1);
80  RCLCPP_INFO(
81  get_logger(), "EV button detected goal pose * '%s'",
82  ev_detected_button_goal_pose_.c_str());
83 }

90 int EvButtonDetector::load_network()
91 {
92  using ModelState = perception_interfaces::srv::GetModelState::Response;
93 
94  if (segmentation_.use_count() == 0) {
95  RCLCPP_INFO(
96  get_logger(), "Segmentation model loading...");
97  segmentation_ = std::make_shared<ml_model::Segmentation>(ev_img_seg_model_name_);
98  int seg_state = segmentation_->initialize();
99  if (seg_state >= PERCEPTION_STATE::SUCCESS_NO_ISSUE) {
100  RCLCPP_INFO(
101  get_logger(), "Segmentation model successfully loaded");
102  } else if (seg_state == PERCEPTION_STATE::ERROR_MODEL_READ) {
103  RCLCPP_ERROR(
104  this->get_logger(), "No segmentation model exists");
105  return ModelState::MODEL_DOES_NOT_EXIST;
106  } else {
107  RCLCPP_ERROR(
108  this->get_logger(), "Failed to load segmentation model * '%s'",
109  get_error_log(seg_state));
110  return ModelState::MODEL_LOAD_FAILED;
111  }
112  } else {
113  RCLCPP_INFO(
114  get_logger(), "Already segmentation model loaded");
115  }
116 
117  if (number_classification_.use_count() == 0) {
118  RCLCPP_INFO(
119  get_logger(), "Number classification model loading...");
120  number_classification_ = std::make_shared<ml_model::Classification>(
121  ev_num_cls_model_name_, 50, num_cls_labels_);
122  int num_cls_state = number_classification_->initialize();
123  if (num_cls_state >= PERCEPTION_STATE::SUCCESS_NO_ISSUE) {
124  RCLCPP_INFO(
125  get_logger(), "Number classification model successfully loaded");
126  } else if (num_cls_state == PERCEPTION_STATE::ERROR_MODEL_READ) {
127  RCLCPP_ERROR(
128  this->get_logger(), "No number classification model exists");
129  return ModelState::MODEL_DOES_NOT_EXIST;
130  } else {
131  RCLCPP_ERROR(
132  this->get_logger(), "Failed to load number classification model * '%s'",
133  get_error_log(num_cls_state));
134  return ModelState::MODEL_LOAD_FAILED;
135  }
136  } else {
137  RCLCPP_INFO(
138  get_logger(), "Already number classification model loaded");
139  }
140 
141  if (state_classification_.use_count() == 0) {
142  RCLCPP_INFO(
143  get_logger(), "State classification model loading...");
144  state_classification_ = std::make_shared<ml_model::Classification>(
145  ev_state_cls_model_name_, 50, num_cls_labels_);
146  int state_cls_state = state_classification_->initialize();
147  if (state_cls_state >= PERCEPTION_STATE::SUCCESS_NO_ISSUE) {
148  RCLCPP_INFO(
149  get_logger(), "State classification model successfully loaded");
150  } else if (state_cls_state == PERCEPTION_STATE::ERROR_MODEL_READ) {
151  RCLCPP_ERROR(
152  this->get_logger(), "No state classification model exists");
153  return ModelState::MODEL_DOES_NOT_EXIST;
154  } else {
155  RCLCPP_ERROR(
156  this->get_logger(), "Failed to load state classification model * '%s'",
157  get_error_log(state_cls_state));
158  return ModelState::MODEL_LOAD_FAILED;
159  }
160  } else {
161  RCLCPP_INFO(
162  get_logger(), "Already state classification model loaded");
163  }
164 
165  this->set_model_state(ModelState::LOAD);
166 
167  return ModelState::SUCCESS;
168 }

170 int EvButtonDetector::unload_network()
171 {
172  using ModelState = perception_interfaces::srv::GetModelState::Response;
173 
174  try {
175  segmentation_.reset();
176  number_classification_.reset();
177  state_classification_.reset();
178  RCLCPP_INFO(get_logger(), "Models successfully deleted");
179 
180  this->set_model_state(ModelState::UNLOAD);
181  } catch (const std::exception & e) {
182  RCLCPP_ERROR_STREAM(get_logger(), "Exception: %s" << e.what());
183  return ModelState::MODEL_UNLOAD_FAILED;
184  }
185 
186  return ModelState::SUCCESS;
187 }

215 bool EvButtonDetector::update()
216 {
217  // rclcpp::Time on_calc_start = rclcpp::Clock().now();
218  cv::Mat cv_left_image, cv_right_image;
219  rclcpp::Time left_stamp, right_stamp;
220  std::string dummy;
221  bool ack_left_img, ack_right_img;
222 
223  if (this->is_data_available(left_img_sub_msg_name_) &&
224  this->is_data_available(right_img_sub_msg_name_))
225  {
226  ack_left_img =
227  this->get_data_image(left_img_sub_msg_name_, cv_left_image, left_stamp, dummy);
228  ack_right_img =
229  this->get_data_image(right_img_sub_msg_name_, cv_right_image, right_stamp, dummy);
230  } else {
231  return false;
232  }
233 
234  if (ack_left_img && ack_right_img) {
235  camera_status_ = true;
236  bool target_btn_state = false;
237  cv::Mat cv_vconcat_image;
238  cv_vconcat_image = visualizer_->convert_rgba_imgs_to_rgb_img(cv_left_image, cv_right_image);
239  cv::Mat resized_image;
240  std::map<std::string, cv::Point2i> tartget_2d_points;
241 
242  if (is_model_loaded() == true) {
243  segmentation_->model_run(cv_vconcat_image);
244 
245  if (segmentation_->get_patch_size() > 0) {
246  number_classification_->model_run(
247  segmentation_->patched_images_,
248  segmentation_->feature_pts_,
249  segmentation_->get_patch_size());
250  } else {
251  number_classification_->left_detected_buttons.clear();
252  number_classification_->right_detected_buttons.clear();
253  RCLCPP_WARN(
254  this->get_logger(),
255  "No segmented target (zero size)");
256  }
257  if (enable_recog_win_) {
258  segmentation_->show_segmentation_images();
259  visualizer_->show_classification_result(
260  cv_vconcat_image,
261  number_classification_->left_detected_buttons,
262  number_classification_->right_detected_buttons);
263  }
264  // inside, outside 각각 상황에서 나올 수 없는 label에 대한 필터링
265  geometry_msgs::msg::PoseStamped pose_stamp;
266  if (number_classification_->left_detected_buttons.size() > 0 &&
267  number_classification_->right_detected_buttons.size() > 0 &&
268  ev_cls_filter_mode_)
269  {
270  if (get_from_queue(left_stamp, pose_stamp)) {
271  if (project_to_image(pose_stamp, tartget_2d_points)) {
272  cv::circle(
273  cv_vconcat_image,
274  tartget_2d_points[CameraGeometry::left_zed], 30, cv::Scalar(0, 0, 255), -1);
275  cv::circle(
276  cv_vconcat_image,
277  tartget_2d_points[CameraGeometry::right_zed] +
278  cv::Point2i(0, cv_vconcat_image.size().height * 0.5), 30, cv::Scalar(255, 0, 0), -1);
279  filter_cls_using_target_point(
280  number_classification_->left_detected_buttons,
281  target_btn_str_,
282  tartget_2d_points[CameraGeometry::left_zed],
283  cv_vconcat_image.size().height * 0.5);
284  filter_cls_using_target_point(
285  number_classification_->right_detected_buttons,
286  target_btn_str_,
287  tartget_2d_points[CameraGeometry::right_zed],
288  cv_vconcat_image.size().height * 0.5);
289  }
290  }
291  }
292 
293  sensor_msgs::msg::PointCloud2 dummy;
294  dummy.header.stamp = left_stamp;
295  pose_estimator_->set_input(
296  number_classification_->left_detected_buttons,
297  number_classification_->right_detected_buttons,
298  dummy);
299 
300  geometry_msgs::msg::PoseStamped target_btn_pose;
301  int calc_state = pose_estimator_->calculate();
302  if (calc_state > -1) {
303  if (pose_estimator_->get_pose(target_btn_str_, target_btn_pose)) {
304  cv::cuda::GpuMat patch_img = cv::cuda::GpuMat();
305  if (visualizer_->get_closest_patch_image(
306  target_btn_str_,
307  pose_estimator_->get_pose_map(),
308  pose_estimator_->get_button_left(),
309  patch_img,
310  segmentation_->device_gray_mat_,
311  btn_lens_[0],
312  btn_lens_[1]))
313  {
314  target_btn_state = state_classification_->model_run(patch_img, 1) ? true : false;
315  } else {
316  RCLCPP_WARN(
317  this->get_logger(),
318  "Failed to on/off state check (no patch image), forcing false state");
319  }
320 
321  // detected btn goal pose - visual servoing
322  target_btn_pose.header.stamp = left_stamp;
323  target_btn_pose.header.frame_id = frame_id_;
324 
325  Eigen::Quaterniond q(
326  target_btn_pose.pose.orientation.w,
327  target_btn_pose.pose.orientation.x,
328  target_btn_pose.pose.orientation.y,
329  target_btn_pose.pose.orientation.z);
330  q.normalize();
331  target_btn_pose.pose.orientation.w = q.w();
332  target_btn_pose.pose.orientation.x = q.x();
333  target_btn_pose.pose.orientation.y = q.y();
334  target_btn_pose.pose.orientation.z = q.z();
335 
336  RCLCPP_INFO_STREAM(
337  this->get_logger(),
338  "frame time stamp: " << std::fixed << left_stamp.seconds());
339  RCLCPP_INFO_STREAM(
340  this->get_logger(),
341  "target button status: " << (target_btn_state ? "on" : "off"));
342  RCLCPP_INFO_STREAM(
343  this->get_logger(),
344  "target button name: " << target_btn_str_);
345  RCLCPP_INFO_STREAM(
346  this->get_logger(),
347  "target button position: " <<
348  target_btn_pose.pose.position.x << ", " <<
349  target_btn_pose.pose.position.y << ", " <<
350  target_btn_pose.pose.position.z);
351  RCLCPP_INFO_STREAM(
352  this->get_logger(),
353  "target button orientation: " <<
354  target_btn_pose.pose.orientation.w << ", " <<
355  target_btn_pose.pose.orientation.x << ", " <<
356  target_btn_pose.pose.orientation.y << ", " <<
357  target_btn_pose.pose.orientation.z);
358 
359  ev_detected_btn_goal_pose_pub_->publish(target_btn_pose);
360  } else {
361  RCLCPP_WARN(
362  this->get_logger(),
363  "Failed to get target pose (invalid target name, or no target in the acquired image)");
364  }
365 
366  // detected btn array
367  if (ev_detected_btn_arr_pub_->get_subscription_count() > 0) {
368  perception_interfaces::msg::DetectedButtonArray detected_btn_arr;
369  detected_btn_arr =
370  pose_estimator_->get_detected_button_array(target_btn_str_, target_btn_state);
371  detected_btn_arr.header.stamp = left_stamp;
372  detected_btn_arr.header.frame_id = frame_id_;
373  ev_detected_btn_arr_pub_->publish(detected_btn_arr);
374  } else {
375  RCLCPP_WARN(
376  this->get_logger(), "No detected button array subscriber!");
377  }
378  } else {
379  RCLCPP_WARN(
380  this->get_logger(), get_error_log(calc_state));
381  }
382  }
383 
384  visualizer_->overlay_stereo_images(
385  cv_vconcat_image,
386  pose_estimator_->get_pose_map(),
387  target_btn_str_,
388  target_btn_state,
389  button_shape_,
390  btn_lens_[0],
391  btn_lens_[1]);
392 
393  this->run_streaming(
394  cv_vconcat_image,
395  left_stamp,
396  frame_id_,
397  zed_streaming_pub_msg_name_,
398  zed_streaming_image_ratio_);
399 
400  tartget_2d_points.clear();
401  }
402  // rclcpp::Duration dur = rclcpp::Clock().now() - on_calc_start;
403  return true;
404 }

406 void EvButtonDetector::get_params()
407 {
408  RCLCPP_INFO_STREAM(this->get_logger(), "[Ros Parameters]");
409 
410  get_param(
411  "ros_topic_option.frame_id",
412  frame_id_,
413  std::string());
414  get_param(
415  "ros_topic_option.ev_button_mode",
416  ev_button_mode_,
417  std::string());
418  get_param(
419  "ros_topic_option.ev_detected_button_server",
420  ev_detected_button_server_,
421  std::string());
422  get_param(
423  "ros_topic_option.ev_target_pose_pub_msg_name",
424  ev_target_pose_msg_name_,
425  std::string());
426  get_param(
427  "zed_cam_ros_topic.left_rect_image_pub_msg_name",
428  left_img_sub_msg_name_,
429  std::string());
430  get_param(
431  "zed_cam_ros_topic.right_rect_image_pub_msg_name",
432  right_img_sub_msg_name_,
433  std::string());
434  get_param(
435  "zed_cam_ros_topic.left_rect_cam_info_pub_msg_name",
436  left_cam_info_sub_msg_name_,
437  std::string());
438  get_param(
439  "zed_cam_ros_topic.right_rect_cam_info_pub_msg_name",
440  right_cam_info_sub_msg_name_,
441  std::string());
442  get_param(
443  "inter_process.detected_button_goal_pose",
444  ev_detected_button_goal_pose_,
445  std::string());
446  get_param(
447  "inter_process.detected_button_arr",
448  ev_detected_button_arr_,
449  std::string());
450 
451  RCLCPP_INFO_STREAM(this->get_logger(), "[Pose Estimator Parameters]");
452 
453  double x_len, y_len;
454  get_param("pose_estimator.pat_btn_x_len", x_len, 40.0);
455  get_param("pose_estimator.pat_btn_y_len", y_len, 40.0);
456  btn_lens_[0] = x_len / 2;
457  btn_lens_[1] = y_len / 2;
458 
459  if (this->get_name() == inside_detector_name_) {
460  this->declare_parameter("pose_estimator.pat_btn_lbl");
461  std::vector<std::string> template_button_names =
462  this->get_parameter("pose_estimator.pat_btn_lbl").as_string_array();
463  if (template_button_names.empty()) {
464  RCLCPP_ERROR(this->get_logger(), "template button names parameters are empty!");
465  return;
466  }
467 
468  RCLCPP_INFO_STREAM(this->get_logger(), "pose_estimator.pat_btn_lbl: ");
469  for (auto lbl : template_button_names) {
470  RCLCPP_INFO_STREAM(this->get_logger(), lbl);
471  }
472 
473  bool is_auto_fill_enabled;
474  get_param("pose_estimator.pat_btn_auto_fill", is_auto_fill_enabled, false);
475 
476  std::map<std::string, Eigen::Vector3f> btn_pos;
477  if (is_auto_fill_enabled) {
478  double x_interval, y_interval;
479  get_param("pose_estimator.pat_x_space", x_interval, 0.0);
480  get_param("pose_estimator.pat_y_space", y_interval, 0.0);
481  int rows, cols;
482  get_param("pose_estimator.pat_btn_rows", rows, 0);
483  get_param("pose_estimator.pat_btn_cols", cols, 0);
484 
485  int next_id = 0;
486  for (int i = 0; i < rows; i++) {
487  for (int j = 0; j < cols; j++) {
488  btn_pos[template_button_names[next_id++]] =
489  Eigen::Vector3f(
490  static_cast<float>(i * y_interval),
491  static_cast<float>(j * x_interval),
492  0.0);
493  }
494  }
495  } else {
496  this->declare_parameter("pose_estimator.pat_btn_pos.x");
497  std::vector<double> button_coordi_x =
498  this->get_parameter(
499  "pose_estimator.pat_btn_pos.x").as_double_array();
500  this->declare_parameter("pose_estimator.pat_btn_pos.y");
501  std::vector<double> button_coordi_y =
502  this->get_parameter(
503  "pose_estimator.pat_btn_pos.y").as_double_array();
504  this->declare_parameter("pose_estimator.pat_btn_pos.z");
505  std::vector<double> button_coordi_z =
506  this->get_parameter(
507  "pose_estimator.pat_btn_pos.z").as_double_array();
508 
509  if (button_coordi_x.size() != button_coordi_y.size() &&
510  button_coordi_y.size() != button_coordi_z.size() &&
511  button_coordi_z.size() != button_coordi_x.size())
512  {
513  RCLCPP_ERROR(
514  this->get_logger(),
515  "different size of template_button_coordi arrays");
516  return;
517  }
518 
519  if (template_button_names.size() != button_coordi_x.size()) {
520  RCLCPP_ERROR(
521  this->get_logger(),
522  "different size of template_button_coordi and template_button_names arrays");
523  return;
524  }
525 
526  for (size_t i = 0; i < template_button_names.size(); i++) {
527  btn_pos[template_button_names[i]] =
528  Eigen::Vector3f(
529  button_coordi_x[i],
530  button_coordi_y[i],
531  button_coordi_z[i]);
532  }
533  }
534 
535  std::map<std::string, Eigen::Vector3f>::iterator iter;
536  for (iter = btn_pos.begin(); iter != btn_pos.end(); iter++) {
537  Eigen::MatrixXf btn_features(5, 3);
538  btn_features.row(0) = iter->second + Eigen::Vector3f(0, btn_lens_[1], 0);
539  btn_features.row(1) = iter->second + Eigen::Vector3f(-btn_lens_[0], 0, 0);
540  btn_features.row(2) = iter->second;
541  btn_features.row(3) = iter->second + Eigen::Vector3f(btn_lens_[0], 0, 0);
542  btn_features.row(4) = iter->second + Eigen::Vector3f(0, -btn_lens_[1], 0);
543  template_pts_[iter->first] = btn_features;
544  }
545 
546  double nor_x, nor_y, nor_z;
547  get_param("pose_estimator.pat_btn_nor.x", nor_x, 0.0);
548  get_param("pose_estimator.pat_btn_nor.y", nor_y, 0.0);
549  get_param("pose_estimator.pat_btn_nor.z", nor_z, 1.0);
550 
551  template_nor_ << nor_x, nor_y, nor_z;
552  }
553 
554  RCLCPP_INFO_STREAM(this->get_logger(), "[DL Inference Parameters]");
555 
556  std::string model_base_path;
557  get_param(
558  "dl_inference.model_path",
559  model_base_path,
560  std::string());
561  this->declare_parameter(
562  "button_list");
563  ev_button_list_ =
564  this->get_parameter(
565  "button_list").as_string_array();
566  for (auto btn_list : ev_button_list_) {
567  RCLCPP_INFO_STREAM(
568  this->get_logger(),
569  "Btn List : " << btn_list);
570  }
571  get_param(
572  "geometry_filter_mode",
573  ev_cls_filter_mode_, false);
574 
575  std::string path_prefix;
576  if (this->get_name() == inside_detector_name_) {
577  path_prefix = "dl_inference.ev_in_";
578  } else {
579  path_prefix = "dl_inference.ev_out_";
580  }
581 
582  std::string ev_btn_seg_path;
583  get_param(
584  path_prefix + "btn_seg",
585  ev_btn_seg_path,
586  std::string());
587  ev_btn_seg_path = model_base_path + ev_btn_seg_path;
588 
589  if (trt_inference::get_full_file_path(ev_btn_seg_path, ".engine", ev_img_seg_model_name_)) {

703 void EvButtonDetector::filter_cls_using_target_point(
704  std::vector<RecognizedText> & cls_results,
705  std::string target_name,
706  cv::Point2i target_2d_pts,
707  int offset)
708 {
709  std::vector<int> none_index;
710  int closest_index = -1;
711  double closest_distance = 1000000.0;
712  bool is_target_name_included = false;
713 
714  for (size_t i = 0; i < cls_results.size(); ++i) {
715  if (cls_results[i].features[2].y >= offset) {
716  cls_results[i].features[0].y -= offset;
717  cls_results[i].features[1].y -= offset;
718  cls_results[i].features[2].y -= offset;
719  cls_results[i].features[3].y -= offset;
720  cls_results[i].features[4].y -= offset;
721  }
722  if (cls_results[i].class_name == target_name) {
723  is_target_name_included = true;
724  }
725  if (!is_target_name_included) {
726  auto button_list_it = std::find(
727  ev_button_list_.begin(),
728  ev_button_list_.end(),
729  cls_results[i].class_name);
730  if (button_list_it == ev_button_list_.end() || cls_results[i].class_name == "none") {
731  cls_results[i].class_name = "none";
732  none_index.push_back(i);
733  double tmp_distance = cv::norm(target_2d_pts - cls_results[i].features[2]);
734  if (tmp_distance < closest_distance) {
735  closest_index = i;
736  closest_distance = tmp_distance;
737  }
738  }
739  }
740  }
741  if (is_target_name_included) {
742  RCLCPP_INFO_STREAM(
743  this->get_logger(),
744  "Target detected successfully");
745  } else {
746  // target으로 받은 2d pts
747  RCLCPP_INFO_STREAM(
748  this->get_logger(),
749  "Target detection fails and applies filtering to the target");
750  if (none_index.size() > 0) {
751  if (closest_index != -1) {
752  cls_results[closest_index].class_name = target_name;
753  }
754  // Filter된 None 라벨 외에 None 불필요한 Label이기 때문에 모두 삭제 처리함
755  std::sort(none_index.begin(), none_index.end(), std::greater<int>());
756  for (size_t i = 0; i < none_index.size(); i++) {
757  if (none_index[i] != closest_index) {
758  cls_results.erase(cls_results.begin() + none_index[i]);
759  }
760  }
761  }
762  }
763 }

776 bool EvButtonDetector::get_from_queue(
777  rclcpp::Time t, geometry_msgs::msg::PoseStamped & output)
778 {
779  if (msgs_.size() == 0) {
780  std::cout << "fail to get data" << std::endl;
781  return false;
782  }
783 
784  detector_mutex_.lock();
785  std::vector<geometry_msgs::msg::PoseStamped> tmp_msgs(msgs_);
786  detector_mutex_.unlock();
787 
788  double min_tol = 0.001; // 1 ms
789  double max_tol = 3.0;
790  double min_diff = 1000000.0;
791  double min_id = -1;
792 
793  // 갖고 있는 배열 중에 입력 time_stamp와 가장 가까이에 있는 obj를 찾음
794  for (int i = 0; i < tmp_msgs.size(); i++) {
795  rclcpp::Time msg_time(
796  tmp_msgs[i].header.stamp.sec,
797  tmp_msgs[i].header.stamp.nanosec);
798  double tmp = fabs(msg_time.seconds() - t.seconds());
799 
800  if (tmp <= min_diff) {
801  min_diff = tmp;
802  min_id = i;
803  output = tmp_msgs[i];
804  // 중간에 tolerance보다 낮은 obj가 찾고자 하는 obj라 간주하고 이를 리턴
805  if (min_diff < min_tol) {
806  return true;
807  }
808  } else { // 차이가 작아지다가 커지는 시점에서 리턴
809  output = tmp_msgs[min_id];
810  return min_diff > max_tol ? false : true;
811  }
812  }
813  return true;
814 }

816 bool EvButtonDetector::project_to_image(
817  geometry_msgs::msg::PoseStamped & msg,
818  std::map<std::string, cv::Point2i> & proj_pts)
819 {
820  using Util = detector_lib::PerceptionUtils;
821  CameraGeometry * geo = CameraGeometry::get_instance();
822 
823  std::string frame_id = msg.header.frame_id;
824  std::string cam_name = geo->get_cam_name_from(frame_id);
825  if (cam_name == "") {
826  RCLCPP_INFO_STREAM(this->get_logger(), "No frame id: " << frame_id);
827  return false;
828  }
829 
830  Eigen::Vector3f trans_pt =
831  geo->get_cam_to_opt().transpose() *
832  Util::ros_to_eigen(msg.pose.position) *
833  1000.0f;
834  Eigen::Matrix<float, 3, 4> left_proj_mat = geo->get_cam_proj_mat(cam_name);
835  Eigen::Vector2f left_proj_pt = geo->project_to_image(left_proj_mat, trans_pt);
836  Eigen::Matrix<float, 3, 4> right_proj_mat = geo->get_cam_proj_mat(
837  CameraGeometry::right_to_left_zed);
838  Eigen::Vector2f right_proj_pt = geo->project_to_image(right_proj_mat, trans_pt);
839 
840  cv::Point2i cv_left_proj_pt;
841  cv_left_proj_pt.x = left_proj_pt(0);
842  cv_left_proj_pt.y = left_proj_pt(1);
843  cv::Point2i cv_right_proj_pt;
844  cv_right_proj_pt.x = right_proj_pt(0);
845  cv_right_proj_pt.y = right_proj_pt(1);
846 
847  proj_pts[CameraGeometry::left_zed] = cv_left_proj_pt;
848  proj_pts[CameraGeometry::right_zed] = cv_right_proj_pt;
849 
850  return true;
