275 void RosNode::load_config()
276 {
277  std::string config_file_path =
278  ament_index_cpp::get_package_share_directory("aimbot_demo_gui") + "/res/";
279  std::string texts_file_path =
280  params_path_ + "/site/" + site_name_ + "/gui/localizations/texts.yaml";

282  texts_file_path = params_path_ + "/site/_default/gui/localizations/texts.yaml";
283  }
284 
285  try {
286  YAML::Node node = YAML::LoadFile(config_file_path + CONFIG_FILE_NAME);
287  YAML::Node config_node = node["config"];
288 
289  auto volume = config_node["volume"];
290  if (volume.IsDefined() == true && volume.IsNull() == false) {
291  config_volume_ = config_node["volume"].as<int>();
292  }
293 
294  auto volume2 = config_node["volume2"];
295  if (volume2.IsDefined() == true && volume2.IsNull() == false) {
296  config_volume2_ = config_node["volume2"].as<int>();
297  }
298 
299  auto volume3 = config_node["volume3"];
300  if (volume3.IsDefined() == true && volume3.IsNull() == false) {
301  config_volume3_ = config_node["volume3"].as<int>();
302  }
303 
304  auto password = config_node["password"];
305  if (password.IsDefined() == true && password.IsNull() == false) {
306  config_password_ = config_node["password"].as<std::string>();
307  }
308 
309  auto language = config_node["language"];
310  if (language.IsDefined() == true && language.IsNull() == false) {
311  config_language_ = config_node["language"].as<std::string>();
312  }
313 
314  auto cargo_merge_12 = config_node["cargo_merge_12"];
315  if (cargo_merge_12.IsDefined() == true && cargo_merge_12.IsNull() == false) {
316  config_cargo_merge_12_ = config_node["cargo_merge_12"].as<std::string>();

323  } catch (const std::exception & e) {
324  RCLCPP_ERROR_STREAM(get_logger(), "Fail to load config data : " << e.what());

326 
327  try {
328  YAML::Node node = YAML::LoadFile(texts_file_path.c_str());

337  RCLCPP_ERROR_STREAM(get_logger(), "Fail to load texts data : " << e.what());
338  RCLCPP_INFO(get_logger(), "texts_file: %s ", texts_file_path.c_str());
339  }

342 void RosNode::save_config()
343 {
344  std::string config_file_path =
345  ament_index_cpp::get_package_share_directory("aimbot_demo_gui") + "/res/";
346 
347  struct stat directory_info;

354  }

473 void RosNode::add_task()
474 {
475  if (cargo_type_.compare("drawer") == 0) {
476  int idx = 0;
477  if (cargo_goal_[TRAY_1].compare("") != 0 && cargo_goal_[TRAY_1] == cargo_goal_[TRAY_2]) {
478  add_transfer_task(++idx, cargo_goal_[TRAY_1], "0", cargo_phonenum_[TRAY_1]);
479  } else {
480  for (int i = 0; i < 2; i++) {
481  if (cargo_goal_[i].compare("") != 0) {
482  add_transfer_task(++idx, cargo_goal_[i], std::to_string(i + 1), cargo_phonenum_[i]);
483  }
484  }
485  }
486  } else {
487  std::vector<std::tuple<std::string, std::string, std::string>> task_list;
488 
489  const int TUPLE_TARGET_NAME_INDEX = 0;
490  const int TUPLE_CARGO_NUM_INDEX = 1;
491  const int TUPLE_PHONE_NUM_INDEX = 2;
492 
493  for (int i = 0; i < 4; i++) {
494  if ((config_cargo_merge_12_.compare("Y") == 0 && i == 1) ||
495  (config_cargo_merge_34_.compare("Y") == 0 && i == 3))
496  {
497  continue;
