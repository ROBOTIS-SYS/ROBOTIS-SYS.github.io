362 void PageArrivedDocent::on_action_plan_timeout()
363 {
364  action_index_++;
365  action_plan_timer_->stop();
366 
367  if (is_end_of_action(action_index_)) {
368  if (action_plan_repeat_cnt_ != -1) {
369  action_plan_repeat_cnt_--;
370  }
371  if (action_plan_repeat_cnt_ > 0 || action_plan_repeat_cnt_ == -1) {
372  action_index_ = 1;
373  do_actions(action_index_);
374  if (!is_curr_docent_done_) {
375  action_plan_timer_->start(static_cast<int>(get_action_time(action_index_) * 1000));
376  }
377  return;
378  }
379  is_curr_docent_done_ = true;
380  on_timeout();
381  } else {
382  do_actions(action_index_);
383  if (!is_curr_docent_done_) {
384  action_plan_timer_->start(static_cast<int>(get_action_time(action_index_) * 1000));
385  }
386  }

589 bool PageArrivedDocent::is_end_of_action(int action_index)
590 {
591  std::string action_plan_file_path =
592  doc_["destination_config"][docent_dest_index_]["action_plan"]["file"].as<std::string>();
593  if (!action_plan_file_path.empty() && action_plan_file_path.front() == '/') {
594  action_plan_file_path = action_plan_file_path.substr(1);
595  }
596  std::string file_path = action_plan_folder_path_ + action_plan_file_path;
597 
598  YAML::Node action_plan_doc;
599  try {
600  action_plan_doc = YAML::LoadFile(file_path);
601  } catch (const std::exception & e) {
602  RCLCPP_ERROR(
603  ros_node_->get_logger(), "Failed to load action plan file : %s",
604  file_path.c_str());
605  }
606 
607  for (const auto & iter : action_plan_doc[action_index]["action"]) {
608  // std::cout << iter.first.as<std::string>() << std::endl;
609  if (iter.first.as<std::string>().compare("end_of_action") == 0) {
610  return true;
611  }
612  }
613  if (static_cast<bool>(action_plan_doc[action_index + 1]) == false) {
614  RCLCPP_ERROR(
615  ros_node_->get_logger(),
