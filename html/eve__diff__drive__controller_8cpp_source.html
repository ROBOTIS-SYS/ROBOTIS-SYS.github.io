159 controller_interface::return_type EveDiffDriveController::update()
160 {
161  auto logger = node_->get_logger();
162  if (get_current_state().id() == State::PRIMARY_STATE_INACTIVE) {
163  if (!is_halted) {
164  halt();
165  is_halted = true;
166  }
167  return controller_interface::return_type::OK;
168  }
169 
170  const auto current_time = node_->get_clock()->now();
171 
172  std::shared_ptr<Twist> last_msg;
173  received_velocity_msg_ptr_.get(last_msg);
174 
175  if (last_msg == nullptr) {
176  RCLCPP_WARN(logger, "Velocity message received was a nullptr.");
177  return controller_interface::return_type::ERROR;
178  }
179 
180  const auto dt = current_time - last_msg->header.stamp;
181  // Brake if cmd_vel has timeout, override the stored command
182  if (dt > cmd_vel_timeout_) {
183  last_msg->twist.linear.x = 0.0;
184  last_msg->twist.angular.z = 0.0;
185  }
186 
187  Twist command = *last_msg;
188  double & linear_command = command.twist.linear.x;
189  double & angular_command = command.twist.angular.z;
190 
191  // Apply (possibly new) multipliers:
192  const auto wheels = wheel_params_;
193  const double wheel_separation = wheels.separation_multiplier * wheels.separation;
194  const double left_wheel_radius = wheels.left_radius_multiplier * wheels.radius;
195  const double right_wheel_radius = wheels.right_radius_multiplier * wheels.radius;
196 
197  if (odom_params_.open_loop) {
198  odometry_.updateOpenLoop(linear_command, angular_command, current_time);
199  } else {
200  double left_position_mean = 0.0;
201  double right_position_mean = 0.0;
202  for (size_t index = 0; index < wheels.wheels_per_side; ++index) {
203  const double left_position =
204  registered_left_wheel_handles_[index].position.get().get_value();
205  const double right_position =
206  registered_right_wheel_handles_[index].position.get().get_value();
207 
208  if (std::isnan(left_position) || std::isnan(right_position)) {
209  RCLCPP_ERROR(
210  logger, "Either the left or right wheel position is invalid for index [%zu]", index);
211  return controller_interface::return_type::ERROR;
212  }
213 
214  left_position_mean += left_position;
215  right_position_mean += right_position;
216  }
217  left_position_mean /= wheels.wheels_per_side;
218  right_position_mean /= wheels.wheels_per_side;
219 
220  odometry_.update(left_position_mean, right_position_mean, current_time);
221  }
222 
223  tf2::Quaternion orientation;
224  orientation.setRPY(0.0, 0.0, odometry_.getHeading());
225 
226  if (previous_publish_timestamp_ + publish_period_ < current_time) {
227  previous_publish_timestamp_ += publish_period_;
228 
229  if (realtime_odometry_publisher_->trylock()) {
230  auto & odometry_message = realtime_odometry_publisher_->msg_;
231  odometry_message.header.stamp = current_time;
232  odometry_message.pose.pose.position.x = odometry_.getX();
233  odometry_message.pose.pose.position.y = odometry_.getY();
234  odometry_message.pose.pose.orientation.x = orientation.x();
235  odometry_message.pose.pose.orientation.y = orientation.y();
236  odometry_message.pose.pose.orientation.z = orientation.z();
237  odometry_message.pose.pose.orientation.w = orientation.w();
238  odometry_message.twist.twist.linear.x = odometry_.getLinear();
239  odometry_message.twist.twist.angular.z = odometry_.getAngular();
240  realtime_odometry_publisher_->unlockAndPublish();
241  }
242 
243  if (odom_params_.enable_odom_tf && realtime_odometry_transform_publisher_->trylock()) {
244  auto & transform = realtime_odometry_transform_publisher_->msg_.transforms.front();
245  transform.header.stamp = current_time;
246  transform.transform.translation.x = odometry_.getX();
247  transform.transform.translation.y = odometry_.getY();
248  transform.transform.rotation.x = orientation.x();
249  transform.transform.rotation.y = orientation.y();
250  transform.transform.rotation.z = orientation.z();
251  transform.transform.rotation.w = orientation.w();
252  realtime_odometry_transform_publisher_->unlockAndPublish();
253  }
254  }
255 
256  const auto update_dt = current_time - previous_update_timestamp_;
257  previous_update_timestamp_ = current_time;
258 
259  auto & last_command = previous_commands_.back().twist;
260  auto & second_to_last_command = previous_commands_.front().twist;
261  limiter_linear_.limit(
262  linear_command, last_command.linear.x, second_to_last_command.linear.x, update_dt.seconds());
263  limiter_angular_.limit(
264  angular_command, last_command.angular.z, second_to_last_command.angular.z, update_dt.seconds());
265 
266  previous_commands_.pop();
267  previous_commands_.emplace(command);
268 
269  // Publish limited velocity
270  if (publish_limited_velocity_ && realtime_limited_velocity_publisher_->trylock()) {
271  auto & limited_velocity_command = realtime_limited_velocity_publisher_->msg_;
272  limited_velocity_command.header.stamp = current_time;
273  limited_velocity_command.twist = command.twist;
274  realtime_limited_velocity_publisher_->unlockAndPublish();
275  }
276 
277 
279  SafetyFilter::UltrasonicRawData ultrasonic_left;
280  ultrasonic_left.range =
281  ultrasonic_handles_.at("ultrasonic_left").range.get().get_value();
282  ultrasonic_left.field_of_view =
283  ultrasonic_handles_.at("ultrasonic_left").fov.get().get_value();
284 
285  SafetyFilter::UltrasonicRawData ultrasonic_front_left;
286  ultrasonic_front_left.range =
287  ultrasonic_handles_.at("ultrasonic_front_left").range.get().get_value();
288  ultrasonic_front_left.field_of_view =
289  ultrasonic_handles_.at("ultrasonic_front_left").fov.get().get_value();
290 
291  SafetyFilter::UltrasonicRawData ultrasonic_front_right;
292  ultrasonic_front_right.range =
293  ultrasonic_handles_.at("ultrasonic_front_right").range.get().get_value();
294  ultrasonic_front_right.field_of_view =
295  ultrasonic_handles_.at("ultrasonic_front_right").fov.get().get_value();
296 
297  SafetyFilter::UltrasonicRawData ultrasonic_right;
298  ultrasonic_right.range =
299  ultrasonic_handles_.at("ultrasonic_right").range.get().get_value();
300  ultrasonic_right.field_of_view =
301  ultrasonic_handles_.at("ultrasonic_right").fov.get().get_value();
302 
303  SafetyFilter::UltrasonicRawData ultrasonic_back_right;
304  ultrasonic_back_right.range =
305  ultrasonic_handles_.at("ultrasonic_back_right").range.get().get_value();
306  ultrasonic_back_right.field_of_view =
307  ultrasonic_handles_.at("ultrasonic_back_right").fov.get().get_value();
308 
309  SafetyFilter::UltrasonicRawData ultrasonic_back;
310  ultrasonic_back.range =
311  ultrasonic_handles_.at("ultrasonic_back").range.get().get_value();
312  ultrasonic_back.field_of_view =
313  ultrasonic_handles_.at("ultrasonic_back").fov.get().get_value();
314 
315  SafetyFilter::UltrasonicRawData ultrasonic_back_left;
316  ultrasonic_back_left.range =
317  ultrasonic_handles_.at("ultrasonic_back_left").range.get().get_value();
318  ultrasonic_back_left.field_of_view =
319  ultrasonic_handles_.at("ultrasonic_back_left").fov.get().get_value();
320 
321 
322  safety_filter_.update_ultrasonic_data(
323  ultrasonic_left,
324  ultrasonic_front_left,
325  ultrasonic_front_right,
326  ultrasonic_right,
327  ultrasonic_back_right,
328  ultrasonic_back,
329  ultrasonic_back_left
330  );
331 
332  // command may be limited further by SpeedLimit,
333  // without affecting the stored twist command
334  SafetyFilter::FilterResult filter_result = safety_filter_.filtering_cmd_vel(command);
335  command = filter_result.cmd_vel;
336 
338  static std::stringstream log;
339  static double log_most_ct = 100.0;
340  static int log_cnt = 0;
341  static bool detected = false;
342  if (filter_result.decelration) {
343  if (log_most_ct > filter_result.clash_time) {
344  log.str("");
345  log.setf(std::ios::fixed);
346  log.precision(1);
347  log << "Filterd Vel " << (filter_result.deceleration_rate * 100.0) << "%";
348  log << "(clash time is " << filter_result.clash_time << "s";
349  log.precision(2);
350  for (auto detected : filter_result.detected_sensor_data) {
351  log << " /" << detected.first << ":" << detected.second << "m";
352  }
353  log << ")";
354  log_most_ct = filter_result.clash_time;
355  }
356  detected = true;
357  }
358  log_cnt++;
359  if (log_cnt > 50 && detected) {
360  publish_log(
361  log.str(),
362  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
363  aimbot_log_interfaces::msg::Log::TYPE_WARN);
364  log_cnt = 0;
365  detected = false;
366  }
368 
369  // Compute wheels velocities:
370  const double velocity_left =
371  (linear_command - angular_command * wheel_separation / 2.0) / left_wheel_radius;
372  const double velocity_right =
373  (linear_command + angular_command * wheel_separation / 2.0) / right_wheel_radius;
374 
375  // Set wheels velocities:
376  for (size_t index = 0; index < wheels.wheels_per_side; ++index) {
377  registered_left_wheel_handles_[index].velocity.get().set_value(velocity_left);
378  registered_right_wheel_handles_[index].velocity.get().set_value(velocity_right);
379  }
380 
381  return controller_interface::return_type::OK;
382 }

384 CallbackReturn EveDiffDriveController::on_configure(const rclcpp_lifecycle::State &)
385 {
386  auto logger = node_->get_logger();
387 
388  // update parameters
389  left_wheel_names_ = node_->get_parameter("left_wheel_names").as_string_array();
390  right_wheel_names_ = node_->get_parameter("right_wheel_names").as_string_array();
391 
392  if (left_wheel_names_.size() != right_wheel_names_.size()) {
393  RCLCPP_ERROR(
394  logger, "The number of left wheels [%zu] and the number of right wheels [%zu] are different",
395  left_wheel_names_.size(), right_wheel_names_.size());
396  return CallbackReturn::ERROR;
397  }
398 
399  if (left_wheel_names_.empty()) {
400  RCLCPP_ERROR(logger, "Wheel names parameters are empty!");
401  return CallbackReturn::ERROR;
402  }
403 
404  wheel_params_.separation = node_->get_parameter("wheel_separation").as_double();
405  wheel_params_.wheels_per_side =
406  static_cast<size_t>(node_->get_parameter("wheels_per_side").as_int());
407  wheel_params_.radius = node_->get_parameter("wheel_radius").as_double();
408  wheel_params_.separation_multiplier =
409  node_->get_parameter("wheel_separation_multiplier").as_double();
410  wheel_params_.left_radius_multiplier =
411  node_->get_parameter("left_wheel_radius_multiplier").as_double();
412  wheel_params_.right_radius_multiplier =
413  node_->get_parameter("right_wheel_radius_multiplier").as_double();
414 
415  const auto wheels = wheel_params_;
416 
417  const double wheel_separation = wheels.separation_multiplier * wheels.separation;
418  const double left_wheel_radius = wheels.left_radius_multiplier * wheels.radius;
419  const double right_wheel_radius = wheels.right_radius_multiplier * wheels.radius;
420 
421  odometry_.setWheelParams(wheel_separation, left_wheel_radius, right_wheel_radius);
422  odometry_.setVelocityRollingWindowSize(
423  node_->get_parameter("velocity_rolling_window_size").as_int());
424 
425  odom_params_.odom_frame_id = node_->get_parameter("odom_frame_id").as_string();
426  odom_params_.base_frame_id = node_->get_parameter("base_frame_id").as_string();
427 
428  auto pose_diagonal = node_->get_parameter("pose_covariance_diagonal").as_double_array();
429  std::copy(
430  pose_diagonal.begin(), pose_diagonal.end(), odom_params_.pose_covariance_diagonal.begin());
431 
432  auto twist_diagonal = node_->get_parameter("twist_covariance_diagonal").as_double_array();
433  std::copy(
434  twist_diagonal.begin(), twist_diagonal.end(), odom_params_.twist_covariance_diagonal.begin());
435 
436  odom_params_.open_loop = node_->get_parameter("open_loop").as_bool();
437  odom_params_.enable_odom_tf = node_->get_parameter("enable_odom_tf").as_bool();
438 
439  cmd_vel_timeout_ = std::chrono::milliseconds{
440  static_cast<int>(node_->get_parameter("cmd_vel_timeout").as_double() * 1000.0)};
441  publish_limited_velocity_ = node_->get_parameter("publish_limited_velocity").as_bool();
442  use_stamped_vel_ = node_->get_parameter("use_stamped_vel").as_bool();
443 
444  try {
445  limiter_linear_ = SpeedLimiter(
446  node_->get_parameter("linear.x.has_velocity_limits").as_bool(),
447  node_->get_parameter("linear.x.has_acceleration_limits").as_bool(),
448  node_->get_parameter("linear.x.has_jerk_limits").as_bool(),
449  node_->get_parameter("linear.x.min_velocity").as_double(),
450  node_->get_parameter("linear.x.max_velocity").as_double(),
451  node_->get_parameter("linear.x.min_acceleration").as_double(),
452  node_->get_parameter("linear.x.max_acceleration").as_double(),
453  node_->get_parameter("linear.x.min_jerk").as_double(),
454  node_->get_parameter("linear.x.max_jerk").as_double());
455  } catch (const std::runtime_error & e) {
456  RCLCPP_ERROR(node_->get_logger(), "Error configuring linear speed limiter: %s", e.what());
457  }
458 
459  try {
460  limiter_angular_ = SpeedLimiter(
461  node_->get_parameter("angular.z.has_velocity_limits").as_bool(),
462  node_->get_parameter("angular.z.has_acceleration_limits").as_bool(),
463  node_->get_parameter("angular.z.has_jerk_limits").as_bool(),
464  node_->get_parameter("angular.z.min_velocity").as_double(),
465  node_->get_parameter("angular.z.max_velocity").as_double(),
466  node_->get_parameter("angular.z.min_acceleration").as_double(),
467  node_->get_parameter("angular.z.max_acceleration").as_double(),
468  node_->get_parameter("angular.z.min_jerk").as_double(),
469  node_->get_parameter("angular.z.max_jerk").as_double());
470  } catch (const std::runtime_error & e) {
471  RCLCPP_ERROR(node_->get_logger(), "Error configuring angular speed limiter: %s", e.what());
472  }
473 
474  safety_filter_params_.check_time = node_->get_parameter("check_time").as_double();
475  RCLCPP_INFO_STREAM(
476  node_->get_logger(), "Check time is " << safety_filter_params_.check_time << "s.");
477  safety_filter_params_.stop_time = node_->get_parameter("stop_time").as_double();
478  RCLCPP_INFO_STREAM(
479  node_->get_logger(),
480  "Stop time is " << safety_filter_params_.stop_time << "s.");
481  safety_filter_params_.base_radius = node_->get_parameter("base_radius").as_double();
482  RCLCPP_INFO_STREAM(
483  node_->get_logger(), "Base radius is " << safety_filter_params_.base_radius << "m.");
484 
485  ultrasonic_names_ = node_->get_parameter("ultrasonic_names").as_string_array();
486 
487  safety_filter_.set_parameters(safety_filter_params_);
488 
489  if (!reset()) {
490  return CallbackReturn::ERROR;
491  }
492 
493  // Safety Filter enable state serice server
494  safety_filter_.set_safety_filter(true);
495  safety_filter_enable_srv_ = node_->create_service<std_srvs::srv::SetBool>(
496  DEFAULT_SAFETY_FILTER_SERVICE, [this](
497  const std::shared_ptr<std_srvs::srv::SetBool::Request> request,
498  std::shared_ptr<std_srvs::srv::SetBool::Response> response) {
499  response->message = this->safety_filter_.set_safety_filter(request->data);
500  response->success = true;
501  });
502 
503  // Debuging Path publisher
504  debug_path_pub_ = node_->create_publisher<nav_msgs::msg::Path>(
505  DEFAULT_DEBUG_PATH_TOPIC, rclcpp::SystemDefaultsQoS());
506 
507  // LOG publisher
508  log_pub_ = node_->create_publisher<aimbot_log_interfaces::msg::Log>("/aimbot_log", 10);
509 
510  // left and right sides are both equal at this point
511  wheel_params_.wheels_per_side = left_wheel_names_.size();
512 
513  if (publish_limited_velocity_) {
514  limited_velocity_publisher_ =
515  node_->create_publisher<Twist>(DEFAULT_COMMAND_OUT_TOPIC, rclcpp::SystemDefaultsQoS());
516  realtime_limited_velocity_publisher_ =
517  std::make_shared<realtime_tools::RealtimePublisher<Twist>>(limited_velocity_publisher_);
518  }
519 
520  const Twist empty_twist;
521  received_velocity_msg_ptr_.set(std::make_shared<Twist>(empty_twist));
522 
523  // Fill last two commands with default constructed commands
524  previous_commands_.emplace(empty_twist);
525  previous_commands_.emplace(empty_twist);
526 
527  // initialize command subscriber
528  if (use_stamped_vel_) {
529  velocity_command_subscriber_ = node_->create_subscription<Twist>(
530  DEFAULT_COMMAND_TOPIC, rclcpp::SystemDefaultsQoS(),
531  [this](const std::shared_ptr<Twist> msg) -> void {
532  if (!subscriber_is_active_) {
533  RCLCPP_WARN(node_->get_logger(), "Can't accept new commands. subscriber is inactive");
534  return;
535  }
536  if ((msg->header.stamp.sec == 0) && (msg->header.stamp.nanosec == 0)) {
537  RCLCPP_WARN_ONCE(
538  node_->get_logger(),
539  "Received TwistStamped with zero timestamp, setting it to current "
540  "time, this message will only be shown once");
541  msg->header.stamp = node_->get_clock()->now();
542  }
543  received_velocity_msg_ptr_.set(std::move(msg));
544  });
545  } else {
546  velocity_command_unstamped_subscriber_ = node_->create_subscription<geometry_msgs::msg::Twist>(
547  DEFAULT_COMMAND_UNSTAMPED_TOPIC, rclcpp::SystemDefaultsQoS(),
548  [this](const std::shared_ptr<geometry_msgs::msg::Twist> msg) -> void {
549  if (!subscriber_is_active_) {
550  RCLCPP_WARN(node_->get_logger(), "Can't accept new commands. subscriber is inactive");
551  return;
552  }
553 
554  // Write fake header in the stored stamped command
555  std::shared_ptr<Twist> twist_stamped;
556  received_velocity_msg_ptr_.get(twist_stamped);
557  twist_stamped->twist = *msg;
558  twist_stamped->header.stamp = node_->get_clock()->now();
559  });
560  }
561 
562  // initialize odometry publisher and messasge
563  odometry_publisher_ = node_->create_publisher<nav_msgs::msg::Odometry>(
564  DEFAULT_ODOMETRY_TOPIC, rclcpp::SystemDefaultsQoS());
565  realtime_odometry_publisher_ =
566  std::make_shared<realtime_tools::RealtimePublisher<nav_msgs::msg::Odometry>>(
567  odometry_publisher_);
568 
569  auto & odometry_message = realtime_odometry_publisher_->msg_;
570  odometry_message.header.frame_id = odom_params_.odom_frame_id;
571  odometry_message.child_frame_id = odom_params_.base_frame_id;
572 
573  // limit the publication on the topics /odom and /tf
574  publish_rate_ = node_->get_parameter("publish_rate").as_double();
575  publish_period_ = rclcpp::Duration::from_seconds(1.0 / publish_rate_);
576  previous_publish_timestamp_ = node_->get_clock()->now();
577 
578  // initialize odom values zeros
579  odometry_message.twist =
580  geometry_msgs::msg::TwistWithCovariance(rosidl_runtime_cpp::MessageInitialization::ALL);
581 
582  constexpr size_t NUM_DIMENSIONS = 6;
583  for (size_t index = 0; index < 6; ++index) {
584  // 0, 7, 14, 21, 28, 35
585  const size_t diagonal_index = NUM_DIMENSIONS * index + index;
586  odometry_message.pose.covariance[diagonal_index] = odom_params_.pose_covariance_diagonal[index];
587  odometry_message.twist.covariance[diagonal_index] =
588  odom_params_.twist_covariance_diagonal[index];
589  }
590 
591  // initialize transform publisher and message
592  odometry_transform_publisher_ = node_->create_publisher<tf2_msgs::msg::TFMessage>(
593  DEFAULT_TRANSFORM_TOPIC, rclcpp::SystemDefaultsQoS());
594  realtime_odometry_transform_publisher_ =
595  std::make_shared<realtime_tools::RealtimePublisher<tf2_msgs::msg::TFMessage>>(
596  odometry_transform_publisher_);
597 
598  // keeping track of odom and base_link transforms only
599  auto & odometry_transform_message = realtime_odometry_transform_publisher_->msg_;
600  odometry_transform_message.transforms.resize(1);
601  odometry_transform_message.transforms.front().header.frame_id = odom_params_.odom_frame_id;
602  odometry_transform_message.transforms.front().child_frame_id = odom_params_.base_frame_id;
603 
604  previous_update_timestamp_ = node_->get_clock()->now();
605  return CallbackReturn::SUCCESS;

699 CallbackReturn EveDiffDriveController::configure_side(
700  const std::string & side, const std::vector<std::string> & wheel_names,
701  std::vector<WheelHandle> & registered_handles)
702 {
703  auto logger = node_->get_logger();
704 
705  if (wheel_names.empty()) {
706  RCLCPP_ERROR(logger, "No '%s' wheel names specified", side.c_str());
707  return CallbackReturn::ERROR;
708  }
709 
710  // register handles
711  registered_handles.reserve(wheel_names.size());
712  for (const auto & wheel_name : wheel_names) {
713  const auto state_handle = std::find_if(
714  state_interfaces_.cbegin(), state_interfaces_.cend(), [&wheel_name](const auto & interface) {
715  return interface.get_name() == wheel_name &&
716  interface.get_interface_name() == HW_IF_POSITION;
717  });
718 
719  if (state_handle == state_interfaces_.cend()) {
720  RCLCPP_ERROR(logger, "Unable to obtain joint state handle for %s", wheel_name.c_str());
721  return CallbackReturn::ERROR;
722  }
723 
724  const auto command_handle = std::find_if(
725  command_interfaces_.begin(), command_interfaces_.end(),
726  [&wheel_name](const auto & interface) {
727  return interface.get_name() == wheel_name &&
728  interface.get_interface_name() == HW_IF_VELOCITY;
729  });
730 
731  if (command_handle == command_interfaces_.end()) {
732  RCLCPP_ERROR(logger, "Unable to obtain joint command handle for %s", wheel_name.c_str());
733  return CallbackReturn::ERROR;
734  }
735 
736  registered_handles.emplace_back(
737  WheelHandle{std::ref(*state_handle), std::ref(*command_handle)});
738  }
739 
740  return CallbackReturn::SUCCESS;
741 }

743 CallbackReturn EveDiffDriveController::configure_ultrasonic_handles(
744  std::vector<std::string> sensor_names)
745 {
746  if (sensor_names.empty()) {
747  RCLCPP_ERROR(node_->get_logger(), "No sensor names specified");
748  return CallbackReturn::ERROR;
749  }
750 
751  // register handles
752  ultrasonic_handles_.clear();
753  for (const auto & sensor_name : sensor_names) {
754  // find range
755  const auto range_handle = std::find_if(
756  state_interfaces_.cbegin(),
757  state_interfaces_.cend(),
758  [&sensor_name](const auto & interface) {
759  return interface.get_name() == sensor_name &&
760  interface.get_interface_name() == "range";
761  });
762  if (range_handle == state_interfaces_.cend()) {
763  RCLCPP_ERROR(
764  node_->get_logger(), "Unable to obtain Range state handle for %s", sensor_name.c_str());
765  return CallbackReturn::ERROR;
766  }
767 
768  // find fov
769  const auto fov_handle = std::find_if(
770  state_interfaces_.cbegin(),
771  state_interfaces_.cend(),
772  [&sensor_name](const auto & interface) {
773  return interface.get_name() == sensor_name &&
774  interface.get_interface_name() == "field_of_view";
775  });
776  if (fov_handle == state_interfaces_.cend()) {
777  RCLCPP_ERROR(
778  node_->get_logger(), "Unable to obtain fov state handle for %s", sensor_name.c_str());
779  return CallbackReturn::ERROR;
780  }
781 
782  ultrasonic_handles_.insert(
783  std::make_pair(
784  sensor_name,
785  UltrasonicHandle{std::ref(*range_handle), std::ref(*fov_handle)}));
786  }
787 
788  return CallbackReturn::SUCCESS;

791 void EveDiffDriveController::pub_debug_path(geometry_msgs::msg::Twist cmd_vel)
792 {
793  nav_msgs::msg::Path path;
794  path.header.stamp = node_->now();
795  path.header.frame_id = "base_link";
796  for (double check_time = 0.0;
797  check_time < safety_filter_params_.check_time;
798  check_time += 0.1)
799  {
800  geometry_msgs::msg::PoseStamped single_pose;
801 
802  double v = cmd_vel.linear.x;
803  double w = cmd_vel.angular.z;
804 
805  double delta_x = 0.0;
806  double delta_y = 0.0;
807  double delta_theta = 0.0;
808  if (w == 0.0) {
809  delta_x = v * check_time;
810  delta_y = 0.0;
811  } else {
812  double r = v / w;
813  delta_theta = w * check_time;
814  delta_x = r * sin(delta_theta);
815  delta_y = r * (1 - cos(delta_theta));
816  }
817 
818  single_pose.header.stamp = node_->now();
819  single_pose.header.frame_id = "base_link";
820  single_pose.pose.position.x = delta_x;
821  single_pose.pose.position.y = delta_y;
822  single_pose.pose.position.z = 0.0;
823  single_pose.pose.orientation.w = cos(delta_theta / 2);
824  single_pose.pose.orientation.x = 0.0;
825  single_pose.pose.orientation.y = 0.0;
826  single_pose.pose.orientation.z = sin(delta_theta / 2);
827  path.poses.push_back(single_pose);
828  }
829 
830  debug_path_pub_->publish(path);
831 }
