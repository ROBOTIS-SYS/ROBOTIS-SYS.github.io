89 bool V4L2Driver::query_capabilities()
90 {
91  struct v4l2_capability capability;
92  // 연결된 device의 이름, 수행 가능한 동작 등 장치의 정보를 받아오는 명령
93  if (ioctl(driver_, VIDIOC_QUERYCAP, &capability) == -1) {
94  std::cout << "Failed to query capability" << std::endl;
95  close(driver_);
96  return false;
97  }
98  std::cout << "[Driver Caps]" << std::endl;
99  std::cout << "Driver: " << capability.driver << std::endl;
100  std::cout << "Card: " << capability.card << std::endl;
101  std::cout << "Bus: " << capability.bus_info << std::endl;
102  std::cout << "Version: " << capability.version << std::endl;

125 bool V4L2Driver::request_buffers()
126 {
127  struct v4l2_requestbuffers req_buffers;
128  memset(&req_buffers, 0, sizeof(req_buffers));
129  req_buffers.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
130  req_buffers.memory = V4L2_MEMORY_MMAP;
131  req_buffers.count = buffer_count_;
132 
133  // video device에서 받아온 데이터를 저장하기 위한 buffer를 할당하는 명령
134  if (ioctl(driver_, VIDIOC_REQBUFS, &req_buffers) == -1) {
135  std::cout << "Failed to request buffers" << std::endl;
136  close(driver_);
137  return false;
138  }
139 
140  buffers_ = (struct Buffer *)calloc(buffer_count_, sizeof(*buffers_));
141  return true;
142 }

144 bool V4L2Driver::map_memory()
145 {
146  // device 내 사용하는 buffer를 사용자 영역의 buffer와 맵핑
147  for (int i = 0; i < buffer_count_; ++i) {
148  struct v4l2_buffer buffer;
149  memset(&buffer, 0, sizeof(buffer));
150  buffer.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
151  buffer.memory = V4L2_MEMORY_MMAP;
152  buffer.index = i;
153 
154  if (!query_buffer(buffer)) {
155  return false;
156  }
157 
158  buffers_[i].length = buffer.length;
159  buffers_[i].start = mmap(
160  NULL, buffer.length, PROT_READ | PROT_WRITE, MAP_SHARED, driver_, buffer.m.offset);
161 
162  if (buffers_[i].start == MAP_FAILED) {
163  std::cout << "Failed to mmap" << std::endl;
164  close(driver_);
165  return false;
166  }
167 
168  if (!queue_buffer(buffer)) {
169  return false;
170  }
171  }
172 
173  return true;
174 }

198 bool V4L2Driver::dequeue_buffer(struct v4l2_buffer & buffer)
199 {
200  // VIDIOC_QBUF 명령 수행으로 얻은 프레임의 index를 토대로,

202  if (ioctl(driver_, VIDIOC_DQBUF, &buffer) == -1) {
203  std::cout << "Failed to dequeue buffer" << std::endl;
204  close(driver_);
205  return false;
206  }

258 bool V4L2Driver::grab()
259 {
260  struct v4l2_buffer buffer;
261  memset(&buffer, 0, sizeof(buffer));
262  buffer.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
263  buffer.memory = V4L2_MEMORY_MMAP;
264 
265  if (!dequeue_buffer(buffer)) {
266  return false;
267  }
268 
269  cv::Mat color_YUV_image(format_.fmt.pix.height, format_.fmt.pix.width, CV_8UC2);
270  memcpy(color_YUV_image.data, buffers_[buffer.index].start, format_.fmt.pix.sizeimage);
271  cv::cvtColor(color_YUV_image, color_image_, cv::COLOR_YUV2BGR_UYVY);
272 
273  if (!queue_buffer(buffer)) {
274  return false;
275  }
276 
277  return true;
278 }
