50 template<typename Iter, typename Getter>
51 Iter min_by(Iter begin, Iter end, Getter getCompareVal)
52 {
53  if (begin == end) {
54  return end;
55  }
56  auto lowest = getCompareVal(*begin);
57  Iter lowest_it = begin;
58  for (Iter it = ++begin; it != end; ++it) {
59  auto comp = getCompareVal(*it);
60  if (comp < lowest) {
61  lowest = comp;
62  lowest_it = it;
63  }
64  }
65  return lowest_it;
66 }

426 double ElevatorRegulatedPurePursuitController::getLookAheadDistance(
427  const geometry_msgs::msg::Twist & speed)
428 {
429  // If using velocity-scaled look ahead distances, find and clamp the dist
430  // Else, use the static look ahead distance
431  double lookahead_dist = lookahead_dist_;
432  if (use_velocity_scaled_lookahead_dist_) {
433  lookahead_dist = speed.linear.x * lookahead_time_;
434  lookahead_dist = std::clamp(lookahead_dist, min_lookahead_dist_, max_lookahead_dist_);
435  }
436 
437  return lookahead_dist;
438 }

440 geometry_msgs::msg::TwistStamped ElevatorRegulatedPurePursuitController::computeVelocityCommands(
441  const geometry_msgs::msg::PoseStamped & pose,
442  const geometry_msgs::msg::Twist & speed,
443  const geometry_msgs::msg::Pose & goal_pose)
444 {
445  // Transform path to robot base frame
446  auto transformed_plan = transformGlobalPlan(pose);
447 
448  // Choose a feasible path from the path candidates (localization error compensation)
449  if (use_local_error_compensation_) {
450  transformed_plan = compensateLocalError(transformed_plan, pose, speed);
451  }
452 
453  // Find look ahead distance and point on path and publish
454  const double lookahead_dist = getLookAheadDistance(speed);
455  auto carrot_pose = getLookAheadPoint(lookahead_dist, transformed_plan);
456  carrot_pub_->publish(std::move(createCarrotMsg(carrot_pose)));
457 
458  double linear_vel, angular_vel;
459 
460  // Find distance^2 to look ahead point (carrot) in robot base frame
461  // This is the chord length of the circle
462  const double carrot_dist2 =
463  (carrot_pose.pose.position.x * carrot_pose.pose.position.x) +
464  (carrot_pose.pose.position.y * carrot_pose.pose.position.y);
465 
466  // Find curvature of circle (k = 1 / R)
467  double curvature = 0.0;
468  if (carrot_dist2 > 0.001) {
469  curvature = 2.0 * carrot_pose.pose.position.y / carrot_dist2;
470  }
471  linear_vel = desired_linear_vel_;
472 
473  // Make sure we're in compliance with basic constraints
474  double angle_to_heading;
475  if (shouldRotateToGoalHeading(carrot_pose)) {
476  // double angle_to_goal = tf2::getYaw(transformed_plan.poses.back().pose.orientation);
477  double angle_to_goal = angles::shortest_angular_distance(
478  tf2::getYaw(pose.pose.orientation),
479  tf2::getYaw(goal_pose.orientation));
480  rotateToHeading(linear_vel, angular_vel, angle_to_goal, speed);
481  RCLCPP_INFO_STREAM(logger_, "[ev controller] --> Rotational moving to goal");
482  RCLCPP_INFO_STREAM(logger_, "[ev controller] yaw difference to goal: " << angle_to_goal);
483  } else if (shouldRotateToPath(carrot_pose, angle_to_heading) && check_path_heading_) {
484  rotateToHeading(linear_vel, angular_vel, angle_to_heading, speed);
485  RCLCPP_INFO_STREAM(logger_, "[ev controller] --> Rotational moving to path");
486  RCLCPP_INFO_STREAM(logger_, "[ev controller] yaw difference to path: " << angle_to_heading);
487  } else {
488  applyConstraints(
489  fabs(lookahead_dist - sqrt(carrot_dist2)),
490  lookahead_dist, curvature, speed,
491  pose,
492  transformed_plan,
493  linear_vel);
494 
495  // Apply curvature to angular velocity after constraining linear velocity

614  const geometry_msgs::msg::PoseStamped & robot_pose,
615  const double & linear_vel, const double & angular_vel)
616 {
617  // Note(stevemacenski): This may be a bit unusual, but the robot_pose is in
618  // odom frame and the carrot_pose is in robot base frame.
619 
620  // check current point is OK
621  if (inCollision(robot_pose.pose.position.x, robot_pose.pose.position.y)) {
622  return true;
623  }
624 
625  // visualization messages
626  nav_msgs::msg::Path arc_pts_msg;
627  arc_pts_msg.header.frame_id = costmap_ros_->getGlobalFrameID();
628  arc_pts_msg.header.stamp = robot_pose.header.stamp;
629  geometry_msgs::msg::PoseStamped pose_msg;
630  pose_msg.header.frame_id = arc_pts_msg.header.frame_id;
631  pose_msg.header.stamp = arc_pts_msg.header.stamp;
632

643  if (i * projection_time > max_allowed_time_to_collision_) {
644  break;
645  }
646 
647  i++;
648 
649  // apply velocity at curr_pose over distance
650  curr_pose.x += projection_time * (linear_vel * cos(curr_pose.theta));
651  curr_pose.y += projection_time * (linear_vel * sin(curr_pose.theta));
652  curr_pose.theta += projection_time * angular_vel;

654  // store it for visualization
655  pose_msg.pose.position.x = curr_pose.x;
656  pose_msg.pose.position.y = curr_pose.y;
657  pose_msg.pose.position.z = 0.01;
658  arc_pts_msg.poses.push_back(pose_msg);

660  // check for collision at this point
661  if (inCollision(curr_pose.x, curr_pose.y)) {
662  carrot_arc_pub_->publish(arc_pts_msg);
663  return true;
664  }
665  }
666 
667  carrot_arc_pub_->publish(arc_pts_msg);
668 
669  return false;
670 }

695 void ElevatorRegulatedPurePursuitController::applyConstraints(
696  const double & dist_error, const double & lookahead_dist,
697  const double & curvature, const geometry_msgs::msg::Twist & curr_speed,
698  const geometry_msgs::msg::PoseStamped & robot_pose,
699  const nav_msgs::msg::Path & path,
700  double & linear_vel)
701 {
702  double curvature_vel = linear_vel;
703  double cost_vel = linear_vel;
704  double approach_vel = linear_vel;
705 
706  // ************************************ CURVATURE ********************************** //
707  const double radius = fabs(1.0 / curvature);
708  const double & min_rad = regulated_linear_scaling_min_radius_;
709  RCLCPP_INFO_STREAM(logger_, "[ev controller] current path radius: " << radius);
710  RCLCPP_INFO_STREAM(logger_, "[ev controller] mininum radius: " << min_rad);
711  bool result = fabs(curvature) > (1.0 / min_rad);
712  RCLCPP_INFO_STREAM(logger_, "[ev controller] radius < min radius = " << result);
713  if (use_regulated_linear_velocity_scaling_ && radius < min_rad) {
714  RCLCPP_INFO_STREAM(logger_, "[ev controller] CURVATURE HEURISTICS ACTIV !!! ");
715  curvature_vel *= 1.0 - (fabs(radius - min_rad) / min_rad);
716  }
717 
718  // ************************************ OBSTACLES ********************************** //
719 
720  // path base_footprint -> odom 으로 변환
721  nav_msgs::msg::Path transformed_path = getOdomBasedPath(path, robot_pose);
722 
723  // 가장 짧은 거리에 있는 pose 찾기
724  double min_distance = std::numeric_limits<double>::max();
725  geometry_msgs::msg::PoseStamped closest_pose;
726  for (const auto & pose : transformed_path.poses) {
727  double distance = sqrt(
728  pow(pose.pose.position.x - robot_pose.pose.position.x, 2) +
729  pow(pose.pose.position.y - robot_pose.pose.position.y, 2));
730  if (distance < min_distance) {
731  min_distance = distance;
732  closest_pose = pose;
733  }
734  }
735 
736  //
737  nav_msgs::msg::Path local_plan;
738  local_plan.header = robot_pose.header;
739  for (size_t i = 0; i < transformed_path.poses.size(); ++i) {
740  double distance = sqrt(
741  pow(transformed_path.poses[i].pose.position.x - closest_pose.pose.position.x, 2) +
742  pow(transformed_path.poses[i].pose.position.y - closest_pose.pose.position.y, 2));
743  if (distance <= obstacle_check_dist_) {
744  local_plan.poses.push_back(transformed_path.poses[i]);
745  } else {
746  break;
747  }
748  }
749  local_path_pub_->publish(local_plan);
750 
751  // local plan cost check
752  for (const auto & pose : local_plan.poses) {
753  double cost_at_pose = costAtPose(pose.pose.position.x, pose.pose.position.y);
754  if (use_cost_regulated_linear_velocity_scaling_ &&
755  cost_at_pose != static_cast<double>(NO_INFORMATION) &&
756  cost_at_pose > cost_threshold_)
757  {
758  RCLCPP_INFO_STREAM(logger_, "[ev controller] PROXIMITY HEURISTICS ACTIV !!! ");
759  RCLCPP_INFO_STREAM(logger_, "[ev controller] cost on path: " << cost_at_pose);
760  cost_vel = 0.0;
761  break;
762  }
763  }
764  linear_vel = std::min(cost_vel, curvature_vel);
765 
766  // ************************************ APPROACH ********************************** //
767  RCLCPP_INFO_STREAM(logger_, "[ev controller] lookahead_dist - carrot_dist: " << dist_error);
768  RCLCPP_INFO_STREAM(
769  logger_,
770  "[ev controller] costmap resolution : " << costmap_->getResolution());
771  result = dist_error > 2.0 * costmap_->getResolution();
772  RCLCPP_INFO_STREAM(logger_, "[ev controller] (lookahead_dist - carrot_dist) > 0.1 = " << result);
773  if (use_approach_vel_scaling_ && dist_error > 2.0 * costmap_->getResolution()) {
774  double velocity_scaling = 1.0 - (dist_error / lookahead_dist);
775  double unbounded_vel = approach_vel * velocity_scaling;
776  RCLCPP_INFO_STREAM(logger_, "[ev controller] APPROACH HEURISTICS ACTIV !!! ");
777  if (unbounded_vel < min_approach_linear_velocity_) {
778  approach_vel = min_approach_linear_velocity_;
779  } else {
780  approach_vel *= velocity_scaling;
781  }
782 
783  // Use the lowest velocity between approach and other constraints, if all overlapping
784  linear_vel = std::min(linear_vel, approach_vel);
785  }
786 
787  // ************************************ LIMITS ********************************** //
788  double & dt = control_duration_;
789  const double max_feasible_linear_speed = curr_speed.linear.x + max_linear_accel_ * dt;
790  const double min_feasible_linear_speed = curr_speed.linear.x - max_linear_decel_ * dt;
791  if (cost_vel == desired_linear_vel_) {
792  linear_vel = std::clamp(linear_vel, min_feasible_linear_speed, max_feasible_linear_speed);
793  linear_vel = std::clamp(linear_vel, 0.0, desired_linear_vel_);
794  }
795 
796  RCLCPP_INFO_STREAM(
797  logger_,
798  "[ev controller] min vel on curved path: " <<
799  regulated_linear_scaling_min_speed_);
800  RCLCPP_INFO_STREAM(logger_, "[ev controller] max feasible vel: " << max_feasible_linear_speed);
801  RCLCPP_INFO_STREAM(logger_, "[ev controller] min feasible vel: " << min_feasible_linear_speed);
802  RCLCPP_INFO_STREAM(logger_, "[ev controller] desired vel: " << desired_linear_vel_);
803  RCLCPP_INFO_STREAM(logger_, "[ev controller] curvature vel: " << curvature_vel);
804  RCLCPP_INFO_STREAM(logger_, "[ev controller] proximity vel: " << cost_vel);
805  RCLCPP_INFO_STREAM(logger_, "[ev controller] approach vel: " << approach_vel);
806  RCLCPP_INFO_STREAM(logger_, "[ev controller] ---------------------------");
807 }

814 nav_msgs::msg::Path ElevatorRegulatedPurePursuitController::transformGlobalPlan(
815  const geometry_msgs::msg::PoseStamped & pose)
816 {
817  if (global_plan_.poses.empty()) {
818  throw antis_nav_core::PlannerException("[ev controller] Received plan with zero length");
819  }
820 
821  // let's get the pose of the robot in the frame of the plan
822  geometry_msgs::msg::PoseStamped robot_pose;
823  if (!transformPose(global_plan_.header.frame_id, pose, robot_pose)) {
824  throw antis_nav_core::PlannerException(
825  "[ev controller] Unable to transform robot pose into global plan's frame");
826  }
827 
828  // We'll discard points on the plan that are outside the local costmap
829  antis_nav_costmap_2d::Costmap2D * costmap = costmap_ros_->getCostmap();
830  const double max_costmap_dim = std::max(costmap->getSizeInCellsX(), costmap->getSizeInCellsY());
831  const double max_transform_dist = max_costmap_dim * costmap->getResolution() / 2.0;
832 
833  // First find the closest pose on the path to the robot
834  auto transformation_begin =
835  min_by(
836  global_plan_.poses.begin(), global_plan_.poses.end(),
837  [&robot_pose](const geometry_msgs::msg::PoseStamped & ps) {
838  return euclidean_distance(robot_pose, ps);
839  });
840 
841  // Find points definitely outside of the costmap so we won't transform them.
842  auto transformation_end = std::find_if(
843  transformation_begin, end(global_plan_.poses),
844  [&](const auto & global_plan_pose) {
845  return euclidean_distance(robot_pose, global_plan_pose) > max_transform_dist;
846  });
847 
848  // Lambda to transform a PoseStamped from global frame to local
849  auto transformGlobalPoseToLocal = [&](const auto & global_plan_pose) {
850  geometry_msgs::msg::PoseStamped stamped_pose, transformed_pose;
851  stamped_pose.header.frame_id = global_plan_.header.frame_id;
852  stamped_pose.header.stamp = robot_pose.header.stamp;
853  stamped_pose.pose = global_plan_pose.pose;
854  transformPose(costmap_ros_->getBaseFrameID(), stamped_pose, transformed_pose);
855  return transformed_pose;
856  };
857 
858  // Transform the near part of the global plan into the robot's frame of reference.
859  nav_msgs::msg::Path transformed_plan;
860  std::transform(
861  transformation_begin, transformation_end,
862  std::back_inserter(transformed_plan.poses),
863  transformGlobalPoseToLocal);
864  transformed_plan.header.frame_id = costmap_ros_->getBaseFrameID();
865  transformed_plan.header.stamp = robot_pose.header.stamp;
866 
867  // Remove the portion of the global plan that we've already passed so we don't
868  // process it on the next iteration (this is called path pruning)
869  global_plan_.poses.erase(begin(global_plan_.poses), transformation_begin);
870  global_path_pub_->publish(transformed_plan);
871 
872  if (transformed_plan.poses.empty()) {
873  throw antis_nav_core::PlannerException("[ev controller] Resulting plan has 0 poses in it.");
874  }
875 
876  return transformed_plan;
877 }
