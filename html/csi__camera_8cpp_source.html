72 bool CSICamera::update()
73 {
74  bool is_successed = false;
75  for (auto & iter : cam_param_map_) {
76  is_successed = iter.second.v4l2_driver->grab() ? true : false;
77  }
78 
79  if (is_successed) {
80  for (auto & iter : cam_param_map_) {
81  iter.second.mat_img = iter.second.v4l2_driver->get_color_image();
82  if (is_rotated_) {
83  cv::rotate(iter.second.mat_img, iter.second.mat_img, cv::ROTATE_90_CLOCKWISE);
84  }
85  this->set_pub_data_image(
86  iter.second.img_pub_msg_name,
87  perception_tools::cvmat_to_ros(
88  iter.second.mat_img, iter.second.frame_id, rclcpp::Clock().now()));
89  }
90  } else {
91  return false;
92  }
93 
94  cv::Mat streaming_img;
95  if (cam_param_map_.size() == 2) {
96  if (is_rotated_) {
97  cv::hconcat(
98  cam_param_map_[cam_names_.at(0)].mat_img,
99  cam_param_map_[cam_names_.at(1)].mat_img,
100  streaming_img);
101  } else {
102  cv::vconcat(
103  cam_param_map_[cam_names_.at(0)].mat_img,
104  cam_param_map_[cam_names_.at(1)].mat_img,
105  streaming_img);
106  }
107  } else {
108  streaming_img = cam_param_map_[cam_names_.at(0)].mat_img;
109  }
110 
111  if (capture_continuous_onoff_ || capture_oneshot_onoff_) {
112  capture_oneshot_onoff_ = false;
113  detector_lib::PerceptionUtils::save_image(streaming_img, "csi");
114  RCLCPP_INFO_STREAM(get_logger(), "Save images : CSI Cam");
115  detector_lib::PerceptionUtils::draw_recording_mark(streaming_img);
116  cv::Mat tmp_vis;
117  cv::resize(streaming_img, tmp_vis, cv::Size(0, 0), 0.25, 0.25);
118  cv::imshow("Capture mode [CSI]", tmp_vis);
119  cv::waitKey(1);
120  }
121 
122  std::string streaming_frame_id = "left_csi_camera_joint";
123  this->run_streaming(
124  streaming_img,
125  rclcpp::Clock().now(),
126  streaming_frame_id,
127  csi_streaming_pub_msg_name_,
128  csi_streaming_image_ratio_);
129 
130  return true;
131 }

133 void CSICamera::get_ros_params()
134 {
135  int qos_hist;
136  get_param("qos_history", qos_hist, 1);
137  if (qos_hist == 1) {
138  qos_.history(RMW_QOS_POLICY_HISTORY_KEEP_LAST);
139  RCLCPP_INFO_STREAM(get_logger(), "qos_history : RMW_QOS_POLICY_HISTORY_KEEP_LAST");
140  } else if (qos_hist == 2) {
141  qos_.history(RMW_QOS_POLICY_HISTORY_KEEP_ALL);
142  RCLCPP_INFO_STREAM(get_logger(), "qos_history : RMW_QOS_POLICY_HISTORY_KEEP_ALL");
143  } else {
144  RCLCPP_WARN_STREAM(
145  get_logger(), "The parameter 'qos_history' is not available, using the default value");
146  qos_.history(RMW_QOS_POLICY_HISTORY_KEEP_LAST);
147  }
148 
149  int qos_de;
150  get_param("qos_depth", qos_de, 1);
151  qos_.keep_last(static_cast<size_t>(qos_de));
152  RCLCPP_INFO_STREAM(get_logger(), "qos_depth : " << qos_de);
153 
154  int qos_re;
155  get_param("ros_topic_option.qos_reliability", qos_re, 1);
156  if (qos_re == 1) {
157  qos_.reliability(RMW_QOS_POLICY_RELIABILITY_RELIABLE);
158  RCLCPP_INFO_STREAM(get_logger(), "qos_reliability : RMW_QOS_POLICY_RELIABILITY_RELIABLE");
159  } else if (qos_re == 2) {
160  qos_.reliability(RMW_QOS_POLICY_RELIABILITY_BEST_EFFORT);
161  RCLCPP_INFO_STREAM(get_logger(), "qos_reliability : RMW_QOS_POLICY_RELIABILITY_BEST_EFFORT");
162  } else {
163  RCLCPP_WARN_STREAM(
164  get_logger(), "The parameter 'qos_reliability' is not available, using the default value");
165  qos_.reliability(RMW_QOS_POLICY_RELIABILITY_RELIABLE);
166  }
167 
168  int qos_du;
169  get_param("ros_topic_option.qos_durability", qos_du, 2);
170  if (qos_du == 1) {
171  qos_.durability(RMW_QOS_POLICY_DURABILITY_TRANSIENT_LOCAL);
172  RCLCPP_INFO_STREAM(get_logger(), "qos_durability : RMW_QOS_POLICY_DURABILITY_TRANSIENT_LOCAL");
173  } else if (qos_du == 2) {
174  qos_.durability(RMW_QOS_POLICY_DURABILITY_VOLATILE);
175  RCLCPP_INFO_STREAM(get_logger(), "qos_durability : RMW_QOS_POLICY_DURABILITY_VOLATILE");
176  } else {
177  RCLCPP_WARN_STREAM(
178  get_logger(), "The parameter 'qos_durability' is not available, using the default value");
179  qos_.durability(RMW_QOS_POLICY_DURABILITY_VOLATILE);
180  }
181 
182  get_param(
183  "streaming_param.csi_streaming_pub_msg_name",
184  csi_streaming_pub_msg_name_,
185  std::string());
186  get_param("streaming_param.csi_streaming_image_ratio", csi_streaming_image_ratio_, 1.0);
187 
188  this->declare_parameter<std::vector<std::string>>("camera_list", std::vector<std::string>());
189  cam_names_ = this->get_parameter("camera_list").as_string_array();
190 
191  get_param("camera_option.frame_rate_hz", frame_rate_hz_, 10);
192 
193  int vl4_buffer_size = 0;
194  get_param("camera_option.vl4_buffer_size", vl4_buffer_size, 1);
195 
196  using Util = detector_lib::PerceptionUtils;
197  CameraGeometry * geo = CameraGeometry::get_instance();
198  for (std::string name : cam_names_) {
199  CameraParam cam_param;
200 
201  get_param(name + ".v4l_by_path", cam_param.cam_port, std::string());
202  get_param("camera_option." + name + ".frame_id", cam_param.frame_id, std::string());
203  get_param("camera_option." + name + ".image_width", cam_param.img_width, 1920);
204  get_param("camera_option." + name + ".image_height", cam_param.img_height, 1080);
205  get_param(
206  "camera_option." + name + ".image_pub_msg_name",
207  cam_param.img_pub_msg_name,
208  std::string());
209 
210  cv::Size image_size;
211  image_size.width = cam_param.img_width;
212  image_size.height = cam_param.img_height;
213  cam_param.v4l2_driver = std::make_shared<V4L2Driver>(image_size, vl4_buffer_size);
214 
215  std::string yaw_joint = name + ".yaw_joint";
216  this->declare_parameter<std::vector<double>>(yaw_joint, std::vector<double>());
217  cam_param.yaw_joint = this->get_parameter(yaw_joint).as_double_array();
218 
219  std::string pitch_joint = name + ".pitch_joint";
220  this->declare_parameter<std::vector<double>>(pitch_joint, std::vector<double>());
221  cam_param.pitch_joint = this->get_parameter(pitch_joint).as_double_array();
222 
223  std::string cam_joint = name + ".cam_joint";
224  this->declare_parameter<std::vector<double>>(cam_joint, std::vector<double>());
225  cam_param.cam_joint = this->get_parameter(cam_joint).as_double_array();
226 
227  std::string calib_joint = name + ".calib_joint";
228  this->declare_parameter<std::vector<double>>(calib_joint, std::vector<double>());
229  cam_param.calib_joint = this->get_parameter(calib_joint).as_double_array();
230 
231  std::string intrinsic = name + ".intrinsic";
232  this->declare_parameter<std::vector<double>>(intrinsic, std::vector<double>());
233  cam_param.intrinsic = this->get_parameter(intrinsic).as_double_array();
234 
235  Eigen::Matrix4d to_camera = Eigen::Matrix4d::Identity();
236  to_camera.block<3, 3>(0, 0) = geo->get_cam_to_opt().cast<double>();
237 
238  Eigen::Matrix4d mat =
239  Util::rpyxyz_to_tmat(cam_param.yaw_joint) *
240  Util::rpyxyz_to_tmat(cam_param.pitch_joint) *
241  Util::rpyxyz_to_tmat(cam_param.cam_joint) *
242  Util::rpyxyz_to_tmat(cam_param.calib_joint) *
243  to_camera;
244 
245  cam_param.extrinsic = mat.inverse();
246 
247  geo->set_cam_info(
248  name,
249  cam_param.img_width,
250  cam_param.img_height,
251  cam_param.intrinsic,
252  cam_param.extrinsic,
253  cam_param.frame_id);
254 
255  geo->create_frustum(name);
256 
257  cam_param_map_.insert(std::pair<std::string, CameraParam>(name, cam_param));
258  }
259 }

261 void CSICamera::open_csi_cam()
262 {
263  for (auto & iter : cam_param_map_) {
264  // trying to open CSI camera
265  bool is_opened = false;
266  while (!is_opened) {
267  iter.second.v4l2_driver->init(iter.second.cam_port);
268  if (!iter.second.v4l2_driver->is_opened()) {
269  RCLCPP_ERROR_STREAM(
270  this->get_logger(),
271  "Can't open the camera : Video " << iter.second.cam_port);
272  if (fms_notify_once_) {
273  robotis_server_interfaces::msg::FmsNotify fms_alarm;
274  fms_alarm.header.stamp = this->now();
275  fms_alarm.level = robotis_server_interfaces::msg::FmsNotify::ERROR;
276  fms_alarm.source = "PerceptionManager";
277  fms_alarm.keyword = "SHOULDER_CAM_CONNECTION_FAILED";
278  this->notify(fms_alarm);
279  fms_notify_once_ = false;
280  exit(-1);
281  }
282  } else {
283  fms_notify_once_ = true;
284  is_opened = true;
285  iter.second.v4l2_driver->start();
286  }
287  rclcpp::sleep_for(std::chrono::seconds(10));
288  }
289  }
