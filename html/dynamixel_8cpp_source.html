47 DxlError Dynamixel::InitDxlComm(
48  std::vector<uint8_t> id_arr,
49  std::string port_name,
50  std::string baudrate)
51 {
52  // Open communication port
53  port_handler_ = dynamixel::PortHandler::getPortHandler(port_name.c_str()); // port name
54  packet_handler_ = dynamixel::PacketHandler::getPacketHandler();
55 
56  // Open port
57  if (port_handler_->openPort()) {
58  fprintf(stderr, "Succeeded to open the port!\n");
59  } else {
60  fprintf(stderr, "Failed to open the port!\n");
61  return DxlError::OPEN_PORT_FAIL;
62  }
63 
64  // Set port baudrate
65  if (port_handler_->setBaudRate(stoi(baudrate))) { // baudrate
66  fprintf(stderr, "Succeeded to change the [%d] baudrate!\n", stoi(baudrate));
67  } else {
68  fprintf(stderr, "Failed to change the baudrate!\n");
69  return DxlError::OPEN_PORT_FAIL;
70  }
71 
72  // Ping to dxl
73  uint16_t dxl_model_number;
74  uint8_t dxl_error = 0; // Dynamixel error
75 
76  for (auto it_id : id_arr) {
77  fprintf(stderr, "[ID:%03d] Request ping\t", it_id);
78  int dxl_comm_result = packet_handler_->ping(
79  port_handler_, it_id, &dxl_model_number, &dxl_error);
80  if (dxl_comm_result != COMM_SUCCESS) {
81  fprintf(stderr, " - COMM_ERROR : %s\n", packet_handler_->getTxRxResult(dxl_comm_result));
82  return DxlError::CANNOT_FIND_CONTROL_ITEM;
83  } else if (dxl_error != 0) {
84  fprintf(stderr, " - RX_PACKET_ERROR : %s\n", packet_handler_->getRxPacketError(dxl_error));
85  uint32_t err = 0;
86  ReadItem(it_id, "Hardware Error Status", err);
87  fprintf(stderr, "Read Hardware Error Status : %x\n", err);
88  return DxlError::CANNOT_FIND_CONTROL_ITEM;
89  } else {
90  fprintf(stderr, " - Ping succeeded. Dynamixel model number : %d\n", dxl_model_number);
91  }
92 
93  // save the dxl model file
94  dxl_info_.ReadDxlModelFile(it_id, dxl_model_number);
95  }
96 
97  read_data_list_.clear();
98  write_data_list_.clear();
99 
100  for (auto it_id : id_arr) {
101  torque_state_[it_id] = TORQUE_OFF;
102  }
103 
104  return DxlError::OK;
105 }

381  packet_handler_->write1ByteTxRx(

390  port_handler_, id, addr,

392  }
393 
394  if (dxl_comm_result != COMM_SUCCESS) {
395  fprintf(
396  stderr,
397  "[ID:%03d] COMM_ERROR : %s\n",
398  id,
399  packet_handler_->getTxRxResult(dxl_comm_result));
400  return DxlError::ITEM_WRITE_FAIL;
401  } else if (dxl_error != 0) {
402  fprintf(
403  stderr,
404  "[ID:%03d] RX_PACKET_ERROR : %s\n",
405  id,
406  packet_handler_->getRxPacketError(dxl_error));
407  return DxlError::ITEM_WRITE_FAIL;
408  }
409  return DxlError::OK;
