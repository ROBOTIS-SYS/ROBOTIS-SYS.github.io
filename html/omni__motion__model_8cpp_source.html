44 OmniMotionModel::odometryUpdate(pf_t * pf, const pf_vector_t & pose, const pf_vector_t & delta)
45 {
46  // Compute the new sample poses
47  pf_sample_set_t * set;
48 
49  set = pf->sets + pf->current_set;
50  pf_vector_t old_pose = pf_vector_sub(pose, delta);
51 
52  double delta_trans, delta_rot, delta_bearing;
53  double delta_trans_hat, delta_rot_hat, delta_strafe_hat;
54 
55  delta_trans = sqrt(
56  delta.v[0] * delta.v[0] +
57  delta.v[1] * delta.v[1]);
58  delta_rot = delta.v[2];
59 
60  // Precompute a couple of things
61  double trans_hat_stddev = sqrt(
62  alpha3_ * (delta_trans * delta_trans) +
63  alpha4_ * (delta_rot * delta_rot) );
64  double rot_hat_stddev = sqrt(
65  alpha1_ * (delta_rot * delta_rot) +
66  alpha2_ * (delta_trans * delta_trans) );
67  double strafe_hat_stddev = sqrt(
68  alpha4_ * (delta_rot * delta_rot) +
69  alpha5_ * (delta_trans * delta_trans) );
70 
71  for (int i = 0; i < set->sample_count; i++) {
72  pf_sample_t * sample = set->samples + i;
73 
74  delta_bearing = angleutils::angle_diff(
75  atan2(delta.v[1], delta.v[0]),
76  old_pose.v[2]) + sample->pose.v[2];
77  double cs_bearing = cos(delta_bearing);
78  double sn_bearing = sin(delta_bearing);
79 
80  // Sample pose differences
81  delta_trans_hat = delta_trans + pf_ran_gaussian(trans_hat_stddev);
82  delta_rot_hat = delta_rot + pf_ran_gaussian(rot_hat_stddev);
83  delta_strafe_hat = 0 + pf_ran_gaussian(strafe_hat_stddev);
84  // Apply sampled update to particle pose
85  sample->pose.v[0] += (delta_trans_hat * cs_bearing +
86  delta_strafe_hat * sn_bearing);
87  sample->pose.v[1] += (delta_trans_hat * sn_bearing -
88  delta_strafe_hat * cs_bearing);
89  sample->pose.v[2] += delta_rot_hat;
90  }
91 }
