218 bool GetTargetAction::call_get_saved_target(std::string key)
219 {
220  // check server
221  if (get_saved_target_cli_->service_is_ready() == false) {
222  RCLCPP_ERROR_STREAM(get_logger(), "[GetTargetAction] Failed to connect service server");
223  return false;
224  }
225 
226  auto request =
227  std::make_shared<aimbot_state_interfaces::srv::GetAimbotSavedTarget::Request>();
228  request->key = key;
229 
230  using GetAimbotSavedTargetFuture =
231  rclcpp::Client<aimbot_state_interfaces::srv::GetAimbotSavedTarget>::SharedFuture;
232  auto response_received_callback = [this](GetAimbotSavedTargetFuture future) {
233  if (status() != BT::NodeStatus::RUNNING) {
234  return;
235  }
236 
237  auto result = future.get();
238 
239  geometry_msgs::msg::Pose2D goal_msg;
240  std::string floor, poi, sector;
241 
242  if (result->result == false) {
243  setStatus(BT::NodeStatus::FAILURE);
244  return;
245  } else {
246  goal_msg = result->target.pose;
247  floor = result->target.floor;
248  poi = result->target.type;
249  sector = result->target.sector;
250 
251  setOutput("goal", goal_msg);
252  setOutput("floor", floor);
253  setOutput("poi", poi);
254  setOutput("sector", sector);
255  }
256  };
257  // request to server
258  auto future = get_saved_target_cli_->async_send_request(request, response_received_callback);
259 
260  // wait for response
261  // method 1 : use std::future --> no spin no response
262  // auto result = future.wait_for(std::chrono::seconds(static_cast<int>(ROS_TIMEOUT)));
263  double check_time = 1000.0;
264  double loop_rate = 100.0; // ms
265  int loop_count = 0;
266 
267  rclcpp::WallRate loopRate(std::chrono::milliseconds(100));
268 
269  std::future_status future_result = std::future_status::deferred;
270  while (rclcpp::ok() && isHaltRequested() == false) {
271  future_result = future.wait_for(std::chrono::milliseconds(0));
272  if (future_result == std::future_status::ready) {
273  break;
274  }
275 
276  // sleep
277  loopRate.sleep();
278 
279  if ((loop_rate * ++loop_count) > check_time) {
280  break;
281  }
282  }
283 
284  if (isHaltRequested()) {
285  return false;
286  }
287 
288  switch (future_result) {
289  case std::future_status::ready:
290  std::cout << "service future : ready" << std::endl;
291  break;
292 
293  case std::future_status::timeout:
294  std::cout << "service future : timeout" << std::endl;
295  return false;
296 
297  default:
298  return false;
299  }
300  return true;
301 }
