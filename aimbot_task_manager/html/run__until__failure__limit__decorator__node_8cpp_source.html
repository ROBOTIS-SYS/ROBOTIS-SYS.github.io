55  bool reset_value = config().blackboard->get<bool>(reset_key);
56  if (reset_value == true) {
57  failure_count_ = 0;
58  config().blackboard->set<bool>(reset_key, false);
59  }
60  }
61 
62  BT::NodeStatus return_over_limit = BT::NodeStatus::FAILURE;
63  getInput<BT::NodeStatus>("return_over_limit", return_over_limit);
64 
65  // check limit
66  if (failure_count_ >= num_limit_) {
67  RCLCPP_WARN_STREAM(get_logger(), "Failure count : " << failure_count_);
68  return return_over_limit;
69  }
70 
71  // excute tick child
72  child()->executeTick();
73
