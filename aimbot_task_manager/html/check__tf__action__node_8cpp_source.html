40 bool CheckTFAction::on_init()
41 {
42  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
43  RCLCPP_INFO_STREAM(get_logger(), "Init : CheckTFAction - " << name());
44  }
45 
46  // init tf
47  try {
48  // init ros
49  // tf buffer
50  if (tf_buffer_ == nullptr) {
51  bool on_blackboard = config().blackboard->get<std::shared_ptr<tf2_ros::Buffer>>(
52  "tf_buffer", tf_buffer_);
53  if (on_blackboard == false) {
54  tf_buffer_ = std::make_shared<tf2_ros::Buffer>(node_->get_clock());
55  config().blackboard->set<std::shared_ptr<tf2_ros::Buffer>>("tf_buffer", tf_buffer_);
56  }
57  }
58  // tf listener
59  if (tf_listener_ == nullptr) {
60  bool on_blackboard = config().blackboard->get<std::shared_ptr<tf2_ros::TransformListener>>(
61  "tf_listener", tf_listener_);
62  if (on_blackboard == false) {
63  tf_listener_ = std::make_shared<tf2_ros::TransformListener>(*tf_buffer_);
64  config().blackboard->set<std::shared_ptr<tf2_ros::TransformListener>>(
65  "tf_listener", tf_listener_);
66  }
67  }
68  } catch (std::exception & ex) {
69  RCLCPP_ERROR_STREAM(get_logger(), "Init error : " << ex.what());
70  return false;
71  }
72 
73  return true;
74 }
