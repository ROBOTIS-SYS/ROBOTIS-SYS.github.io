97 bool SetServiceOptionAction::parse_yaml(std::string file_path)
98 {
99  try {
100  YAML::Node file = YAML::LoadFile(file_path.c_str());
101 
102  for (YAML::iterator it = file.begin(); it != file.end(); it++) {
103  std::string opt_name = it->first.as<std::string>();
104  std::cout << "\n----------" << std::endl;
105  std::cout << "opt_name : " << opt_name << std::endl;
106  for (YAML::iterator itt = it->second.begin(); itt != it->second.end(); itt++) {
107  std::string param_name = itt->first.as<std::string>();
108  std::cout << "-- " << param_name << " : ";
109  if (itt->second.IsScalar()) { // scalar
110  set_blackboard(param_name, itt->second.as<std::string>());
111 
112  } else if (itt->second.IsMap()) { // map
113  YAML::Node sub_node = itt->second;
114  auto sub_node_first_it = sub_node.begin();
115  std::string type = sub_node_first_it->first.as<std::string>();
116 
117  if (type == "ros_param") {
118  std::string ros_param_value = "";
119  for (YAML::iterator sub_it = sub_node_first_it; sub_it != sub_node.end(); sub_it++) {
120  make_ros_param_value(sub_it->second, ros_param_value);
121  }
122  if (ros_param_value != "") {
123  set_blackboard(param_name, ros_param_value);
124  }
125  } else if (type == "geometry_point") {
126  set_blackboard_point(
127  param_name,
128  sub_node_first_it->second["x"].as<double>(),
129  sub_node_first_it->second["y"].as<double>(),
130  sub_node_first_it->second["z"].as<double>());
131  } else if (type == "geometry_pose2d") {
132  set_blackboard_2d_pose(
133  param_name,
134  sub_node_first_it->second["x"].as<double>(),
135  sub_node_first_it->second["y"].as<double>(),
136  sub_node_first_it->second["theta"].as<double>());
137  } else {
138  continue;
139  }
140  } else if (itt->second.IsSequence()) { // array
141  } else {
142  }
143  }
144  }
145  } catch (const std::exception & e) {
146  std::stringstream log_stream;
147  log_stream << "Fail to load process data file. : " << e.what();
148  RCLCPP_WARN_STREAM(get_logger(), log_stream.str());
149  return false;
150  }
151  return true;
152 }
