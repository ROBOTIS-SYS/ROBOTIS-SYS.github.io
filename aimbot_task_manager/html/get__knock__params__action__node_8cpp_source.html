140 bool GetKnockParamsAction::parseYaml(std::string room_name, std::string floor)
141 {
142  try {
143  // 파일을 Parsing
144  YAML::Node node = YAML::LoadFile(file_path_.c_str());
145  for (YAML::iterator it = node.begin(); it != node.end(); it++) {
146  // room_name이 파일안에 있는지 확인. 없으면 if안에 들어가지 않으므로 아무것도 하지않음.
147  if (it->second["room_name"].as<std::string>() == room_name) {
148  auto floor_list_from_yaml = it->second["floor"].as<std::vector<std::string>>();
149  for (std::string floor_yaml : floor_list_from_yaml) {
150  // room_name이 같은 node안에서 현재 floor가 포함되어있는지 확인.
151  // 없으면 if안에 들어가지 않으므로 아무것도 하지않음.
152  if (floor_yaml == floor) {
153  // 파라미터 설정이 필요하므로 params를 받음.
154  YAML::Node params = it->second["params"];
155  for (YAML::iterator param_it = params.begin(); param_it != params.end();
156  param_it++)
157  {
158  // iterator로 돌리다가 해당 파라미터가 있을때 등록 없으면 등록안함.
159  std::string param_name = param_it->first.as<std::string>();
160  if (param_name == "count") {
161  if (
162  config().blackboard->get("enable_log", enable_log_) == false ||
163  enable_log_ == true)
164  {
165  RCLCPP_INFO_STREAM(get_logger(), "[GetKnockParamsAction] set param count");
166  }
167  count_ = param_it->second.as<int>();
168  } else if (param_name == "position_rate") {
169  if (
170  config().blackboard->get("enable_log", enable_log_) == false ||
171  enable_log_ == true)
172  {
173  RCLCPP_INFO_STREAM(
174  get_logger(), "[GetKnockParamsAction] set param position_rate");
175  }
176  position_rate_ = param_it->second.as<double>();
177  } else if (param_name == "direction") {
178  if (
179  config().blackboard->get("enable_log", enable_log_) == false ||
180  enable_log_ == true)
181  {
182  RCLCPP_INFO_STREAM(
183  get_logger(), "[GetKnockParamsAction] set param direction");
184  }
185  direction_.x = param_it->second["x"].as<double>();
186  direction_.y = param_it->second["y"].as<double>();
187  direction_.theta = param_it->second["theta"].as<double>();
188  } else {
189  // Yaml 파일에는 적혀있지만, 세팅할수 없는 변수가 있을경우, 실패하고 log를
190  // publish.
191  publish_log(
192  "유효하지 않은 파라미터 이름 사용 : " + param_name,
193  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
194  aimbot_log_interfaces::msg::Log::TYPE_ERROR);
195  std::cout << "유효하지 않은 파라미터 이름 사용 : " << param_name << std::endl;
196  return false;
197  }
198  }
199  } else {
200  // room_name이 같은 node안에서 현재 floor가 포함되어있는지 확인.
201  // 없으면 if안에 들어가지 않으므로 아무것도 하지않음.
202  }
203  } // for (std::string single_floor : floor_list)
204  } else {
205  // room_name이 파일안에 있는지 확인. 없으면 if안에 들어가지 않으므로 아무것도 하지않음.
206  }
207  } // for (YAML::iterator it = node.begin(); it != node.end(); it++)
208  } catch (const std::exception & e) {
209  std::stringstream log_stream;
210  log_stream << "Fail to load process data file. : " << e.what();
211  RCLCPP_ERROR_STREAM(get_logger(), log_stream.str());
212  return false;
213  }
214  return true;
215 }
