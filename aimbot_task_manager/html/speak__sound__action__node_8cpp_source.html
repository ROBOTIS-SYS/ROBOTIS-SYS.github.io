48 BT::NodeStatus SpeakSoundAction::tick()
49 {
50  std::cout << std::endl;
51  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
52  RCLCPP_INFO_STREAM(get_logger(), "<TICK> SpeakSoundAction");
53  }
54 
55  std::string package_name, folder_name, sound_file_name;
56  try {
57  if (!getInput<std::string>("package_name", package_name) || package_name.empty()) {
58  // return BT::NodeStatus::FAILURE;
59  publish_tree_status(name(), "음성 안내를 실패했습니다.");
60  RCLCPP_ERROR(get_logger(), "failed to play sound : " + name());
61  return BT::NodeStatus::SUCCESS;
62  }
63 
64  if (!getInput<std::string>("sound_file_name", sound_file_name)) {
65  // return BT::NodeStatus::FAILURE;
66  publish_tree_status(name(), "음성 안내를 실패했습니다.");
67  RCLCPP_ERROR(get_logger(), "failed to play sound : " + name());
68  return BT::NodeStatus::SUCCESS;
69  }
70 
71  if (getInput<std::string>("folder_name", folder_name)) {
72  folder_name = folder_name + "/";
73  }
74 
75  std::string sound_file_path =
76  ament_index_cpp::get_package_prefix(package_name) + "/" + folder_name + sound_file_name;
77 
78  // init publisher
79  std::string topic_name = "play_sound_file";
80  if (play_sound_pub_ == nullptr) {
81  bool on_blackboard =
82  config().blackboard->get<decltype(play_sound_pub_)>(topic_name, play_sound_pub_);
83 
84  if (on_blackboard == false) {
85  play_sound_pub_ = node_->create_publisher<std_msgs::msg::String>(topic_name, 10);
86  config().blackboard->set<decltype(play_sound_pub_)>(topic_name, play_sound_pub_);
87  }
88  }
89 
90  play_sound(sound_file_path);
91  } catch (std::exception & ex) {
92  RCLCPP_ERROR_STREAM(get_logger(), "Error : " << ex.what());
93  return BT::NodeStatus::FAILURE;
94  }
95  return BT::NodeStatus::SUCCESS;
96 }
