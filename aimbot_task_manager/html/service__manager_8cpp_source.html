236 void ServiceManager::task_thread_func()
237 {
238  rclcpp::WallRate loop_rate(std::chrono::milliseconds(100));
239 
240  while (rclcpp::ok()) {
241  // 작업 상태 체크
242  switch (task_state_) {
243  case IDLE: // start next task
244  break;
245 
246  case RUNNING: // keep running
247  // check running time
248  if (service_delayed_.load() == false &&
249  execution_timer_.current_elapsed_time_in_seconds() > service_delay_threshold_)
250  {
251  service_delayed_.store(true);
252  RCLCPP_ERROR_STREAM(
253  get_logger(),
254  "Task is running over " << execution_timer_.current_elapsed_time_in_seconds());
255 
256  publish_fms_alarm(FmsNotify::WARN, "SERVICE_DELAYED", true, CURRENT_TASK);
257  }
258 
259  // sleep and go next
260  loop_rate.sleep();
261  continue;
262 
263  case FINISHED: // start next task
264  execution_timer_.end();
265  task_state_ = IDLE;
266  break;
267 
268  case STOPPED: // wait for start command
269  if (start_requested_.load() == true) {
270  task_state_ = IDLE;
271  start_requested_.store(false);
272  break;
273  } else {
274  // clear current task and publish empty task list
275  if (clear_current_task(true)) {
276  publish_task_list(true);
277  }
278  // sleep and go next
279  loop_rate.sleep();
280  continue;
281  }
282 
283  case FAILED: // wait for start command
284  case CANCELED: // start next task
285  task_state_ = IDLE;
286  break;
287 
288  case PAUSED:
289  break;
290 
291  default:
292  break;
293  }
294 
295  // if not, check task list to run
296  bool set_task_result = get_task_from_queue(current_task_);
297 
298  // get task and run
299  if (set_task_result == true) {
300  // publish task list
301  publish_task_list(true);
302 
303  if (current_task_ == nullptr) {
304  std::cout << "current_task_ is null" << std::endl;
305  return;
306  }
307 
308  // clear last task
309  if (last_task_ != nullptr) {
310  last_task_.reset();
311  }
312 
313  // start task
314  auto task_thread = std::thread{
315  std::bind(&task_manager::ServiceManager::run_task, this, std::placeholders::_1),
316  current_task_};
317 
318  // set start time
319  execution_timer_.start();
320  service_delayed_.store(false);
321 
322  task_thread.detach();
323 
324  } else {
325  // clear current task and publish empty task list
326  if (clear_current_task()) {
327  publish_task_list(true);
328  }
329  }
330 
331  // sleep
332  loop_rate.sleep();
333  }
334 }

336 bool ServiceManager::get_task_from_queue(TaskItem::SharedPtr & current_task)
337 {
338  std::lock_guard<std::mutex> lock(mutex_);
339 
340  if (remained_task_list_.size() == 0) {

345

348  remained_task_list_.erase(remained_task_list_.begin());
349 
350  return true;

416 void ServiceManager::set_task_command(task_command_t command)
417 {
418  if (start_requested_.load() || stop_requested_.load() || cancel_requested_.load()) {
419  RCLCPP_WARN_STREAM(get_logger(), "Command ignored, last command does not performed.");
420  return;
421  }
422 
423  RCLCPP_INFO_STREAM(get_logger(), "Task Command : " << command);
424 
425  if (command == Task::START || command == Task::RESUME) {
426  // check service tree : initialized
427  bool init;
428  bool result = main_blackboard_->get<bool>("initialized", init);
429 
430  if (result == true && init == false) {
431  RCLCPP_WARN_STREAM(get_logger(), "Command ignored, it was not initialized");
432  return;
433  }
434  }
435 
436  switch (command) {
437  case Task::START: // works when task was failed/canceled
438  if (task_state_.load() == FAILED || task_state_.load() == STOPPED) {
439  start_requested_.store(true);
440  }
441  // clear last task
442  if (last_task_ != nullptr) {
443  last_task_.reset();
444  }
445  publish_log(
446  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
447  aimbot_log_interfaces::msg::Log::TYPE_INFO, "Task Command : START");
448  break;
449 
450  case Task::STOP: // works when task is running
451  if (task_state_.load() == RUNNING) {
452  // stop task
453  stop_requested_.store(true);
454  // fms
455  publish_fms_alarm(FmsNotify::INFO, "TASK_STOPPED", false, CURRENT_TASK);
456  } else {
457  task_state_.store(TaskItem::STOPPED);
458  }
459  publish_log(
460  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
461  aimbot_log_interfaces::msg::Log::TYPE_INFO, "Task Command : STOP");
462  break;
463 
464  case Task::STOP_ALL: // works when task is running
465  // fms 알림
466  if (get_remained_task_number() != 0 || task_state_.load() == RUNNING) {
467  publish_fms_alarm(FmsNotify::INFO, "TASK_STOPPED_ALL", false, ALL_TASK);
468  }
469 
470  // 큐의 모든 작업 삭제
471  if (get_remained_task_number() != 0) {
472  std::lock_guard<std::mutex> lock(mutex_);
473  // clear remained tasks
474  remained_task_list_.clear();
475  }
476 
477  // 진행 중인 작업 정지
478  if (task_state_.load() == RUNNING) {
479  // stop task
480  stop_requested_.store(true);
481  } else {
482  publish_task_list(true);
483  task_state_.store(TaskItem::CANCELED);
484  }
485 
486  publish_log(
487  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
488  aimbot_log_interfaces::msg::Log::TYPE_INFO, "Task Command : STOP ALL");
489  break;
490 
491  case Task::PAUSE:
492  // not yet
493  break;
494 
495  case Task::RESUME: // works when the last task was calceled or failed
496  if (last_task_ != nullptr) {
497  if (task_state_.load() == FAILED || task_state_.load() == STOPPED) {
498  // restore the last task to the task queue
499  remained_task_list_.insert(remained_task_list_.begin(), last_task_);
500  last_task_.reset();
501  // set a flag to start
502  start_requested_.store(true);
503  }
504  } else {
505  if (task_state_.load() == FAILED || task_state_.load() == STOPPED) {
506  start_requested_.store(true);
507  }
508  }
509  publish_log(
510  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
511  aimbot_log_interfaces::msg::Log::TYPE_INFO, "Task Command : RESUME");
512  break;
513 
514  case Task::CANCEL: // works when task is running
515  if (task_state_.load() == RUNNING) {
516  // cancel task
517  cancel_requested_.store(true);
518  // } else {
519  // task_state_.store(TaskItem::CANCELED);
520  }
521  publish_log(
522  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
523  aimbot_log_interfaces::msg::Log::TYPE_INFO, "Task Command : CANCEL");
524  break;
525 
526  default:
527  break;
528  }
529 }

531 void ServiceManager::set_user_command(task_command_t command, const std::string & user_id)
532 {
533  switch (command) {
534  default:
535  break;
536 
537  case Task::CANCEL: // works when task is running
538  if (get_remained_task_number() != 0) {
539  std::lock_guard<std::mutex> lock(mutex_);
540  // remove task using user_id
541  for (auto it = remained_task_list_.begin(); it != remained_task_list_.end(); ) {
542  if ((*it)->task.task_name == "TRANSFER" || (*it)->task.task_name == "CALL") {
543  if ((*it)->user_id == user_id) {
544  remained_task_list_.erase(it);
545  continue;
546  }
547  }
548  ++it;
549  }
550  }
551 
552  TaskItem::SharedPtr current_task;
553  if (
554  get_current_task(current_task) && task_state_.load() == RUNNING &&
555  current_task->user_id == user_id)
556  {
557  // stop task
558  cancel_requested_.store(true);

560  publish_task_list(true);
561  // task_state_.store(TaskItem::CANCELED);
562  }
563 
564  publish_log(
565  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
566  aimbot_log_interfaces::msg::Log::TYPE_INFO, "User Task Command : STOP");
567  break;
568  }

599 void ServiceManager::service_thread_func(const std::string & tree_name)
600 {
601  // execute main service tree
602  // create tree from xml file //
603  DEBUG_PRINT("blackboard(main) : " << main_blackboard_);
604  bool has_ext = tree_name.find("xml") != std::string::npos;
605  std::string tree_file_path = ament_index_cpp::get_package_prefix("aimbot_task_manager") +
606  "/config/" + tree_name + (has_ext ? "" : ".xml");
607  auto tree = std::make_shared<BT::Tree>(
608  bt_factory_->createTreeFromFile(tree_file_path, main_blackboard_));
609  main_blackboard_->set<bool>("initialized", false);
610 
611  // Iterate through all the nodes and call init() if it uses ros node
612  rclcpp::Time start = this->get_clock()->now();
613  for (auto & tree_node : tree->nodes) {
614  if (auto ros_node = dynamic_cast<task_manager::RosTreeNode *>(tree_node.get())) {
