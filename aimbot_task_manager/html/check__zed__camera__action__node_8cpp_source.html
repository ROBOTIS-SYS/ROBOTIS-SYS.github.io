42 BT::NodeStatus CheckZedCameraAction::tick()

45 
46  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {

48  }
49 
50  is_topic_alive_ = false;
51 
52  // check topic name
53  std::string topic_name;
54  if (!getInput<std::string>("topic_name", topic_name)) {
55  RCLCPP_ERROR_STREAM(get_logger(), "Empty input port [topic_name]");
56  return BT::NodeStatus::FAILURE;
57  }
58 
59  double check_time;
60  if (!getInput<double>("check_time", check_time)) {
61  check_time = 1.0;
62  }
63 
64  // init ros
65  if (topic_sub_ == nullptr) {
66  topic_sub_ = node_->create_subscription<sensor_msgs::msg::CompressedImage>(
67  topic_name, 0, std::bind(
68  &CheckZedCameraAction::topic_callback, this,
69  std::placeholders::_1));
70  }
71 
72  // s -> ms
73  check_time *= 1000.0;
74  double loop_rate = 100.0; // ms
75  int loop_count = 0;
76 
77  rclcpp::WallRate loopRate(std::chrono::milliseconds(100));
78 
79  while (rclcpp::ok() && is_topic_alive_ == false && isHaltRequested() == false) {
80  // sleep
81  loopRate.sleep();
82 
83  if ((loop_rate * ++loop_count) > check_time) {
84  break;
85  }
86  }
87 
88  if (isHaltRequested()) {
89  return BT::NodeStatus::IDLE;
90  }
91 
92  if (is_topic_alive_ == true) {
93  publish_tree_status(name(), "토픽이 정상적으로 출력되고 있습니다.");
94 
95  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
96  RCLCPP_INFO_STREAM(get_logger(), "[" << topic_name << "] is alive.");
97  }
98  } else {
99  publish_tree_status(name(), "토픽이 정상적으로 출력되지 않고 있습니다.");
100  RCLCPP_WARN_STREAM(get_logger(), "[" << topic_name << "] is not alive.");
101  }
102 
103  return is_topic_alive_ ? BT::NodeStatus::SUCCESS : BT::NodeStatus::FAILURE;
104 }
