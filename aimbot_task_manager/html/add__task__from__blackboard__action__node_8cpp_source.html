79  RCLCPP_ERROR_STREAM(get_logger(), "Init error : " << ex.what());
80  return BT::NodeStatus::FAILURE;
81  }
82  }
83 
84  int8_t index = task_manager_interfaces::srv::AddTask::Request::END;
85  std::string add_front;
86  if (getInput<std::string>("add_front", add_front) && add_front == "true") {
87  index = task_manager_interfaces::srv::AddTask::Request::FRONT;
