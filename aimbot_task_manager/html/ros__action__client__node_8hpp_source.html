38 template<class ActionT>
39 class ROSActionClientNode : public BT::AsyncActionNode
40 {
41 public:
42  ROSActionClientNode(
43  const std::string & name, const BT::NodeConfiguration & conf)
44  : AsyncActionNode(name, conf)
45  {
46  // Ros Node를 Blackboard에서 받아옵니다.
47  // Blackboard에 Node를 꼭 넣어 주어야합니다.
48  node_ = config().blackboard->get<rclcpp::Node::SharedPtr>("node");
49 
50  RCLCPP_INFO(node_->get_logger(), "\"%s\" ROSActionClientNode initialized", name.c_str());
51  }

53  ~ROSActionClientNode() override
54  {
55  action_client_.reset();
56  }
57 
59  // 파생 클래스는 on_tick와 on_wait_for_result, on_success 등의 함수를 정의하여
60  // Processing에 적용할 수 있습니다.
61 
62  // Tick의 시작부분에 처리 할 내용을 작성합니다.
63  // 이 함수는 Tick내에서 처음 단 한번만 호출됩니다.
64  // 함수는 Action을 호출하기 전에 호출됩니다.
65  // 여기서 Blackboard에서 데이터를 받아와, Ros Action Goal "goal_"을 설정하십시오.
66  // Return value가 'BT::NodeStatus::RUNNING'이 아닐 경우,
67  // Action을 호출하지 않고, Return value로 Tick이 종료됩니다.
68  virtual BT::NodeStatus on_tick()
69  {return BT::NodeStatus::RUNNING;}
70 
71  // 액션을 호출하고, Result를 기다리는동안 호출됩니다.
72  // 이 함수는 async를 이용해 Ros action과는 별개의 Thread에서 동작합니다.
73  // 반복적으로 동작을 실행하려면 내부에 loop를 만드십시오.
74  // Return value가 'BT::NodeStatus::RUNNING'이 아닐 경우, Action을 Cancel하고,
75  // Return value로 Tick이 종료됩니다.
76  virtual BT::NodeStatus on_wait_for_result()
77  {return status();}
78 
79  // 액션이 성공적으로 완료되면 호출됩니다.
80  // (ResultCode == rclcpp_action::ResultCode::SUCCEEDED)
81  // 파생 클래스는 예를 들어 blackboard의 value를 입력하는 것과 같이 이 함수를 재정의할 수 있습니다.
82  virtual BT::NodeStatus on_success()
83  {return BT::NodeStatus::SUCCESS;}
84 
85  // 작업이 중단될 때 호출됩니다. 기본적으로 노드는 FAILURE를 반환합니다.
86  // (ResultCode == rclcpp_action::ResultCode::ABORTED)
87  // 파생 클래스는 다른 value을 반환하도록 재정의할 수 있습니다
88  virtual BT::NodeStatus on_aborted()
89  {return BT::NodeStatus::FAILURE;}
90 
91  // 액션이 취소되면 호출됩니다. 기본적으로 노드는 SUCCESS를 반환합니다.
92  // (ResultCode == rclcpp_action::ResultCode::CANCELED)
93  // 파생 클래스는 다른 value을 반환하도록 재정의할 수 있습니다
94  virtual BT::NodeStatus on_cancelled()
95  {return BT::NodeStatus::SUCCESS;}
96 
97 // // 액션의 Response Callback으로 사용되어 기본적으로 아무것도 하지않습니다.
98 // // 파생 클래스에서는 추가적인 다른 처리를 할 수 있습니다.
99 // virtual void response_callback(
100 // std::shared_future<typename rclcpp_action::ClientGoalHandle<ActionT>::SharedPtr> future) {}
101 
102  // 액션의 Feedback Callback으로 사용되어 기본적으로 아무것도 하지않습니다.
103  // 파생 클래스에서는 추가적인 다른 처리를 할 수 있습니다.
104  virtual void feedback_callback(
105  typename rclcpp_action::ClientGoalHandle<ActionT>::SharedPtr,
106  const std::shared_ptr<const typename ActionT::Feedback>) {}

108  BT::NodeStatus tick() final
109  {
110  std::cout << std::endl;
111  RCLCPP_INFO_STREAM(node_->get_logger(), "[TICK] " << name());
112  // 액션 입력 및 출력 초기화
113  action_status_ = NONE;
114  goal_ = typename ActionT::Goal();
115  result_ = typename rclcpp_action::ClientGoalHandle<ActionT>::WrappedResult();
116  BT::NodeStatus node_status;
117 
118  try {
119  // 유저의 on_tick()함수 실행
120  // on_tick()함수의 결과가 BT::NodeStatus::RUNNING이 아닐경우 액션을 호출하지않고 종료.
121  node_status = on_tick();
122  if (node_status != BT::NodeStatus::RUNNING) {return node_status;}
123 
124  // on_tick함수를 실행하는 동안 발생한 halt처리
125  if (isHaltRequested()) {
126  return BT::NodeStatus::IDLE;
127  }
128  rclcpp::spin_some(node_);
129  } catch (std::exception & ex) {
130  RCLCPP_ERROR_STREAM(
131  node_->get_logger(),
132  "[" << name() << "] on_tick Exception - " << ex.what());
133  return BT::NodeStatus::FAILURE;
134  }
135 
136  try {
137  // Ros 액션을 호출
138  call_action_goal();
139  // 액션을 호출하는 동안 발생한 Halt처리
140  if (isHaltRequested()) {
141  action_cancel();
142  return BT::NodeStatus::IDLE;
143  }
144 
145  // 유저가 정의한 on_wait_for_result()함수 thread 생성
146  on_wait_future_handle_ =
147  std::async(std::launch::async, std::bind(&ROSActionClientNode::on_wait_for_result, this));
148  if (on_wait_future_handle_.valid() == false) {
149  throw std::runtime_error("[" + name() + "] Failed to get 'on_wait_for_result' future.");
150  }
151  std::future_status on_wait_future_result;
152 
153  // 액션 전체의 timeout 처리
154  start_time_ = node_->get_clock()->now();
155  rclcpp::Duration timeout_dur(total_action_timeout_);
156  // 액션 및 Ros spin 시작
157  rclcpp::WallRate loop_rate(std::chrono::milliseconds(static_cast<int>(loop_rate_ms_)));
158  while (rclcpp::ok() && action_status_ == REQUESTED) {
159 // RCLCPP_INFO_STREAM(node_->get_logger(), "waitng loop.");
160  // 루프 초반에 Hlat 체크
161  if (isHaltRequested()) {
162  action_cancel();
163  return BT::NodeStatus::IDLE;
164  }
165  // ROS spin
166  rclcpp::spin_some(node_);
167 
168  // on_wait_for_result() 함수의 결과 확인.
169  // on_wait_for_result() 함수가 이미 종료 되었을 경우,
170  // on_wait_future.valid() = false이므로 실행되지않음?
171  if (on_wait_future_handle_.valid()) {
172  on_wait_future_result = on_wait_future_handle_.wait_for(std::chrono::milliseconds(1));
173  if (on_wait_future_result == std::future_status::ready) {
174  node_status = on_wait_future_handle_.get();
175  // 결과가 BT::NodeStatus::RUNNING이 아닐경우 액션을 취소하고 Tick종료.
176  if (node_status != BT::NodeStatus::RUNNING && node_status != BT::NodeStatus::IDLE) {
177  // on_wait_future_result()의 결과확인중 Halt가 발생했을경우, 바로 Tick 종료
178  // Halt내부에서 액션은 cancel되었을 것임.
179  if (isHaltRequested()) {
180  action_cancel();
181  return BT::NodeStatus::IDLE;
182  } else {
183  // on_wait_for_result()의 결과가 BT::NodeStatus::RUNNING이
184  // 아닐경우 액션을 취소하고 Tick종료.
185  // skip_action_cancel가 true이면 cancel을 날리지 않고 skip
186  // halt 상황일때는 모두 cancel을 날리고 끊냄.
187  if (!skip_action_cancel_) {
188  action_cancel();
189  }
190  return node_status;
191  }
192  }
193  }
194  }
195 
196  // on_wait_future_result()의 결과확인중 Halt가 발생했을경우, 바로 Tick 종료
197  if (isHaltRequested()) {
198  action_cancel();
199  return BT::NodeStatus::IDLE;
200  }
201 
202  // 액션 결과를 받지 못했을 경우 액션 타임아웃 체크
203  rclcpp::Duration dur = node_->get_clock()->now() - start_time_;
204  // 액션의 타임아웃시 실패로 종료
205  if (dur > timeout_dur) {
206  RCLCPP_ERROR_STREAM(node_->get_logger(), "[" << name() << "] TIMEOUT");
207  action_cancel();
208  return BT::NodeStatus::FAILURE;
209  }
210  // 루프 슬립
211  loop_rate.sleep();
212  }
213 
214  // 액션 결과를 받았을 경우 액션 결과에 따라 리턴
215  if (action_status_ == RESULT_AVAILABLE) {
216  switch (result_.code) {
217  case rclcpp_action::ResultCode::SUCCEEDED:
218  RCLCPP_INFO_STREAM(node_->get_logger(), name() << " : SUCCEEDED");
219  return on_success();
220 
221  case rclcpp_action::ResultCode::ABORTED:
222  RCLCPP_WARN_STREAM(node_->get_logger(), name() << " : ABORTED");
223  return on_aborted();
224 
225  case rclcpp_action::ResultCode::CANCELED:
226  RCLCPP_WARN_STREAM(node_->get_logger(), name() << " : CANCELED");
227  return on_cancelled();
228 
229  default:
230  throw std::logic_error("ROSActionClientNode::Tick: invalid status value");
231  }
232  }
233  } catch (std::exception & ex) {
234  RCLCPP_ERROR_STREAM(node_->get_logger(), "[" << name() << "] Exception - " << ex.what());
235  action_cancel();
236  return BT::NodeStatus::FAILURE;
237  }
238  RCLCPP_ERROR_STREAM(
239  node_->get_logger(),
240  "[" << name() << "] ROS error or Action was not activated.");
241  return BT::NodeStatus::FAILURE;
242  }
243 
244  void halt() final
245  {
246  RCLCPP_WARN_STREAM(node_->get_logger(), name() << " Halted.");
247  // Halt 내부에서 Action을 건드리거나 Spin을 돌리게되면, 충돌이 일어날 수 있다.
248  // 따라서, Tcik이 돌고 있지 않을때만 spin이 돌지 않기 때문에, RUNNING 체크를 한다.
249  if (action_status_ == REQUESTED && status() != BT::NodeStatus::RUNNING) {
250  RCLCPP_INFO_STREAM(node_->get_logger(), name() << "Waiting for receive the action result");
251 
252  rclcpp::WallRate loop_rate(std::chrono::milliseconds(static_cast<int>(loop_rate_ms_)));
253  int wait_cnt = 0;
254 
255  rclcpp::Duration timeout_dur(total_action_timeout_);
256  while (action_status_ == REQUESTED) {
257  if (wait_cnt % (1000 / loop_rate_ms_) == 0) {
258  RCLCPP_INFO_STREAM(
259  node_->get_logger(),
260  name() << "Waiting... " + std::to_string(wait_cnt * loop_rate_ms_ / 1000) + "s.");
261  }
262  // ROS spin
263  rclcpp::spin_some(node_);
264  loop_rate.sleep(); // 루프 슬립
265  wait_cnt++;
266  // 타임 아웃 (처음 action을 보낸 시간으로 부터 timeout 체크)
267  rclcpp::Duration dur = node_->get_clock()->now() - start_time_;
268  if (dur > timeout_dur) {
269  RCLCPP_ERROR_STREAM(node_->get_logger(), "[" << name() << "] HALT TIMEOUT");
270  action_status_ = UNKNOWN;
271  return;
272  }
273  }
274 
275  RCLCPP_INFO_STREAM(node_->get_logger(), name() << "Receive the action result!!");
276  switch (result_.code) {
277  case rclcpp_action::ResultCode::SUCCEEDED:
278  RCLCPP_INFO_STREAM(node_->get_logger(), name() << " : SUCCEEDED");
279  case rclcpp_action::ResultCode::ABORTED:
280  RCLCPP_INFO_STREAM(node_->get_logger(), name() << " : ABORTED");
281  case rclcpp_action::ResultCode::CANCELED:
282  RCLCPP_INFO_STREAM(node_->get_logger(), name() << " : CANCELED");
283  default:
284  RCLCPP_INFO_STREAM(node_->get_logger(), name() << " : UNKNOWN");
285  }
286  }
287 
288  AsyncActionNode::halt();
289  }

294  // 액션 클라이언트를 생성합니다. 자동으로 생성되지 않으므로,
295  // 생성자 혹은 on_init()에서 초기화해야합니다.
296  bool create_action_client(
297  std::string action_name, std::chrono::milliseconds total_action_timeout,
298  std::chrono::milliseconds request_timeout = std::chrono::milliseconds(500))
299  {
300  try {
301  action_name_ = action_name;
302  total_action_timeout_ = total_action_timeout;
303  request_timeout_ = request_timeout;
304 
305 // bool on_blackboard = config().blackboard->get<decltype(action_client_)>(
306 // action_name_, action_client_);
307 
308 // if (on_blackboard == false) {
309  // Now that we have the ROS node to use, create the action client for this BT action
310  action_client_ = rclcpp_action::create_client<ActionT>(node_, action_name_);
311 // config().blackboard->set<decltype(action_client_)>(action_name_, action_client_);
312 // }
313  } catch (std::exception & ex) {
314  RCLCPP_ERROR(
315  node_->get_logger(), "Failed to create \"%s\" action client",
316  action_name_.c_str());
317  action_status_ = UNKNOWN;
318  return false;
319  }
320 
321  // 계속하기 전에 서버가 실제로 있는지 확인합니다.
322  try {
323  RCLCPP_INFO_STREAM(
324  node_->get_logger(), "Waiting for \"" << action_name_ <<
325  "\" action server. (request timeout : " << request_timeout_.count() << "ms)");
326  if (!action_client_->wait_for_action_server(request_timeout_)) {
327  RCLCPP_ERROR(
328  node_->get_logger(), "Failed to connect \"%s\" action server.", action_name_.c_str());
329  action_status_ = UNKNOWN;
330  return false;
331  }
332  return true;
333  } catch (std::exception & ex) {
334  RCLCPP_ERROR(
335  node_->get_logger(), "Failed to wait_for_action_server in \"%s\" action",
336  action_name_.c_str());
337  action_status_ = UNKNOWN;
338  return false;
339  }
340  }
341 
342  bool action_cancel()
343  {
344  if (should_cancel_goal()) {
345  try {
346  auto future_cancel = action_client_->async_cancel_goal(goal_handle_);
347  if (rclcpp::spin_until_future_complete(node_, future_cancel, request_timeout_) !=
348  rclcpp::FutureReturnCode::SUCCESS)
349  {
350  RCLCPP_ERROR(
351  node_->get_logger(), "Failed to cancel action server for %s", action_name_.c_str());
352  action_status_ = UNKNOWN;
353  return false;
354  }
355  RCLCPP_INFO_STREAM(node_->get_logger(), "[" << name() << "] Action canceled.");
356  action_status_ = NONE;
357  } catch (std::exception & ex) {
358  RCLCPP_ERROR(
359  node_->get_logger(), "Failed to async_cancel_goal in \"%s\" action",
360  action_name_.c_str());
361  action_status_ = UNKNOWN;
362  return false;
363  }
364  }
365  return true;
366  }

368  enum ActionStatus
369  {
370  NONE = 0,
371  REQUESTED = 1,
372  RESULT_AVAILABLE = 2,
373  UNKNOWN = 3
374  };
375 
376  // 모든 ROS 작업에 사용될 노드
377  rclcpp::Node::SharedPtr node_;
378  int loop_rate_ms_{10};
379 
380  // Action 메세지 이름
381  std::string action_name_;
382  // 액션의 goal cancel 등의 요청을 했을때 응답이 오는 때까지의 타임아웃
383  std::chrono::milliseconds request_timeout_;
384  // 액션 전체가 실행되고 끝나는데까지의 타임아웃
385  std::chrono::milliseconds total_action_timeout_;
386 
387  // Action Client
388  typename std::shared_ptr<rclcpp_action::Client<ActionT>> action_client_;
389  // Action Goal gandle
390  typename rclcpp_action::ClientGoalHandle<ActionT>::SharedPtr goal_handle_;
391  // Action 서버로 요청하는 Goal
392  typename ActionT::Goal goal_;
393  // Action을 call하고 시작된 시간
394  rclcpp::Time start_time_;
395 
396  // Action Status
397  ActionStatus action_status_{NONE};
398  // Action을 call 하지 않았으면 NONE
399  // Action call을 실행했으면 REQUESTED
400  // 결과를 받으면 RESULT_AVAIABLE
401 
402  // Action의 결과 (메세지 결과 result_.result)
403  typename rclcpp_action::ClientGoalHandle<ActionT>::WrappedResult result_;
404 
405  // on_wait_for_result의 Handle
406  std::future<BT::NodeStatus> on_wait_future_handle_;
407 
408  bool skip_action_cancel_{false};
409  bool enable_log_{true};
410 
411 private:
412  // 액션의 Result Callback으로 사용됩니다 사용자가 정의할 수 없습니다.
413  // 결과는 result_ 변수에서 확인할 수 있고,
414  // 그때 결과 처리는 on_success(), on_aborted(), on_cancelled()에서 처리할 수 있습니다.
415  void result_callback(
416  const typename rclcpp_action::ClientGoalHandle<ActionT>::WrappedResult & result)
417  {
418  RCLCPP_INFO_STREAM(node_->get_logger(), "[" << name() << "] it received result.");
419  // Goal ID가 일치하지 않으면 이전 Goal이 이 콜백을 호출하므로 결과를 무시해야한다.
420  // 일치하는 경우 처리해야 함(Abort 포함)
421  if (this->goal_handle_->get_goal_id() == result.goal_id) {
422  action_status_ = RESULT_AVAILABLE;
423  result_ = result;
424  } else {
425  RCLCPP_ERROR_STREAM(
426  node_->get_logger(), "[" << name() << "] it received result with wrong goal id.");
427  action_status_ = UNKNOWN;
428  }
429  }
430 
431  // 액션을 호출하는 함수입니다. on_tick()함수가 실행된 이후 자동으로 실행 됩니다.
432  void call_action_goal()
433  {
434  RCLCPP_INFO_STREAM(node_->get_logger(), "[" << name() << "] Call Action goal : " << status());
435 
436  // Goal Option 세팅 (Callback 함수)
437  auto send_goal_options = typename rclcpp_action::Client<ActionT>::SendGoalOptions();
438 // send_goal_options.goal_response_callback =
439 // std::bind(&ROSActionClientNode<ActionT>::response_callback, this, _1);
440  send_goal_options.feedback_callback =
441  std::bind(
442  &ROSActionClientNode<ActionT>::feedback_callback,
443  this,
444  std::placeholders::_1,
445  std::placeholders::_2);
446  send_goal_options.result_callback =
447  std::bind(&ROSActionClientNode<ActionT>::result_callback, this, std::placeholders::_1);
448 
449  // Action 호출합니다.
450  auto goal_handle_future = action_client_->async_send_goal(goal_, send_goal_options);
451  RCLCPP_INFO_STREAM(
452  node_->get_logger(), "[" << name() << "] Call async action : " << status());
453 
454  // Action의 future가 공유상태인지 확인합니다.
455  if (goal_handle_future.valid() == false) {
456  RCLCPP_ERROR_STREAM(
457  node_->get_logger(), "[" << name() << "] Failed to get action future");
458  action_status_ = UNKNOWN;
459  throw std::runtime_error(action_name_ + " : failed to get action future");
460  }
461 
462  // Action의 Goal의 결과가 나오기를 기다립니다.
463  // 지정된 ROS_TIMEOUT이 경과하거나 결과를 사용할 수 있을 때까지 차단합니다.
464  // 반환 Value는 결과의 상태를 나타냅니다.
465  auto future_result =
466  rclcpp::spin_until_future_complete(node_, goal_handle_future, request_timeout_);
467  if (future_result != rclcpp::FutureReturnCode::SUCCESS) {
468  action_status_ = UNKNOWN;
469  throw std::runtime_error(action_name_ + " : send_goal failed");
470  }
471  RCLCPP_INFO_STREAM(
472  node_->get_logger(),
