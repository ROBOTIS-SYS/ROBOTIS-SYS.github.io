44 BT::NodeStatus IsManipulatorTargetInRangeCondition::tick()
45 {
46  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
47  RCLCPP_INFO_STREAM(get_logger(), "<TICK> IsManipulatorTargetInRangeCondition");
48  }
49 
50  std::string manipulator_namespace = "gc_manipulator";
51  std::string service_name = manipulator_namespace +
52  "/manipulator_controller/workspace";
53 
54  gc_workspace_client_ =
55  node_->create_client<manipulator_interfaces::srv::GcWorkspace>(service_name);
56 
57  recved_flag_ = false;
58 
59  std::string msg_name;
60  if (getInput<std::string>("target_msg_name", msg_name)) {
61  std::string manipulator_namespace = "gc_manipulator/";
62  // create msg sub.
63  tracking_target_sub_ = node_->create_subscription<geometry_msgs::msg::PoseStamped>(
64  manipulator_namespace + msg_name, 0,
65  std::bind(
66  &IsManipulatorTargetInRangeCondition::tracking_target_callback, this,
67  std::placeholders::_1));
68  } else {
69  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
70  RCLCPP_INFO_STREAM(get_logger(), "tracking target msg name is empty!!, Check Pose");
71  }
72 
73  geometry_msgs::msg::Pose target_pose;
74  std::string target_frame_id;
75  if (!getInput<geometry_msgs::msg::Pose>("target_pose", target_pose)) {
76  RCLCPP_ERROR_STREAM(get_logger(), "Fail to get target pose");
77  return BT::NodeStatus::FAILURE;
78  }
79  if (!getInput<std::string>("target_pose_frame_id", target_frame_id)) {
80  RCLCPP_ERROR_STREAM(get_logger(), "Fail to get target pose frame id");
81  return BT::NodeStatus::FAILURE;
82  }
83 
84  target_pose_.header.stamp = node_->now();
85  target_pose_.header.frame_id = target_frame_id;
86  target_pose_.pose = target_pose;
87  recved_flag_ = true;
88  }
89 
90  // wait to detected ev button data sub.
91  double check_time = 5.0; // s
92  double loop_rate = 10.0; // ms
93  int loop_cnt = 0;
94  check_time *= 1000.0;
95  rclcpp::WallRate loopRate(std::chrono::milliseconds(static_cast<int>(loop_rate)));
96  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
97  RCLCPP_INFO_STREAM(get_logger(), "Waiting for detected ev button data");
98  }
99  while (rclcpp::ok() && recved_flag_ == false) {
100  // sleep
101  loopRate.sleep();
102  if ((loop_rate * ++loop_cnt) >= check_time) {
103  break;
104  }
105  }
106 
107  // fail to recv data
108  if (recved_flag_ == false) {
109  RCLCPP_ERROR_STREAM(get_logger(), "Fail to subscribe the detected ev button data");
110  return BT::NodeStatus::FAILURE;
111 // return BT::NodeStatus::SUCCESS;
112  } else {
113  tracking_target_sub_ = nullptr;
114  return send_service_client();
115  }
116 }

126 BT::NodeStatus IsManipulatorTargetInRangeCondition::send_service_client()
127 {
128  // check server
129  if (!gc_workspace_client_->wait_for_service(std::chrono::seconds(5))) {
130  RCLCPP_ERROR_STREAM(
131  get_logger(),
132  "Gc Manipulator Workspace Service Server not available after waiting");
133  return BT::NodeStatus::FAILURE;
134  }
135 
136  geometry_msgs::msg::Point delta_position;
137  if (!getInput<geometry_msgs::msg::Point>("delta_position", delta_position)) {
138  RCLCPP_ERROR_STREAM(get_logger(), "tracking target delta positon is empty!!");
139  return BT::NodeStatus::FAILURE;
140  }
141 
142  // call service server
143  auto request = std::make_shared<manipulator_interfaces::srv::GcWorkspace::Request>();
144  request->header = target_pose_.header;
145  request->component_name = "tip";
146  request->pose = target_pose_.pose;
147  request->delta_position = delta_position;
148 
149  auto future = gc_workspace_client_->async_send_request(request);
150 
151  future.wait_for(std::chrono::seconds(1));
152  auto result = future.get();
153 
154  std::string output_result = "fail";
155  if (result->success == false) {
156  output_result = "fail";
157  RCLCPP_WARN_STREAM(
158  get_logger(), "The goal is failed / " << output_result);
159  } else {
160  if (result->distance.x == 0.0 && result->distance.y == 0.0 && result->distance.theta == 0.0) {
161  output_result = "tracking";
162  RCLCPP_WARN_STREAM(
163  get_logger(), "The goal is within the workspace. / " << output_result);
164  } else {
165  output_result = "move_base";
166  RCLCPP_WARN_STREAM(
167  get_logger(), "The goal is not within the workspace. / " << output_result);
168 
169  RCLCPP_WARN_STREAM(
170  get_logger(),
171  "Move to base : " << result->distance.x << ", " << result->distance.y << ", " <<
172  result->distance.theta);
173  }
174  }
175 
176  setOutput<geometry_msgs::msg::Pose2D>("base_goal", result->distance);
177  setOutput<std::string>("moved_frame_id", result->frame_id);
178  setOutput<geometry_msgs::msg::Pose>("moved_pose", result->moved_pose);
179  setOutput<std::string>("result", output_result);
180 
181  return BT::NodeStatus::SUCCESS;
182 }
