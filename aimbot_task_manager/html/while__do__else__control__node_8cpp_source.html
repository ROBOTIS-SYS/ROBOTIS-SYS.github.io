37 {

39 
40  if (children_count != 2 && children_count != 3) {
41  throw std::logic_error("WhileDoElseControl must have either 2 or 3 children");
42  }
43 
44  setStatus(BT::NodeStatus::RUNNING);
45 
46  while (current_child_idx_ < children_count) {
47  TreeNode * condition_node = children_nodes_[0];
48  TreeNode * child_node = children_nodes_[current_child_idx_];
49 
50  if (current_child_idx_ == 1) { // do child
51  // check condition
52  const BT::NodeStatus condition_status = condition_node->executeTick();
53 
54  if (condition_status == BT::NodeStatus::FAILURE) {
55  child_node->halt();
56  condition_node->halt();
57  current_child_idx_++;
58  } else {
59  const BT::NodeStatus status = child_node->executeTick();
60 
61  switch (status) {
62  case BT::NodeStatus::SUCCESS:
63  case BT::NodeStatus::FAILURE: {
64  halt();
65  return status;
66  }
67 
68  case BT::NodeStatus::RUNNING: {
69  return status;
70  }
71 
72  default: {
73  halt();
74  return BT::NodeStatus::FAILURE;
75  }
76  } // end switch
77  }
78  } else if (current_child_idx_ == 2) { // recovery child
79  const BT::NodeStatus status = child_node->executeTick();
80 
81  switch (status) {
82  case BT::NodeStatus::SUCCESS:
83  case BT::NodeStatus::FAILURE: {
84  halt();
85  return status;
86  }
87 
88  case BT::NodeStatus::RUNNING: {
89  return status;
90  }
91 
92  default: {
93  halt();
94  return BT::NodeStatus::FAILURE;
95  }
96  }
97  }
98  } // end while loop
