62 {
63  std::cout << std::endl;
64  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
65  RCLCPP_INFO_STREAM(get_logger(), "<TICK> SetParameterAction");
66  }
67 
68  // input port
69  std::vector<rclcpp::Parameter> parameters;
70  if (!getInput<std::vector<rclcpp::Parameter>>("parameters", parameters)) {
71  RCLCPP_ERROR_STREAM(
72  get_logger(), "Failed to read parameters to set : " << parameters.size());
73  return BT::NodeStatus::FAILURE;
74  }
75 
76  if (!getInput<std::string>("node_name", remote_node_name_)) {
77  RCLCPP_ERROR(
78  get_logger(), "[SetParameterAction] parameter server's node name is required.");
79  return BT::NodeStatus::FAILURE;
80  }
81 
82  // Init Param service
83  try {
84  if (param_service_ == nullptr) {
85  std::string blackboard_key = "param_client_" + remote_node_name_;
86  bool on_blackboard =
87  config().blackboard->get<decltype(param_service_)>(blackboard_key, param_service_);
88 
89  if (on_blackboard == false) {
90  param_service_ =
91  std::make_shared<rclcpp::AsyncParametersClient>(node_, remote_node_name_);
92  config().blackboard->set<decltype(param_service_)>(blackboard_key, param_service_);
93  }
94  }
95  } catch (std::exception & ex) {
96  RCLCPP_ERROR_STREAM(get_logger(), "Init error : " << ex.what());
97  return BT::NodeStatus::FAILURE;
98  }
99 
100  // Wait Param service : check param service is available
101  bool result = param_service_->wait_for_service(std::chrono::milliseconds(ROS_TIMEOUT));
102  if (isHaltRequested()) {
103  return BT::NodeStatus::IDLE;
104  }
105 
106  if (result == false) {
107  RCLCPP_ERROR(get_logger(), "Set param service is not available");
108  return BT::NodeStatus::FAILURE;
109  }
110 
111  result = set_parameters(parameters);
112  if (isHaltRequested()) {
113  return BT::NodeStatus::IDLE;
114  }
115 
116  return result ? BT::NodeStatus::SUCCESS : BT::NodeStatus::FAILURE;
117 }
