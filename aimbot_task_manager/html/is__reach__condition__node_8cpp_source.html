76 BT::NodeStatus IsReachCondition::tick()
77 {
78  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
79  RCLCPP_INFO_STREAM(get_logger(), "<TICK> IsReachCondition");
80  }
81  std::string goal_reach;
82  geometry_msgs::msg::Pose2D goal_pose;
83 
84  // use reached flag
85  if (getInput<std::string>("reached", goal_reach)) {
86  if (goal_reach == "true") {
87  std::cout << "[ Is Reach: YES ]" << std::endl;
88  return BT::NodeStatus::SUCCESS;
89  }
90  }
91 
92  // check diff between current pose and goal pose
93  if (!getInput<geometry_msgs::msg::Pose2D>("goal", goal_pose)) {
94  // no goal pose
95  return BT::NodeStatus::FAILURE;
96  }
97 
98  double tolerance = 0.2;
99  getInput<double>("threshold", tolerance);
100  geometry_msgs::msg::Pose2D current_pose;
101  bool result = get_current_pose(current_pose);
102 
103  // fail to get current pose
104  if (result == false) {
105  RCLCPP_ERROR_STREAM(get_logger(), "[IsReachCondition] failed to get current pose");
106  return BT::NodeStatus::FAILURE;
107  }
108 
109  double diff_x = goal_pose.x - current_pose.x;
110  double diff_y = goal_pose.y - current_pose.y;
111  double diff_theta = goal_pose.theta - current_pose.theta;
112  double diff =
113  sqrt(diff_x * diff_x + diff_y * diff_y + diff_theta * diff_theta);
114 
115  if (diff < tolerance) {
116  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
117  RCLCPP_INFO_STREAM(get_logger(), "[IsReachCondition] is reach : OK");
118  }
119  return BT::NodeStatus::SUCCESS;
120  } else {
121  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
122  RCLCPP_INFO_STREAM(
123  get_logger(),
124  "[IsReachCondition] is reach : Failed, tol : " << tolerance << ", diff : " << diff);
125  }
126  return BT::NodeStatus::FAILURE;
127  }
128 
129  return BT::NodeStatus::FAILURE;
130 }

132 bool IsReachCondition::get_current_pose(
133  geometry_msgs::msg::Pose2D & current_pose_2d)
134 {
135  try {
136  geometry_msgs::msg::TransformStamped base_link_transform;
137  std::string map_frame_id;
138  std::string base_frame_id = "base_link";
139  if (!getInput<std::string>("map_frame", map_frame_id)) {
140  map_frame_id = "map";
141  }
142 
143  base_link_transform = tf_buffer_->lookupTransform(
144  map_frame_id, base_frame_id, rclcpp::Time(0),
145  rclcpp::Duration::from_seconds(0.5));
146 
147  auto dur = node_->get_clock()->now() - base_link_transform.header.stamp;
148  RCLCPP_WARN_STREAM(
149  get_logger(),
150  "[" << name() << "] Transform diff : " << dur.seconds() << "s");
151 
152  geometry_msgs::msg::Pose current_pose;
153  current_pose.position.x = base_link_transform.transform.translation.x;
154  current_pose.position.y = base_link_transform.transform.translation.y;
155  current_pose.position.z = base_link_transform.transform.translation.z;
156 
157  current_pose_2d.x = current_pose.position.x;
158  current_pose_2d.y = current_pose.position.y;
159 
160  current_pose.orientation.w = base_link_transform.transform.rotation.w;
161  current_pose.orientation.x = base_link_transform.transform.rotation.x;
162  current_pose.orientation.y = base_link_transform.transform.rotation.y;
163  current_pose.orientation.z = base_link_transform.transform.rotation.z;
164 
165  tf2::Quaternion tf_q;
166  tf2::convert(current_pose.orientation, tf_q);
167  tf2::Matrix3x3 rotation_mat(tf_q);
168  double r, p, y;
169  rotation_mat.getRPY(r, p, y);
170 
171  current_pose_2d.theta = y;
172  } catch (tf2::TransformException & ex) {
173  RCLCPP_ERROR_STREAM(
174  get_logger(), "Failed to get the current pose : " << ex.what());
175  current_pose_2d = geometry_msgs::msg::Pose2D();
176 
177  return false;
178  }
179 
180  return true;
181 }
