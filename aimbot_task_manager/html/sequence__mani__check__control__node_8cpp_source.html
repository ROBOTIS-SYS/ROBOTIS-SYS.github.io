51 {

53

56  while (current_child_idx_ < children_count) {
57 // // manipulator status check

64 // dxl_hw_err_ = false;
65 // return BT::NodeStatus::FAILURE;
66 // }
67 
68  // tick
69  TreeNode * current_child_node = children_nodes_[current_child_idx_];
70  const BT::NodeStatus child_status = current_child_node->executeTick();
71 
72  switch (child_status) {
73  case BT::NodeStatus::RUNNING:
74  {
75  setOutput<std::string>("is_mani_ok", "true");
76  return child_status;
77  }
78  case BT::NodeStatus::FAILURE:
79  {
80  // Reset on failure
81  haltChildren();
82  current_child_idx_ = 0;
83  setOutput<std::string>("is_mani_ok", "true");
84  return child_status;
85  }
86  case BT::NodeStatus::SUCCESS:
87  {
88  current_child_idx_++;
89  }
90  break;
91 
92  case BT::NodeStatus::IDLE:
93  {
94  setOutput<std::string>("is_mani_ok", "true");
95  throw std::logic_error("A child node must never return IDLE");
96  }
97  } // end switch
98  } // end while loop
99 
100  // The entire while loop completed. This means that all the children returned SUCCESS.
101  if (current_child_idx_ == children_count) {
102  haltChildren();
103  current_child_idx_ = 0;
104  }
105  setOutput<std::string>("is_mani_ok", "true");
106  return BT::NodeStatus::SUCCESS;
107 }

109 void SequenceManiCheckControl::dynamixel_state_callback(
110  const dynamixel_interfaces::msg::DynamixelState::SharedPtr msg)
111 {
112  if (status() != BT::NodeStatus::RUNNING) {
113  return;
114  }
115 
116  static int comm_err_cnt = 0;
117  static int hw_err_cnt[NUM_OF_DXL] = {0, };
118 
119  // check comm error
120  if (msg->comm_state != 0 && comm_err_cnt > 5) { // 500ms
121  RCLCPP_ERROR_STREAM(get_logger(), "DXL COMM ERROR [" << msg->comm_state << "]");
122  comm_err_cnt = 0;
123 
124  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
125  RCLCPP_INFO_STREAM(get_logger(), "[SequenceManiCheckControl] Error in DXL");
126  }
127  haltChildren();
128  current_child_idx_ = 0;
129 
130  setOutput<std::string>("is_mani_ok", "false");
131  setStatus(BT::NodeStatus::FAILURE);
132 
133  } else if (msg->comm_state != 0) {
134  comm_err_cnt++;
135  } else if (msg->comm_state == 0) {
136  comm_err_cnt = 0;
137  }
138 
139  // check dxl hw error
140  for (size_t i = 0; i < msg->id.size(); i++) {
141  if (msg->dxl_hw_state.at(i) != 0 && hw_err_cnt[i] > 5) {
142  RCLCPP_ERROR_STREAM(
143  get_logger(), "DXL HW ERROR [id: " << msg->id.at(i) <<
144  "], [" << msg->dxl_hw_state.at(i) << "]");
145  hw_err_cnt[i] = 0;
146 
147  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
148  RCLCPP_INFO_STREAM(get_logger(), "[SequenceManiCheckControl] Error in DXL");
149  }
150  haltChildren();
151  current_child_idx_ = 0;
152 
153  setOutput<std::string>("is_mani_ok", "false");
154  setStatus(BT::NodeStatus::FAILURE);
155 
156  } else if (msg->dxl_hw_state.at(i) != 0) {
157  hw_err_cnt[i]++;
158  } else if (msg->dxl_hw_state.at(i) == 0) {
159  hw_err_cnt[i] = 0;
160  }
161  }
162 }
