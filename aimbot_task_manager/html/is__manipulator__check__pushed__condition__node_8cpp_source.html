43 BT::NodeStatus IsManipulatorCheckPushedCondition::tick()
44 {
45  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
46  RCLCPP_INFO_STREAM(get_logger(), "<TICK> IsManipulatorCheckPushedCondition");
47  }
48 
49  // create msg sub.
50  std::string msg_name = "/gc_manipulator/detection_manager/ev_detected_btn_arr";
51  ev_detected_btn_arr_sub_ =
52  node_->create_subscription<perception_interfaces::msg::DetectedButtonArray>(
53  msg_name, 0,
54  std::bind(
55  &IsManipulatorCheckPushedCondition::ev_detected_btn_arr_callback, this,
56  std::placeholders::_1));
57 
58 
59  if (!getInput<std::string>("target_ev_button", target_ev_btn)) {
60  RCLCPP_ERROR_STREAM(get_logger(), "target_ev_button is empty!!");
61  return BT::NodeStatus::FAILURE;
62  }
63 
64  // wait to detected ev button data sub.
65  double check_time = 1.0; // s
66  double loop_rate = 10.0; // ms
67  int loop_cnt = 0;
68 
69  check_time *= 1000.0;
70 
71  is_checked = false;
72  recved_flag = false;
73 
74  rclcpp::WallRate loopRate(std::chrono::milliseconds(static_cast<int>(loop_rate)));
75  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
76  RCLCPP_INFO_STREAM(get_logger(), "Waiting for detected ev button data");
77  }
78  while (rclcpp::ok() && recved_flag == false) {
79  // sleep
80  loopRate.sleep();
81  if ((loop_rate * ++loop_cnt) >= check_time) {
82  RCLCPP_WARN_STREAM(get_logger(), "timeout");
83  break;
84  }
85  }
86 
87  RCLCPP_WARN_STREAM(
88  get_logger(),
89  "The target button [ " << target_ev_btn << " ] state : " <<
90  (is_checked ? "Pushed" : "Not Pushed"));
91 
92  return is_checked ? BT::NodeStatus::SUCCESS : BT::NodeStatus::FAILURE;
93 }

104  recved_flag = true;
105  }
