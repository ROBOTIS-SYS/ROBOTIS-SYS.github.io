91  // INIT : Set Service
92  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
93  RCLCPP_INFO_STREAM(get_logger(), "Init : ClearCostmapAction - " << name());
94  }
95 
96  try {
97  std::string section_name = "navigation";
98  if (node_->has_parameter(section_name + ".clear_costmap_service") == false) {
99  node_->declare_parameter<std::vector<std::string>>(
100  section_name + ".clear_costmap_service",

102  "/global_costmap/clear_entirely_global_costmap"});
103  }
104 
105  node_->get_parameter<std::vector<std::string>>(
106  section_name + ".clear_costmap_service", clear_costmap_services_);
107 
108  if (clear_costmap_clients_.size() != clear_costmap_services_.size()) {
109  clear_costmap_clients_.clear();
110  for (auto service_name : clear_costmap_services_) {
111  ClearCostmapPtr clear_costmap_client;
112  bool on_blackboard =
113  config().blackboard->get<ClearCostmapPtr>(service_name, clear_costmap_client);
114  if (on_blackboard == false) {
115  clear_costmap_client =
116  node_->create_client<nav2_msgs::srv::ClearEntireCostmap>(service_name);
117  config().blackboard->set<ClearCostmapPtr>(service_name, clear_costmap_client);
118  }
119  clear_costmap_clients_.push_back(clear_costmap_client);
120  }
121  }
122  } catch (std::exception & ex) {
123  RCLCPP_ERROR_STREAM(get_logger(), "Init error : " << ex.what());
124  return BT::NodeStatus::FAILURE;
125  }
126 
127  // Wait Service
128  for (auto client : clear_costmap_clients_) {
129  bool result = client->wait_for_service(std::chrono::milliseconds(ROS_TIMEOUT));
130  if (isHaltRequested()) {
131  return BT::NodeStatus::IDLE;
132  }
133  if (result == false) {
134  return BT::NodeStatus::FAILURE;
135  }
136  }
137 
138  shared_futures_.clear();
139 
140  // Clear COSTMAP
141  bool result = clear_costmap();
142 
143  if (isHaltRequested()) {
144  return BT::NodeStatus::IDLE;
145  }
146 
147  if (result == true) {
148  publish_tree_status(name(), "코스트맵 초기화에 성공했습니다.");
149  } else {
150  publish_tree_status(name(), "코스트맵 초기화에 실패했습니다.");
151  }
152 
153  return result ? BT::NodeStatus::SUCCESS : BT::NodeStatus::FAILURE;
154 }
