59  // [srv client] ev detected btn massage name

61 
62  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
63  RCLCPP_INFO_STREAM(get_logger(), service_name.c_str() << " service server");
64  }
65 
66  log_pub_ = node_->create_publisher<aimbot_log_interfaces::msg::Log>("/aimbot_log", 10);
67 
68  // get input
69  std::string target_ev_button;
70  if (!getInput<std::string>("target_ev_button", target_ev_button)) {
71  RCLCPP_ERROR_STREAM(get_logger(), "target_ev_button is empty!!");
72  return BT::NodeStatus::FAILURE;
73  }
74  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
75  RCLCPP_INFO_STREAM(get_logger(), "target_ev_button : " << target_ev_button);
76  }
77 
78  // check server
79 
80  if (!srv_client_->wait_for_service(std::chrono::milliseconds(ROS_TIMEOUT))) {
81  RCLCPP_ERROR_STREAM(
82  get_logger(),
83  "Detection Manager Service Server not available after waiting");
84  return BT::NodeStatus::FAILURE;
85  }
86 
87  // call service server
88  auto request = std::make_shared<perception_interfaces::srv::DetectedButton::Request>();
89  request->target_button_name = target_ev_button;
90 
91  auto future = srv_client_->async_send_request(request);
92  future.wait_for(std::chrono::seconds(1));
93  auto result = future.get();
94 
95  std::string output_str = result->pose_msg_name;
96  bool camera_status = result->camera_status;
97  if (camera_status) {
98  publish_log("SUCCESS!!!");
99  } else {
100  publish_log(
101  "Fail to set target button.",
102  aimbot_log_interfaces::msg::Log::LEVEL_DEBUG,
103  aimbot_log_interfaces::msg::Log::TYPE_ERROR);
104  }
105 
106 // std::string output_str = "/test_tracking"; // for test
107 
108  setOutput<std::string>("target_msg_name", output_str);
109  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
110  RCLCPP_INFO_STREAM(get_logger(), "[output] target_msg_name : " << output_str);
111  }
112 
113  setOutput<bool>("camera_status", camera_status);
114  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
115  RCLCPP_INFO_STREAM(get_logger(), "[output] camera_status : " << camera_status);
116  }
117  if (camera_status == false) {
118  RCLCPP_ERROR_STREAM(get_logger(), "[Zed camera] Failed to read image");
119  return BT::NodeStatus::FAILURE;
120  }
121 
122  return output_str.empty() ? BT::NodeStatus::FAILURE : BT::NodeStatus::SUCCESS;
123 }
