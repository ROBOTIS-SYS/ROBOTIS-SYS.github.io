36 BT::NodeStatus PursuitPathAction::on_tick()
37 {
38  bool result_blackboard = config().blackboard->get<bool>("no_cancel_goal", skip_action_cancel_); // NOLINT
39  if (result_blackboard == false) {
40  skip_action_cancel_ = false;
41  }
42 
43  if (!getInput<nav_msgs::msg::Path>("path", goal_.path)) {
44  RCLCPP_ERROR_STREAM(node_->get_logger(), "path is empty!");
45  }
46 
47  if (!getInput<std::string>("controller_id", goal_.controller_id)) {
48  RCLCPP_ERROR_STREAM(node_->get_logger(), "controller id is empty!");
49  }
50 
51  double time;
52  if (!getInput<double>("server_timeout", time)) {
53  time = 1800;
54  RCLCPP_ERROR_STREAM(node_->get_logger(), "server_timeout is empty!");
55  }
56 
57  std::string action_name = "pursuit_path";
58 
59  std::chrono::milliseconds action_timeout = std::chrono::milliseconds(
60  static_cast<int>(time * 1000) + 1000);
61 
62  // Create Action [.action]
63  if (!create_action_client(action_name, action_timeout, action_timeout)) {
64  return BT::NodeStatus::FAILURE;
65  }
66 
67  return BT::NodeStatus::RUNNING;
68 }

70 BT::NodeStatus PursuitPathAction::on_wait_for_result()
71 {
72  // Check if the goal has been updated
73  if (config().blackboard->get<bool>("path_updated")) {
74  // Reset the flag in the blackboard
75  config().blackboard->set("path_updated", false);
76 
77  // Grab the new goal and set the flag so that we send the new goal to
78  // the action server on the next loop iteration
79  getInput("path", goal_.path);
80  goal_updated_ = true;
81  return status();
82  }
83  return status();
84 }
