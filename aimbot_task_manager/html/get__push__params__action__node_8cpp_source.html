98  RCLCPP_ERROR_STREAM(get_logger(), "[GetPushParamsAction] missing required input floor");

104  }
105

107  if (!getInput<double>("default_push_thres", push_thres_)) {
108  RCLCPP_ERROR_STREAM(
109  get_logger(), "[GetPushParamsAction] missing required input default push_thres");

118  RCLCPP_ERROR_STREAM(
119  get_logger(), "[GetPushParamsAction] missing required input default goal_from_target");
120  return BT::NodeStatus::FAILURE;

123  RCLCPP_ERROR_STREAM(
124  get_logger(), "[GetPushParamsAction] missing required input default target_offset");

126  }
127 
128  // Parsing
129  if (parseYaml(target_name, floor, dest)) {
130  // 안에서 yaml에 파라미터가 포함되어있으면 맴버변수를 변경해둠.
131  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
132  RCLCPP_INFO_STREAM(get_logger(), "[GetPushParamsAction] set params");
133  }
134  setOutput<double>("push_thres", push_thres_);
135  setOutput<double>("push_pos_x", push_pos_x_);
136  setOutput<geometry_msgs::msg::Point>("goal_from_target", goal_from_target_);
137  setOutput<geometry_msgs::msg::Point>("target_offset", target_offset_);
138  return BT::NodeStatus::SUCCESS;
139  } else {
140  setOutput<double>("push_thres", push_thres_);
141  setOutput<double>("push_pos_x", push_pos_x_);
142  setOutput<geometry_msgs::msg::Point>("goal_from_target", goal_from_target_);
143  setOutput<geometry_msgs::msg::Point>("target_offset", target_offset_);
144  return BT::NodeStatus::FAILURE;
145  }

150 {
151  try {
152  // 파일을 Parsing
153  YAML::Node node = YAML::LoadFile(file_path_.c_str());
154  for (YAML::iterator it = node.begin(); it != node.end(); it++) {
155  // room_name이 파일안에 있는지 확인. 없으면 if안에 들어가지 않으므로 아무것도 하지않음.
156  if (it->second["target_name"].as<std::string>() == target_name) {
157  auto dest_list_yaml = it->second["dest"].as<std::vector<std::string>>();
158  if (dest_list_yaml.empty()) {
159  auto floor_list_from_yaml = it->second["floor"].as<std::vector<std::string>>();
160  for (std::string floor_yaml : floor_list_from_yaml) {
161  // room_name이 같은 node안에서 현재 floor가 포함되어있는지 확인.
162  // 없으면 if안에 들어가지 않으므로 아무것도 하지않음.
163  if (floor_yaml == floor) {
164  // 파라미터 설정이 필요하므로 params를 받음.
165  YAML::Node params = it->second["params"];
166  for (YAML::iterator param_it = params.begin(); param_it != params.end();
167  param_it++)
168  {
169  // iterator로 돌리다가 해당 파라미터가 있을때 등록 없으면 등록안함.
170  std::string param_name = param_it->first.as<std::string>();
