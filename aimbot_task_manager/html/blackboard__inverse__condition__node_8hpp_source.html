75 template<typename T>
76 inline BT::NodeStatus BlackboardInverseConditionNode<T>::tick()
77 {
78  T value_A;
79  T value_B;
80  BT::NodeStatus default_return_status = BT::NodeStatus::FAILURE;
81 
82  // 첫 시작시 동작
83  if (status() == BT::NodeStatus::IDLE) {
84  setStatus(BT::NodeStatus::RUNNING);
85 
86  // 포트가 비어 있으면 실패 처리
87  if (!getInput("value_A", value_A) || !getInput("value_B", value_B)) {
88  return BT::NodeStatus::FAILURE;
89  }
90 
91  // 두 포트 모두 있고 다른 경우 자식 노드 실행
92  if (getInput("value_A", value_A) && getInput("value_B", value_B) &&
93  IsNotSame(value_A, value_B))
94  {
95  return child_node_->executeTick();
96  } else {
97  getInput("return_on_match", default_return_status);
98  return default_return_status;
99  }
100  }
101 
102  // RUNNING시
103  return child_node_->executeTick();
104 }
