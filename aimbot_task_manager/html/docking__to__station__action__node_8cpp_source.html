50 BT::NodeStatus DockingToStationAction::tick()
51 {
52  std::cout << std::endl;
53  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
54  RCLCPP_INFO_STREAM(get_logger(), "<TICK> DockingToStationAction");
55  }
56 
57  double time;
58  if (!getInput<double>("server_timeout", time)) {
59  RCLCPP_ERROR_STREAM(node_->get_logger(), "server_timeout is unknown!");
60  RCLCPP_INFO_STREAM(node_->get_logger(), "server_timeout is set to default value");
61  time = 500.0;
62  }
63 
64  // Init ros
65  std::string action_name = "/docking";
66  if (docking_action_client_ == nullptr) {
67  try {
68  bool on_blackboard = config().blackboard->get<decltype(docking_action_client_)>(
69  action_name,
70  docking_action_client_);
71  if (on_blackboard == false) {
72  docking_action_client_ = rclcpp_action::create_client<DockingAction>(
73  node_->get_node_base_interface(), node_->get_node_graph_interface(),
74  node_->get_node_logging_interface(), node_->get_node_waitables_interface(),
75  action_name);
76 
77  config().blackboard->set<decltype(docking_action_client_)>(
78  action_name, docking_action_client_);
79 
80  if (
81  config().blackboard->get("enable_log", enable_log_) == false ||
82  enable_log_ == true)
83  {
84  RCLCPP_INFO_STREAM(
85  get_logger(),
86  "[DockingToStationAction]" << action_name.c_str() << " action server");
87  }
88  }
89  } catch (std::exception & ex) {
90  RCLCPP_ERROR_STREAM(get_logger(), "Init error : " << ex.what());
91  return BT::NodeStatus::FAILURE;
92  }
93  }
94 
95  // Wait Action Server : Make sure the server is actually there before continuing
96  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
97  RCLCPP_INFO(get_logger(), "Waiting for \"%s\" action server", action_name.c_str());
98  }
99 
100  // Wait Action Server
101  request_timeout_ = std::chrono::milliseconds(static_cast<int>(time));
102  if (!docking_action_client_->wait_for_action_server(request_timeout_)) {
103  RCLCPP_ERROR_STREAM(
104  get_logger(), "[DockingToStationAction] Action server not available after waiting");
105  return BT::NodeStatus::FAILURE;
106  }
107 
108  return send_action_client();
109 }

120 BT::NodeStatus DockingToStationAction::send_action_client()
121 {
122  using namespace std::placeholders;
123 
124  auto goal_msg = DockingAction::Goal();
125 
126  auto goal_opt = rclcpp_action::Client<DockingAction>::SendGoalOptions();
127  goal_opt.goal_response_callback =
128  std::bind(&DockingToStationAction::docking_response_callback, this, _1);
129  goal_opt.feedback_callback =
130  std::bind(&DockingToStationAction::docking_feedback_callback, this, _1, _2);
131  goal_opt.result_callback =
132  std::bind(&DockingToStationAction::docking_result_callback, this, _1);
133  auto goal_handle_future = docking_action_client_->async_send_goal(goal_msg, goal_opt);
134 
135  return BT::NodeStatus::RUNNING;
136 }

145  } else {
146  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
147  RCLCPP_INFO_STREAM(get_logger(), "Docking accepted by server, waiting for result");
148  }
149  setStatus(BT::NodeStatus::RUNNING);

164 void DockingToStationAction::docking_result_callback(
165  const rclcpp_action::ClientGoalHandle<DockingAction>::WrappedResult & result)
166 {
167  using RESULT = docking_interfaces::msg::Result;
168 
169  auto value = result.result->result_state.val;
170  if (value == docking_interfaces::msg::Result::SUCCESS) {
171  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
172  RCLCPP_INFO_STREAM(get_logger(), "[Docking] SUCCESS!!!");
173  }
174  setOutput<std::string>("result", "success");
175  setStatus(BT::NodeStatus::SUCCESS);
176  } else {
177  if
178  (value == RESULT::FAILURE_BUMPER_CONTACT || value == RESULT::FAILURE_DISTANCE_ERROR) {
179  if (config().blackboard->get("enable_log", enable_log_) == false || enable_log_ == true) {
180  RCLCPP_INFO_STREAM(get_logger(), "[Docking] FAIL :( RETRY : " << value);
181  }
182  setOutput<std::string>("result", "retry");
183  setStatus(BT::NodeStatus::SUCCESS);
184  } else {
185  RCLCPP_ERROR_STREAM(get_logger(), "[Docking] FAIL :( ERROR : " << value);
186  setOutput<std::string>("result", "fail");
187  setStatus(BT::NodeStatus::SUCCESS);
188  }
189  }
190 }
